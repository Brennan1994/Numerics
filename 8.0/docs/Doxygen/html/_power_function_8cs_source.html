The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

206 public double Function(double x)
207 {
208 // Validate parameters
209 if (_parametersValid == false)
210 ValidateParameters(new[] { Alpha, Beta, Xi, Sigma }, true);
211
212 // Check support
213 if (x <= Xi)
214 x = Xi + Tools.DoubleMachineEpsilon;
215 else if (x >= Maximum)
216 x = Maximum;
217
218 double y = 0;
219 if (IsInverse)
220 {
221 if (IsDeterministic == true || ConfidenceLevel < 0 || ConfidenceLevel > 1)
222 {
223 y = Math.Exp((Math.Log(x) - Math.Log(Alpha)) / Beta) + Xi;
224 }
225 else
226 {
227 y = Math.Exp((Math.Log(x) - Math.Log(Alpha) - _normal.InverseCDF(ConfidenceLevel)) / Beta) + Xi;
228 }
229 }
230 else
231 {
232 if (IsDeterministic == true || ConfidenceLevel < 0 || ConfidenceLevel > 1)
233 {
234 y = Math.Exp(Math.Log(Alpha) + Beta * Math.Log(x - Xi));
235 }
236 else
237 {
238 y = Math.Exp(Math.Log(Alpha) + Beta * Math.Log(x - Xi) + _normal.InverseCDF(ConfidenceLevel));
239 }
240 }
241 return y;
242 }

245 public double InverseFunction(double y)
246 {
247 // Validate parameters
248 if (_parametersValid == false)
249 ValidateParameters(new[] { Alpha, Beta, Xi, Sigma }, true);
250
251 double x = 0;
252 if (IsInverse)
253 {
254 if (IsDeterministic == true || ConfidenceLevel < 0 || ConfidenceLevel > 1)
255 {
256 x = Math.Exp(Math.Log(Alpha) + Beta * Math.Log(y - Xi));
257 }
258 else
259 {
260 x = Math.Exp(Math.Log(Alpha) + Beta * Math.Log(y - Xi) + _normal.InverseCDF(ConfidenceLevel));
261 }
262 }
263 else
264 {
265 if (IsDeterministic == true || ConfidenceLevel < 0 || ConfidenceLevel > 1)
266 {
267 x = Math.Exp((Math.Log(y) - Math.Log(Alpha)) / Beta) + Xi;
268 }
269 else
270 {
271 x = Math.Exp((Math.Log(y) - Math.Log(Alpha) - _normal.InverseCDF(ConfidenceLevel)) / Beta) + Xi;
272 }
273 }
274 if (x < Minimum) return Minimum;
275 if (x > Maximum) return Maximum;
276 return x;
277 }
