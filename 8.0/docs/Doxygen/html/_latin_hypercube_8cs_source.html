78 public static double[,] Random(int sampleSize, int dimension = 1, int seed = -1)
79 {
80
81 int N = sampleSize;
82 var lhs = new double[N, dimension];
83
84 // Create random number generators
85 var rndM = seed > 0 ? new Random(seed) : new Random();
86 var rnd = new Random[dimension, 2];
87 for (int i = 0; i < dimension; i++)
88 {
89 rnd[i, 0] = new Random(rndM.Next(1, dimension * 10000 + 1));
90 rnd[i, 1] = new Random(rndM.Next(1, dimension * 10000 + 1));
91 }
92
93 for (int col = 0; col < dimension; col++)
94 {
95 // Create N bins of equal probability,
96 // and randomly generate a number within that bin
97 var bins = new List<double>();
98 for (int i = 1; i <= N; i++)
99 bins.Add((rnd[col, 0].NextDouble() + i - 1d) / N);
100
101 // Sample random bin without replacement
102 //var bins = populationBins.ToList();
103 for (int row = 0; row < N; row++)
104 {
105 int r = rnd[col, 1].Next(0, bins.Count);
106 lhs[row, col] = bins[r];
107 bins.RemoveAt(r);
108 }
109
110 }
111
112 return lhs;
113 }
