Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

60 public static double GaussLegendre(Func<double, double> f, double a, double b)
61 {
62 var x = new double[] { 0.1488743389816312, 0.4333953941292472, 0.6794095682990244, 0.8650633666889845, 0.9739065285171717 };
63 var w = new double[] { 0.2955242247147529, 0.2692667193099963, 0.2190863625159821, 0.1494513491505806, 0.0666713443086881 };
64 double xm = 0.5 * (b + a);
65 double xr = 0.5 * (b - a);
66 double s = 0;
67 for (int j = 0; j < 5; j++)
68 {
69 double dx = xr * x[j];
70 s += w[j] * (f(xm + dx) + f(xm - dx));
71 }
72 return s *= xr;

83 public static double TrapezoidalRule(Func<double, double> f, double a, double b, int steps = 2)
84 {
85 double h = (b - a) / steps;
86 double x = a;
87 double sum = 0.5d * (f(a) + f(b));
88 for (int i = 1; i <= steps - 1; i++)
89 {
90 x += h;
91 sum += f(x);
92 }
93 return h * sum;
94 }

105 public static double SimpsonsRule(Func<double, double> f, double a, double b, int steps = 2)
106 {
107 double h = (b - a) / steps;
108 double sum1 = f(a + h / 2d);
109 double sum2 = 0d;
110 for (int i = 1; i <= steps - 1; i++)
111 {
112 sum1 += f(a + h * i + h / 2d);
113 sum2 += f(a + h * i);
114 }
115 return h / 6d * (f(a) + f(b) + 4d * sum1 + 2d * sum2);
116 }

126 public static double Midpoint(Func<double, double> f, double a, double b, int steps = 2)
127 {
128 double h = (b - a) / steps;
129 double x = a + h / 2d;
130 double sum = 0;
131 for (int i = 1; i <= steps; i++)
132 {
133 sum += f(x);
134 x += h;
135 }
136 return h * sum;
