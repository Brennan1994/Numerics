253 public void Estimate(IList<double> sample, ParameterEstimationMethod estimationMethod)
254 {
255 if (estimationMethod == ParameterEstimationMethod.MethodOfPercentiles)
256 {
257 var data = sample.ToList();
258 data.Sort();
259 var five = Statistics.Percentile(data, 0.05, true);
260 var fifty = Statistics.Percentile(data, 0.5, true);
261 var ninetyFive = Statistics.Percentile(data, 0.95, true);
262 var pertP = new PertPercentile(five, fifty, ninetyFive);
263 pertP.SolveParameters();
264 var pert = pertP.ToPert();
265 SetParameters(pert.Min, pert.Mode, pert.Max);
266 }
267 else if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
268 {
269 var min = Tools.Min(sample);
270 var max = Tools.Max(sample);
271 var mean = Tools.Mean(sample);
272 var mode = (mean * 6d - max - min) / 4d;
273 mode = Math.Max(min, mode);
274 mode = Math.Min(max, mode);
275 SetParameters(min, mode, max);
276 }
277 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
278 {
279 SetParameters(MLE(sample));
280 }
281 else
282 {
283 throw new NotImplementedException();
284 }
285 }

353 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
354 {
355 var initialVals = new double[NumberOfParameters];
356 var lowerVals = new double[NumberOfParameters];
357 var upperVals = new double[NumberOfParameters];
358 // Estimate initial values using the method of moments (a.k.a product moments).
359 var min = Tools.Min(sample);
360 var max = Tools.Max(sample);
361 var mean = Tools.Mean(sample);
362 var mode = (mean * 6d - max - min) / 4d;
363 initialVals[0] = min - Tools.DoubleMachineEpsilon;
364 initialVals[1] = mode;
365 initialVals[2] = max + Tools.DoubleMachineEpsilon;
366 // Get bounds of min
367 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(min)) + 1d));
368 upperVals[0] = min;
369 // Get bounds of mode
370 lowerVals[1] = min;
371 upperVals[1] = max;
372 // Get bounds of max
373 lowerVals[2] = max;
374 upperVals[2] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(max)) + 1d));
375
376 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
377 }

380 public double[] MLE(IList<double> sample)
381 {
382 // Set constraints
383 var tuple = GetParameterConstraints(sample);
384 var Initials = tuple.Item1;
385 var Lowers = tuple.Item2;
386 var Uppers = tuple.Item3;
387
388 // Solve using Nelder-Mead (Downhill Simplex)
389 double logLH(double[] x)
390 {
391 var N = new Pert();
392 N.SetParameters(x);
393 return N.LogLikelihood(sample);
394 }
395 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
396 solver.ReportFailure = true;
397 solver.Maximize();
398 return solver.BestParameterSet.Values;
399 }
