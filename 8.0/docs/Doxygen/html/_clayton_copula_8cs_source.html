132 public override double GeneratorPrime2(double t)

139 public override double GeneratorPrimeInverse(double t)
140 {

146 public override double CDF(double u, double v)
147 {
148 // Validate parameters
149 if (_parametersValid == false) ValidateParameter(Theta, true);
150 return Math.Pow(Math.Max(Math.Pow(u,-Theta) + Math.Pow(v, -Theta) - 1d,0d), -1d / Theta);
151 }
152
154 public override double[] InverseCDF(double u, double v)
155 {
156 // Validate parameters
157 if (_parametersValid == false) ValidateParameter(Theta, true);
158 v = Math.Pow(Math.Pow(u, -Theta) * (Math.Pow(v, -Theta / (Theta + 1d)) - 1d) + 1d, -1d / Theta);
159 return [u, v];
160 }
