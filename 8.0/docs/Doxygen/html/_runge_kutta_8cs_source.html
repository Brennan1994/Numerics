71 public static double[] SecondOrder(Func<double, double, double> f, double initialValue, double startTime, double endTime, int timeSteps)
72 {
73 double dt = (endTime - startTime) / (timeSteps - 1);
74 double k1 = 0d;
75 double k2 = 0d;
76 double t = startTime;
77 var y = new double[timeSteps];
78 double y0 = initialValue;
79 y[0] = y0;
80 for (int i = 1; i < timeSteps; i++)
81 {
82 k1 = f(t, y0);
83 k2 = f(t + dt, y0 + k1 * dt);
84 y[i] = y0 + dt * 0.5d * (k1 + k2);
85 t += dt;
86 y0 = y[i];
87 }
88
89 return y;
90 }

103 public static double[] FourthOrder(Func<double, double, double> f, double initialValue, double startTime, double endTime, int timeSteps)
104 {
105 double dt = (endTime - startTime) / (timeSteps - 1);
106 double k1 = 0d;
107 double k2 = 0d;
108 double k3 = 0d;
109 double k4 = 0d;
110 double t = startTime;
111 var y = new double[timeSteps];
112 double y0 = initialValue;
113 y[0] = y0;
114 for (int i = 1; i < timeSteps; i++)
115 {
116 k1 = f(t, y0);
117 k2 = f(t + dt / 2d, y0 + k1 * dt / 2d);
118 k3 = f(t + dt / 2d, y0 + k2 * dt / 2d);
119 k4 = f(t + dt, y0 + k3 * dt);
120 y[i] = y0 + dt / 6d * (k1 + 2d * k2 + 2d * k3 + k4);
121 t += dt;
122 y0 = y[i];
123 }
124 return y;

137 public static double FourthOrder(Func<double, double, double> f, double initialValue, double startTime, double dt)
138 {
139 double t = startTime;
140 double y = initialValue;
141 double k1 = f(t, y);
142 double k2 = f(t + dt / 2d, y + k1 * dt / 2d);
143 double k3 = f(t + dt / 2d, y + k2 * dt / 2d);
144 double k4 = f(t + dt, y + k3 * dt);
145 return y + dt / 6d * (k1 + 2d * k2 + 2d * k3 + k4);

167 public static double Fehlberg(Func<double, double, double> f, double initialValue, double startTime, double dt, double dtMin, double tolerance = 1E-3)
168 {
169 double t = startTime;
170 double tf = startTime + dt;
171 double y = initialValue;
172 double k1, k2, k3, k4, k5, k6, rk4, rk5, error;
173 double h = dt;
174 double minTolerance = 1E-2 * tolerance;
175
176 while (t < tf)
177 {
178 // Make sure the time step is constrained
179 if (h < dtMin) h = dtMin;
180 if (h > dt) h = dt;
181 if (t + h > tf) h = tf - t;
182
183 // Compute RK4, RK5, and error = |RK4 - RK5|
184 k1 = h * f(t, y);
185 k2 = h * f(t + 1d / 4d * h, y + 1d / 4d * k1);
186 k3 = h * f(t + 3d / 8d * h, y + 3d / 32d * k1 + 9d / 32d * k2);
187 k4 = h * f(t + 12d / 13d * h, y + 1932d / 2197d * k1 - 7200d / 2197d * k2 + 7296d / 2197d * k3);
188 k5 = h * f(t + 1d / 1d * h, y + 439d / 216d * k1 - 8d * k2 + 3680d / 513d * k3 - 845d / 4104d * k4);
189 k6 = h * f(t + 1d / 2d * h, y - 8d / 27d * k1 + 2d * k2 - 3544d / 2565d * k3 + 1859d / 4104d * k4 - 11d / 40d * k5);
190 rk4 = y + 25d / 216d * k1 + 1408d / 2565d * k3 + 2197d / 4104d * k4 - 1d / 5d * k6;
191 rk5 = y + 16d / 135d * k1 + 6656d / 12825d * k3 + 28561d / 56430d * k4 - 9d / 50d * k5 + 2d / 55d * k6;
192 error = Math.Max(minTolerance, Math.Abs(rk4 - rk5));
193
194 // Check convergence
195 if (error <= tolerance || h <= dtMin)
196 {
197 // accept the step
198 t += h;
199 y = rk5;
200
201 // Determine if the step size should be increased
202 if (error < minTolerance)
203 h *= 2d;
204 }
205 else
206 {
207 // Reject the step and decrease the step size
208 h /= 2d;
209 }
210 }
211
212 return y;
213 }
