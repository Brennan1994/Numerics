200 public Vector Solve(Vector B, double threshold = -1)
201 {
202 // Solve A*x=b for a vector x using the pseudoinverse of A as obtained by SVD. If positive,
203 // the threshold is the threshold value below which singular values are considered zero. If threshold is
204 // negative, a default based on expected roundoff error is used.
205 int i, j, jj;
206 double s;
207 var x = new Vector(n);
208 var tmp = new Vector(n);
209 if (B.Length != m)
210 {
211 throw new ArgumentOutOfRangeException(nameof(B), "The vector b must have the same number of rows as the matrix A.");
212 }
213 Threshold = (threshold >= 0d ? threshold : 0.5 * Math.Sqrt(m + n + 1d) * W[0] * eps);
214 for (j = 0; j < n; j++) // Calculate U^T*B
215 {
216 s = 0.0;
217 if (W[j] > Threshold) // Nonzero result only if W-j is nonzero.
218 {
219 for (i = 0; i < m; i++) s += U[i,j] * B[i];
220 s /= W[j]; // This s is then divided by W-j.
221 }
222 tmp[j] = s;
223 }
224 // Matrix multiply by V to get answer.
225 for (j = 0; j < n; j++)
226 {
227 s = 0.0;
228 for (jj = 0; jj < n; jj++) s += V[j,jj] * tmp[jj];
229 x[j] = s;
230 }
231 return x;
232 }

242 public Matrix Solve(Matrix B, double threshold = -1)
243 {
244 int i, j, p = B.NumberOfColumns;
245 var x = new Matrix(n, p);
246 var xx = new Vector(n);
247 var bcol = new Vector(m);
248 if (B.NumberOfRows != n)
249 {
250 throw new ArgumentOutOfRangeException(nameof(B), "The matrix B must have the same number of rows as columns in the matrix A.");
251 }
252 // Copy and solve each column in turn.
253 for (j = 0; j < p; j++)
254 {
255 for (i = 0; i < m; i++) bcol[i] = B[i, j];
256 xx = Solve(bcol, threshold);
257 for (i = 0; i < n; i++) x[i, j] = xx[i];
258 }
259 return x;
260 }

265 private void Decompose()
266 {
267 bool flag;
268 int i, its, j, jj, k, l=0, nm=0;
269 double anorm, c, f, g, h, s, scale, x, y, z;
270 var rv1 = new Vector(n);
271 g = scale = anorm = 0.0;
272 for (i = 0; i < n; i++)
273 {
274 l = i + 2;
275 rv1[i] = scale * g;
276 g = s = scale = 0.0;
277 if (i < m)
278 {
279 for (k = i; k < m; k++) scale += Math.Abs(U[k,i]);
280 if (scale != 0.0)
281 {
282 for (k = i; k < m; k++)
283 {
284 U[k,i] /= scale;
285 s += U[k,i] * U[k,i];
286 }
287 f = U[i,i];
288 g = -Tools.Sign(Math.Sqrt(s), f);
289 h = f * g - s;
290 U[i,i] = f - g;
291 for (j = l - 1; j < n; j++)
292 {
293 for (s = 0.0, k = i; k < m; k++) s += U[k,i] * U[k,j];
294 f = s / h;
295 for (k = i; k < m; k++) U[k,j] += f * U[k,i];
296 }
297 for (k = i; k < m; k++) U[k,i] *= scale;
298 }
299 }
300 W[i] = scale * g;
301 g = s = scale = 0.0;
302 if (i + 1 <= m && i + 1 != n)
303 {
304 for (k = l - 1; k < n; k++) scale += Math.Abs(U[i,k]);
305 if (scale != 0.0)
306 {
307 for (k = l - 1; k < n; k++)
308 {
309 U[i,k] /= scale;
310 s += U[i,k] * U[i,k];
311 }
312 f = U[i,l - 1];
313 g = -Tools.Sign(Math.Sqrt(s), f);
314 h = f * g - s;
315 U[i,l - 1] = f - g;
316 for (k = l - 1; k < n; k++) rv1[k] = U[i,k] / h;
317 for (j = l - 1; j < m; j++)
318 {
319 for (s = 0.0, k = l - 1; k < n; k++) s += U[j,k] * U[i,k];
320 for (k = l - 1; k < n; k++) U[j,k] += s * rv1[k];
321 }
322 for (k = l - 1; k < n; k++) U[i,k] *= scale;
323 }
324 }
325 anorm = Math.Max(anorm, (Math.Abs(W[i]) + Math.Abs(rv1[i])));
326 }
327 for (i = n - 1; i >= 0; i--)
328 {
329 if (i < n - 1)
330 {
331 if (g != 0.0)
332 {
333 for (j = l; j < n; j++)
334 V[j,i] = (U[i,j] / U[i,l]) / g;
335 for (j = l; j < n; j++)
336 {
337 for (s = 0.0, k = l; k < n; k++) s += U[i,k] * V[k,j];
338 for (k = l; k < n; k++) V[k,j] += s * V[k,i];
339 }
340 }
341 for (j = l; j < n; j++) V[i,j] = V[j,i] = 0.0;
342 }
343 V[i,i] = 1.0;
344 g = rv1[i];
345 l = i;
346 }
347 for (i = Math.Min(m, n) - 1; i >= 0; i--)
348 {
349 l = i + 1;
350 g = W[i];
351 for (j = l; j < n; j++) U[i,j] = 0.0;
352 if (g != 0.0)
353 {
354 g = 1.0 / g;
355 for (j = l; j < n; j++)
356 {
357 for (s = 0.0, k = l; k < m; k++) s += U[k,i] * U[k,j];
358 f = (s / U[i,i]) * g;
359 for (k = i; k < m; k++) U[k,j] += f * U[k,i];
360 }
361 for (j = i; j < m; j++) U[j,i] *= g;
362 }
363 else for (j = i; j < m; j++) U[j,i] = 0.0;
364 ++U[i,i];
365 }
366 for (k = n - 1; k >= 0; k--)
367 {
368 for (its = 0; its < 30; its++)
369 {
370 flag = true;
371 for (l = k; l >= 0; l--)
372 {
373 nm = l - 1;
374 if (l == 0 || Math.Abs(rv1[l]) <= eps * anorm)
375 {
376 flag = false;
377 break;
378 }
379 if (Math.Abs(W[nm]) <= eps * anorm) break;
380 }
381 if (flag)
382 {
383 c = 0.0;
384 s = 1.0;
385 for (i = l; i < k + 1; i++)
386 {
387 f = s * rv1[i];
388 rv1[i] = c * rv1[i];
389 if (Math.Abs(f) <= eps * anorm) break;
390 g = W[i];
391 h = Pythag(f, g);
392 W[i] = h;
393 h = 1.0 / h;
394 c = g * h;
395 s = -f * h;
396 for (j = 0; j < m; j++)
397 {
398 y = U[j,nm];
399 z = U[j,i];
400 U[j,nm] = y * c + z * s;
401 U[j,i] = z * c - y * s;
402 }
403 }
404 }
405 z = W[k];
406 if (l == k)
407 {
408 if (z < 0.0)
409 {
410 W[k] = -z;
411 for (j = 0; j < n; j++) V[j,k] = -V[j,k];
412 }
413 break;
414 }
415 if (its == 99) throw new ArgumentException("There was no convergence in 100 iterations");
416 x = W[l];
417 nm = k - 1;
418 y = W[nm];
419 g = rv1[nm];
420 h = rv1[k];
421 f = ((y - z) * (y + z) + (g - h) * (g + h)) / (2.0 * h * y);
422 g = Pythag(f, 1.0);
423 f = ((x - z) * (x + z) + h * ((y / (f + Tools.Sign(g, f))) - h)) / x;
424 c = s = 1.0;
425 for (j = l; j <= nm; j++)
426 {
427 i = j + 1;
428 g = rv1[i];
429 y = W[i];
430 h = s * g;
431 g = c * g;
432 z = Pythag(f, h);
433 rv1[j] = z;
434 c = f / z;
435 s = h / z;
436 f = x * c + g * s;
437 g = g * c - x * s;
438 h = y * s;
439 y *= c;
440 for (jj = 0; jj < n; jj++)
441 {
442 x = V[jj,j];
443 z = V[jj,i];
444 V[jj,j] = x * c + z * s;
445 V[jj,i] = z * c - x * s;
446 }
447 z = Pythag(f, h);
448 W[j] = z;
449 if (z != 0.0)
450 {
451 z = 1.0 / z;
452 c = f * z;
453 s = h * z;
454 }
455 f = c * g + s * y;
456 x = c * y - s * g;
457 for (jj = 0; jj < m; jj++)
458 {
459 y = U[jj,j];
460 z = U[jj,i];
461 U[jj,j] = y * c + z * s;
462 U[jj,i] = z * c - y * s;
463 }
464 }
465 rv1[l] = 0.0;
466 rv1[k] = f;
467 W[k] = x;
468 }
469 }

475 private void Reorder()
476 {
477 int i, j, k, s, inc = 1;
478 double sw;
479 var su = new Vector(m);
480 var sv = new Vector(n);
481 do { inc *= 3; inc++; } while (inc <= n);
482 do
483 {
484 inc /= 3;
485 for (i = inc; i < n; i++)
486 {
487 sw = W[i];
488 for (k = 0; k < m; k++) su[k] = U[k,i];
489 for (k = 0; k < n; k++) sv[k] = V[k,i];
490 j = i;
491 while (W[j - inc] < sw)
492 {
493 W[j] = W[j - inc];
494 for (k = 0; k < m; k++) U[k,j] = U[k,j - inc];
495 for (k = 0; k < n; k++) V[k,j] = V[k,j - inc];
496 j -= inc;
497 if (j < inc) break;
498 }
499 W[j] = sw;
500 for (k = 0; k < m; k++) U[k,j] = su[k];
501 for (k = 0; k < n; k++) V[k,j] = sv[k];
502
503 }
504 } while (inc > 1);
505 for (k = 0; k < n; k++)
506 {
507 s = 0;
508 for (i = 0; i < m; i++) if (U[i,k] < 0d) s++;
509 for (j = 0; j < n; j++) if (V[j,k] < 0d) s++;
510 if (s > (m + n) / 2)
511 {
512 for (i = 0; i < m; i++) U[i,k] = -U[i,k];
513 for (j = 0; j < n; j++) V[j,k] = -V[j,k];
514 }
515 }
516 }

521 private double Pythag(double a, double b) {
522 double absa = Math.Abs(a), absb = Math.Abs(b);
523 return (absa > absb ? absa * Math.Sqrt(1.0 + Math.Pow(absb / absa, 2)) :
524 (absb == 0.0 ? 0.0 : absb * Math.Sqrt(1.0 + Math.Pow(absa / absb, 2))));
525 }
