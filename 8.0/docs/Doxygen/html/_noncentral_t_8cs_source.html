302 public override double PDF(double x)
303 {
304 // Validate parameters
305 if (_parametersValid == false)
306 ValidateParameters(_degreesOfFreedom, Noncentrality, true);
307 double u = Noncentrality;
308 double v = DegreesOfFreedom;
309 if (x != 0)
310 {
311 double A = NCT_CDF(x * Math.Sqrt(1 + 2 / v), v + 2, u);
312 double B = NCT_CDF(x, v, u);
313 double C = v / x;
314 return C * (A - B);
315 }
316 else
317 {
318 double A = Gamma.Function((v + 1) / 2);
319 double B = Math.Sqrt(Math.PI * v) * Gamma.Function(v / 2);
320 double C = Math.Exp(-(u * u) / 2);
321 return (A / B) * C;
322 }

360 private double NCT_CDF(double t, double df, double delta)
361 {
362 double Z;
363 double ANS = 0d;
364 try
365 {
366 ANS = NCTDist(t, df, delta);
367 }
368 catch (Exception ex)
369 {
370 // If the solver fails, then use approximation
371 if (ex.ToString() == "Maximum number of iterations reached.")
372 {
373 Z = (t * (1.0d - 1.0d / (4.0d * df)) - delta) / Math.Sqrt(1.0d + Math.Pow(t, 2d) / (2.0d * df));
374 ANS = Normal.StandardCDF(Z);
375 }
376
377 if (ANS < 0d)
378 ANS = 0d;
379 if (ANS > 1d)
380 ANS = 1d;
381 }
382
383 return ANS;
384 }

399 private double NCTDist(double t, double df, double delta)
400 {
401
402 // REAL FUNCTION TNC(T, DF, DELTA, IFAULT)
403 //
404 // ALGORITHM AS 243 APPL. STATIST. (1989), VOL.38, NO. 1
405 //
406 // Cumulative probability at T of the non-central t-distribution
407 // with DF degrees of freedom (may be fractional) and non-centrality
408 // parameter DELTA.
409 //
410 // Note - requires the following auxiliary routines
411 // ALOGAM (X) - ACM 291 or AS 245
412 // BETAIN (X, A, B, ALBETA, IFAULT) - AS 63 (updated in ASR 19)
413 // ALNORM (X, UPPER) - AS 66
414 //
415 // Translated by William A. Huber. www.quantdec.com
416 //
417 double a;
418 double ALBETA;
419 double b;
420 double DEL;
421 var N = default(int);
422 double ERRBD;
423 double GEVEN;
424 double GODD;
425 double LAMBDA;
426 double P;
427 double q;
428 double RXB;
429 double s;
430 double TT;
431 double x;
432 double XEVEN;
433 double XODD;
434 double TNC;
435 bool NEGDEL;
436 //
437 // Note - ITRMAX and ERRMAX may be changed to suit one's needs.
438 //
439 const int ITRMAX = 1000;
440 const double Errmax = 0.0000001d;
441
442 // DATA ITRMAX/100.1/, ERRMAX/1.E-06/
443 //
444 // Constants - R2PI = 1/ {GAMMA(1.5) * SQRT(2)} = SQRT(2 / PI)
445 // ALNRPI = Ln(SQRT(Pi))
446 //
447 const double zero = 0d;
448 const double half = 0.5d;
449 const double one = 1.0d;
450 const double two = 2.0d;
451 const double r2pi = 0.797884560802865d;
452 const double alnrpi = 0.5723649429247d;
453 TNC = zero;
454 TT = t;
455 DEL = delta;
456 NEGDEL = false;
457 if (t < zero)
458 {
459 NEGDEL = true;
460 TT = -TT;
461 DEL = -DEL;
462 }
463 //
464 // Initialize twin series (Guenther, J. Statist. Computn. Simuln.
465 // vol.6, 199, 1978).
466 //
467 x = t * t / (t * t + df);
468 if (x <= zero)
469 goto Twenty;
470 LAMBDA = DEL * DEL;
471 P = half * Math.Exp(-half * LAMBDA);
472 q = r2pi * P * DEL;
473 s = half - P;
474 a = half;
475 b = half * df;
476 RXB = Math.Pow(one - x, b);
477 ALBETA = alnrpi + Gamma.LogGamma(b) - Gamma.LogGamma(a + b);
478 XODD = Beta.IncompleteRatio(x, a, b, ALBETA);
479 GODD = two * RXB * Math.Exp(a * Math.Log(x) - ALBETA);
480 XEVEN = one - RXB;
481 GEVEN = b * x * RXB;
482 TNC = P * XODD + q * XEVEN;
483 //
484 // Repeat until convergence
485 //
486 N = 1;
487 do
488 {
489 a = a + one;
490 XODD = XODD - GODD;
491 XEVEN = XEVEN - GEVEN;
492 GODD = GODD * x * (a + b - one) / a;
493 GEVEN = GEVEN * x * (a + b - half) / (a + half);
494 P = P * LAMBDA / (two * N);
495 q = q * LAMBDA / (two * N + one);
496 s = s - P;
497 N = N + 1;
498 TNC = TNC + P * XODD + q * XEVEN;
499 ERRBD = two * s * (XODD - GODD);
500 }
501 while (ERRBD > Errmax & N <= ITRMAX);
502 //
503 Twenty:
504 ;
505 if (N > ITRMAX)
506 {
507 throw new ArgumentException("Max number of iterations were exceeded.");
508 }
509
510 if (NEGDEL)
511 {
512 // TNC = one - TNC
513 TNC = Normal.StandardCDF(DEL) - TNC;
514 }
515 else
516 {
517 TNC = TNC + (1.0d - Normal.StandardCDF(DEL));
518 } // Upper tail area of N(0,1)
519 return TNC;
520 }

528 private double NCT_INV(double p, double df, double delta)
529 {
530 double t0, t1, t2, y0, y1, tInc, Slope;
531 int iter;
532 //double tEstimate;
533 const double ytol = 0.0000001d; // Y-tolerance
534 const double xtol = 0.0000001d; // X-tolerance
535 //const double w = 1.5d; // Over-shooting parameter (>= 1)
536 const int iterMax = 50;
537 t0 = NCTInv0(p, df, delta);
538 //tEstimate = t0; // Default if we run into trouble
539 y0 = NCTDist(t0, df, delta) - p;
540 if (y0 > ytol)
541 {
542 tInc = -1.0d;
543 }
544 else if (y0 < -ytol)
545 {
546 tInc = 1.0d;
547 }
548 else
549 {
550 return t0;
551 }
552 //
553 // Find a bracket through overshooting.
554 //
555 t1 = t0 + tInc;
556 y1 = NCTDist(t1, df, delta) - p;
557 iter = 0;
558 while (y0 < 0d != y1 > 0d & Math.Abs(t1 - t0) > xtol & iter < iterMax)
559 {
560 //
561 // Use secant method to extrapolate a zero, but overshoot by w >= 1.
562 //
563 Slope = (y1 - y0) / (t1 - t0);
564 if (Slope == 0d)
565 {
566 return (t1 + t0) / 2.0d;
567 }
568
569 t2 = t0 - 2.0d * y0 / Slope;
570 t0 = t1;
571 t1 = t2;
572 y0 = y1;
573 y1 = NCTDist(t1, df, delta) - p;
574 iter = iter + 1;
575 }
576 // Solve for T using Brent
577 double ANS = ZBrent(t0, t1, y0, y1, xtol, df, delta, p);
578 return ANS;
579 }

581 private double NCTInv0(double P, double N, double D)
582 {
583 //
584 // Approximates percentage points of the non-central t distribution.
585 // P is the percentage, N is the degrees of freedom, D is the non-centrality parameter.
586 //
587 // Non-central t is the ratio (U + D)/(Chi/Sqrt(n)) where U and Chi are independent
588 // random variables distributed as Normal(0, 1) and Chi(n), respectively.
589 // D is the non-centrality parameter. When equal to 0, TInvNC is Student's t.
590 //
591 // Source: Johnson & Kotz, Continuous Univariate Distributions, Volume 2.
592 //
593 // NB: The number of iterations appears to go quadratically in D. Thus, for large
594 // D, we will be in trouble.
595 //
596 // VB version (c) 2001 Quantitative Decisions. All rights reserved.
597 // Contact William A. Huber. www.quantdec.com
598 //
599 double b;
600 double z;
601 double b2;
602 double u2;
603 double T;
604 //
605 // Establish entry conditions.
606 //
607 // If N < 1.0# Or P <= 0.000001 Or P >= 0.999999 Then
608 // Return Double.NaN
609 // End If
610
611 var StudentT = new StudentT(N);
612 z = Normal.StandardZ(P);
613 //
614 // Jennett & Welch approximation, formula (14.1).
615 // Intended for large values of D^2, such as are used for most tolerance interval calculations.
616 //
617 b = Math.Exp(Gamma.LogGamma((N + 1d) / 2d) - Gamma.LogGamma(N / 2d)) * Math.Sqrt(2d / N);
618 u2 = z * z;
619 b2 = b * b;
620 T = b2 + (1d - b2) * (Math.Pow(D, 2d) - u2);
621 if (T > 0d)
622 {
623 T = (D * b + z * Math.Sqrt(T)) / (b2 - u2 * (1d - b2));
624 }
625 else
626 {
627 T = 0d;
628 }
629
630 return T;
631 }

633 private double ZBrent(double X1, double X2, double y1, double y2, double Tol, double N, double Dnc, double Perc)
634 {
635 double ZBrentRet = default;
636 //
637 // Finds the zero of NCTDist(x, n, d) - p given that [x1, x2] brackets the zero and
638 // Y1 = value at X1, Y2 = value at X2.
639 //
640 // Translated from Numerical Recipes (1986).
641 // William A. Huber, 24 March 2001.
642 //
643 double a;
644 double b;
645 var c = default(double);
646 double fc;
647 var D = default(double);
648 var e = default(double);
649 double tol1;
650 double xm;
651 double s;
652 double P;
653 double q;
654 double r;
655 double fa;
656 double fb;
657 int iter;
658 const int itmax = 100;
659 const double eps = 0.00000003d;
660 a = X1;
661 b = X2;
662 fa = y1;
663 fb = y2;
664 if (fb * fa > 0d)
665 {
666 throw new ArgumentException("Brent's method failed because the root is not bracketed.");
667 }
668
669 fc = fb;
670 for (iter = 1; iter <= itmax; iter++)
671 {
672 if (fb * fc > 0d)
673 {
674 c = a;
675 fc = fa;
676 D = b - a;
677 e = D;
678 }
679
680 if (Math.Abs(fc) < Math.Abs(fb))
681 {
682 a = b;
683 b = c;
684 c = a;
685 fa = fb;
686 fb = fc;
687 fc = fa;
688 }
689
690 tol1 = 2.0d * eps * Math.Abs(b) + 0.5d * Tol;
691 xm = 0.5d * (c - b);
692 if (Math.Abs(xm) <= tol1 | fb == 0d)
693 {
694 return b;
695 }
696
697 if (Math.Abs(e) >= tol1 & Math.Abs(fa) > Math.Abs(fb))
698 {
699 s = fb / fa;
700 if (a == c)
701 {
702 P = 2.0d * xm * s;
703 q = 1.0d - s;
704 }
705 else
706 {
707 q = fa / fc;
708 r = fb / fc;
709 P = s * (2.0d * xm * q * (q - r) - (b - a) * (r - 1.0d));
710 q = (q - 1.0d) * (r - 1.0d) * (s - 1.0d);
711 }
712
713 if (P > 0d)
714 q = -q;
715 P = Math.Abs(P);
716 if (2.0d * P < 3.0d * xm * q - Math.Abs(tol1 * q) & 2.0d * P < Math.Abs(e * q))
717 {
718 e = D;
719 D = P / q;
720 }
721 else
722 {
723 D = xm;
724 e = D;
725 }
726 }
727 else
728 {
729 D = xm;
730 e = D;
731 }
732
733 a = b;
734 fa = fb;
735 if (Math.Abs(D) > tol1)
736 {
737 b = b + D;
738 }
739 else if (xm > 0d) // b = b + SIGN(tol1, xm)
740 {
741 b = b + Math.Abs(tol1);
742 }
743 else
744 {
745 b = b - Math.Abs(tol1);
746 }
747
748 fb = NCTDist(b, N, Dnc) - Perc;
749 }
750
751 ZBrentRet = b;
752 return ZBrentRet;
753 }
