Numerics 1.0.0
Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

185 public ArgumentOutOfRangeException ValidateParameters(IList<double> x1Values, IList<double> x2Values, double[,] pValues, bool throwException)
186 {
187
188 if (x1Values.Count < 2)
189 {
190 if (throwException)
191 throw new ArgumentOutOfRangeException("Primary Values", "There must be at least 2 primary values.");
192 return new ArgumentOutOfRangeException("Primary Values", "There must be at least 2 primary values.");
193 }
194 if (x2Values.Count < 2)
195 {
196 if (throwException)
197 throw new ArgumentOutOfRangeException("Secondary Values", "There must be at least 2 secondary values.");
198 return new ArgumentOutOfRangeException("Secondary Values", "There must be at least 2 secondary values.");
199 }
200
201 // Check if X1 and X2 are in ascending order
202 for (int i = 1; i < x1Values.Count; i++)
203 {
204 if (x1Values[i] <= x1Values[i - 1])
205 {
206 if (throwException)
207 throw new ArgumentOutOfRangeException("Primary Values", "Primary values must be in ascending order.");
208 return new ArgumentOutOfRangeException("Primary Values", "Primary values must be in ascending order.");
209 }
210 }
211 for (int i = 1; i < x2Values.Count; i++)
212 {
213 if (x2Values[i] <= x2Values[i - 1])
214 {
215 if (throwException)
216 throw new ArgumentOutOfRangeException("Secondary Values", "Secondary values must be in ascending order.");
217 return new ArgumentOutOfRangeException("Secondary Values", "Secondary values must be in ascending order.");
218 }
219 }
220
221 // Check if probabilities are between 0 and 1 and in ascending order
222 if (pValues.GetLength(0) != x1Values.Count)
223 {
224 if (throwException)
225 throw new ArgumentOutOfRangeException(nameof(ProbabilityValues), "The number of rows in the probability array must be the same length as the X1 array.");
226 return new ArgumentOutOfRangeException(nameof(ProbabilityValues), "The number of rows in the probability array must be the same length as the X1 array.");
227 }
228 if (pValues.GetLength(1) != x2Values.Count)
229 {
230 if (throwException)
231 throw new ArgumentOutOfRangeException(nameof(ProbabilityValues), "The number of columns in the probability array must be the same length as the X2 array.");
232 return new ArgumentOutOfRangeException(nameof(ProbabilityValues), "The number of columns in the probability array must be the same length as the X2 array.");
233 }
234
235 for (int i = 0; i < pValues.GetLength(0); i++) // Rows
236 {
237 for (int j = 0; j < pValues.GetLength(1); j++) // Columns
238 {
239 if (pValues[i, j] < 0.0d || pValues[i, j] > 1.0d)
240 {
241 if (throwException) throw new ArgumentOutOfRangeException(nameof(ProbabilityValues), "Probability values must be equal to or between 0 and 1.");
242 return new ArgumentOutOfRangeException(nameof(ProbabilityValues), "Probability values must be equal to or between 0 and 1.");
243 }
244 }
245 }
246
247 return null;
248 }

290 public double CDF(double x1, double x2)
291 {
292 // Validate parameters
293 if (_parametersValid == false) ValidateParameters(X1Values, X2Values, ProbabilityValues, true);
294 // Make sure the transforms are up-to-date.
295 if (bilinear == null) bilinear = new Bilinear(X1Values, X2Values, ProbabilityValues);
296 bilinear.X1Transform = X1Transform;
297 bilinear.X2Transform = X2Transform;
298 bilinear.YTransform = ProbabilityTransform;
299 return bilinear.Interpolate(x1, x2);
300 }
