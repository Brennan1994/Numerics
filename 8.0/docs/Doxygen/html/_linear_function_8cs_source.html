The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

179 public double Function(double x)
180 {
181 // Validate parameters
182 if (_parametersValid == false)
183 ValidateParameters(new[] { Alpha, Beta, Sigma }, true);
184
185 // Check support
186 if (x <= Minimum)
187 {
188 if (IsDeterministic == true || ConfidenceLevel < 0 || ConfidenceLevel > 1) return Alpha + Beta * Minimum;
189 return Alpha + Beta * Minimum + _normal.InverseCDF(ConfidenceLevel);
190 }
191 if (x >= Maximum)
192 {
193 if (IsDeterministic == true || ConfidenceLevel < 0 || ConfidenceLevel > 1) return Alpha + Beta * Maximum;
194 return Alpha + Beta * Maximum + _normal.InverseCDF(ConfidenceLevel);
195 }
196
197 if (IsDeterministic == true || ConfidenceLevel < 0 || ConfidenceLevel > 1) return Alpha + Beta * x;
198 return Alpha + Beta * x + _normal.InverseCDF(ConfidenceLevel);
199 }

202 public double InverseFunction(double y)
203 {
204 // Validate parameters

206 ValidateParameters(new[] { Alpha, Beta, Sigma }, true);
207
208 double x = 0;
209 if (IsDeterministic == true || ConfidenceLevel < 0 || ConfidenceLevel > 1)
210 {
211 x = (y - Alpha) / Beta;
212 }
213 else
214 {
215 x = (y - Alpha - _normal.InverseCDF(ConfidenceLevel)) / Beta;
216 }

218 if (x > Maximum) return Maximum;
