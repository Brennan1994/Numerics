212 public static double Variance(IList<double> sampleData)
213 {
214 if (sampleData.Count <= 1) return double.NaN;
215
216 double variance_ = 0d;
217 double t = sampleData[0];
218 for (int i = 1; i < sampleData.Count; i++)
219 {
220 t += sampleData[i];
221 double diff = (i + 1) * sampleData[i] - t;
222 variance_ += diff * diff / ((i + 1.0d) * i);
223 }
224 return variance_ / (sampleData.Count - 1);
225 }

306 public static double Skewness(IList<double> sampleData)
307 {
308 if (sampleData.Count == 0) return double.NaN;
309
310 double mean = Mean(sampleData);
311 int n = sampleData.Count;
312 double s2 = 0d, s3 = 0d;
313 for (int i = 0; i < n; i++)
314 {
315 double xm = sampleData[i] - mean;
316 s2 += xm * xm;
317 s3 += xm * xm * xm;
318 }
319 double m2 = s2 / n;
320 double m3 = s3 / n;
321 double g = m3 / Math.Pow(m2, 3.0d / 2.0d);
322 double a = Math.Sqrt(n * (n - 1));
323 double b = n - 2;
324 return a / b * g;
325 }

332 public static double JackKnifeStandardError(IList<double> sampleData, Func<IList<double>, double> statistic)
333 {
334 int N = sampleData.Count;
335 double theta = statistic(sampleData);
336 double I = 0d;
337 Parallel.For(0, N, () => 0d, (i, loop, subI) =>
338 {
339 // Remove data point
340 var jackSample = new List<double>(sampleData);
341 jackSample.RemoveAt(i);
342 // Compute statistic
343 subI += Tools.Sqr(statistic(jackSample) - theta);
344 return subI;
345 }, z => Tools.ParallelAdd(ref I, z));
346 return Math.Sqrt((N - 1) / (double)N * I);
347 }

375 public static double Kurtosis(IList<double> sampleData)
376 {
377 if (sampleData.Count == 0) return double.NaN;
378
379 double mean = Mean(sampleData);
380 int n = sampleData.Count;
381 double s2 = 0d, s4 = 0d;
382 for (int i = 0; i < n; i++)
383 {
384 double xm = sampleData[i] - mean;
385 s2 += xm * xm;
386 s4 += xm * xm * xm * xm;
387 }
388 double m2 = s2 / n;
389 double m4 = s4 / n;
390 double v = s2 / (n - 1);
391 double a = n * (n + 1) / (double)((n - 1) * (n - 2) * (n - 3));
392 double b = s4 / (v * v);
393 double c = (n - 1) * (n - 1) / (double)((n - 2) * (n - 3));
394 return a * b - 3d * c;
395 }

405 {

449 public static double[] ProductMoments(IList<double> sampleData)
450 {
451 if (sampleData.Count == 0)
452 {
453 return [double.NaN, double.NaN, double.NaN, double.NaN];
454 }
455
456 // Create variables
457 double M; // mean
458 double V; // variance
459 double S; // standard deviation
460 double G; // skew
461 double K; // kurtosis
462 double X = 0; // sum
463 double X2 = 0; // sum of X^2
464 double X3 = 0; // sum of X^3
465 double X4 = 0; // sum of X^4
466 double U1, U2, U3, U4;
467
468 // Get sample size
469 double N = sampleData.Count;
470
471 // Compute sum
472 for (int i = 0; i <= (int)(N - 1L); i++)
473 {
474 X += sampleData[i];
475 X2 += Math.Pow(sampleData[i], 2d);
476 X3 += Math.Pow(sampleData[i], 3d);
477 X4 += Math.Pow(sampleData[i], 4d);
478 }
479
480 // Compute averages
481 U1 = X / N;
482 U2 = X2 / N;
483 U3 = X3 / N;
484 U4 = X4 / N;
485 // Get mean
486 M = U1;
487 // Compute variance
488 V = (U2 - Math.Pow(U1, 2d)) * (N / (N - 1));
489 // Compute sample standard deviation
490 S = Math.Sqrt(V);
491 // Compute sample skew
492 G = Math.Pow(N, 2d) * (U3 - 3d * U1 * U2 + 2d * Math.Pow(U1, 3d)) / ((N - 1L) * (N - 2L) * Math.Pow(S, 3d));
493 // Compute sample kurtosis
494 K = Math.Pow(N, 2d) * (N + 1L) * (U4 - 4d * U1 * U3 + 6d * U2 * Math.Pow(U1, 2d) - 3d * Math.Pow(U1, 4d)) / ((N - 1L) * (N - 2L) * (N - 3L) * Math.Pow(S, 4d)) - 3d * Math.Pow(N - 1L, 2d) / ((N - 2L) * (N - 3L));
495 return [M, S, G, K];
496 }

502 public static double[] LinearMoments(IList<double> sampleData)
503 {
504 var sample = sampleData.ToArray();
505 if (sample.Count() == 0)
506 {
507 return [double.NaN, double.NaN, double.NaN, double.NaN];
508 }
509 //
510 double B0 = 0d;
511 double B1 = 0d;
512 double B2 = 0d;
513 double B3 = 0d;
514 double N = sample.Count();
515 Array.Sort(sample);
516
517 for (int i = 1; i <= N; i++)
518 {
519 B0 += sample[i - 1];
520 if (i > 1)
521 B1 += (i - 1) / (N - 1) * sample[i - 1];
522 if (i > 2)
523 B2 += (i - 2) * (i - 1) / ((N - 2) * (N - 1)) * sample[i - 1];
524 if (i > 3)
525 B3 += (i - 3) * (i - 2) * (i - 1) / ((N - 3) * (N - 2) * (N - 1)) * sample[i - 1];
526 }
527
528 B0 /= N;
529 B1 /= N;
530 B2 /= N;
531 B3 /= N;
532 // L-Mean (λ1)
533 // L-Scale (λ2)
534 // L-Skewness (τ3)
535 // L-Kurtosis (τ4)
536 double L1 = B0;
537 double L2 = 2d * B1 - B0;
538 double T3 = 2d * (3d * B2 - B0) / (2d * B1 - B0) - 3d;
539 double T4 = 5d * (2d * (2d * B3 - 3d * B2) + B0) / (2d * B1 - B0) + 6d;
540 return [L1, L2, T3, T4];
541 }

550 public static double Percentile(IList<double> sampleData, double k, bool sampleDataIsSorted = false)
551 {
552 int n = sampleData.Count;
553 double m = (n - 1) * k + 1d;
554 //
555 if (sampleDataIsSorted == false)
556 {
557 var sortedSampleData = sampleData.ToArray(); // allows the original sampleData to not get sorted.
558 Array.Sort(sortedSampleData);
559 //
560 if (m == 1.0d)
561 {
562 return sortedSampleData[0];
563 }
564 else if (m == n)
565 {
566 return sortedSampleData[n - 1];
567 }
568 else
569 {
570 int i = (int)Math.Truncate(m);
571 double d = m - i;
572 return sortedSampleData[i - 1] + d * (sortedSampleData[i] - sortedSampleData[i - 1]);
573 }
574 }
575 else if (m == 1.0d)
576 {
577 return sampleData[0];
578 }
579 else if (m == n)
580 {
581 return sampleData[n - 1];
582 }
583 else
584 {
585 int i = (int)Math.Truncate(m);
586 double d = m - i;
587 return sampleData[i - 1] + d * (sampleData[i] - sampleData[i - 1]);
588 }
589 }

638 public static double[] RanksInplace(double[] data)
639 {
640
641 var ranks = new double[data.Length];
642 var index = new int[data.Length];
643 for (int i = 0; i < index.Length; i++)
644 {
645 index[i] = i;
646 }
647
648 Array.Sort(data, index);
649 int previousIndex = 0;
650 for (int i = 1; i < data.Length; i++)
651 {
652
653 if (Math.Abs(data[i] - data[previousIndex]) <= 0d)
654 {
655 continue;
656 }
657
658 if (i == previousIndex + 1)
659 {
660 ranks[index[previousIndex]] = i;
661 }
662 else
663 {
664 RanksTies(ranks, index, previousIndex, i);
665 }
666
667 previousIndex = i;
668 }
669
670 RanksTies(ranks, index, previousIndex, data.Length);
671 return ranks;
672 }

679 public static double[] RanksInPlace(double[] data, out double [] ties)
680 {
681
682 var ranks = new double[data.Length];
683 ties = new double[data.Length];
684 var index = new int[data.Length];
685 for (int i = 0; i < index.Length; i++)
686 {
687 index[i] = i;
688 }
689
690 Array.Sort(data, index);
691 int previousIndex = 0;
692 int t = 0;
693 for (int i = 1; i < data.Length; i++)
694 {
695 if (data[i].AlmostEquals(data[previousIndex], Tools.DoubleMachineEpsilon))
696 {
697 t += 1;
698 continue;
699 }
700
701 if (i == previousIndex + 1)
702 {
703 ranks[index[previousIndex]] = i;
704 t = 0;
705 }
706 else
707 {
708 RanksTies(ranks, index, previousIndex, i);
709 ties[i - 1] = t;
710 t = 0;
711 }
712
713 previousIndex = i;
714 }
715
716 RanksTies(ranks, index, previousIndex, data.Length);
717 return ranks;
718 }

723 private static void RanksTies(double[] ranks, int[] index, int a, int b)
724 {
725
726 double rank = (b + a - 1) / 2d + 1;
727 for (int k = a; k < b; k++)
728 {
729 ranks[index[k]] = rank;
