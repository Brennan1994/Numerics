200 private DecisionNode GrowTree(Matrix xTrain, Vector yTrain, int depth = 0)
201 {
202 int numberOfSamples = xTrain.NumberOfRows;
203 int numberOfLabels = IsRegression ? yTrain.Length : yTrain.ToList().Distinct().Count();
204
205 // Find the best split
206 var featureIdxs = Random.NextIntegers(0, Dimensions, Features, false);
207 int bestIndex = 0; double bestThreshold = 0;
208 BestSplit(xTrain, yTrain.ToArray(), featureIdxs, out bestIndex, out bestThreshold);
209
210 // Check stopping criteria
211 if (bestIndex == -1 || depth >= MaxDepth || numberOfLabels <= 1 || numberOfSamples < MinimumSplitSize)
212 {
213 if (IsRegression)
214 {
215 // If regression return the average of Y
216 var avg = Tools.Mean(yTrain.ToArray());
217 return new DecisionNode() { Value = avg, IsLeafNode = true };
218 }
219 else
220 {
221 // If classification, return the most common value
222 var most = yTrain.ToList().GroupBy(i => i).OrderByDescending(grp => grp.Count()).Select(grp => grp.Key).First();
223 return new DecisionNode() { Value = most, IsLeafNode = true };
224 }
225 }
226
227
228 // Create child nodes
229 var leftIdxs = new List<int>();
230 var rightIdxs = new List<int>();
231 Split(xTrain.Column(bestIndex), bestThreshold, out leftIdxs, out rightIdxs);
232
233 // Split to the left
234 var xLeft = new Matrix(leftIdxs.Count, xTrain.NumberOfColumns);
235 var yLeft = new Vector(leftIdxs.Count);
236 for (int i = 0; i < leftIdxs.Count; i++)
237 {
238 yLeft[i] = yTrain[leftIdxs[i]];
239 for (int j = 0; j < xTrain.NumberOfColumns; j++)
240 {
241 xLeft[i, j] = xTrain[leftIdxs[i], j];
242 }
243 }
244 var left = GrowTree(xLeft, yLeft, depth + 1);
245
246 // Split to the right
247 var xRight = new Matrix(rightIdxs.Count, xTrain.NumberOfColumns);
248 var yRight = new Vector(rightIdxs.Count);
249 for (int i = 0; i < rightIdxs.Count; i++)
250 {
251 yRight[i] = yTrain[rightIdxs[i]];
252 for (int j = 0; j < xTrain.NumberOfColumns; j++)
253 {
254 xRight[i, j] = xTrain[rightIdxs[i], j];
255 }
256 }
257 var right = GrowTree(xRight, yRight, depth + 1);
258
259 // Return decision node
260 return new DecisionNode() { FeatureIndex = bestIndex, Threshold = bestThreshold, Left = left, Right = right };
261 }

271 private void BestSplit(Matrix xTrain, double[] yTrain, int[] indices, out int bestFeatureIndex, out double bestThreshold)
272 {
273 double best = double.MinValue;
274 bestFeatureIndex = -1;
275 bestThreshold = 0;
276
277 for (int i = 0; i < indices.Length; i++)
278 {
279 var x = xTrain.Column(indices[i]);
280 var thresholds = IsRegression ? x : x.Distinct().ToArray();
281 for (int j = 0; j < thresholds.Count(); j++)
282 {
283 // Test if the split variance reduction or information gain
284 double performance = IsRegression ? VarianceReduction(x, yTrain, thresholds[j]) : InformationGain(x, yTrain, thresholds[j]);
285 // Keep track of the best value
286 if (performance > best)
287 {
288 best = performance;
289 bestFeatureIndex = indices[i];
290 bestThreshold = thresholds[j];
291 }
292 }
293 }

303 private double VarianceReduction(double[] x, double[] y, double threshold)
304 {
305 // parent entropy
306 var parentVariance = Statistics.PopulationVariance(y);
307
308 // create children
309 var leftIdxs = new List<int>();
310 var rightIdxs = new List<int>();
311 Split(x, threshold, out leftIdxs, out rightIdxs);
312
313 if (leftIdxs.Count == 0 || rightIdxs.Count == 0)
314 return double.MinValue;
315
316 // calculate the weighted average variance of the children
317 var n = (double)y.Length;
318 var nLeft = (double)leftIdxs.Count;
319 var nRight = (double)rightIdxs.Count;
320 var yLeft = new double[leftIdxs.Count];
321 for (int i = 0; i < leftIdxs.Count; i++)
322 yLeft[i] = y[leftIdxs[i]];
323 var yRight = new double[rightIdxs.Count];
324 for (int i = 0; i < rightIdxs.Count; i++)
325 yRight[i] = y[rightIdxs[i]];
326 var varLeft = Statistics.PopulationVariance(yLeft);
327 var varRight = Statistics.PopulationVariance(yRight);
328 var childVariance = nLeft / n * varLeft + nRight / n * varRight;
329
330 // Return variance reduction
331 return parentVariance - childVariance;
332 }

340 private double InformationGain(double[] x, double[] y, double threshold)
341 {
342 // parent entropy
343 var parentE = Entropy(y);
344
345 // create children
346 var leftIdxs = new List<int>();
347 var rightIdxs = new List<int>();
348 Split(x, threshold, out leftIdxs, out rightIdxs);
349
350 if (leftIdxs.Count == 0 || rightIdxs.Count == 0)
351 return double.MinValue;
352
353 // calculate the weighted average entropy of children
354 var n = (double)y.Length;
355 var nl = (double)leftIdxs.Count;
356 var nr = (double)rightIdxs.Count;
357 var yl = new double[leftIdxs.Count];
358 for (int i = 0; i < leftIdxs.Count; i++)
359 yl[i] = y[leftIdxs[i]];
360 var yr = new double[rightIdxs.Count];
361 for (int i = 0; i < rightIdxs.Count; i++)
362 yr[i] = y[rightIdxs[i]];
363 var el = Entropy(yl);
364 var er = Entropy(yr);
365 var childrenE = nl / n * el + nr / n * er;
366
367 return parentE - childrenE;
368 }

374 private double Entropy(double[] y)
375 {
376 if (IsRegression == true)
377 {
378 // use kernel density
379 var kde = new KernelDensity(y);
380 return Statistics.Entropy(y, kde.PDF);
381 }
382 else
383 {
384 // Use histogram
385 return Statistics.Entropy(y, (x) =>
386 {
387 double n = 0;
388 for (int i = 0; i < y.Length; i++)
389 {
390 if (x == y[i]){ n++; }
391 }
392 return n / y.Length;
393 });
394 }
