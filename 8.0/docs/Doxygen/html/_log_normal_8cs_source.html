379 public double[] IndirectMethodOfMoments(IList<double> sample)
380 {
381 // Transform the sample
382 var transformedSample = new List<double>();
383 for (int i = 0; i < sample.Count; i++)
384 {
385 if (sample[i] > 0d)
386 {
387 transformedSample.Add(Math.Log(sample[i], Base));
388 }
389 else
390 {
391 transformedSample.Add(Math.Log(0.1d, Base));
392 }
393 }
394 return Statistics.ProductMoments(transformedSample);
395 }

425 public double[] DirectMethodOfMoments(double mean, double standardDeviation)
426 {
427 double variance = Math.Pow(standardDeviation, 2d);
428 double mu = Math.Log(Math.Pow(mean, 2d) / Math.Sqrt(variance + Math.Pow(mean, 2d)), Base);
429 double sigma = Math.Sqrt(Math.Log(1.0d + variance / Math.Pow(mean, 2d), Base));
430 return [mu, sigma];
431 }

475 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
476 {
477 var initialVals = new double[NumberOfParameters];
478 var lowerVals = new double[NumberOfParameters];
479 var upperVals = new double[NumberOfParameters];
480 // Estimate initial values using the method of moments (a.k.a product moments).
481 var mom = IndirectMethodOfMoments(sample);
482 initialVals = new double[] { mom[0], mom[1] };
483 // Get bounds of mean
484 double real = Math.Exp(initialVals[0] / K);
485 lowerVals[0] = Tools.DoubleMachineEpsilon;
486 upperVals[0] = Math.Ceiling(Math.Log(Math.Pow(10d, Math.Ceiling(Math.Log10(real) + 1d)), Base));
487 upperVals[0] = double.IsNaN(upperVals[0]) ? 5 : upperVals[0];
488 // Get bounds of standard deviation
489 real = Math.Exp(initialVals[1] / K);
490 lowerVals[1] = Tools.DoubleMachineEpsilon;
491 upperVals[1] = Math.Ceiling(Math.Log(Math.Pow(10d, Math.Ceiling(Math.Log10(real) + 1d)), Base));
492 upperVals[1] = double.IsNaN(upperVals[1]) ? 4 : upperVals[1];
493 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
494 }

497 public double[] MLE(IList<double> sample)
498 {
499 // Set constraints
500 var tuple = GetParameterConstraints(sample);
501 var Initials = tuple.Item1;
502 var Lowers = tuple.Item2;
503 var Uppers = tuple.Item3;
504
505 // Solve using Nelder-Mead (Downhill Simplex)
506 double logLH(double[] x)
507 {
508 var LogN = new LogNormal() { Base = Base };
509 LogN.SetParameters(x);
510 return LogN.LogLikelihood(sample);
511 }
512 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
513 solver.ReportFailure = true;
514 solver.Maximize();
515 return solver.BestParameterSet.Values;
516
517 }

520 public override double PDF(double x)
521 {
522 if (_parametersValid == false)
523 ValidateParameters(Mu, Sigma, true);
524 if (x <= Minimum) return 0.0d;
525 double d = (Math.Log(x, Base) - Mu) / Sigma;
526 return Math.Exp(-0.5d * d * d) / (Tools.Sqrt2PI * Sigma) * (K / x);
527 }
528
530 public override double CDF(double x)
531 {
532 if (_parametersValid == false)
533 ValidateParameters(Mu, Sigma, true);
534 if (x <= Minimum)
535 return 0d;
536 return 0.5d * (1.0d + Erf.Function((Math.Log(x, Base) - Mu) / (Sigma * Math.Sqrt(2.0d))));
537 }

540 public override double InverseCDF(double probability)
541 {
542 // Validate probability
543 if (probability < 0.0d || probability > 1.0d)
544 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
545 if (probability == 0.0d)
546 return Minimum;
547 if (probability == 1.0d)
548 return Maximum;
549 // Validate parameters
550 if (_parametersValid == false)
551 ValidateParameters(Mu, Sigma, true);
552 return Math.Exp((Mu - Sigma * Math.Sqrt(2.0d) * Erf.InverseErfc(2.0d * probability)) / K);
553 }

565 public double[,] MonteCarloConfidenceIntervals(int sampleSize, int realizations, IList<double> quantiles, IList<double> percentiles)
566 {
567 // validate parameters
568 if (_parametersValid == false)
569 ValidateParameters(Mu, _sigma, true);
570 // Dimension output array
571 int q = quantiles.Count;
572 int p = percentiles.Count;
573 var Output = new double[q, p];
574
575 // Variables
576 double OriginalMean = Mu;
577 double OriginalStdDev = Sigma;
578
579 // Create random numbers for mean and standard deviation
580 var r = new MersenneTwister(12345);
581 var rndMean = r.NextDoubles(realizations);
582 r = new MersenneTwister(45678);
583 var rndStdDev = r.NextDoubles(realizations);
584
585
586 // Create list of Monte Carlo distributions
587 var MonteCarloDistributions = new UnivariateDistributionBase[realizations];
588 Parallel.For(0, realizations, idx =>
589 {
590
591 // Generate new mean
592 var Normal = new Normal(OriginalMean, OriginalStdDev / Math.Sqrt(sampleSize));
593 double NewMu = Normal.InverseCDF(rndMean[idx]);
594 // Generate new standard deviation
595 var Chi = new ChiSquared(sampleSize - 1);
596 double NewSigma = Math.Sqrt((sampleSize - 1) * Math.Pow(OriginalStdDev, 2d) / Chi.InverseCDF(rndStdDev[idx]));
597 // Create a new distribution with the new parameters
598 MonteCarloDistributions[idx] = new LogNormal(NewMu, NewSigma);
599 });
600
601 // Create confidence intervals
602 for (int i = 0; i < q; i++)
603 {
604 // Create array of X values across user-defined probabilities
605 var XValues = new double[realizations];
606 // Record X values
607 Parallel.For(0, realizations, idx => XValues[idx] = MonteCarloDistributions[idx].InverseCDF(1d - quantiles[i]));
608 // Record percentiles for user-defined probabilities
609 for (int j = 0; j < p; j++)
610 Output[i, j] = Statistics.Percentile(XValues, percentiles[j]);
611 }
612
613 // Return confidence percentile output
614 return Output;
615 }

624 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
625 {
626 if (estimationMethod != ParameterEstimationMethod.MethodOfMoments &&
627 estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
628 {
629 throw new NotImplementedException();
630 }
631 // Validate parameters
632 if (_parametersValid == false)
633 ValidateParameters(Mu, _sigma, true);
634 // Compute covariance
635 double u2 = Sigma;
636 var covar = new double[2, 2];
637 covar[0, 0] = Math.Pow(u2, 2d) / sampleSize; // location
638 covar[1, 1] = 2d * Math.Pow(u2, 4d) / sampleSize; // scale
639 covar[0, 1] = 0.0;
640 covar[1, 0] = covar[0, 1];
641 return covar;
642 }

645 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
646 {
647 var covar = ParameterCovariance(sampleSize, estimationMethod);
648 var grad = QuantileGradient(probability);
649 double varA = covar[0, 0];
650 double varB = covar[1, 1];
651 double covAB = covar[1, 0];
652 double dQx1 = grad[0];
653 double dQx2 = grad[1];
654 double varQ = Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + 2d * dQx1 * dQx2 * covAB;
655 return varQ * Math.Pow(InverseCDF(probability) / K, 2d);
656 }

674 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
675 {
676 if (probabilities.Count != NumberOfParameters)
677 {
678 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
679 }
680 // Get gradients
681 var dQp1 = QuantileGradient(probabilities[0]);
682 var dQp2 = QuantileGradient(probabilities[1]);
683 // Compute determinant
684 // |a b|
685 // |c d|
686 // |A| = ad − bc
687 double p0 = InverseCDF(probabilities[0]) / K;
688 double p1 = InverseCDF(probabilities[1]) / K;
689 double a = dQp1[0] * p0;
690 double b = dQp1[1] * p0;
691 double c = dQp2[0] * p1;
692 double d = dQp2[1] * p1;
693 determinant = a * d - b * c;
694 // Return Jacobian
695 var jacobian = new double[,] { { a, b }, { c, d } };
696 return jacobian;
697 }
