Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

60 public Interpolater(IList<double> xValues, IList<double> yValues, SortOrder sortOrder = SortOrder.Ascending)
61 {
62 Count = xValues.Count;
63
64 // Validate
65 if (yValues.Count != Count) throw new ArgumentException(nameof(xValues), "The x and y lists must be the same length.");
66 if (Count < 2) throw new ArgumentException(nameof(xValues), "The x list is too small. It must have at least 2 values.");
67 for (int i = 1; i < xValues.Count; ++i)
68 {
69 if (xValues[i] == xValues[i - 1]) throw new ArgumentException(nameof(xValues), "All x values should be unique.");
70 if (sortOrder == SortOrder.Ascending && xValues[i] < xValues[i - 1]) throw new ArgumentException(nameof(xValues), "The x values are not in ascending order.");
71 if (sortOrder == SortOrder.Descending && xValues[i] > xValues[i - 1]) throw new ArgumentException(nameof(xValues), "The x values are not in descending order.");
72 }
73 this.XValues = xValues;
74 this.YValues = yValues;
75 deltaStart = Math.Min(1, (int)Math.Pow((double)Count, 0.25));
76 SortOrder = sortOrder;
77
78 }

156 public int Search(double x)
157 {
158 int start = 0;
159 if (UseSmartSearch)
160 {
161 start = correlated ? HuntSearch(x) : BisectionSearch(x);
162 }
163 else
164 {
165 start = SequentialSearch(x);
166 }
167 correlated = Math.Abs(start - SearchStart) > deltaStart ? false : true;
168 SearchStart = start < 0 || start >= Count ? 0 : start;
169 return start;
170 }

179 public int SequentialSearch(double x)
180 {
181 int jl = SearchStart;
182 if ((SortOrder == SortOrder.Ascending && x < XValues[0]) ||
183 (SortOrder == SortOrder.Descending && x > XValues[0]))
184 {
185 return 0;
186 }
187 else if ((SortOrder == SortOrder.Ascending && x > XValues[Count - 1]) ||
188 (SortOrder == SortOrder.Descending && x < XValues[Count - 1]))
189 {
190 return Count - 2;
191 }
192 else if ((SortOrder == SortOrder.Ascending && x < XValues[SearchStart]) ||
193 (SortOrder == SortOrder.Descending && x > XValues[SearchStart]))
194 {
195 jl = 0;
196 }
197 for (int i = jl; i < Count; i++)
198 {
199 if ((SortOrder == SortOrder.Ascending && x <= XValues[i]) ||
200 (SortOrder == SortOrder.Descending && x >= XValues[i]))
201 {
202 jl = i - 1;
203 break;
204 }
205 }
206 return jl;
207 }

216 public int BisectionSearch(double x)
217 {
218 int ju = Count - 1, jm, jl = 0;
219 bool ascnd = SortOrder == SortOrder.Ascending;
220 while (ju - jl > 1)
221 {
222 jm = (ju + jl) >> 1;
223 if (x >= XValues[jm] == ascnd)
224 jl = jm;
225 else
226 ju = jm;
227 }
228 return jl;
229 }

238 public int HuntSearch(double x)
239 {
240 int jl = SearchStart, jm, ju, inc = 1;
241 bool ascnd = SortOrder == SortOrder.Ascending;
242 if (jl < 0 || jl > Count - 1)
243 {
244 jl = 0;
245 ju = Count - 1;
246 }
247 else
248 {
249 if (x >= XValues[jl] == ascnd)
250 {
251 for (; ; )
252 {
253 ju = jl + inc;
254 if (ju >= Count - 1) { ju = Count - 1; break; }
255 else if (x < XValues[ju] == ascnd) break;
256 else
257 {
258 jl = ju;
259 inc += inc;
260 }
261 }
262 }
263 else
264 {
265 ju = jl;
266 for (; ; )
267 {
268 jl = jl - inc;
269 if (jl <= 0) { jl = 0; break; }
270 else if (x >= XValues[jl] == ascnd) break;
271 else
272 {
273 ju = jl;
274 inc += inc;
275 }
276 }
277 }
278 }
279 while (ju - jl > 1)
280 {
281 jm = (ju + jl) >> 1;
282 if (x >= XValues[jm] == ascnd)
283 jl = jm;
284 else
285 ju = jm;
286 }
287 return jl;
288 }
