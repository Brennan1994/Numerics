Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

64 public StratificationBin(double lowerBound, double upperBound, double weight = -1)
65 {
66 // validate inputs
67 if (lowerBound > upperBound)
68 {
69 throw new ArgumentOutOfRangeException(nameof(LowerBound), "The upper bound must be greater than or equal to the lower bound.");
70 }
71
72 LowerBound = lowerBound;
73 UpperBound = upperBound;
74 if (weight < 0d)
75 {
76 Weight = upperBound - lowerBound;
77 }
78 else
79 {
80 Weight = weight;
81 }
82 }

151 public int CompareTo(StratificationBin other)
152 {
153 if (UpperBound > other.LowerBound && LowerBound < other.LowerBound)
154 {
155 throw new ArgumentException(nameof(other), "The bins cannot be overlapping.");
156 }
157
158 if (UpperBound.Equals(other.UpperBound) && LowerBound.Equals(other.LowerBound))
159 return 0;
160 if (other.UpperBound <= LowerBound)
161 return 1;
162 return -1;
