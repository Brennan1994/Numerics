Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

186 protected override ParameterSet ChainIteration(int index, ParameterSet state)
187 {
188
189 // Update the sample count
190 SampleCount[index] += 1;
191
192 // Jigger the step size and number of steps
193 var _stepSize = _stepSizeU.InverseCDF(_chainPRNGs[index].NextDouble());
194 var _steps = (int)Math.Ceiling(_stepsU.InverseCDF(_chainPRNGs[index].NextDouble()));
195
196 // Step 1. Sample phi from a N~(0,M)
197 var phi = new Vector(NumberOfParameters);
198 for (int i = 0; i < NumberOfParameters; i++)
199 phi[i] = Math.Sqrt(Mass[i]) * Normal.StandardZ(_chainPRNGs[index].NextDouble());
200
201 // Get kinetic energy of the current state
202 var logKi = -0.5 * Statistics.Sum((_inverseMass * phi * phi).ToArray());
203
204 // Step 2. Perform leapfrog steps to get proposal vector
205 var xp = new Vector(state.Values.ToArray());
206 phi += GradientFunction(xp.ToArray()) * _stepSize * 0.5;
207 for (int i = 0; i < _steps; i++)
208 {
209 xp += _inverseMass * phi * _stepSize;
210
211 // Ensure the parameters are feasible (within the constraints)
212 for (int j = 0; j < NumberOfParameters; j++)
213 {
214 if (xp[j] < PriorDistributions[j].Minimum)
215 xp[j] = PriorDistributions[j].Minimum + Tools.DoubleMachineEpsilon;
216 if (xp[j] > PriorDistributions[j].Maximum)
217 xp[j] = PriorDistributions[j].Maximum - Tools.DoubleMachineEpsilon;
218 }
219
220 phi += GradientFunction(xp.ToArray()) * _stepSize * (i == _steps-1 ? 0.5: 1.0);
221 }
222 phi *= -1d;
223
224 // Get kinetic energy of the proposal state
225 var logKp = -0.5 * Statistics.Sum((_inverseMass * phi * phi).ToArray());
226
227 // Evaluate fitness
228 var logLHp = LogLikelihoodFunction(xp.ToArray());
229 var logLHi = state.Fitness;
230
231 // Calculate the Metropolis ratio
232 var logRatio = logLHp - logLHi + logKp - logKi;
233
234 // Accept the proposal with probability min(1,r)
235 // otherwise leave xi unchanged
236 var logU = Math.Log(_chainPRNGs[index].NextDouble());
237 if (logU <= logRatio)
238 {
239 // The proposal is accepted
240 AcceptCount[index] += 1;
241 return new ParameterSet(xp.ToArray(), logLHp);
242 }
243 else
244 {
245 return state;
246 }
247
248 }
