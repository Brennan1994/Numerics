111 public EmpiricalDistribution(OrderedPairedData orderedPairedData)
112 {
113 if (orderedPairedData.OrderX != SortOrder.Ascending) throw new ArgumentException("The x values must be in ascending order", nameof(orderedPairedData));
114 opd = orderedPairedData;
115 _xValues = orderedPairedData.Select(v => v.X).ToArray();
116
117 // Probability values can be in ascending or descending order.
118 // If the sort order is "None", then we cannot use smart search.
119
120 // Check if the probability values are actually in ascending order
121 if (opd.OrderY == SortOrder.None)
122 {
123 bool isAsc = true;
124 _pValues = new double[opd.Count];
125 for (int i = 0; i < opd.Count; i++)
126 {
127 _pValues[i] = opd[i].Y;
128 if (i > 0 && _pValues[i] <= _pValues[i - 1])
129 {
130 isAsc = false;
131 }
132 }
133 if (isAsc == true)
134 {
135 opd = new OrderedPairedData(_xValues, _pValues, true, SortOrder.Ascending, true, SortOrder.Ascending);
136 }
137 else
138 {
139 opd.UseSmartSearch = false;
140 }
141 }
142 else
143 {
144 _pValues = orderedPairedData.Select(v => v.Y).ToArray();
145 }
146 }

153 public EmpiricalDistribution(IList<double> sample, PlottingPositions.PlottingPostionType plottingPostionType = PlottingPositions.PlottingPostionType.Weibull)
154 {
155 _xValues = sample.ToArray();
156 Array.Sort(_xValues);
157 _pValues = PlottingPositions.Function(_xValues.Count(), plottingPostionType);
158 opd = new OrderedPairedData(_xValues, _pValues, true, SortOrder.Ascending, true, SortOrder.Ascending);
159 }

411 public override double PDF(double X)
412 {
413 if (X < Minimum || X > Maximum) return 0.0d;
414 var d = NumericalDerivative.Derivative(CDF, X);
415 return d < 0d ? 0d : d;
416 }
417
423 public double PDF(double xl, double xu)
424 {
425 if (xu == xl) return PDF(xu);
426 if (xu < xl) return 0.0d;
427 if (xl < Minimum || xl > Maximum || xu < Minimum || xu > Maximum) return 0.0d;
428 var d = (CDF(xu) - CDF(xl)) / (xu - xl);
429 return d < 0d ? 0d : d;
430 }

433 public override double CDF(double X)
434 {
435 double p = 0;
436 if (opd.OrderY == SortOrder.Ascending || opd.OrderY == SortOrder.None)
437 {
438 p = opd.GetYFromX(X, XTransform, ProbabilityTransform);
439 }
440 else
441 {
442 // If descending then it is a survival function
443 p = 1d - opd.GetYFromX(X, XTransform, ProbabilityTransform);
444 }
445 return p < 0d ? 0d : p > 1d ? 1d : p;
446 }
447
449 public override double InverseCDF(double probability)
450 {
451 // Validate probability
452 if (probability < 0.0d || probability > 1.0d)
453 throw new ArgumentOutOfRangeException(nameof(probability), "Probability must be between 0 and 1.");
454
455 double min = Minimum;
456 double max = Maximum;
457 if (probability <= 1E-16) return min;
458 if (probability >= 1 - 1E-16) return max;
459 double x = 0;
460 if (opd.OrderY == SortOrder.Ascending || opd.OrderY == SortOrder.None)
461 {
462 x = opd.GetXFromY(probability, XTransform, ProbabilityTransform);
463 }
464 else
465 {
466 // If descending then it is a survival function
467 x = opd.GetXFromY(1d - probability, XTransform, ProbabilityTransform);
468 }
469 return x < min ? min : x > max ? max : x;
470 }
