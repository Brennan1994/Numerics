Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

263 public override double PDF(double k)
264 {
265 // Validate parameters
266 if (_parametersValid == false)
267 ValidateParameters([ProbabilityOfSuccess, NumberOfTrials], true);
268 k = Math.Floor(k);
269 if (k < Minimum || k > Maximum) return 0.0d;
270 if (ProbabilityOfSuccess == 0.0d)
271 return k == 0d ? 1.0d : 0.0d;
272 if (ProbabilityOfSuccess == 1.0d)
273 return k == NumberOfTrials ? 1.0d : 0.0d;
274 return Factorial.BinomialCoefficient(NumberOfTrials, (int)k) * Math.Pow(ProbabilityOfSuccess, k) * Math.Pow(Complement, NumberOfTrials - k);
275 }
276
278 public override double CDF(double k)
279 {
280 if (_parametersValid == false)
281 ValidateParameters([ProbabilityOfSuccess, NumberOfTrials], true);
282 k = Math.Floor(k);
283 if (k < Minimum)
284 return 0.0d;
285 if (k > Maximum)
286 return 1.0d;
287 return Beta.Incomplete(NumberOfTrials - k, k + 1d, Complement);
288 }
289
291 public override double InverseCDF(double probability)
292 {
293 // Validate probability
294 if (probability < 0.0d || probability > 1.0d)
295 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
296 if (probability == 0.0d)
297 return Minimum;
298 if (probability == 1.0d)
299 return Maximum;
300 // Validate parameters
301 if (_parametersValid == false)
302 ValidateParameters([probability, NumberOfTrials], true);
303 double k = 0d;
304 for (int i = 0; i < NumberOfTrials; i++)
305 {
306 if (CDF(i) >= probability)
307 {
308 k = i;
309 break;
310 }
311 }
312 return k;
313 }
