The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

120 {
121 try
122 {
123 using (var memStream = new MemoryStream())
124 {
125 var binForm = new BinaryFormatter();
126 memStream.Write(bytes, 0, bytes.Length);
127 memStream.Seek(0, SeekOrigin.Begin);
128 var obj = binForm.Deserialize(memStream);
129 return (UncertaintyAnalysisResults)obj;

134 // An error can occur because of differences in versions.

175 public static UncertaintyAnalysisResults FromXElement(XElement xElement)
176 {
177 var ua = new UncertaintyAnalysisResults();
178 // Parent distribution
179 if (xElement.Element("Distribution") != null)
180 ua.ParentDistribution = UnivariateDistributionFactory.CreateDistribution(xElement.Element("Distribution"));
181
182 double outDbl = 0;
183 // AIC
184 if (xElement.Attribute(nameof(AIC)) != null)
185 {
186 double.TryParse(xElement.Attribute(nameof(AIC)).Value, out outDbl);
187 ua.AIC = outDbl;
188 }
189 // BIC
190 if (xElement.Attribute(nameof(BIC)) != null)
191 {
192 double.TryParse(xElement.Attribute(nameof(BIC)).Value, out outDbl);
193 ua.BIC = outDbl;
194 }
195 // RMSE
196 if (xElement.Attribute(nameof(RMSE)) != null)
197 {
198 double.TryParse(xElement.Attribute(nameof(RMSE)).Value, out outDbl);
199 ua.RMSE = outDbl;
200 }
201 // ERL
202 if (xElement.Attribute(nameof(ERL)) != null)
203 {
204 double.TryParse(xElement.Attribute(nameof(ERL)).Value, out outDbl);
205 ua.ERL = outDbl;
206 }
207
208 // Mode Curve
209 if (xElement.Attribute(nameof(ua.ModeCurve)) != null)
210 {
211 var vals = xElement.Attribute(nameof(ua.ModeCurve)).Value.Split('|');
212 if (vals.Length > 0)
213 {
214 ua.ModeCurve = new double[vals.Length];
215 for (int i = 0; i < vals.Length; i++)
216 {
217 double.TryParse(vals[i], out ua.ModeCurve[i]);
218 }
219 }
220 }
221 // Mean Curve
222 if (xElement.Attribute(nameof(ua.MeanCurve)) != null)
223 {
224 var vals = xElement.Attribute(nameof(ua.MeanCurve)).Value.Split('|');
225 if (vals.Length > 0)
226 {
227 ua.MeanCurve = new double[vals.Length];
228 for (int i = 0; i < vals.Length; i++)
229 {
230 double.TryParse(vals[i], out ua.MeanCurve[i]);
231 }
232 }
233 }
234 // Confidence Intervals
235 if (xElement.Attribute(nameof(ua.ConfidenceIntervals)) != null)
236 {
237 var vals = xElement.Attribute(nameof(ua.ConfidenceIntervals)).Value.Split('|');
238 if (vals.Length > 0)
239 {
240 ua.ConfidenceIntervals = new double[vals.Length, vals[0].Split(',').Length];
241 for (int i = 0; i < vals.Length; i++)
242 {
243 var jVals = vals[i].Split(',');
244 for (int j = 0; j < jVals.Length; j++)
245 {
246 double.TryParse(jVals[j], out ua.ConfidenceIntervals[i, j]);
247 }
248 }
249 }
250 }
251 return ua;
252 }

262 public static UncertaintyAnalysisResults FromMCMCResults(MCMCResults results, UnivariateDistributionBase distribution, IList<double> probabilities, double alpha = 0.1)
263 {
264 if (results == null|| results.Output == null ||results.Output.Count == 0) return null;
265 if (results.Output[0].Values.Length != distribution.NumberOfParameters) return null;
266
267 int realz = results.Output.Count;
268 var dists = new UnivariateDistributionBase[results.Output.Count];
269
270 Parallel.For(0, realz, idx =>
271 {
272 var d = distribution.Clone();
273 d.SetParameters(results.Output[idx].Values);
274 dists[idx] = d;
275 });
276
277 // Set up dummy bootstrap analysis
278 var boot = new BootstrapAnalysis(distribution, ParameterEstimationMethod.MaximumLikelihood, 100, realz);
279 var uar = boot.Estimate(probabilities, alpha, dists);
280 uar.ParameterSets = null;
281 return uar;
282 }
