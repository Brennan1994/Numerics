66 public RWMH(List<IUnivariateDistribution> priorDistributions, LogLikelihood logLikelihoodFunction, Matrix proposalSigma) : base(priorDistributions, logLikelihoodFunction)

80 protected override void ValidateCustomSettings()
81 {
82 if (ProposalSigma == null) throw new ArgumentException(nameof(ProposalSigma), "The proposal covariance matrix cannot be null.");
83 if (ProposalSigma.NumberOfRows != ProposalSigma.NumberOfColumns) throw new ArgumentException(nameof(ProposalSigma), "The proposal covariance matrix must be square.");
84 if (ProposalSigma.NumberOfRows != NumberOfParameters) throw new ArgumentException(nameof(ProposalSigma), "The proposal covariance matrix must have the same number of rows and columns as the number of parameters.");
85 }
86
88 protected override void InitializeCustomSettings()
89 {
90 // Set up multivariate Normal distributions for each chain
91 mvn = new MultivariateNormal[NumberOfChains];
92 for (int i = 0; i < NumberOfChains; i++)
93 {
94 mvn[i] = new MultivariateNormal(NumberOfParameters);
95 }
96 // Set up proposal matrix
97 if (Initialize == InitializationType.MAP && _mapSuccessful)
98 {
99 ProposalSigma = new Matrix(base._MVN.Covariance);
100 }
101 }
102
104 protected override ParameterSet ChainIteration(int index, ParameterSet state)
105 {
106 // Update the sample count
107 SampleCount[index] += 1;
108
109 // Get proposal vector
110 mvn[index].SetParameters(state.Values.ToArray(), ProposalSigma.ToArray());
111 var xp = mvn[index].InverseCDF(_chainPRNGs[index].NextDoubles(NumberOfParameters));
112
113 for (int i = 0; i < NumberOfParameters; i++)
114 {
115 // Check if the parameter is feasible (within the constraints)
116 if (xp[i] < PriorDistributions[i].Minimum || xp[i] > PriorDistributions[i].Maximum)
117 {
118 // The proposed parameter vector was infeasible,
119 // so leave xi unchanged.
120 return state;
121 }
122 }
123
124 // Evaluate fitness
125 var logLHp = LogLikelihoodFunction(xp);
126 var logLHi = state.Fitness;
127
128 // Calculate the Metropolis ratio
129 var logRatio = logLHp - logLHi;
130
131 // Accept the proposal with probability min(1,r)
132 // otherwise leave xi unchanged
133 var logU = Math.Log(_chainPRNGs[index].NextDouble());
134 if (logU <= logRatio)
135 {
136 // The proposal is accepted
137 AcceptCount[index] += 1;
138 return new ParameterSet(xp, logLHp);
139 }
140 else
141 {
142 return state;
143 }
144 }
