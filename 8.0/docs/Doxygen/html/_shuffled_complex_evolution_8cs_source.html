95 public ShuffledComplexEvolution(Func<double[], double> objectiveFunction, int numberOfParameters, IList<double> lowerBounds, IList<double> upperBounds) : base(objectiveFunction, numberOfParameters)
96 {
97 // Check if the length of the lower and upper bounds equal the number of parameters
98 if (lowerBounds.Count != numberOfParameters || upperBounds.Count != numberOfParameters)
99 {
100 throw new ArgumentOutOfRangeException(nameof(lowerBounds), "The lower and upper bounds must be the same length as the number of parameters.");
101 }
102 // Check if the lower bounds are less than the upper bounds
103 for (int i = 0; i < lowerBounds.Count; i++)
104 {
105 if (upperBounds[i] <= lowerBounds[i])
106 {
107 throw new ArgumentOutOfRangeException(nameof(upperBounds), "The upper bound cannot be less than or equal to the lower bound.");
108 }
109 }
110 LowerBounds = lowerBounds.ToArray();
111 UpperBounds = upperBounds.ToArray();
112 CCEIterations = 2 * NumberOfParameters + 1;
113 }

146 protected override void Optimize()
147 {
148 if (Complexes < 1) throw new ArgumentOutOfRangeException(nameof(Complexes), "The number of complexes must be greater than 0.");
149 if (CCEIterations < 1) throw new ArgumentOutOfRangeException(nameof(CCEIterations), "The number of CCE iterations must be greater than 0.");
150 if (ToleranceSteps < 1) throw new ArgumentOutOfRangeException(nameof(ToleranceSteps), "The number of tolerance steps must be greater than 0.");
151
152 int i, j, D = NumberOfParameters;
153 bool cancel = false;
154
155 int populationSize = Complexes * CCEIterations;
156 int converge = 0, alpha = 1;
157 double oldFit, newFit;
158
159 // Set lower and upper bounds and
160 // create uniform distributions for each parameter
161 var uniformDists = new List<Uniform>();
162 for (i = 0; i < D; i++)
163 {
164 uniformDists.Add(new Uniform(LowerBounds[i], UpperBounds[i]));
165 }
166
167 // Initialize the population of points
168 var r = new Random(PRNGSeed);
169 var Dpoints = new List<PointFitness>();
170 for (i = 0; i < populationSize; i++)
171 {
172 var values = new double[D];
173 for (j = 0; j < D; j++)
174 values[j] = uniformDists[j].InverseCDF(r.NextDouble());
175
176 Dpoints.Add(new PointFitness { ParameterSet = new ParameterSet(values, Evaluate(values, ref cancel)), Index = i });
177 if (cancel == true) return;
178 }
179 oldFit = Dpoints[0].ParameterSet.Fitness;
180
181 Iterations += 1;
182
183 // Initialize shuffled complex evolution (SCE) algorithm
184 // Create list of complex prngs
185 var complexPRNGs = new List<Random>();
186 for (i = 0; i < Complexes; i++)
187 complexPRNGs.Add(new Random(r.Next()));
188
189 // create trapezoidal cumulative probability for points in complex
190 var cdf = Trapezoidal(CCEIterations);
191
192 // Perform the Shuffled complex evolution loop
193 while (Iterations < MaxIterations)
194 {
195
196 // Partition populations into p complexes distributing points evenly between complexes
197 var Acomplex = new List<List<PointFitness>>(); // The A complex
198 for (i = 0; i < Complexes; i++)
199 {
200 Acomplex.Add(new List<PointFitness>());
201 for (j = 0; j < CCEIterations; j++)
202 Acomplex[i].Add(Dpoints[i + Complexes * j].Clone());
203 }
204
205 // Evolve each complex according to competitive complex evolution (CCE) algorithm
206 for (i = 0; i < Complexes; i++)
207 {
208 var tmp = Acomplex;
209 var argAcomplex = tmp[i];
210 EvolveComplex(alpha, ref argAcomplex, cdf, complexPRNGs[i], ref cancel);
211 tmp[i] = argAcomplex;
212 // Check cancel
213 if (cancel == true) return;
214 }
215
216 // Replace A into D
217 for (i = 0; i < Complexes; i++)
218 {
219 for (j = 0; j < CCEIterations; j++)
220 Dpoints[i + Complexes * j] = Acomplex[i][j].Clone();
221 }
222
223 // Rank the points in terms of fitness (i.e., order of increasing function value).
224 Dpoints.Sort((x, y) => x.ParameterSet.Fitness.CompareTo(y.ParameterSet.Fitness));
225
226 // Next reset the indexes, so that i = 0 represents the point With the smallest function value.
227 for (i = 0; i <= Dpoints.Count - 1; i++)
228 Dpoints[i].Index = i;
229
230 newFit = Dpoints.First().ParameterSet.Fitness;
231
232 // Check convergence
233 if (CheckConvergence(oldFit, newFit))
234 {
235 // If no improvement increase alpha to increase chance of sub-complexes evolving
236 converge += 1;
237 alpha = Math.Min(3, converge + 1);
238 oldFit = newFit;
239
240 if (converge >= ToleranceSteps)
241 {
242 UpdateStatus(OptimizationStatus.Success);
243 return;
244 }
245
246 }
247 else
248 {
249 oldFit = newFit;
250 converge = 0;
251 alpha = 1;
252 }
253
254 Iterations += 1;
255 }
256
257 // If we made it to here, the maximum iterations were reached.
258 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
259 }

270 private void EvolveComplex(int alpha, ref List<PointFitness> Acomplex, double[] cdf, Random prng, ref bool cancel)
271 {
272 int beta = Acomplex.Count;
273 int q = NumberOfParameters + 1;
274 int i, j, k;
275 var g = new double[NumberOfParameters]; // centroid g
276 var p = new double[NumberOfParameters];
277 double fitness;
278
279 // The Beta loop. Allow sub-complex to evolve beta times.
280 for (i = 0; i < beta; i++)
281 {
282
283 // Select parents by randomly choosing q distinct points from A complex
284 var B = new List<PointFitness>();
285 for (j = 1; j <= q; j++)
286 {
287 double rnd = prng.NextDouble();
288 int rndi = cdf.Count() - 1;
289 for (k = 0; k <= cdf.Count() - 1; k++)
290 {
291 if (rnd >= cdf[k])
292 {
293 rndi = k;
294 break;
295 }
296 }
297
298 B.Add(Acomplex[rndi].Clone());
299 B.Last().Index = rndi;
300 }
301
302 // Alpha loop. The worst point in the complex is reflected or contracted to seek an improvement alpha times.
303 for (j = 0; j < alpha; j++)
304 {
305
306 // Rank the B points in terms of fitness (i.e., order of increasing function value.
307 B.Sort((x, y) => x.ParameterSet.Fitness.CompareTo(y.ParameterSet.Fitness));
308
309 // Find centroid g excluding worst point and compute reflection of worst point about centroid r = 2g - u(worst)
310 Reflection(ref B, ref g, ref p);
311
312 // Check if r is feasible
313 if (IsFeasible(p) == true)
314 {
315 // The parameter set is feasible, so evaluate it.
316 fitness = Evaluate(p, ref cancel);
317 if (cancel == true) return;
318 }
319 else
320 {
321 // The parameters were infeasible, so perform mutation
322 // Compute the smallest hypercube enclosing A complex and randomly sample a point within it.
323 SmallestHypercube(ref Acomplex, ref prng, ref p);
324 // Evaluate random point
325 fitness = Evaluate(p, ref cancel);
326 if (cancel == true) return;
327 }
328
329 // Now, either replace worst point with better point
330 if (fitness < B.Last().ParameterSet.Fitness)
331 {
332 B.Last().ParameterSet = new ParameterSet(p, fitness);
333 }
334 else
335 {
336 // Or contract to midpoint between centroid and worst point
337 Contraction(ref B, ref g, ref p);
338 // Evaluate contracted point
339 fitness = Evaluate(p, ref cancel);
340 if (cancel == true) return;
341
342 // If better than worst point replace worst point
343 if (fitness < B.Last().ParameterSet.Fitness)
344 {
345 B.Last().ParameterSet = new ParameterSet(p, fitness);
346 }
347 else
348 {
349 // Otherwise perform mutation
350 // Compute the smallest hypercube enclosing A complex and randomly sample a point within it.
351 SmallestHypercube(ref Acomplex, ref prng, ref p);
352 // Evaluate random point
353 fitness = Evaluate(p, ref cancel);
354 if (cancel == true) return;
355 // Replace worst point with new point regardless of its value
356 B.Last().ParameterSet = new ParameterSet(p, fitness);
357 }
358 }
359 }
360
361 // Replace B into A according to L And sort A in order of increasing function value
362 for (j = 0; j < Acomplex.Count; j++)
363 {
364 var best = Acomplex[j].Clone();
365 for (k = 0; k <= q - 1; k++)
366 {
367 if (B[k].Index == j)
368 {
369 if (B[k].ParameterSet.Fitness < best.ParameterSet.Fitness)
370 {
371 best = B[k].Clone();
372 }
373 }
374 }
375 Acomplex[j].ParameterSet = best.ParameterSet.Clone();
376 }
377
378 Acomplex.Sort((x, y) => x.ParameterSet.Fitness.CompareTo(y.ParameterSet.Fitness));
379 }
380 }

386 private bool IsFeasible(double[] point)
387 {
388 bool feasible = true;
389 for (int i = 0; i <= point.Count() - 1; i++)
390 {
391 if (point[i] < LowerBounds[i] || point[i] > UpperBounds[i])
392 {
393 feasible = false;
394 break;
395 }
396 }
397 return feasible;
398 }

406 private void Reflection(ref List<PointFitness> U, ref double[] g, ref double[] r)
407 {
408 for (int i = 0; i < NumberOfParameters; i++)
409 {
410 g[i] = 0.0d;
411 for (int j = 0; j < U.Count - 1; j++)
412 g[i] += U[j].ParameterSet.Values[i];
413 g[i] /= U.Count - 1;
414 r[i] = 2d * g[i] - U.Last().ParameterSet.Values[i]; // u(worst)
415 }
416 }

424 private void Contraction(ref List<PointFitness> U, ref double[] g, ref double[] c)
425 {
426 for (int i = 0; i < NumberOfParameters; i++)
427 c[i] = 0.5d * (g[i] + U.Last().ParameterSet.Values[i]); // u(worst)
428 }

436 private void SmallestHypercube(ref List<PointFitness> Acomplex, ref Random prng, ref double[] z)
437 {
438 double low, high;
439 for (int i = 0; i < NumberOfParameters; i++)
440 {
441 low = Acomplex[0].ParameterSet.Values[i];
442 high = low;
443 for (int j = 1; j < Acomplex.Count; j++)
444 {
445 low = Math.Min(low, Acomplex[j].ParameterSet.Values[i]);
446 high = Math.Max(high, Acomplex[j].ParameterSet.Values[i]);
447 }
448 z[i] = low + (high - low) * prng.NextDouble();
449 }
450 }

457 private double[] Trapezoidal(int N)
458 {
459 var PP = new double[N];
460 for (int i = 1; i <= N; i++)
461 PP[i - 1] = 2 * (N + 1 - i) / (double)(N * (N + 1));
462 return PP;
463 }
