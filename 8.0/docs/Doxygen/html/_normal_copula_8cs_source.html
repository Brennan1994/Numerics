149 public override double PDF(double u, double v)
150 {
151 // Validate parameters
152 if (_parametersValid == false) ValidateParameter(Theta, true);
153 var r = _theta;
154 var s = Normal.StandardZ(u);
155 var t = Normal.StandardZ(v);
156 return 1d / Math.Sqrt(1d - r * r) * Math.Exp(-(r * r * s * s + r * r * t * t - 2 * r * s * t) / (2 * (1d - r * r)));
157 }
158
160 public override double CDF(double u, double v)
161 {
162 // Validate parameters
163 if (_parametersValid == false) ValidateParameter(Theta, true);
164 return MultivariateNormal.BivariateCDF(Normal.StandardZ(1 - u), Normal.StandardZ(1 - v), _theta);
165 }
166
168 public override double[] InverseCDF(double u, double v)
169 {
170 // Validate parameters
171 if (_parametersValid == false) ValidateParameter(Theta, true);
172 double z1 = Normal.StandardZ(u);
173 double z2 = Normal.StandardZ(v);
174 double r = _theta;
175 double w2 = r * z1 + Math.Sqrt(1d - r * r) * z2;
176 v = Normal.StandardCDF(w2);
177 return [u, v];
178 }
