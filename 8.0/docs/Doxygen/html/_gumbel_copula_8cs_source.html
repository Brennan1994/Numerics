126 public override double GeneratorPrime(double t)
127 {
128 double a = Math.Log(t);
129 return -Theta * Math.Sign(a) * Math.Pow(Math.Abs(a), Theta - 1.0d) / t;
130 }

133 public override double GeneratorPrime2(double t)
134 {
135 double a = -Math.Log(t);
136 return Theta * Math.Sign(a) * Math.Pow(Math.Abs(a), Theta - 2.0d) * (-Theta + Math.Log(t) + 1.0d) / (t * t);
137 }

146 public override double[] InverseCDF(double u, double v)
147 {
148 // Validate parameters
149 if (_parametersValid == false) ValidateParameter(Theta, true);
150
151 // Use conditional probability function
152 double p = v;
153 v = Brent.Solve(x =>
154 {
155 double vu = Math.Pow(-Math.Log(u), Theta - 1d) * Math.Exp(-Math.Pow(Math.Pow(-Math.Log(u), Theta) + Math.Pow(-Math.Log(x), Theta), 1d / Theta)) * Math.Pow(Math.Pow(-Math.Log(u), Theta) + Math.Pow(-Math.Log(x), Theta), 1d / Theta - 1d) / u;
156 return vu - p;
157 }, 0d, 1d);
158 return [u, v];
159 }
