Numerics 1.0.0
Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

84 public StratificationOptions(double lowerBound, double upperBound, int numberOfBins, bool isProbability = false)
85 {
86 // Set the options
87 LowerBound = lowerBound;
88 UpperBound = upperBound;
89 NumberOfBins = numberOfBins;
90 IsProbability = isProbability;
91 // Determine if the options are valid
92 IsValid = true;
93 if (lowerBound >= upperBound || numberOfBins < 2)
94 IsValid = false;
95 if (isProbability == true && (lowerBound < 0d || upperBound > 1d))
96 IsValid = false;
97 }

205 public List<string> GetErrors()
206 {
207 var result = new List<string>();
208 if (IsValid == true)
209 return result;
210 if (LowerBound >= UpperBound)
211 {
212 result.Add("The upper bound '" + UpperBound + "' must be greater than the lower bound '" + LowerBound + "'.");
213 }
214
215 if (NumberOfBins < 2.0d)
216 {
217 result.Add("The number of bins '" + NumberOfBins + "' must be greater than 1.");
218 }
219
220 if (IsProbability == true && LowerBound < 0.0d)
221 {
222 result.Add("The lower bound must be greater than or equal to 0.");
223 }
224
225 if (IsProbability == true && UpperBound > 1.0d)
226 {
227 result.Add("The upper bound must be less than or equal to 1.");
228 }
229
