59 public JenksCluster(double[] data, int startIndex, int endIndex)
60 {
61 StartIndex = startIndex;
62 EndIndex = endIndex;
63 MinValue = data[startIndex];
64 MaxValue = data[endIndex];
65
66 // Compute summary statistics
67 double X = 0; // sum
68 double X2 = 0; // sum of X^2
69 for (int i = startIndex; i <= endIndex; i++)
70 {
71 X += data[i];
72 X2 += Math.Pow(data[i], 2d);
73 }
74
75 double U1 = X / Count;
76 double U2 = X2 / Count;
77 Sum = X;
78 Average = Count == 1 ? X : U1;
79 Variance = Count == 1 ? 0 : (U2 - Math.Pow(U1, 2d)) * (Count / (double)(Count - 1));
80 SumOfSquaredDeviations = Variance * (double)(Count - 1);
81 }
