Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

76 public ADAM(Func<double[], double> objectiveFunction, int numberOfParameters,
77 IList<double> initialValues, IList<double> lowerBounds, IList<double> upperBounds, double alpha = 0.001,
78 Func<double[], double[]> gradient = null) : base(objectiveFunction, numberOfParameters)
79 {
80 // Check if the length of the initial, lower and upper bounds equal the number of parameters
81 if (initialValues.Count != numberOfParameters || lowerBounds.Count != numberOfParameters || upperBounds.Count != numberOfParameters)
82 {
83 throw new ArgumentOutOfRangeException(nameof(lowerBounds), "The initial values and lower and upper bounds must be the same length as the number of parameters.");
84 }
85 // Check if the initial values are between the lower and upper values
86 for (int j = 0; j < initialValues.Count; j++)
87 {
88 if (upperBounds[j] < lowerBounds[j])
89 {
90 throw new ArgumentOutOfRangeException(nameof(upperBounds), "The upper bound cannot be less than the lower bound.");
91 }
92 if (initialValues[j] < lowerBounds[j] || initialValues[j] > upperBounds[j])
93 {
94 throw new ArgumentOutOfRangeException(nameof(initialValues), "The initial values must be between the upper and lower bounds.");
95 }
96 }
97 InitialValues = initialValues.ToArray();
98 LowerBounds = lowerBounds.ToArray();
99 UpperBounds = upperBounds.ToArray();
100 Alpha = alpha;
101 Gradient = gradient;
102 }

140 protected override void Optimize()
141 {
142 int D = NumberOfParameters;
143 bool cancel = false;
144 var p = InitialValues.ToArray();
145 var g = new double[D];
146 var m = new double[D];
147 var v = new double[D];
148 double mhat, vhat;
149 double f0 = Evaluate(p, ref cancel);
150 double f1;
151
152 while (Iterations < MaxIterations)
153 {
154 // Get gradient with respect to objective function
155 g = Gradient != null ? Gradient(p) : NumericalDerivative.Gradient((x) => Evaluate(x, ref cancel), p);
156 if (cancel) return;
157
158 // Update parameters
159 for (int i = 0; i < D; i++)
160 {
161 // Update biased first and second moment estimates
162 m[i] = Beta1 * m[i] + (1d - Beta1) * g[i];
163 v[i] = Beta2 * v[i] + (1d - Beta2) * g[i] * g[i];
164 // Compute biased corrected moment estimates
165 mhat = m[i] / (1d - Math.Pow(Beta1, Iterations + 1));
166 vhat = v[i] / (1d - Math.Pow(Beta2, Iterations + 1));
167 // Update parameter
168 p[i] = p[i] - Alpha * mhat / (Math.Sqrt(vhat) + Tools.DoubleMachineEpsilon);
169 // Make sure the parameter is within the bounds.
170 p[i] = RepairParameter(p[i], LowerBounds[i], UpperBounds[i]);
171 }
172
173 // Evaluate the objective function at the new point
174 f1 = Evaluate(p, ref cancel);
175 if (cancel) return;
176
177 // Check convergence.
178 if (CheckConvergence(f0, f1))
179 {
180 UpdateStatus(OptimizationStatus.Success);
181 return;
182 }
183
184 f0 = f1;
185 Iterations += 1;
186 }
187
188 // If we made it to here, the maximum iterations were reached.
189 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
190 }
191 }
