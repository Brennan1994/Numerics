113 public static GeneralizedBeta PERT(double min, double mode, double max, double scale)
114 {
115 // validate parameters
116 if (min > max)
117 {
118 throw new ArgumentOutOfRangeException(nameof(max), "The maximum value must be greater than the minimum value.");
119 }
120 else if (mode < min || mode > max)
121 {
122 throw new ArgumentOutOfRangeException(nameof(mode), "The mode must be between the minimum and maximum values.");
123 }
124 // get alpha and beta
125 double mean = (min + scale * mode + max) / (scale + 2d);
126 double alpha = 1d + scale / 2d;
127 if (mean != mode)
128 {
129 alpha = (mean - min) * (2d * mode - min - max) / ((mode - mean) * (max - min));
130 }
131 double beta = alpha * (max - mean) / (mean - min);
132 // return beta
133 return new GeneralizedBeta(alpha, beta, min, max);
134 }

361 public void SetParametersFromMoments(double mu, double sigma, double min, double max)
362 {
363 var s2 = sigma * sigma;
364 _alpha = (min - mu) * (min * max - min * mu - max * mu + mu * mu + s2) / (s2 * (max - min));
365 _beta = -(max - mu) * (min * max - min * mu - max * mu + mu * mu + s2) / (s2 * (max - min));
366 _min = min;
367 _max = max;
368 // validate parameters
369 _parametersValid = ValidateParameters(_alpha, _beta, _min, _max, false) is null;
370 }

412 public override double PDF(double X)
413 {
414 // Validate parameters
415 if (_parametersValid == false)
416 ValidateParameters(Alpha, Beta, Min, Max, true);
417 if (X < Minimum || X > Maximum) return 0.0d;
418
419 if (Min == Max) return 0d;
420 double constant = 1.0d / Mathematics.SpecialFunctions.Beta.Function(Alpha, Beta);
421 double z = (X - Min) / (Max - Min);
422 double a = Math.Pow(z, Alpha - 1d);
423 double b = Math.Pow(1d - z, Beta - 1d);
424 return constant * a * b / (Max - Min);
425 }
426
428 public override double CDF(double X)
429 {
430 // Validate parameters
431 if (_parametersValid == false)
432 ValidateParameters(Alpha, Beta, Min, Max, true);
433
434 if (Min == Max) return 1d;
435 if (X <= Minimum) return 0d;
436 if (X >= Maximum) return 1d;
437 double z = (X - Min) / (Max - Min);
438 return Mathematics.SpecialFunctions.Beta.Incomplete(Alpha, Beta, z);
439 }

442 public override double InverseCDF(double probability)
443 {
444 // Validate probability
445 if (probability < 0.0d || probability > 1.0d)
446 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
447 if (probability == 0.0d) return Minimum;
448 if (probability == 1.0d) return Maximum;
449 // Validate parameters
450 if (_parametersValid == false)
451 ValidateParameters(Alpha, Beta, Min, Max, true);
452 double z = Mathematics.SpecialFunctions.Beta.IncompleteInverse(Alpha, Beta, probability);
453 return z * (Max - Min) + Min;
454 }
