276 public override double PDF(double x)
277 {
278 // validate parameters
279 if (_parametersValid == false)
280 ValidateParameters([DegreesOfFreedom, Sigma], true);
281 if (x < Minimum) return 0.0d;
282 double v = DegreesOfFreedom;
283 double t2 = Sigma;
284 double a = Math.Pow(t2 * v / 2d, v / 2d);
285 double b = Gamma.Function(v / 2d);
286 double c = Math.Exp(-v * t2 / (2d * x));
287 double d = Math.Pow(x, 1d + v / 2d);
288 return a / b * (c / d);
289 }

292 public override double CDF(double x)
293 {
294 // Validate parameters
295 if (_parametersValid == false)
296 ValidateParameters([DegreesOfFreedom, Sigma], true);
297 if (x <= Minimum)
298 return 0d;
299 double v = DegreesOfFreedom;
300 double t2 = Sigma;
301 return Gamma.UpperIncomplete(v / 2.0d, v * t2 / 2.0d * x);
302 }

305 public override double InverseCDF(double probability)
306 {
307 // Validate probability
308 if (probability < 0.0d || probability > 1.0d)
309 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
310 if (probability == 0.0d)
311 return Minimum;
312 if (probability == 1.0d)
313 return Maximum;
314 // validate parameters
315 if (_parametersValid == false)
316 ValidateParameters([DegreesOfFreedom, Sigma], true);
317 double v = DegreesOfFreedom;
318 double t2 = Sigma;
319 return Gamma.InverseUpperIncomplete(v / 2.0d, probability) / (t2 * v / 2.0d);
320 }
