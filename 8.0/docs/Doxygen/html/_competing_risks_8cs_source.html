426 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
427 {
428 var initialVals = new double[NumberOfParameters];
429 var lowerVals = new double[NumberOfParameters];
430 var upperVals = new double[NumberOfParameters];
431
432 int t = 0;
433 for (int i = 0; i < Distributions.Count; i++)
434 {
435 var tuple = ((IMaximumLikelihoodEstimation)Distributions[i]).GetParameterConstraints(sample);
436 var initials = tuple.Item1;
437 var lowers = tuple.Item2;
438 var uppers = tuple.Item3;
439
440 for (int j = t; j < t + Distributions[i].NumberOfParameters; j++)
441 {
442 initialVals[j] = initials[j - t];
443 lowerVals[j] = lowers[j - t];
444 upperVals[j] = uppers[j - t];
445 }
446 t += Distributions[i].NumberOfParameters;
447 }
448
449 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
450 }

453 public double[] MLE(IList<double> sample)
454 {
455 // Set constraints
456 var tuple = GetParameterConstraints(sample);
457 var Initials = tuple.Item1;
458 var Lowers = tuple.Item2;
459 var Uppers = tuple.Item3;
460
461 // Solve using Nelder-Mead (Downhill Simplex)
462 double logLH(double[] x)
463 {
464 var dist = (CompetingRisks)Clone();
465 dist.SetParameters(x);
466 return dist.LogLikelihood(sample);
467 }
468 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
469 solver.Maximize();
470 return solver.BestParameterSet.Values;
471 }

474 public override double PDF(double x)
475 {
476 double f = double.NaN;
477
478 // Only compute the exact PDF for independent random variables
479 if (Dependency == Probability.DependencyType.Independent)
480 {
481 double hf = 0d;
482 double sf = 1d;
483 for (int i = 0; i < Distributions.Count; i++)
484 {
485
486 if (MinimumOfRandomVariables == true)
487 {
488 hf += Distributions[i].HF(x);
489 sf *= Distributions[i].CCDF(x);
490 }
491 else
492 {
493 hf += Distributions[i].PDF(x) / Distributions[i].CDF(x);
494 sf *= Distributions[i].CDF(x);
495 }
496 }
497 f = hf * sf;
498 }
499 else
500 {
501 // Compute the PDF using numerical differentiation
502 f = NumericalDerivative.Derivative(CDF, x);
503 }
504
505 return f < 0d ? 0d : f;
506 }

509 public override double CDF(double x)
510 {
511 double p = double.NaN;
512 var ind = new int[Distributions.Count];
513 var cdf = new double[Distributions.Count];
514 for (int i = 0; i < Distributions.Count; i++)
515 {
516 ind[i] = 1;
517 cdf[i] = Distributions[i].CDF(x);
518 }
519
520 if (MinimumOfRandomVariables == true)
521 {
522
523 if (Dependency == Probability.DependencyType.PerfectlyNegative || Dependency == Probability.DependencyType.CorrelationMatrix)
524 {
525 if (_mvnCreated == false)
526 CreateMultivariateNormal();
527 p = Probability.Union(cdf, _mvn);
528 }
529 else
530 {
531 p = Probability.Union(cdf, Dependency);
532 }
533 }
534 else
535 {
536 if (Dependency == Probability.DependencyType.CorrelationMatrix)
537 {
538 if (_mvnCreated == false)
539 CreateMultivariateNormal();
540 p = Probability.JointProbability(cdf, ind, _mvn);
541 }
542 else
543 {
544 p = Probability.JointProbability(cdf, Dependency);
545 }
546
547 }
548 return p < 0d ? 0d : p > 1d ? 1d : p;
549 }

552 public override double InverseCDF(double probability)
553 {
554 // Validate probability
555 if (probability < 0.0d || probability > 1.0d)
556 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
557 if (probability == 0.0d) return Minimum;
558 if (probability == 1.0d) return Maximum;
559 // Validate parameters
560 if (_parametersValid == false)
561 ValidateParameters([0], true);
562
563 // If there is only one distribution, return its inverse CDF
564 if (Distributions.Count() == 1)
565 {
566 return Distributions[0].InverseCDF(probability);
567 }
568
569 // Otherwise use a root finder to solve the inverse CDF
570 var xVals = Distributions.Select(d => d.InverseCDF(probability));
571 double minX = xVals.Min();
572 double maxX = xVals.Max();
573 double x = 0;
574 try
575 {
576 Brent.Bracket((y) => { return probability - CDF(y); }, ref minX, ref maxX, out var f1, out var f2);
577 x = Brent.Solve((y) => { return probability - CDF(y); }, minX, maxX, 1E-6, 100, true);
578 }
579 catch (Exception)
580 {
581 // If the root finder fails, create an empirical inverse CDF
582 if (_inverseCDFCreated == false)
583 CreateInverseCDF();
584 x = _inverseCDF.InverseCDF(probability);
585 }
586 double min = Minimum;
587 double max = Maximum;
588 return x < min ? min : x > max ? max : x;
589 }

595 public List<EmpiricalDistribution> CumulativeIncidenceFunctions(List<StratificationBin> bins = null)
596 {
597 // Get stratification bins
598 if (bins == null)
599 {
600 double minP = 1E-16;
601 double maxP = 1 - 1E-16;
602 double minX = Distributions.Min(d => d.InverseCDF(minP));
603 double maxX = Distributions.Max(d => d.InverseCDF(maxP));
604 bins = Stratify.XValues(new StratificationOptions(minX, maxX, 200, false), XTransform == Transform.Logarithmic ? true : false);
605 }
606
607 var D = Distributions.Count();
608 var x = new List<double[]>();
609 var p = new List<double[]>();
610 var dF = new List<double[]>();
611
612 if (Dependency == Probability.DependencyType.PerfectlyNegative || Dependency == Probability.DependencyType.CorrelationMatrix)
613 {
614 /*
615 * For perfect negative dependency or a custom correlation matrix,
616 * use the Genz Method. This method is slow but accurate.
617 */
618
619 if (_mvnCreated == false)
620 CreateMultivariateNormal();
621
622 var lower = new double[D];
623 var upper = new double[D];
624 for (int i = 0; i < D; i++)
625 {
626 x.Add(new double[bins.Count + 1]);
627 p.Add(new double[bins.Count + 1]);
628 dF.Add(new double[bins.Count + 1]);
629
630 // Record the first bin
631 x[i][0] = bins[0].LowerBound;
632 for (int k = 0; k < D; k++)
633 {
634 if (MinimumOfRandomVariables == true)
635 {
636 lower[k] = k == i ? Normal.StandardZ(1E-16) : Normal.StandardZ(Distributions[k].CDF(bins[0].LowerBound));
637 upper[k] = k == i ? Normal.StandardZ(Distributions[i].CDF(bins[0].LowerBound)) : Normal.StandardZ(1 - 1E-16);
638 }
639 else
640 {
641 lower[k] = Normal.StandardZ(1E-16);
642 upper[k] = Normal.StandardZ(Distributions[i].CDF(bins[0].LowerBound));
643 }
644 }
645 dF[i][0] = _mvn.Interval(lower, upper);
646 if (double.IsNaN(dF[i][0])) dF[i][0] = 0;
647 dF[i][0] = Math.Max(0, Math.Min(1, dF[i][0]));
648 p[i][0] = dF[i][0];
649
650 // Record the remaining bins
651 for (int j = 0; j < bins.Count; j++)
652 {
653 x[i][j + 1] = bins[j].UpperBound;
654 for (int k = 0; k < D; k++)
655 {
656 if (MinimumOfRandomVariables == true)
657 {
658 lower[k] = k == i ? Normal.StandardZ(Distributions[i].CDF(bins[j].LowerBound)) : Normal.StandardZ(Distributions[k].CDF(bins[j].Midpoint));
659 upper[k] = k == i ? Normal.StandardZ(Distributions[i].CDF(bins[j].UpperBound)) : Normal.StandardZ(1 - 1E-16);
660 }
661 else
662 {
663 lower[k] = k == i ? Normal.StandardZ(Distributions[i].CDF(bins[j].LowerBound)) : Normal.StandardZ(1E-16);
664 upper[k] = k == i ? Normal.StandardZ(Distributions[i].CDF(bins[j].UpperBound)) : Normal.StandardZ(Distributions[k].CDF(bins[j].Midpoint));
665 }
666 }
667 dF[i][j + 1] = _mvn.Interval(lower, upper);
668 if (double.IsNaN(dF[i][j + 1])) dF[i][j + 1] = 0;
669 dF[i][j + 1] = Math.Max(0, Math.Min(1, dF[i][j + 1]));
670 }
671 }
672 }
673 else if (Dependency == Probability.DependencyType.Independent || Dependency == Probability.DependencyType.PerfectlyPositive)
674 {
675 /*
676 * For perfect independent or perfectly positive,
677 * use the "Delta Method" developed by Haden Smith and Dave Margo.
678 * It is fast and accurate.
679 */
680
681 double F1 = 0, F2 = 0;
682 var pm = new double[D];
683 var pl = new double[D];
684 var pu = new double[D];
685 var ind = new int[D];
686 ind.Fill(1);
687
688 for (int i = 0; i < D; i++)
689 {
690 x.Add(new double[bins.Count + 1]);
691 p.Add(new double[bins.Count + 1]);
692 dF.Add(new double[bins.Count + 1]);
693
694 // Record first bin
695 x[i][0] = bins[0].LowerBound;
696 for (int k = 0; k < D; k++)
697 {
698 if (MinimumOfRandomVariables == true)
699 {
700 pl[k] = k == i ? Distributions[k].CCDF(bins[0].LowerBound) : Distributions[k].CCDF(bins[0].LowerBound);
701 pu[k] = k == i ? 1.0 : Distributions[k].CCDF(bins[0].LowerBound);
702 }
703 else
704 {
705 pu[k] = Distributions[k].CDF(bins[0].LowerBound);
706 }
707 }
708 F1 = Probability.JointProbability(pl, Dependency);
709 F2 = Probability.JointProbability(pu, Dependency);
710 dF[i][0] = F2 - F1;
711 if (double.IsNaN(dF[i][0]))
712 dF[i][0] = 0;
713 dF[i][0] = Math.Max(0, Math.Min(1, dF[i][0]));
714 p[i][0] = dF[i][0];
715
716 // Record remaining bins
717 for (int j = 0; j < bins.Count; j++)
718 {
719 x[i][j + 1] = bins[j].UpperBound;
720 for (int k = 0; k < D; k++)
721 {
722 if (MinimumOfRandomVariables == true)
723 {
724 pl[k] = k == i ? Distributions[k].CCDF(bins[j].UpperBound) : Distributions[k].CCDF(bins[j].Midpoint);
725 pu[k] = k == i ? Distributions[k].CCDF(bins[j].LowerBound) : Distributions[k].CCDF(bins[j].Midpoint);
726 }
727 else
728 {
729 pl[k] = k == i ? Distributions[k].CDF(bins[j].LowerBound) : Distributions[k].CDF(bins[j].Midpoint);
730 pu[k] = k == i ? Distributions[k].CDF(bins[j].UpperBound) : Distributions[k].CDF(bins[j].Midpoint);
731 }
732
733 }
734 F1 = Probability.JointProbability(pl, Dependency);
735 F2 = Probability.JointProbability(pu, Dependency);
736 dF[i][j + 1] = F2 - F1;
737 if (double.IsNaN(dF[i][j + 1]))
738 dF[i][j + 1] = 0;
739 dF[i][j + 1] = Math.Max(0, Math.Min(1, dF[i][j + 1]));
740 }
741 }
742 }
743
744 // Get cumulative probabilities and make sure they sum <= 1 across D
745 bool fixDF = false;
746 var sum = new double[bins.Count + 1];
747 for (int j = 1; j <= bins.Count; j++)
748 {
749 for (int i = 0; i < D; i++)
750 {
751 sum[j] += p[i][j - 1] + dF[i][j];
752 p[i][j] = Math.Max(0, Math.Min(1, p[i][j - 1] + dF[i][j]));
753 }
754 if (sum[j] > 1 && sum[j] != sum[j - 1] && fixDF == false)
755 {
756 double s = 0;
757 for (int i = 0; i < D; i++)
758 {
759 dF[i][j] *= (1 - sum[j - 1]) / (sum[j] - sum[j - 1]);
760 s += p[i][j - 1] + dF[i][j];
761
762 p[i][j] = Math.Max(0, Math.Min(1, p[i][j - 1] + dF[i][j]));
763 }
764 sum[j] = s;
765 fixDF = true;
766 }
767 else if (fixDF == true)
768 {
769 for (int i = 0; i < D; i++)
770 {
771 dF[i][j] = 0;
772 p[i][j] = Math.Max(0, Math.Min(1, p[i][j - 1] + dF[i][j]));
773 }
774 }
775 }
776
777 // Return CIFs
778 var CIFs = new List<EmpiricalDistribution>();
779 for (int i = 0; i < D; i++)
780 CIFs.Add(new EmpiricalDistribution(x[i], p[i]));
781
782 return CIFs;
783
784 }

789 private void CreateMultivariateNormal()
790 {
791 var D = Distributions.Count();
792 var mu = new double[D];
793 var sigma = new double[D, D];
794 if (Dependency == Probability.DependencyType.PerfectlyNegative)
795 {
796 CorrelationMatrix = new double[D, D];
797 double rho = -1d / (D - 1d) + Math.Sqrt(Tools.DoubleMachineEpsilon);
798 for (int i = 0; i < D; i++)
799 {
800 mu[i] = 0d;
801 for (int j = 0; j < D; j++)
802 sigma[i, j] = i == j ? 1d : rho;
803 }
804 }
805 else
806 {
807 for (int i = 0; i < D; i++)
808 {
809 mu[i] = 0d;
810 for (int j = 0; j < D; j++)
811 sigma[i, j] = CorrelationMatrix[i, j];
812 }
813 }
814 _mvn = new MultivariateNormal(mu, sigma);
815 _mvnCreated = true;
816 }

822 private void CreateInverseCDF()
823 {
824 // Get min & max
825 double minP = 1E-16;
826 double maxP = 1 - 1E-16;
827 double minX = Distributions.Min(d => d.InverseCDF(minP));
828 double maxX = Distributions.Max(d => d.InverseCDF(maxP));
829 // Get number of bins
830 double shift = 0;
831 if (minX <= 0) shift = Math.Abs(minX) + 1d;
832 double min = minX + shift;
833 double max = maxX + shift;
834 int order = (int)Math.Floor(Math.Log10(max) - Math.Log10(min));
835 int binN = Math.Max(200, 100 * order) - 1;
836 // Create bins
837 var bins = Stratify.XValues(new StratificationOptions(minX, maxX, binN, false), XTransform == Transform.Logarithmic ? true : false);
838 var xValues = new List<double>();
839 var pValues = new List<double>();
840 var x = bins.First().LowerBound;
841 var p = CDF(bins.First().LowerBound);
842 xValues.Add(x);
843 pValues.Add(p);
844 for (int i = 1; i < bins.Count; i++)
845 {
846 x = bins[i].LowerBound;
847 p = CDF(x);
848 if (x > xValues.Last() && p > pValues.Last())
849 {
850 xValues.Add(x);
851 pValues.Add(p);
852 }
853 }
854 x = maxX;
855 p = CDF(x);
856 if (x > xValues.Last() && p > pValues.Last())
857 {
858 xValues.Add(x);
859 pValues.Add(p);
860 }
861 _inverseCDF = new EmpiricalDistribution(xValues, pValues) { XTransform = XTransform, ProbabilityTransform = ProbabilityTransform };
862 _inverseCDFCreated = true;
863 }

881 }
882 sample[i] = MinimumOfRandomVariables == true ? xMin : xMax;
883 }
