309 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
310 {
311 var initialVals = new double[NumberOfParameters];
312 var lowerVals = new double[NumberOfParameters];
313 var upperVals = new double[NumberOfParameters];
314 // Get initial values
315 initialVals = ParametersFromMoments(Statistics.ProductMoments(sample));
316 // Get bounds of location
317 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[0]) + 1d));
318 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[0]) + 1d));
319 // Get bounds of scale
320 lowerVals[1] = Tools.DoubleMachineEpsilon;
321 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[1]) + 1d));
322 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
323 }

326 public double[] MLE(IList<double> sample)
327 {
328 // Set constraints
329 var tuple = GetParameterConstraints(sample);
330 var Initials = tuple.Item1;
331 var Lowers = tuple.Item2;
332 var Uppers = tuple.Item3;
333
334 // Solve using Nelder-Mead (Downhill Simplex)
335 double logLH(double[] x)
336 {
337 var LO = new Logistic();
338 LO.SetParameters(x);
339 return LO.LogLikelihood(sample);
340 }
341 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
342 solver.ReportFailure = true;
343 solver.Maximize();
344 return solver.BestParameterSet.Values;
345 }

366 public override double InverseCDF(double probability)
367 {
368 // Validate probability
369 if (probability < 0.0d || probability > 1.0d)
370 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
371 if (probability == 0.0d)
372 return Minimum;
373 if (probability == 1.0d)
374 return Maximum;
375 // Validate parameters
376 if (_parametersValid == false)
377 ValidateParameters([Xi, Alpha], true);
378 return Xi + Alpha * Math.Log(probability / (1d - probability));

388 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
389 {
390 if (estimationMethod != ParameterEstimationMethod.MethodOfMoments &&
391 estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
392 {
393 throw new NotImplementedException();
394 }
395 // Validate parameters
396 if (_parametersValid == false)
397 ValidateParameters([Xi, _alpha], true);
398
399 // Compute covariance
400 double a = Alpha;
401 var covar = new double[2, 2];
402 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
403 {
404 covar[0, 0] = Math.PI * Math.PI / 3d * (a * a / sampleSize); // location
405 covar[1, 1] = 4d / 5d * (a * a / sampleSize); // scale
406 covar[0, 1] = 0.0; // location & scale
407 covar[1, 0] = covar[0, 1];
408 }
409 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
410 {
411 covar[0, 0] = 3d * (a * a / sampleSize); // location
412 covar[1, 1] = 9d / (3d + Math.PI * Math.PI) * (a * a / sampleSize); // scale
413 covar[0, 1] = 0.0; // location & scale
414 covar[1, 0] = covar[0, 1];
415 }
416 return covar;
417 }

420 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
421 {
422 var covar = ParameterCovariance(sampleSize, estimationMethod);
423 var grad = QuantileGradient(probability);
424 double varA = covar[0, 0];
425 double varB = covar[1, 1];
426 double covAB = covar[1, 0];
427 double dQx1 = grad[0];
428 double dQx2 = grad[1];
429 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + 2d * dQx1 * dQx2 * covAB;
430 }

433 public double[] QuantileGradient(double probability)
434 {
435 // Validate parameters
436 if (_parametersValid == false)
437 ValidateParameters([Xi, _alpha], true);
438 var gradient = new double[]
439 {
440 1.0d, // location
441 Math.Log(probability / (1d - probability)) // scale
442 };
443 return gradient;
444 }

447 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
448 {
449 if (probabilities.Count != NumberOfParameters)
450 {
451 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
452 }
453 // Get gradients
454 var dQp1 = QuantileGradient(probabilities[0]);
455 var dQp2 = QuantileGradient(probabilities[1]);
456 // Compute determinant
457 // |a b|
458 // |c d|
459 // |A| = ad − bc
460 double a = dQp1[0];
461 double b = dQp1[1];
462 double c = dQp2[0];
463 double d = dQp2[1];
464 determinant = a * d - b * c;
465 // Return Jacobian
466 var jacobian = new double[,] { { a, b }, { c, d } };
467 return jacobian;
468 }
