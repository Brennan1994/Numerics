The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

122 public static int[] NextIntegers(this Random random, int minValue, int maxValue, int length, bool replace = true)
123 {
124 if (replace == true)
125 {
126 var values = new int[length];
127 for (int i = 0; i < length; i++)
128 values[i] = random.Next(minValue, maxValue);
129 return values;
130 }
131 else
132 {
133 if (length > maxValue - minValue)
134 throw new ArgumentException("When sampling without replacement, the length must be less than or equal to the range of values.");
135
136 var bins = new List<int>();
137 for (int i = minValue; i < maxValue; i++)
138 bins.Add(i);
139
140 // Sample random bin without replacement
141 var values = new int[length];
142 for (int i = 0; i < length; i++)
143 {
144 int r = random.Next(0, bins.Count);
145 values[i] = bins[r];
146 bins.RemoveAt(r);
147 }
148 return values;
149 }

172 public static double[,] NextDoubles(this Random random, int length, int dimension)
173 {
174 var values = new double[length, dimension];
175 var prngs = new Random[dimension];
176 for (int i = 0; i < dimension; i++)
177 {
178 prngs[i] = random.GetType() == typeof(MersenneTwister) ? new MersenneTwister(random.Next()) : new Random(random.Next());
179 for (int j = 0; j < length; j++)
180 {
181 values[j, i] = prngs[i].NextDouble();
182 }
183 }
184 return values;
185 }
