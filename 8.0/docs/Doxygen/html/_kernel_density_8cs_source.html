447 {

497 public override double PDF(double x)
498 {
499 double total = 0d;
500 Parallel.For(0, SampleSize, () => 0d, (i, loop, subtotal) =>
501 {
502 subtotal += _kernel.Function((x - _sampleData[i]) / Bandwidth);
503 return subtotal;
504 }, z => Tools.ParallelAdd(ref total, z));
505 return total / (SampleSize * Bandwidth);
506 }

512 if (x > Maximum) return 1.0;

518 public override double InverseCDF(double probability)
519 {
520 // Validate probability
521 if (probability < 0.0d || probability > 1.0d)
522 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
523 if (probability == 0.0) return Minimum;
524 if (probability == 1.0) return Maximum;
525 if (!_cdfCreated) CreateCDF();
526 return opd.GetXFromY(probability, XTransform, ProbabilityTransform);

538 private void CreateCDF()
539 {
540 // Create wide bins
541 int n = 1000;
542 var bins = Stratify.XValues(new StratificationOptions(Minimum, Maximum, n));
543 var xValues = new double[n];
544 var pValues = new double[n];
545
546 // Create empirical cdf
547 xValues[0] = bins[0].Midpoint;
548 pValues[0] = PDF(xValues[0]) * bins[0].Weight;
549 for (int i = 1; i < n; i++)
550 {
551 xValues[i] = bins[i].Midpoint;
552 pValues[i] = pValues[i - 1] + PDF(xValues[i]) * bins[i].Weight;
553 }
554
555 // Normalize the cdf to sum to 1
556 for (int i = 0; i < n; i++)
557 {
558 pValues[i] /= pValues.Last();
559 }
560
561 opd = new OrderedPairedData(xValues, pValues, true, SortOrder.Ascending, true, SortOrder.Ascending);
562 _cdfCreated = true;
563 }
