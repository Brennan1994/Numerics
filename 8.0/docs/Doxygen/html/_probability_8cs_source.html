84 public static double AAndB(double A, double B, double rho = 0d)
85 {
86 if (Math.Abs(rho) <= 1E-3) return A * B;
87 if (rho >= 0.999) return Math.Min(A, B);
88 return MultivariateNormal.BivariateCDF(Normal.StandardZ(1 - A), Normal.StandardZ(1 - B), rho);
89 }
90
97 public static double AOrB(double A, double B, double rho = 0d)
98 {
99 return A + B - AAndB(A, B, rho);
100 }
101
108 public static double ANotB(double A, double B, double rho = 0d)
109 {
110 return A - AAndB(A, B, rho);
111 }
112
119 public static double BNotA(double A, double B, double rho = 0d)

121 return B - AAndB(A, B, rho);

270 public static double JointProbabilityHPCM(IList<double> probabilities, int[] indicators, double[,] correlationMatrix, double[] conditionalProbabilities = null)
271 {
272 // Get z-values
273 int n = probabilities.Count;
274 var R = new double[n, n];
275 Array.Copy(correlationMatrix, R, correlationMatrix.Length);
276 int i, j, k, ir, ic;
277 double pdf, cdf, A, B, z1, z2, r12, z21, p21, jp;
278 for (i = 0; i < n; i++)
279 {
280 if (indicators[i] == 0)
281 {
282 R[i, i] = Normal.StandardZ(1);
283 }
284 else
285 {
286 R[i, i] = Normal.StandardZ(probabilities[i]);
287 }
288 }
289 // Update the conditional correlation matrix
290 // First cycle
291 z1 = R[0, 0];
292 pdf = Normal.StandardPDF(z1);
293 cdf = Normal.StandardCDF(z1);
294 A = pdf / cdf;
295 B = A * (z1 + A);
296 for (k = 1; k < n; k++)
297 {
298 z2 = R[k, k];
299 r12 = R[0, k];
300 r12 = Math.Abs(r12) < 1E-3 ? 0: r12;
301 p21 = MultivariateNormal.BivariateCDF(-z1, -z2, r12) / cdf;
302 p21 = Math.Max(0, Math.Min(1, p21));
303 z21 = Normal.StandardZ(p21);
304 R[k, 0] = z21;
305 }
306 // update condition correlation r[ir|ic] and store them in R[ir,ic]
307 for (ir = 1; ir < n - 1; ir++)
308 {
309 for (ic = ir + 1; ic < n; ic++)
310 {
311 R[ir, ic] = (R[ir, ic] - R[0, ir] * R[0, ic] * B) / Math.Sqrt((1d - R[0, ir] * R[0, ir] * B) * (1d - R[0, ic] * R[0, ic] * B));
312 }
313 }
314 // Remaining cycles
315 for (j = 1; j < n - 1; j++)
316 {
317 z1 = R[j, j - 1];
318 pdf = Normal.StandardPDF(z1);
319 cdf = Normal.StandardCDF(z1);
320 A = pdf / cdf;
321 B = A * (z1 + A);
322 for (k = j + 1; k < n; k++)
323 {
324 z2 = R[k, j - 1];
325 r12 = R[j, k];
326 r12 = Math.Abs(r12) < 1E-3 ? 0 : r12;
327 p21 = MultivariateNormal.BivariateCDF(-z1, -z2, r12) / cdf;
328 p21 = Math.Max(0, Math.Min(1, p21));
329 z21 = Normal.StandardZ(p21);
330 R[k, j] = z21;
331
332 }
333 for (ir = j + 1; ir < n - 1; ir++)
334 {
335 for (ic = ir + 1; ic < n; ic++)
336 {
337 R[ir, ic] = (R[ir, ic] - R[j, ir] * R[j, ic] * B) / Math.Sqrt((1d - R[j, ir] * R[j, ir] * B) * (1d - R[j, ic] * R[j, ic] * B));
338 }
339 }
340 }
341
342 // Calculate the product of conditional marginals (PCM)
343 jp = Math.Log(Normal.StandardCDF(R[0, 0]));
344 if (conditionalProbabilities != null && conditionalProbabilities.Length == n)
345 conditionalProbabilities[0] = Normal.StandardCDF(R[0, 0]);
346 for (i = 1; i < n; i++)
347 {
348 jp += Math.Log(Normal.StandardCDF(R[i, i - 1]));
349 if (conditionalProbabilities != null && conditionalProbabilities.Length == n)
350 conditionalProbabilities[i] = Normal.StandardCDF(R[i, i - 1]);
351 }
352 jp = Math.Exp(jp);
353 jp = Math.Min(1, Math.Max(0, jp));
354 if (double.IsNaN(jp)) jp = 0;
355 return jp;
356 }

365 public static double JointProbabilityPCM(IList<double> probabilities, int[] indicators, double[,] correlationMatrix, double[] conditionalProbabilities = null)
366 {
367 // Get z-values
368 int n = probabilities.Count;
369 var R = new double[n, n];
370 Array.Copy(correlationMatrix, R, correlationMatrix.Length);
371 int i, j, k, ir, ic;
372 double A, B, z1, z2, z21, r12;
373 for (i = 0; i < n; i++)
374 {
375 if (indicators[i] == 0)
376 {
377 R[i, i] = Normal.StandardZ(1);
378 }
379 else
380 {
381 R[i, i] = Normal.StandardZ(probabilities[i]);
382 }
383 }
384 // Update the conditional correlation matrix
385 // First cycle
386 z1 = R[0, 0];
387 A = Normal.StandardPDF(z1) / Normal.StandardCDF(z1);
388 B = A * (z1 + A);
389 // calculate z[k|0] and store them in R[k,0], k = 1,...,n
390 for (k = 1; k < n; k++)
391 {
392 z2 = R[k, k];
393 r12 = R[0, k];
394 z21 = (z2 + r12 * A) / Math.Sqrt(1d - r12 * r12 * B);
395 R[k, 0] = z21;
396 }
397 // update r[ir|ic] and store them in R[ir,ic]
398 for (ir = 1; ir < n - 1; ir++)
399 {
400 for (ic = ir + 1; ic < n; ic++)
401 {
402 R[ir, ic] = (R[ir, ic] - R[0, ir] * R[0, ic] * B) / Math.Sqrt((1d - R[0, ir] * R[0, ir] * B) * (1d - R[0, ic] * R[0, ic] * B));
403 }
404 }
405 // Remaining cycles
406 for (j = 1; j < n - 1; j++)
407 {
408 z1 = R[j, j - 1];
409 A = Normal.StandardPDF(z1) / Normal.StandardCDF(z1);
410 B = A * (z1 + A);
411 for (k = j + 1; k < n; k++)
412 {
413 z2 = R[k, j - 1];
414 r12 = R[j, k];
415 z21 = (z2 + r12 * A) / Math.Sqrt(1d - r12 * r12 * B);
416 R[k, j] = z21;
417 }
418 for (ir = j + 1; ir < n - 1; ir++)
419 {
420 for (ic = ir + 1; ic < n; ic++)
421 {
422 R[ir, ic] = (R[ir, ic] - R[j, ir] * R[j, ic] * B) / Math.Sqrt((1d - R[j, ir] * R[j, ir] * B) * (1d - R[j, ic] * R[j, ic] * B));
423 }
424 }
425 }
426 // Calculate the product of conditional marginals (PCM)
427 double jp = Math.Log(Normal.StandardCDF(R[0, 0]));
428 if (conditionalProbabilities != null && conditionalProbabilities.Length == n)
429 conditionalProbabilities[0] = Normal.StandardCDF(R[0, 0]);
430 for (i = 1; i < n; i++)
431 {
432 jp += Math.Log(Normal.StandardCDF(R[i, i - 1]));
433 if (conditionalProbabilities != null && conditionalProbabilities.Length == n)
434 conditionalProbabilities[i] = Normal.StandardCDF(R[i, i - 1]);
435 }
436 jp = Math.Exp(jp);
437 jp = Math.Min(1, Math.Max(0, jp));
438 if (double.IsNaN(jp)) jp = 0;
439 return jp;
440 }

476 for (int i = 0; i < indicators.Length; i++)

484 zVals[i] = Normal.StandardZ(probabilities[i]);

608 public static double UnionPCM(IList<double> probabilities, int[] binomialCombinations, int[,] indicators, double[,] correlationMatrix, double absoluteTolerance = 1E-4, double relativeTolerance = 1E-4)
609 {
610
611 double result = 0;
612 double s = 1;
613 int j = 0;
614 int c = binomialCombinations[j];
615 double inc = double.NaN;
616 double exc = double.NaN;
617
618 for (int i = 0; i < indicators.GetLength(0); i++)
619 {
620 if (i == c)
621 {
622 if (j > 0 && s == 1)
623 {
624 inc = result;
625 }
626 else if (j > 0 && s == -1)
627 {
628 exc = result;
629 }
630
631 // Check for convergence
632 // Add the tolerance to the last joint event
633 double diff = Math.Abs(inc - exc);
634 if (j > 0 && j < binomialCombinations.Length && diff <= absoluteTolerance && diff <= relativeTolerance * Math.Min(inc, exc))
635 {
636 return result + 0.5 * diff;
637 }
638
639 s *= -1;
640 j++;
641 if (j < binomialCombinations.Length)
642 {
643 c += binomialCombinations[j];
644 }
645
646 }
647 if (i < probabilities.Count)
648 {
649 result += s * probabilities[i];
650 }
651 else
652 {
653 result += s * JointProbability(probabilities, indicators.GetRow(i), correlationMatrix);
654 }
655
656 }
657
658 return result;
659 }
