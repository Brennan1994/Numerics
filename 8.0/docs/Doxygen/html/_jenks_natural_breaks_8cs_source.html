141 public double GoodnessOfVarianceFit
142 {
143 get
144 {
145 double gvf = 0;
146 double mean = Tools.Mean(SortedData);
147 // Sum of squared deviations from data
148 double sdam = 0;
149 for (int i = 0; i < SortedData.Length; i++)
150 {
151 sdam += Tools.Sqr(SortedData[i] - mean);
152 }
153 // Sum of squared deviations from clusters
154 double sdcm = 0;
155 for (int i = 0; i < Clusters.Length; i++)
156 {
157 sdcm += Clusters[i].SumOfSquaredDeviations;
158 }
159 gvf = (sdam - sdcm) / sdam;

168 private void Estimate()
169 {
170 // Initialize the matrices
171 var lowerClassLimits = new int[SortedData.Length + 1, NumberOfClusters + 1];
172 var varianceCombinations = new double[SortedData.Length + 1, NumberOfClusters + 1];
173 for (int i = 1; i <= NumberOfClusters; i++)
174 {
175 lowerClassLimits[1, i] = 1;
176 varianceCombinations[1, i] = 0;
177 for (int j = 2; j <= SortedData.Length; j++)
178 varianceCombinations[j, i] = double.MaxValue;
179 }
180
181 double v = 0;
182 for (int l = 2; l <= SortedData.Length; l++)
183 {
184 double s1 = 0;
185 double s2 = 0;
186 double w = 0;
187 for (int m = 1; m <= l; m++)
188 {
189 int i3 = l - m + 1;
190 double val = SortedData[i3 - 1];
191
192 s2 += val * val;
193 s1 += val;
194
195 w++;
196 v = s2 - (s1 * s1) / w;
197 int i4 = i3 - 1;
198 if (i4 != 0)
199 {
200 for (int j = 2; j <= NumberOfClusters; j++)
201 {
202 if (varianceCombinations[l, j] >= (v + varianceCombinations[i4, j - 1]))
203 {
204 lowerClassLimits[l, j] = i3;
205 varianceCombinations[l, j] = v + varianceCombinations[i4, j - 1];
206
207 };
208 };
209 };
210 };
211 lowerClassLimits[l, 1] = 1;
212 varianceCombinations[l, 1] = v;
213 };
214
215 // Create clusters
216 int k = SortedData.Length;
217 int[] kclass = new int[NumberOfClusters];
218 kclass[NumberOfClusters - 1] = SortedData.Length - 1;
219 for (int j = NumberOfClusters; j >= 2; j--)
220 {
221 int id = lowerClassLimits[k, j] - 2;
222 kclass[j - 2] = id;
223 k = lowerClassLimits[k, j] - 1;
224 };
225
226 Clusters = new JenksCluster[NumberOfClusters];
227 Breaks = new double[NumberOfClusters];
228 Clusters[0] = new JenksCluster(SortedData, 0, kclass[0]);
229 Breaks[0] = Clusters[0].MaxValue;
230 for (int i = 1; i < NumberOfClusters; i++)
231 {
232 Clusters[i] = new JenksCluster(SortedData, kclass[i - 1] + 1, kclass[i]);
233 Breaks[i] = Clusters[i].MaxValue;
234 }
235 }
