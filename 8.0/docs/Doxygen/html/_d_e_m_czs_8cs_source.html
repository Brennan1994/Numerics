126 protected override ParameterSet ChainIteration(int index, ParameterSet state)
127 {
128 // Update the sample count
129 SampleCount[index] += 1;
130
131 // 10% snooker updates and 90% parallel direction updates.
132 if (_chainPRNGs[index].NextDouble() <= SnookerThreshold && SampleCount[index] > 5 * ThinningInterval)
133 {
134 return SnookerUpdate(index, state);
135 }
136 else
137 {
138
139 // The adaptation for the algorithm to allow for
140 // jumps from one mode region to another in the target
141 // distribution.
142 var G = _chainPRNGs[index].NextDouble() <= JumpThreshold ? 1.0d : Jump;
143
144 // Sample uniformly at random without replacement two numbers R1 and R2
145 // from the numbers 1, 2, ..., M.
146 int r1, r2, M = PopulationMatrix.Count;
147 r1 = _chainPRNGs[index].Next(0, M);
148 do r2 = _chainPRNGs[index].Next(0, M); while (r2 == r1);
149
150 // Calculate the proposal vector
151 // x* ← xi + γ(zR1 − zR2) + e
152 // where zR1 and zR2 are rows R1 and R2 of the population matrix.
153 var xp = new double[NumberOfParameters];
154 for (int i = 0; i < NumberOfParameters; i++)
155 {
156 var xi = state.Values[i];
157 var zr1 = PopulationMatrix[r1].Values[i];
158 var zr2 = PopulationMatrix[r2].Values[i];
159 var e = Math.Pow(_b.InverseCDF(_chainPRNGs[index].NextDouble()), NumberOfParameters);
160 xp[i] = xi + G * (zr1 - zr2) + e;
161
162 // Check if the parameter is feasible (within the constraints)
163 if (xp[i] < PriorDistributions[i].Minimum || xp[i] > PriorDistributions[i].Maximum)
164 {
165 // The proposed parameter vector was infeasible,
166 // so leave xi unchanged.
167 return state;
168 }
169 }
170
171 // Evaluate fitness
172 var logLHp = LogLikelihoodFunction(xp);
173 var logLHi = state.Fitness;
174
175 // Calculate the Metropolis ratio
176 var logRatio = logLHp - logLHi;
177
178 // Accept the proposal with probability min(1,r)
179 // otherwise leave xi unchanged
180 var logU = Math.Log(_chainPRNGs[index].NextDouble());
181 if (logU <= logRatio)
182 {
183 // The proposal is accepted
184 AcceptCount[index] += 1;
185 return new ParameterSet(xp, logLHp);
186 }
187 else
188 {
189 return state;
190 }
191
192 }

200 private ParameterSet SnookerUpdate(int index, ParameterSet state)
201 {
202 // Do snooker update
203 // Get Jump -- uniform random number between 1.2 and 2.2
204 double G = _g.InverseCDF(_chainPRNGs[index].NextDouble());
205 //double G = 1.7;
206
207 // Select another chain, which is in state z
208 int c = index;
209 do c = _chainPRNGs[index].Next(0, NumberOfChains); while (c == index);
210 // Select two other random chains, zR1 and zR2
211 int c1 = c, c2 = c;
212 do c1 = _chainPRNGs[index].Next(0, NumberOfChains); while (c1 == c);
213 do c2 = _chainPRNGs[index].Next(0, NumberOfChains); while (c2 == c1 || c2 == c);
214
215 // Define z
216 int n = MarkovChains[c].Count();
217 var z = new Vector(MarkovChains[c][n - 1].Values);
218 var xi = new Vector(state.Values);
219 // Define line xi - z
220 var line = xi - z;
221 // Orthogonally project zR1 and zR2 onto the line xi - z
222 var zr1 = new Vector(MarkovChains[c1][n - 1].Values);
223 var zr2 = new Vector(MarkovChains[c2][n - 1].Values);
224 var zp1 = Vector.Project(zr1, line);
225 var zp2 = Vector.Project(zr2, line);
226
227 // Calculate the proposal vector
228 // x* ← xi + γ(zP1 − zP2)
229 var xp = new Vector(NumberOfParameters);
230 for (int i = 0; i < NumberOfParameters; i++)
231 {
232 xp[i] = xi[i] + G * (zp1[i] - zp2[i]);
233
234 // Check if the parameter is feasible (within the constraints)
235 if (xp[i] < PriorDistributions[i].Minimum || xp[i] > PriorDistributions[i].Maximum)
236 {
237 // The proposed parameter vector was infeasible,
238 // so leave xi unchanged.
239 return state;
240 }
241 }
242
243 // Evaluate fitness
244 var logLHp = LogLikelihoodFunction(xp.Array);
245 var logLHi = state.Fitness;
246
247 // Get the Euclidean distance
248 var logEDp = Math.Log(Math.Pow(Vector.Distance(xp, z), NumberOfParameters - 1));
249 var logEDi = Math.Log(Math.Pow(Vector.Distance(xi, z), NumberOfParameters - 1));
250
251 // Calculate the Metropolis ratio
252 var logRatio = logLHp + logEDp - logLHi - logEDi;
253 // Accept the proposal with probability min(1,r)
254 // otherwise leave xi unchanged
255 var logU = Math.Log(_chainPRNGs[index].NextDouble());
256 if (logU <= logRatio)
257 {
258 // The proposal is accepted
259 AcceptCount[index] += 1;
260 return new ParameterSet(xp.Array, logLHp);
261 }
262 else
263 {
264 return state;
265 }
266 }
