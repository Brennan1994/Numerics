234 public void Estimate(IList<double> sample, ParameterEstimationMethod estimationMethod)
235 {
236 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
237 {
238 // Estimate using the method of moments(a.k.a product moments).
239 var parms = IndirectMethodOfMoments(sample);
240 Mu = parms[0];
241 Sigma = parms[1];
242 }
243 else if (estimationMethod == ParameterEstimationMethod.MethodOfLinearMoments)
244 {
245 var parms = ParametersFromLinearMoments(IndirectMethodOfLinearMoments(sample));
246 Mu = parms[0];
247 Sigma = parms[1];
248 }
249 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
250 {
251 SetParameters(MLE(sample));
252 }
253 else
254 {
255 throw new NotImplementedException();
256 }
257 }

328 public static double[] IndirectMethodOfMoments(IList<double> sample)
329 {
330 // Transform the sample
331 var transformedSample = new List<double>();
332 for (int i = 0; i < sample.Count; i++)
333 {
334 if (sample[i] > 0d)
335 {
336 transformedSample.Add(Math.Log(sample[i]));
337 }
338 else
339 {
340 transformedSample.Add(Math.Log(0.1d));
341 }
342 }
343 return Statistics.ProductMoments(transformedSample);
344 }

374 public static double[] DirectMethodOfMoments(double mean, double standardDeviation)
375 {
376 if (standardDeviation <= 0)
377 return [double.NaN, double.NaN];
378 double variance = Math.Pow(standardDeviation, 2d);
379 double mu = Math.Log(Math.Pow(mean, 2d) / Math.Sqrt(variance + Math.Pow(mean, 2d)));
380 double sigma = Math.Sqrt(Math.Log(1.0d + variance / Math.Pow(mean, 2d)));
381 if (sigma < 1E-16 && Math.Sign(sigma) != -1) sigma = Tools.DoubleMachineEpsilon;
382 return [mu, sigma];
383 }

430 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
431 {
432 var initialVals = new double[NumberOfParameters];
433 var lowerVals = new double[NumberOfParameters];
434 var upperVals = new double[NumberOfParameters];
435 // Get initial values
436 var moments = Statistics.ProductMoments(sample);
437 initialVals[0] = moments[0];
438 initialVals[1] = moments[1];
439 // Get bounds of mean
440 lowerVals[0] = Tools.DoubleMachineEpsilon;
441 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[0]) + 1d));
442 // Get bounds of standard deviation
443 lowerVals[1] = Tools.DoubleMachineEpsilon;
444 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[1]) + 1d));
445 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
446 }

449 public double[] MLE(IList<double> sample)
450 {
451 // Set constraints
452 var tuple = GetParameterConstraints(sample);
453 var Initials = tuple.Item1;
454 var Lowers = tuple.Item2;
455 var Uppers = tuple.Item3;
456
457 // Solve using Nelder-Mead (Downhill Simplex)
458 double logLH(double[] x)
459 {
460 var LN = new LnNormal();
461 LN.SetParameters(x);
462 return LN.LogLikelihood(sample);
463 }
464 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
465 solver.ReportFailure = true;
466 solver.Maximize();
467 return solver.BestParameterSet.Values;
468 }

471 public override double PDF(double x)
472 {
473 // Validate parameters
474 if (_parametersValid == false)
475 ValidateParameters(Mu, Sigma, true);
476 if (x <= Minimum) return 0.0d;
477 double d = (Math.Log(x) - Mu) / Sigma;
478 return Math.Exp(-0.5d * d * d) / (Tools.Sqrt2PI * Sigma * x);
479 }

482 public override double CDF(double x)
483 {
484 // Validate parameters
485 if (_parametersValid == false)
486 ValidateParameters(Mu, Sigma, true);
487 if (x <= Minimum)
488 return 0d;
489 return 0.5d * (1.0d + Erf.Function((Math.Log(x) - Mu) / (Sigma * Math.Sqrt(2.0d))));
490 }

493 public override double InverseCDF(double probability)
494 {
495 // Validate probability
496 if (probability < 0.0d || probability > 1.0d)
497 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
498 if (probability == 0.0d)
499 return Minimum;
500 if (probability == 1.0d)
501 return Maximum;
502 // Validate parameters
503 if (_parametersValid == false)
504 ValidateParameters(Mu, Sigma, true);
505 return Math.Exp(Mu - Sigma * Math.Sqrt(2.0d) * Erf.InverseErfc(2.0d * probability));
506 }

515 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
516 {
517 if (estimationMethod != ParameterEstimationMethod.MethodOfMoments &&
518 estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
519 {
520 throw new NotImplementedException();
521 }
522 // Validate parameters
523 if (_parametersValid == false)
524 ValidateParameters(Mu, _sigma, true);
525 // Compute covariance
526 double u2 = Sigma;
527 var covar = new double[2, 2];
528 covar[0, 0] = Math.Pow(u2, 2d) / sampleSize; // location
529 covar[1, 1] = 2d * Math.Pow(u2, 4d) / sampleSize; // scale
530 covar[0, 1] = 0.0;
531 covar[1, 0] = covar[0, 1];
532 return covar;
533 }

536 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
537 {
538 var covar = ParameterCovariance(sampleSize, estimationMethod);
539 var grad = QuantileGradient(probability);
540 double varA = covar[0, 0];
541 double varB = covar[1, 1];
542 double covAB = covar[1, 0];
543 double dQx1 = grad[0];
544 double dQx2 = grad[1];
545 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + 2d * dQx1 * dQx2 * covAB;
546 }

549 public double[] QuantileGradient(double probability)
550 {
551 // Validate parameters
552 if (_parametersValid == false)
553 ValidateParameters(Mu, _sigma, true);
554 double u1 = Mu;
555 double u2 = Sigma;
556 double z = Normal.StandardZ(probability);
557 var gradient = new double[]
558 {
559 Math.Exp(u1 + z * u2), // location
560 z * Math.Exp(u1 + z * u2) / (2d * u2) // scale
561 };
562 return gradient;
563 }

566 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
567 {
568 if (probabilities.Count != NumberOfParameters)
569 {
570 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
571 }
572 // Get gradients
573 var dQp1 = QuantileGradient(probabilities[0]);
574 var dQp2 = QuantileGradient(probabilities[1]);
575 // Compute determinant
576 // |a b|
577 // |c d|
578 // |A| = ad − bc
579 double a = dQp1[0];
580 double b = dQp1[1];
581 double c = dQp2[0];
582 double d = dQp2[1];
583 determinant = a * d - b * c;
584 // Return Jacobian
585 var jacobian = new double[,] { { a, b }, { c, d } };
586 return jacobian;
587 }
