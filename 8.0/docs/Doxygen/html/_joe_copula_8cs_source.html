111 public override double Generator(double t)

113 double a = 1.0d - t;

118 public override double GeneratorInverse(double t)
119 {
120 double a = 1.0d - Math.Exp(-t);
121 return 1.0d - Math.Sign(a) * Math.Pow(Math.Abs(a), 1.0d / Theta);
122 }

125 public override double GeneratorPrime(double t)
126 {
127 double a = 1.0d - t;
128 return -(Theta * Math.Sign(a) * Math.Pow(Math.Abs(a), Theta - 1.0d)) / (1.0d - Math.Sign(a) * Math.Pow(Math.Abs(a), Theta));
129 }
130
132 public override double GeneratorPrime2(double t)
133 {
134 double a = 1.0d - t;
135 double num = Theta * (Theta + Math.Sign(a) * Math.Pow(Math.Abs(a), Theta) - 1.0d) * Math.Sign(a) * Math.Pow(Math.Abs(a), Theta - 2.0d);
136 double aa = 1.0d - Math.Sign(a) * Math.Pow(Math.Abs(a), Theta);
137 double den = Math.Sign(aa) * Math.Pow(Math.Abs(aa), 2d);
138 return num / den;
139 }

142 public override double GeneratorPrimeInverse(double t)
143 {
144 return Brent.Solve(x => GeneratorPrime(x) - t, 0d, 1d);
145 }

148 public override double[] InverseCDF(double u, double v)
149 {
150 // Validate parameters
151 if (_parametersValid == false) ValidateParameter(Theta, true);
152
153 // Use conditional probability function
154 double p = v;
155 v = Brent.Solve(x =>
156 {
157 double vu = -(Math.Pow(1d - x, Theta) - 1d) * Math.Pow(Math.Pow(1d - u, Theta) - Math.Pow(1d - u, Theta) * Math.Pow(1d - x, Theta) + Math.Pow(1d - x, Theta), (-Theta + 1d) / Theta) * Math.Pow(1d - u, Theta - 1d);
158 return vu - p;
159 }, 0d, 1d);
160 return [u, v];
161 }
