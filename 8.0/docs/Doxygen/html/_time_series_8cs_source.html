548 public void InterpolateMissingData(int maxNumberOfMissing)
549 {
550 SuppressCollectionChanged = true;
551 SortByTime();
552 double x;
553 //double y;
554 double x1;
555 double x2;
556 double y1;
557 double y2;
558 int upper;
559 //
560 for (int i = 1; i < Count; i++)
561 {
562 // Find missing value
563 if (double.IsNaN(this[i].Value))
564 {
565 // ok we found one
566 x = this[i].Index.ToOADate();
567 x1 = this[i - 1].Index.ToOADate();
568 y1 = this[i - 1].Value;
569 upper = i + maxNumberOfMissing;
570 // Find the next non-missing value
571 for (int j = i; j <= Math.Min(Count - 1, upper); j++)
572 {
573 // ok we found one
574 // the interpolation case
575 if (!double.IsNaN(this[j].Value))
576 {
577 x2 = this[j].Index.ToOADate();
578 y2 = this[j].Value;
579 this[i].Value = y1 + (x - x1) / (x2 - x1) * (y2 - y1);
580 break;
581 }
582 // the extrapolation case
583 if (j == Count - 1)
584 {
585 x1 = this[i - 2].Index.ToOADate();
586 x2 = this[i - 1].Index.ToOADate();
587 y1 = this[i - 2].Value;
588 y2 = this[i - 1].Value;
589 this[i].Value = y1 - (x1 - x) * (y2 - y1) / (x2 - x1);
590 }

592 }

1047 public TimeSeries ConvertTimeInterval(TimeInterval timeInterval, bool average = true)
1048 {
1049 var TS = TimeSeries.TimeIntervalInHours(TimeInterval); // The time step in hours
1050 var newTS = TimeSeries.TimeIntervalInHours(timeInterval); // The new time step in hours
1051 int blockDuration = (int)Math.Floor(newTS / TS);
1052 double N = EndDate.Subtract(StartDate).TotalHours / newTS + 1;
1053
1054 if (newTS == TS)
1055 {
1056 return Clone();
1057 }
1058 else if (newTS < TS && average == true)
1059 {
1060 // Create interpolater with existing data set.
1061 double t = 0, value = 0;
1062 var x = new double[Count];
1063 var y = new double[Count];
1064 for (int i = 0; i < Count; i++)
1065 {
1066 x[i] = t;
1067 y[i] = this[i].Value;
1068 t += TS;
1069 }
1070 var linInt = new Linear(x, y);
1071
1072 // Now Interpolate values for a smaller time step
1073 t = 0;
1074 var timeSeries = new TimeSeries(timeInterval);
1075 timeSeries.Add(new SeriesOrdinate<DateTime, double>(StartDate, this[0].Value));
1076 for (int i = 1; i < N; i++)
1077 {
1078 t += newTS;
1079 value = linInt.Interpolate(t);
1080 timeSeries.Add(new SeriesOrdinate<DateTime, double>(AddTimeInterval(timeSeries[i - 1].Index, timeInterval), value));
1081 }
1082 return timeSeries;
1083
1084 }
1085 else if (newTS > TS && average == true)
1086 {
1087 // Calculate block average
1088 int t = 0;
1089 var timeSeries = new TimeSeries(timeInterval);
1090 for (int i = 0; i < N; i++)
1091 {
1092 double avg = 0;
1093 for (int j = t; j < Math.Min(t + blockDuration, Count); j++)
1094 avg += this[j].Value;
1095 avg /= Math.Min(t + blockDuration, Count) - t;
1096 t += blockDuration;
1097 timeSeries.Add(new SeriesOrdinate<DateTime, double>(i == 0 ? StartDate : AddTimeInterval(timeSeries[i - 1].Index, timeInterval), avg));
1098 }
1099 return timeSeries;
1100 }
1101 else if (newTS < TS && average == false)
1102 {
1103 // Create interpolater with existing data set.
1104 double t = 0, value = 0, rate = TS / newTS;
1105 var x = new double[Count];
1106 var y = new double[Count];
1107 for (int i = 0; i < Count; i++)
1108 {
1109 x[i] = t;
1110 y[i] = this[i].Value;
1111 t += TS;
1112 }
1113 var linInt = new Linear(x, y);
1114
1115 // Now disaggregate values for a smaller time step
1116 t = 0;
1117 var timeSeries = new TimeSeries(timeInterval);
1118 timeSeries.Add(new SeriesOrdinate<DateTime, double>(StartDate, y[0] / rate));
1119 for (int i = 1; i < N; i++)
1120 {
1121 t += newTS;
1122 int idx = linInt.Search(t);
1123 value = y[idx] / rate;
1124 timeSeries.Add(new SeriesOrdinate<DateTime, double>(AddTimeInterval(timeSeries[i - 1].Index, timeInterval), value));
1125 }
1126 return timeSeries;
1127 }
1128 else if (newTS > TS && average == false)
1129 {
1130 // Calculate block sum
1131 int t = 0;
1132 var timeSeries = new TimeSeries(timeInterval);
1133 for (int i = 0; i < N; i++)
1134 {
1135 double sum = 0;
1136 for (int j = t; j < Math.Min(t + blockDuration, Count); j++)
1137 sum += this[j].Value;
1138 t += blockDuration;
1139 timeSeries.Add(new SeriesOrdinate<DateTime, double>(i == 0 ? StartDate : AddTimeInterval(timeSeries[i - 1].Index, timeInterval), sum));
1140 }
1141 return timeSeries;
1142 }
1143
1144 return null;
