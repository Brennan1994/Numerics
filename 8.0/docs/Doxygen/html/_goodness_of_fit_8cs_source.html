98 public static double[] AICWeights(IList<double> aicValues)
99 {
100 var min = Tools.Min(aicValues);
101 var weights = new double[aicValues.Count];
102 var num = new double[aicValues.Count];
103 double sum = 0;
104 for (int i = 0; i < aicValues.Count; i++)
105 {
106 num[i] = Math.Exp(-0.5 * (aicValues[i] - min));
107 sum += num[i];
108 }
109 for (int i = 0; i < aicValues.Count; i++)
110 {
111 weights[i] = num[i] / sum;
112 }
113 return weights;

145 public static double RMSE(IList<double> observedValues, IList<double> modeledValues, int k = 0)
146 {
147 // Check if the lists are the same size
148 if (observedValues.Count != modeledValues.Count)
149 throw new ArgumentOutOfRangeException(nameof(observedValues), "The number of observed values must equal the number of modeled values.");
150
151 int n = observedValues.Count - k;
152 double sse = 0d;
153 for (int i = 0; i < n; i++)
154 sse += Tools.Sqr(modeledValues[i] - observedValues[i]);
155 return Math.Sqrt(sse / n);
156 }

190 public static double[] RMSEWeights(IList<double> rmseValues)
191 {
192 var weights = new double[rmseValues.Count];
193 var invMSE = new double[rmseValues.Count];
194 double sum = 0;
195 for (int i = 0; i < rmseValues.Count; i++)
196 {
197 invMSE[i] = 1 / (rmseValues[i] * rmseValues[i]);
198 sum += invMSE[i];

201 {
202 weights[i] = invMSE[i] / sum;

257 public static double KolmogorovSmirnov(IList<double> observedValues, UnivariateDistributionBase model)
258 {
259 // Check if the lists are the same size
260 if (observedValues.Count < 1)
261 throw new ArgumentOutOfRangeException(nameof(observedValues), "There must be more than one observed value.");
262
263 int n = observedValues.Count;
264 double D = double.MinValue;
265 for (int i = 1; i <= n; i++)
266 {
267 double left = model.CDF(observedValues[i - 1]) - ((double)i - 1) / n;
268 double right = (double)i / n - model.CDF(observedValues[i - 1]);
269 D = Math.Max(D, Math.Max(left, right));
270 }
271 return D;
272 }

284 public static double ChiSquared(IList<double> observedValues, UnivariateDistributionBase model)
285 {
286 // Check if the lists are the same size
287 if (observedValues.Count < 1)
288 throw new ArgumentOutOfRangeException(nameof(observedValues), "There must be more than one observed value.");
289
290 int n = observedValues.Count;
291 var hist = new Histogram(observedValues);
292 double x2 = 0;
293 for (int i = 0; i < hist.NumberOfBins; i++)
294 {
295 double e = n * (model.CDF(hist[i].UpperBound) - model.CDF(hist[i].LowerBound));
296 x2 += Math.Pow(hist[i].Frequency - e, 2) / e;
297 }
298 return x2;
299 }

311 public static double AndersonDarling(IList<double> observedValues, UnivariateDistributionBase model)
312 {
313 // Check if the lists are the same size
314 if (observedValues.Count < 1)
315 throw new ArgumentOutOfRangeException(nameof(observedValues), "There must be more than one observed value.");
316
317 int n = observedValues.Count;
318 double S = 0;
319 for (int i = 1; i <= n; i++)
320 S += (2 * (double)i - 1) * (model.LogCDF(observedValues[i - 1]) + model.LogCCDF(observedValues[n - i]));
321
322 return -n - S / n;
323 }
