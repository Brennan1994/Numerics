200 private double Simpsons(Func<double, double> f, double a, double fa, double b, double fb, ref double m, ref double fm)
201 {
202 m = (a + b) / 2d;
203 fm = f(m);
204 FunctionEvaluations++;
205 return Math.Abs(b - a) / 6d * (fa + 4d * fm + fb);
206 }

225 private double AdaptiveSimpsons(Func<double, double> f, double a, double fa, double b, double fb, double epsilon, int depth, double whole, double m, double fm)
226 {
227
228 double lm = 0, flm = 0, left = Simpsons(f, a, fa, m, fm, ref lm, ref flm);
229 double rm = 0, frm = 0, right = Simpsons(f, m, fm, b, fb, ref rm, ref frm);
230 double delta = (left + right - whole) / 15d;
231
232 // Check tolerance
233 // - Depth is less than or equal to 0 (max recursions have been reached)
234 // - Abs(a-b) is smaller than machine epsilon
235 // - Maximum number of function evaluations
236 // - Minimum number of function evaluations, the minimum depth, and also delta less than tolerance
237 if (depth <= 0 || Math.Abs(a - b) <= Tools.DoubleMachineEpsilon || FunctionEvaluations >= MaxFunctionEvaluations ||
238 (FunctionEvaluations >= MinFunctionEvaluations && depth <= MaxDepth - MinDepth && Math.Abs(delta) <= epsilon + epsilon * Math.Abs(whole)))
239 {
240 // convergence is reached
241 // Terminate recursion
242 StandardError += Math.Abs(delta);
243 return left + right + delta;
244 }
245 else
246 {
247 // Subdivide interval
248 var l = AdaptiveSimpsons(f, a, fa, m, fm, epsilon, depth - 1, left, lm, flm);
249 var r = AdaptiveSimpsons(f, m, fm, b, fb, epsilon, depth - 1, right, rm, frm);
250 return l + r;
251 }
252 }
