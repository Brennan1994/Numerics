408 public double[] ParametersFromLinearMoments(IList<double> moments)
409 {
410 // This routine is taken and converted directly from Fortran code
411 //***********************************************************************
412 //* *
413 //* FORTRAN CODE WRITTEN FOR INCLUSION IN IBM RESEARCH REPORT RC20525, *
414 //* 'FORTRAN ROUTINES FOR USE WITH THE METHOD OF L-MOMENTS, VERSION 3' *
415 //* *
416 //* J.R.M.HOSKING *
417 //* IBM RESEARCH DIVISION *
418 //* T.J.WATSON RESEARCH CENTER *
419 //* YORKTOWN HEIGHTS *
420 //* NEW YORK 10598, U.S.A. *
421 //* *
422 //* VERSION 3 AUGUST 1996 *
423 //* *
424 //***********************************************************************
425
426 double L1 = moments[0];
427 double L2 = moments[1];
428 double T3 = moments[2];
429 double T4 = moments[3];
430 double eps = 1E-6;
431 int maxit = 20, maxsr = 10;
432 double xi = 0, alpha = 0, kappa = 0, hondo = 0;
433
434 // TEST FOR FEASIBILITY
435
436 if (L2 <= 0d) throw new ArgumentException("L-moments invalid.");
437 if (Math.Abs(T3) >= 1d || Math.Abs(T4) >= 1d) throw new ArgumentException("L-moments invalid.");
438 if (T4 <= (5d * T3 * T3 - 1d) / 4d) throw new ArgumentException("L-moments invalid.");
439 if (T4 >= (5d * T3 * T3 + 1d) / 6d) throw new ArgumentException("(TAU-3, TAU-4) lies above the Generalized Logistic (suggests that L-moments are not constant with any Kappa distribution with hondo > -1).");
440
441 // SET STARTING VALUES FOR N-R ITERATION:
442 // G IS CHOSEN TO GIVE THE CORRECT VALUE OF TAU - 3 ON THE
443 // ASSUMPTION THAT H = 1(I.E.A GENERALIZED PARETO FIT) -
444 // BUT H IS ACTUALLY SET TO 1.001 TO AVOID NUMERICAL
445 // DIFFICULTIES WHICH CAN SOMETIMES ARISE WHEN H = 1 EXACTLY
446
447 double G = (1d - 3d * T3) / (1d + T3);
448 double H = 1.001d;
449 double Z = G + H * 0.725d;
450 double XDIST = 10d, DIST = 0;
451 double U1 = 0, U2 = 0, U3 = 0, U4 = 0;
452 double ALAM2 = 0, ALAM3, ALAM4;
453 double TAU3 = 0, TAU4 = 0;
454 double E1 = 0, E2 = 0;
455 double DEL1 = 0, DEL2 = 0;
456 double XG = 0, XH = 0, XZ, RHH;
457 double U1G, U2G, U3G, U4G, U1H, U2H, U3H, U4H;
458 double DL2G, DL2H, DL3G, DL3H, DL4G, DL4H;
459 double D11, D12, D21, D22, DET, H11, H12, H21, H22;
460 double FACTOR;
461
462 // START OF NEWTON-RAPHSON ITERATION
463
464 for (int i = 1; i < maxit; i++)
465 {
466
467 // REDUCE STEPLENGTH UNTIL WE ARE NEARER TO THE REQUIRED
468 // VALUES OF TAU-3 AND TAU-4 THAN WE WERE AT THE PREVIOUS STEP
469
470 for (int j = 1; j <= maxsr; j++)
471 {
472 if (G > 53d) throw new Exception("Iteration encountered numerical difficulties - overflow would have been likely to occur.");
473
474 if (H <= 0)
475 {
476 U1 = Math.Exp(Gamma.LogGamma(-1d / H - G) - Gamma.LogGamma(-1d / H + 1d));
477 U2 = Math.Exp(Gamma.LogGamma(-2d / H - G) - Gamma.LogGamma(-2d / H + 1d));
478 U3 = Math.Exp(Gamma.LogGamma(-3d / H - G) - Gamma.LogGamma(-3d / H + 1d));
479 U4 = Math.Exp(Gamma.LogGamma(-4d / H - G) - Gamma.LogGamma(-4d / H + 1d));
480 }
481 else
482 {
483 U1 = Math.Exp(Gamma.LogGamma(1d / H) - Gamma.LogGamma(1d / H + 1d + G));
484 U2 = Math.Exp(Gamma.LogGamma(2d / H) - Gamma.LogGamma(2d / H + 1d + G));
485 U3 = Math.Exp(Gamma.LogGamma(3d / H) - Gamma.LogGamma(3d / H + 1d + G));
486 U4 = Math.Exp(Gamma.LogGamma(4d / H) - Gamma.LogGamma(4d / H + 1d + G));
487 }
488 ALAM2 = U1 - 2d * U2;
489 ALAM3 = -U1 + 6d * U2 - 6d * U3;
490 ALAM4 = U1 - 12d * U2 + 30d * U3 - 20d * U4;
491 if (ALAM2 == 0d) throw new Exception("Iteration encountered numerical difficulties - overflow would have been likely to occur.");
492 TAU3 = ALAM3 / ALAM2;
493 TAU4 = ALAM4 / ALAM2;
494 E1 = TAU3 - T3;
495 E2 = TAU4 - T4;
496
497 // IF NEARER THAN BEFORE, EXIT THIS LOOP
498 DIST = Math.Max(Math.Abs(E1), Math.Abs(E2));
499 if (DIST < XDIST) break;
500
501 // OTHERWISE, HALVE THE STEPLENGTH AND TRY AGAIN
502 DEL1 *= 0.5;
503 DEL2 *= 0.5;
504 G = XG - DEL1;
505 H = XH - DEL2;
506
507 // TOO MANY STEPLENGTH REDUCTIONS
508 if (j == maxsr) throw new Exception("Iteration encountered numerical difficulties - overflow would have been likely to occur.");
509 }
510
511 // TEST FOR CONVERGENCE
512 if (DIST < eps) break;
513
514 // NOT CONVERGED: CALCULATE NEXT STEP
515 // NOTATION:
516 // U1G - DERIVATIVE OF U1 W.R.T.G
517 // DL2G - DERIVATIVE OF ALAM2 W.R.T.G
518 // D.. - MATRIX OF DERIVATIVES OF TAU-3 AND TAU-4 W.R.T.G AND H
519 // H.. - INVERSE OF DERIVATIVE MATRIX
520 // DEL. - STEPLENGTH
521
522 XG = G;
523 XH = H;
524 XZ = Z;
525 XDIST = DIST;
526 RHH = 1d / (H * H);
527
528 if (H > 0)
529 {
530 U1G = -U1 * Gamma.Digamma(1d / H + 1d + G);
531 U2G = -U2 * Gamma.Digamma(2d / H + 1d + G);
532 U3G = -U3 * Gamma.Digamma(3d / H + 1d + G);
533 U4G = -U4 * Gamma.Digamma(4d / H + 1d + G);
534 U1H = RHH * (-U1G - U1 * Gamma.Digamma(1d / H));
535 U2H = 2d * RHH * (-U2G - U2 * Gamma.Digamma(2d / H));
536 U3H = 3d * RHH * (-U3G - U3 * Gamma.Digamma(3d / H));
537 U4H = 4d * RHH * (-U4G - U4 * Gamma.Digamma(4d / H));
538 }
539 else
540 {
541 U1G = -U1 * Gamma.Digamma(-1d / H - G);
542 U2G = -U2 * Gamma.Digamma(-2d / H - G);
543 U3G = -U3 * Gamma.Digamma(-3d / H - G);
544 U4G = -U4 * Gamma.Digamma(-4d/ H - G);
545 U1H = RHH * (-U1G - U1 * Gamma.Digamma(-1d / H + 1d));
546 U2H = 2d * RHH * (-U2G - U2 * Gamma.Digamma(-2d / H + 1d));
547 U3H = 3d * RHH * (-U3G - U3 * Gamma.Digamma(-3d / H + 1d));
548 U4H = 4d * RHH * (-U4G - U4 * Gamma.Digamma(-4d / H + 1d));
549 }
550
551 DL2G = U1G - 2d * U2G;
552 DL2H = U1H - 2d * U2H;
553 DL3G = -U1G + 6d * U2G - 6d * U3G;
554 DL3H = -U1H + 6d * U2H - 6d * U3H;
555 DL4G = U1G - 12d * U2G + 30d * U3G - 20d * U4G;
556 DL4H = U1H - 12d * U2H + 30d * U3H - 20d * U4H;
557 D11 = (DL3G - TAU3 * DL2G) / ALAM2;
558 D12 = (DL3H - TAU3 * DL2H) / ALAM2;
559 D21 = (DL4G - TAU4 * DL2G) / ALAM2;
560 D22 = (DL4H - TAU4 * DL2H) / ALAM2;
561 DET = D11 * D22 - D12 * D21;
562 H11 = D22 / DET;
563 H12 = -D12 / DET;
564 H21 = -D21 / DET;
565 H22 = D11 / DET;
566 DEL1 = E1 * H11 + E2 * H12;
567 DEL2 = E1 * H21 + E2 * H22;
568
569 // TAKE NEXT N-R STEP
570
571 G = XG - DEL1;
572 H = XH - DEL2;
573 Z = G + H * 0.725;
574
575 // REDUCE STEP IF G AND H ARE OUTSIDE THE PARAMETER SPACE
576 FACTOR = 1d;
577 if (G <= -1d) FACTOR = 0.8 * (XG + 1d) / DEL1;
578 if (H <= -1d) FACTOR = Math.Min(FACTOR, 0.8 * (XH + 1d) / DEL2);
579 if (Z <= -1d) FACTOR = Math.Min(FACTOR, 0.8 * (XZ + 1d) / (XZ - Z));
580 if (H <= 0 && G * H <= -1d) FACTOR = Math.Min(FACTOR, 0.8 * (XG * XH + 1d) / (XG * XH - G * H));
581 if (FACTOR != 1d)
582 {
583 DEL1 = DEL1 * FACTOR;
584 DEL2 = DEL2 * FACTOR;
585 G = XG - DEL1;
586 H = XH - DEL2;
587 Z = G + H * 0.725;
588 }
589
590 // NOT CONVERGED
591 if (i == maxit) throw new Exception("Iterations failed to converge.");
592 }
593
594 hondo = H;
595 kappa = G;
596 var TEMP = Gamma.LogGamma(1d + G);
597 if (TEMP > 170d) throw new Exception("Iteration for hondo and kappa converged, but overflow would have occurred when calculating xi and alpha.");
598 var GAM = Math.Exp(TEMP);
599 TEMP = (1d + G) * Math.Log(Math.Abs(H));
600 if (TEMP > 170d) throw new Exception("Iteration for hondo and kappa converged, but overflow would have occurred when calculating xi and alpha.");
601 var HH = Math.Exp(TEMP);
602 alpha = L2 * G * HH / (ALAM2 * GAM);
603 xi = L1 - alpha / G * (1d - GAM * U1 / HH);
604
605 return [xi, alpha, kappa, hondo];
606

610 public double[] LinearMomentsFromParameters(IList<double> parameters)
611 {
612 double xi = parameters[0];
613 double alpha = parameters[1];
614 double kappa = parameters[2];
615 double hondo = parameters[3];
616 if (((kappa < -1) && (hondo >= 0)) || ((hondo < 0) && ((kappa <= -1) || (kappa >= -1 / hondo))))
617 {
618 throw new ArgumentOutOfRangeException(nameof(parameters), "L-moments can only be defined for hondo (h) >= 0 and kappa (k) > -1, or if h < 0 and -1 < k < -1/h.");
619 }
620
621 double L1;
622 double L2;
623 double T3;
624 double T4;
625 if (kappa == 0.0d) kappa = Math.Pow(10d, -100);
626 if (hondo == 0.0d)
627 {
628 L1 = xi + alpha * (1.0d - Gamma.Function(1.0d + kappa)) / kappa;
629 L2 = alpha * (1.0d - Math.Pow(2.0d, -kappa)) * Gamma.Function(1.0d + kappa) / kappa;
630 T3 = 2.0d * (1.0d - Math.Pow(3.0d, -kappa)) / (1.0d - Math.Pow(2.0d, -kappa)) - 3.0d;
631 T4 = (5.0d * (1.0d - Math.Pow(4.0d, -kappa)) - 10.0d * (1.0d - Math.Pow(3.0d, -kappa)) + 6.0d * (1.0d - Math.Pow(2.0d, -kappa))) / (1.0d - Math.Pow(2.0d, -kappa));
632 }
633 else
634 {
635 var g = new double[4];
636 for (int r = 1; r <= 4; r++)
637 {
638 if (hondo > 0.0d)
639 {
640 g[r - 1] = r * Gamma.Function(1d + kappa) * Gamma.Function(r / hondo) / (Math.Pow(hondo, 1d + kappa) * Gamma.Function(1d + kappa + r / hondo));
641 }
642 else
643 {
644 g[r - 1] = r * Gamma.Function(1d + kappa) * Gamma.Function(-kappa - r / hondo) / (Math.Pow(-hondo, 1d + kappa) * Gamma.Function(1d - r / hondo));
645 }
646 }
647
648 L1 = xi + alpha * (1d - g[0]) / kappa;
649 L2 = alpha * (g[0] - g[1]) / kappa;
650 T3 = (-g[0] + 3d * g[1] - 2d * g[2]) / (g[0] - g[1]);
651 T4 = -(-g[0] + 6d * g[1] - 10d * g[2] + 5d * g[3]) / (g[0] - g[1]);
652 }
653 //
654 return [L1, L2, T3, T4];
655 }

659 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
660 {
661 var initialVals = new double[NumberOfParameters];
662 var lowerVals = new double[NumberOfParameters];
663 var upperVals = new double[NumberOfParameters];
664
665 // Get initial values
666 try
667 {
668 initialVals = ParametersFromLinearMoments(Statistics.LinearMoments(sample));
669
670 // Get bounds of location
671 if (initialVals[0] == 0d) initialVals[0] = Tools.DoubleMachineEpsilon;
672 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
673 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
674
675 // Get bounds of scale
676 lowerVals[1] = Tools.DoubleMachineEpsilon;
677 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[1])) + 1d));
678
679 // Get bounds of shape
680 lowerVals[2] = -10d;
681 upperVals[2] = 10d;
682
683 // Get bounds of shape 2
684 lowerVals[3] = -2d;
685 upperVals[3] = 2d;
686
687 // Correct initial value of kappa and hondo if necessary
688 if (initialVals[2] <= lowerVals[2] || initialVals[2] >= upperVals[2])
689 {
690 initialVals[2] = 0d;
691 }
692
693 if (initialVals[3] <= lowerVals[3] || initialVals[3] >= upperVals[3])
694 {
695 initialVals[3] = 0d;
696 }
697
698 }
699 catch
700 {
701
702 // Get constraints from GEV
703 var gev = new GeneralizedExtremeValue();
704 var parms = gev.GetParameterConstraints(sample);
705 for (int i = 0; i < 3; i++)
706 {
707 initialVals[i] = parms.Item1[i];
708 lowerVals[i] = parms.Item2[i];
709 upperVals[i] = parms.Item3[i];
710
711 // Get bounds of shape 2
712 initialVals[3] = 0;

837 public double[] QuantileGradient(double probability)
838 {
839 double a = Alpha, k = Kappa, h = Hondo, F = probability;
840 double dxi = 1d;
841 double da = (1d - Math.Pow((1d - Math.Pow(F, h)) / h, k)) / k;
842 double dk = a * (-(1d - Math.Pow((1d - Math.Pow(F, h)) / h, k)) / (k * k) - Math.Pow((1d - Math.Pow(F, h)) / h, k) * Math.Log((1d - Math.Pow(F, h)) / h) / k);
843 double x = 1d - Math.Pow(F, h);
844 double dh = -(a * (Math.Pow(F, h) - h * Math.Log(F) * Math.Pow(F, h) - 1d) * Math.Sign(x) * Math.Pow(Math.Abs(x), k - 1d)) / (Math.Sign(h) * Math.Pow(Math.Abs(h), k + 1d));
845 return [dxi, da, dk, dh];

849 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
850 {
851 if (probabilities.Count != NumberOfParameters)
852 {
853 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
854 }
855 // |a b c d|
856 // |e f g h|
857 // |i j k l|
858 // |m n o p|
859 var jacobian = new Matrix(4);
860 for (int i = 0; i < 4; i++)
861 {
862 // Get the gradient
863 var dFdx = QuantileGradient(probabilities[i]);
864 // Populate the Jacobian matrix
865 for (int j = 0; j < 4; j++)
866 jacobian[i, j] = dFdx[j];
867 }
868 // Solve determinant with LU decomposition
869 var LU = new LUDecomposition(jacobian);
870 determinant = LU.Determinant();
871 // Return Jacobian
872 return jacobian.ToArray();
873 }
