78 public BFGS(Func<double[], double> objectiveFunction, int numberOfParameters,
79 IList<double> initialValues, IList<double> lowerBounds, IList<double> upperBounds,
80 Func<double[], double[]> gradient = null) : base(objectiveFunction, numberOfParameters)
81 {
82 // Check if the length of the initial, lower and upper bounds equal the number of parameters
83 if (initialValues.Count != numberOfParameters || lowerBounds.Count != numberOfParameters || upperBounds.Count != numberOfParameters)
84 {
85 throw new ArgumentOutOfRangeException(nameof(lowerBounds), "The initial values and lower and upper bounds must be the same length as the number of parameters.");
86 }
87 // Check if the initial values are between the lower and upper values
88 for (int j = 0; j < initialValues.Count; j++)
89 {
90 if (upperBounds[j] < lowerBounds[j])
91 {
92 throw new ArgumentOutOfRangeException(nameof(upperBounds), "The upper bound cannot be less than the lower bound.");
93 }
94 if (initialValues[j] < lowerBounds[j] || initialValues[j] > upperBounds[j])
95 {
96 throw new ArgumentOutOfRangeException(nameof(initialValues), "The initial values must be between the upper and lower bounds.");
97 }
98 }
99 InitialValues = initialValues.ToArray();
100 LowerBounds = lowerBounds.ToArray();
101 UpperBounds = upperBounds.ToArray();
102 Gradient = gradient;
103 }

126 protected override void Optimize()
127 {
128 int i, j, D = NumberOfParameters;
129 bool cancel = false;
130 double EPS = Tools.DoubleMachineEpsilon;
131 double TOLX = 4 * EPS, STPMX = 100.0;
132 bool check = false;
133 double fret = 0, fac, fad, fae, fp, stpmax, sum = 0.0, sumdg, sumxi;
134 var p = InitialValues.ToArray();
135 var dg = new double[D];
136 var g = new double[D];
137 var hdg = new double[D];
138 var pnew = new double[D];
139 var xi = new double[D];
140 var hessin = new Matrix(D, D);
141
142 // Calculate the starting function value and gradient, and initialize the inverse Hessian to the unit matrix.
143 fp = Evaluate(p, ref cancel);
144 g = Gradient != null ? Gradient(p) : NumericalDerivative.Gradient((x) => Evaluate(x, ref cancel), p);
145 for (i = 0; i < D; i++)
146 {
147 for (j = 0; j < D; j++) hessin[i, j] = 0.0;
148 hessin[i, i] = 1.0;
149 xi[i] = -g[i];
150 sum += p[i] * p[i];
151 }
152 stpmax = STPMX * Math.Max(Math.Sqrt(sum), D);
153 while (Iterations < MaxIterations)
154 {
155 // Perform line search
156 LineSearch(p, fp, g, ref xi, ref pnew, ref fret, stpmax, ref check, ref cancel);
157 if (cancel) return;
158 // Check convergence.
159 if (CheckConvergence(fp, fret))
160 {
161 UpdateStatus(OptimizationStatus.Success);
162 return;
163 }
164 // The new function evaluation occurs in line search; save the function value in fp for the next line search.
165 // It is usually safe to ignore the value of check.
166 fp = fret;
167 for (i = 0; i < D; i++)
168 {
169 xi[i] = pnew[i] - p[i];
170 p[i] = pnew[i];
171 }
172 // Save the old gradient, and get the new gradient.
173 for (i = 0; i < D; i++) dg[i] = g[i];
174 g = Gradient != null ? Gradient(p) : NumericalDerivative.Gradient((x) => Evaluate(x, ref cancel), p);
175 if (cancel) return;
176 // Compute difference of gradients.
177 for (i = 0; i < D; i++)
178 dg[i] = g[i] - dg[i];
179 // And difference times current matrix.
180 for (i = 0; i < D; i++)
181 {
182 hdg[i] = 0.0;
183 for (j = 0; j < D; j++) hdg[i] += hessin[i, j] * dg[j];
184 }
185 // Calculate dot products for the denominators.
186 fac = fae = sumdg = sumxi = 0.0;
187 for (i = 0; i < D; i++)
188 {
189 fac += dg[i] * xi[i];
190 fae += dg[i] * hdg[i];
191 sumdg += Tools.Sqr(dg[i]);
192 sumxi += Tools.Sqr(xi[i]);
193 }
194 // Skip update if fac not sufficiently positive.
195 if (fac > Math.Sqrt(EPS * sumdg * sumxi))
196 {
197 fac = 1.0 / fac;
198 fad = 1.0 / fae;
199 for (i = 0; i < D; i++) dg[i] = fac * xi[i] - fad * hdg[i];
200 for (i = 0; i < D; i++)
201 {
202 for (j = i; j < D; j++)
203 {
204 hessin[i, j] += fac * xi[i] * xi[j] - fad * hdg[i] * hdg[j] + fae * dg[i] * dg[j];
205 hessin[j, i] = hessin[i, j];
206 }
207 }
208 }
209 for (i = 0; i < D; i++)
210 {
211 xi[i] = 0.0;
212 for (j = 0; j < D; j++) xi[i] -= hessin[i, j] * g[j];
213 }
214
215 Iterations += 1;
216 }
217
218 // If we made it to here, the maximum iterations were reached.
219 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
220
221 }

235 private void LineSearch(double[] xold, double fold, double[] g, ref double[] p, ref double[] x, ref double f, double stpmax, ref bool check, ref bool cancel)
236 {
237 double ALF = 1.0e-4, TOLX = Tools.DoubleMachineEpsilon;
238 double a, alam, alam2 = 0.0, alamin, b, disc, f2 = 0.0;
239 double rhs1, rhs2, slope = 0.0, sum = 0.0, temp, test, tmplam;
240 int i, n = xold.Length;
241 check = false;
242 for (i = 0; i < n; i++) sum += p[i] * p[i];
243 sum = Math.Sqrt(sum);
244 if (sum > stpmax)
245 for (i = 0; i < n; i++)
246 p[i] *= stpmax / sum;
247 for (i = 0; i < n; i++)
248 slope += g[i] * p[i];
249 if (slope == 0.0) return; // If the slope is zero, it is on a flat spit. Exit the routine
250 if (slope > 0.0) throw new Exception("Roundoff problem in line search.");
251 test = 0.0;
252 for (i = 0; i < n; i++)
253 {
254 temp = Math.Abs(p[i]) / Math.Max(Math.Abs(xold[i]), 1.0);
255 if (temp > test) test = temp;
256 }
257 alamin = TOLX / test;
258 alam = 1.0;
259 for (; ; )
260 {
261 for (i = 0; i < n; i++)
262 {
263 x[i] = xold[i] + alam * p[i];
264 // Make sure the parameters are within the bounds.
265 x[i] = RepairParameter(x[i], LowerBounds[i], UpperBounds[i]);
266 }
267 f = Evaluate(x, ref cancel);
268 if (cancel) return;
269 if (alam < alamin)
270 {
271 for (i = 0; i < n; i++) x[i] = xold[i];
272 check = true;
273 return;
274 }
275 else if (f <= fold + ALF * alam * slope) return;
276 else
277 {
278 if (alam == 1.0)
279 tmplam = -slope / (2.0 * (f - fold - slope));
280 else
281 {
282 rhs1 = f - fold - alam * slope;
283 rhs2 = f2 - fold - alam2 * slope;
284 a = (rhs1 / (alam * alam) - rhs2 / (alam2 * alam2)) / (alam - alam2);
285 b = (-alam2 * rhs1 / (alam * alam) + alam * rhs2 / (alam2 * alam2)) / (alam - alam2);
286 if (a == 0.0) tmplam = -slope / (2.0 * b);
287 else
288 {
289 disc = b * b - 3.0 * a * slope;
290 if (disc < 0.0) tmplam = 0.5 * alam;
291 else if (b <= 0.0) tmplam = (-b + Math.Sqrt(disc)) / (3.0 * a);
292 else tmplam = -slope / (b + Math.Sqrt(disc));
293 }
294 if (tmplam > 0.5 * alam)
295 tmplam = 0.5 * alam;
296 }
297 }
298 alam2 = alam;
299 f2 = f;
300 alam = Math.Max(tmplam, 0.1 * alam);
301 }
302
303 }
