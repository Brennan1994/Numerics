230 private void FitSVD()
231 {
232 double meanY = Statistics.Statistics.Mean(Y.ToArray());
233 var xx = X;
234 // see if we need to add an intercept column
235 if (hasIntercept)
236 {
237 xx = new Matrix(X.NumberOfRows, X.NumberOfColumns + 1);
238 for (int r = 0; r < X.NumberOfRows; r++)
239 {
240 xx[r, 0] = 1;
241 for (int c = 0; c < X.NumberOfColumns; c++)
242 {
243 xx[r, c + 1] = X[r, c];
244 }
245 }
246 }
247
248 int i, j, k, n = xx.NumberOfRows, m = xx.NumberOfColumns;
249 double sse = 0.0, sst = 0.0, sum = 0.0;
250 DegreesOfFreedom = n - m;
251 Residuals = new double[n];
252 Covariance = new Matrix(m);
253
254 // Estimate coefficients
255 var svd = new SingularValueDecomposition(xx);
256 double tol = 1E-12;
257 double thresh = (tol > 0d ? tol * svd.W[0] : -1d);
258 var betas = svd.Solve(Y, thresh); // vector of fitted coefficients
259
260 // Estimate uncertainty in the coefficients
261 for (i = 0; i < n; i++)
262 {
263 sum = 0.0;
264 for (j = 0; j < m; j++) sum += xx[i, j] * betas[j];
265 Residuals[i] = Y[i] - sum;
266 sse += Tools.Sqr(Residuals[i]);
267 sst += Tools.Sqr(Y[i] - meanY);
268 }
269 for (i = 0; i < m; i++)
270 {
271 for (j = 0; j < i + 1; j++)
272 {
273 sum = 0.0;
274 for (k = 0; k < m; k++) if (svd.W[k] > svd.Threshold)
275 sum += svd.V[i, k] * svd.V[j, k] / Tools.Sqr(svd.W[k]);
276 Covariance[j, i] = Covariance[i, j] = sum;
277 }
278 }
279 double SE = Math.Sqrt(sse / DegreesOfFreedom);
280
281 // Set the output
282 Parameters = betas.ToList();
283 ParameterStandardErrors = new List<double>();
284 for (j = 0; j < m; j++) ParameterStandardErrors.Add(Math.Sqrt(Covariance[j, j]) * SE);
285 StandardError = SE;
286 RSquared = 1 - sse / sst;
287 AdjRSquared = 1 - (1 - RSquared) * (n - 1) / (DegreesOfFreedom);
288
289 }

296 public double[] Predict(Matrix x)
297 {
298 var result = new double[x.NumberOfRows];
299 for (int i = 0; i < x.NumberOfRows; i++)
300 {
301 if (hasIntercept == true)
302 {
303 var values = new List<double>() { 1 };
304 values.AddRange(x.Row(i));
305 result[i] = Tools.SumProduct(Parameters, values);
306 }
307 else
308 {
309 result[i] = Tools.SumProduct(Parameters, x.Row(i));
310 }
311 }
312
313 return result;

321 public double[,] PredictionIntervals(Matrix x, double alpha = 0.1)
322 {
323 var percentiles = new double[] { alpha / 2d, 1d - alpha / 2d };
324 var result = new double[x.NumberOfRows, 3]; // lower, upper, mean
325 for (int i = 0; i < x.NumberOfRows; i++)
326 {
327 double mu = 0;
328 if (hasIntercept == true)
329 {
330 var values = new List<double>() { 1 };
331 values.AddRange(x.Row(i));
332 mu = Tools.SumProduct(Parameters, values);
333
334 }
335 else
336 {
337 mu = Tools.SumProduct(Parameters, x.Row(i));
338 }
339 var s = StandardError;
340 var s2 = s * s;
341 var n = DegreesOfFreedom;
342 var t = new StudentT(mu, Math.Sqrt(s2 / n + s2), n);
343 result[i, 0] = t.InverseCDF(percentiles[0]);
344 result[i, 1] = t.InverseCDF(percentiles[1]);
345 result[i, 2] = mu;
346 }
347
348 return result;
349 }
