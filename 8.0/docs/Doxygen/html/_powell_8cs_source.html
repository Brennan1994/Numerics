75 {

115 protected override void Optimize()
116 {
117 // Set variables
118 int i, j, D = NumberOfParameters, ibig;
119 bool cancel = false;
120 double t, fret, fp, fptt, delta;
121 var p = InitialValues.ToArray();
122 var pt = new double[D];
123 var ptt = new double[D];
124 var xi = new double[D]; // Direction vector
125 // Set the initial matrix for directions
126 // and save the initial point
127 var ximat = new double[D, D];
128 for (i = 0; i < D; i++)
129 {
130 ximat[i, i] = 1d;
131 pt[i] = p[i];
132 }
133 // initial function evaluation
134 fret = Evaluate(p, ref cancel);
135 while (Iterations < MaxIterations)
136 {
137 fp = fret;
138 ibig = 0;
139 delta = 0.0; // Will be the biggest function decrease.
140 // In each iteration, loop over all directions in the set.
141 for (i = 0; i < D; i++)
142 {
143 // Copy the direction
144 for (j = 0; j < D; j++) xi[j] = ximat[j, i];
145 fptt = fret;
146 fret = LineMinimization(p, xi, ref cancel);
147 if (cancel == true) return;
148 // And record it if it is the larges decrease so far.
149 if (fptt - fret > delta)
150 {
151 delta = fptt - fret;
152 ibig = i + 1;
153 }
154 }
155 // Check convergence
156 if (CheckConvergence(fp, fret))
157 {
158 UpdateStatus(OptimizationStatus.Success);
159 return;
160 }
161 // Construct the extrapolated point and save the average direction moved.
162 // Save the old starting point.
163 for (j = 0; j < D; j++)
164 {
165 ptt[j] = 2.0 * p[j] - pt[j];
166 xi[j] = p[j] - pt[j];
167 pt[j] = p[j];
168 }
169 // Function evaluated at the extrapolated point
170 fptt = Evaluate(ptt, ref cancel);
171 if (cancel == true) return;
172 if (fptt < fp)
173 {
174 t = 2.0 * (fp - 2.0 * fret + fptt) * Tools.Sqr(fp - fret - delta) - delta * Tools.Sqr(fp - fptt);
175 if (t < 0.0)
176 {
177 // Move to the minimum of the new direction and save the new direction
178 fret = LineMinimization(p, xi, ref cancel);
179 if (cancel == true) return;
180 for (j = 0; j < D; j++)
181 {
182 ximat[j, ibig - 1] = ximat[j, D - 1];
183 ximat[j, D - 1] = xi[j];
184 }
185 }
186 }
187
188 Iterations += 1;
189 }
190
191 // If we made it to here, the maximum iterations were reached.
192 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
193
194 }

202 private double LineMinimization(double[] startPoint, double[] direction, ref bool cancel)
203 {
204 // Line-minimization routine, Given an n-dimensional point p[0..n-1] and an n-dimension
205 // direction xi[0..n-1], moves and resets p to where the function of functor func(p) takes on
206 // a minimum along the direction xi from p, and replaces xi by the actual vector displacement
207 // that p was moved. Also returns the value of func at the return location p. This is actually
208 // all accomplished by calling the Brent minimize routine.
209 int D = NumberOfParameters;
210 bool c = cancel;
211 double func(double alpha)
212 {
213 var x = new double[D];
214 for (int i = 0; i < D; i++)
215 x[i] = startPoint[i] + alpha * direction[i];
216 return Evaluate(x, ref c);
217 }
218 var brent = new BrentSearch(func, 0d, 1d) { RelativeTolerance = RelativeTolerance, AbsoluteTolerance = AbsoluteTolerance };
219 brent.Bracket(0.1);
220 brent.Minimize();
221 cancel = c;
222 if (cancel) return double.NaN;
223 double xmin = brent.BestParameterSet.Values[0];
224 for (int j = 0; j < NumberOfParameters; j++)
225 {
226 direction[j] *= xmin;
227 startPoint[j] += direction[j];
228 // Make sure the parameter is within bounds
229 startPoint[j] = RepairParameter(startPoint[j], LowerBounds[j], UpperBounds[j]);
230 }
231 return brent.BestParameterSet.Fitness;
232 }
