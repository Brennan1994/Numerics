Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

262 public override double PDF(double x)
263 {
264 // Validate parameters
265 if (_parametersValid == false)
266 ValidateParameters([Alpha, Beta], true);
267 //
268 if (x < Minimum || x > Maximum) return 0.0d;
269 double constant = 1.0d / Mathematics.SpecialFunctions.Beta.Function(Alpha, Beta);
270 double a = Math.Pow(x, Alpha - 1d);
271 double b = Math.Pow(1d - x, Beta - 1d);
272 return constant * a * b;
273 }
274
276 public override double CDF(double x)
277 {
278 // Validate parameters
279 if (_parametersValid == false)
280 ValidateParameters([Alpha, Beta], true);
281 //
282 if (x <= Minimum)
283 return 0d;
284 if (x >= Maximum)
285 return 1d;
286 return Mathematics.SpecialFunctions.Beta.Incomplete(Alpha, Beta, x);
287 }

290 public override double InverseCDF(double probability)
291 {
292 // Validate probability
293 if (probability < 0.0d || probability > 1.0d)
294 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
295 if (probability == 0.0d)
296 return Minimum;
297 if (probability == 1.0d)
298 return Maximum;
299 // Validate parameters
300 if (_parametersValid == false)
301 ValidateParameters([Alpha, Beta], true);
302 //
303 return Mathematics.SpecialFunctions.Beta.IncompleteInverse(Alpha, Beta, probability);
304 }
