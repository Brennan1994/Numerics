260 public void Estimate(IList<double> sample, ParameterEstimationMethod estimationMethod)
261 {
262 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
263 {
264 var min = Tools.Min(sample);
265 var max = Tools.Max(sample);
266 var mean = Tools.Mean(sample);
267 var mode = mean * 3 - max - min;
268 mode = Math.Max(min, mode);
269 mode = Math.Min(max, mode);
270 SetParameters(min, mode, max);
271 }
272 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
273 {
274 SetParameters(MLE(sample));
275 }
276 else
277 {
278 throw new NotImplementedException();
279 }
280 }

347 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
348 {
349 var initialVals = new double[NumberOfParameters];
350 var lowerVals = new double[NumberOfParameters];
351 var upperVals = new double[NumberOfParameters];
352 // Estimate initial values using the method of moments (a.k.a product moments).
353 var min = Tools.Min(sample);
354 var max = Tools.Max(sample);
355 var mean = Tools.Mean(sample);
356 var mode = mean * 3 - max - min;
357 initialVals[0] = min - Tools.DoubleMachineEpsilon;
358 initialVals[1] = mode;
359 initialVals[2] = max + Tools.DoubleMachineEpsilon;
360 // Get bounds of min
361 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(min)) + 1d));
362 upperVals[0] = min;
363 // Get bounds of mode
364 lowerVals[1] = min;
365 upperVals[1] = max;
366 // Get bounds of max
367 lowerVals[2] = max;
368 upperVals[2] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(max)) + 1d));
369
370 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
371 }

374 public double[] MLE(IList<double> sample)
375 {
376 // Set constraints
377 var tuple = GetParameterConstraints(sample);
378 var Initials = tuple.Item1;
379 var Lowers = tuple.Item2;
380 var Uppers = tuple.Item3;
381
382 // Solve using Nelder-Mead (Downhill Simplex)
383 double logLH(double[] x)
384 {
385 var N = new Triangular();
386 N.SetParameters(x);
387 return N.LogLikelihood(sample);
388 }
389 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
390 solver.ReportFailure = true;
391 solver.Maximize();
392 return solver.BestParameterSet.Values;
393 }

396 public override double PDF(double x)
397 {
398 // Validate parameters
399 if (_parametersValid == false)
400 ValidateParameters(Min, MostLikely, Max, true);
401 //
402 if (Min == Max && Min == MostLikely) return 0.0d;
403 if (x < Minimum || x > Maximum) return 0.0d;
404 if (x >= Min && x < MostLikely)
405 {
406 return 2.0d * (x - Min) / ((Max - Min) * (MostLikely - Min));
407 }
408 else if (x > MostLikely && x <= Max)
409 {
410 return 2.0d * (Max - x) / ((Max - Min) * (Max - MostLikely));
411 }
412 else if (x == MostLikely)
413 {
414 return 2.0d / (Max - Min);
415 }
416 return double.NaN;
417 }

420 public override double CDF(double x)
421 {
422 // Validate parameters
423 if (_parametersValid == false)
424 ValidateParameters(Min, MostLikely, Max, true);
425 if (Min == Max && Min == MostLikely) return 1d;
426 if (x <= Minimum) return 0d;
427 if (x >= Maximum) return 1d;
428 if (x > Min && x <= MostLikely)
429 {
430 return (x - Min) * (x - Min) / ((Max - Min) * (MostLikely - Min));
431 }
432 else if (x > MostLikely && x < Max)
433 {
434 return 1d - (Max - x) * (Max - x) / ((Max - Min) * (Max - MostLikely));
435 }
436 return double.NaN;
437 }

440 public override double InverseCDF(double probability)
441 {
442 // Validate probability
443 if (probability < 0.0d || probability > 1.0d)
444 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
445 if (Min == Max && Min == MostLikely) return Min;
446 if (probability == 0.0d) return Minimum;
447 if (probability == 1.0d) return Maximum;
448 // Validate parameters
449 if (_parametersValid == false)
450 ValidateParameters(Min, MostLikely, Max, true);
451 if (probability < (MostLikely - Min) / (Max - Min))
452 {
453 return Min + Math.Sqrt(probability * (Max - Min) * (MostLikely - Min));
454 }
455 else if (probability >= (MostLikely - Min) / (Max - Min))
456 {
457 return Max - Math.Sqrt((1.0d - probability) * (Max - Min) * (Max - MostLikely));
458 }
459 else if (Max - Min == 0d)
460 {
461 return MostLikely;
462 }
463 return double.NaN;
464 }
