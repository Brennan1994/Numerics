Numerics 1.0.0
Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

263 private void InvokeProgressHandlers(object state)
264 {
265 double prog = ((double[])state)[0];
266 double prevProg = ((double[])state)[1];
267 if (prevProg < 0d)
268 prevProg = 0d;
269 OnProgressReported(prog);
270 ProgressReported?.Invoke(this, prog, prog - prevProg);
271 }
272
277 private void InvokeMessageHandlers(object state)
278 {
279 MessageContentStruct prog = (MessageContentStruct)state;
280 OnMessageReported(prog);
281 MessageReported?.Invoke(prog);
282 }

297 public SafeProgressReporter CreateProgressModifier(float fractionOfTotal, string subTaskName = "")
298 {
299 if (fractionOfTotal > 1f)
300 fractionOfTotal = 1f;
301 if (fractionOfTotal < 0f)
302 fractionOfTotal = 0f;
303 if (string.IsNullOrEmpty(subTaskName))
304 subTaskName = TaskName;
305 var child = new SafeProgressReporter(subTaskName);
306 child.SetContext(_synchronizationContext);
307 child._previousProgress = 0d;
308 child.ProgressReported += (reporter, prog, progDelta) => ReportProgress(_previousProgress + progDelta * fractionOfTotal);
309 child.MessageReported += msg => ReportMessage(msg);
310 child._cancellationTokenSource = _cancellationTokenSource;
311 _subProgReporterCollection.Add(child);
312 var invokeChildCreatedHandlers = new SendOrPostCallback(state => ChildReporterCreated?.Invoke(child));
313 _synchronizationContext.Post(invokeChildCreatedHandlers, child);
314 return child;
315 }

333 public void ReportProgress(double progress)
334 {
335 if (_previousProgress == progress) return;
336 _synchronizationContext.Post(_invokeProgressHandlers, new double[] { progress, _previousProgress });
337 _previousProgress = progress;
338 }

345 public void ReportMessage(string message, MessageType messageType = MessageType.Status)
346 {
347 _synchronizationContext.Post(_invokeMessageHandlers, new MessageContentStruct(message, messageType, this));
348 MessageCount += 1;
349 _previousMessage = message;
350 _previousMessageType = messageType;
351 }

357 protected void ReportMessage(MessageContentStruct message)
