65 public static double OneSampleTtest(IList<double> sample, double populationMean = 0d)
66 {
67 var meanVar = Statistics.MeanVariance(sample);
68 int N = sample.Count;
69 double se = Math.Sqrt(meanVar.Item2) / Math.Sqrt(N);
70 double t = Math.Abs((meanVar.Item1 - populationMean) / se);
71 var tdist = new StudentT(N - 1);
72 return (1d - tdist.CDF(t)) * 2d;
73 }

82 public static double EqualVarianceTtest(IList<double> sample1, IList<double> sample2)
83 {
84 var meanVar1 = Statistics.MeanVariance(sample1);
85 var meanVar2 = Statistics.MeanVariance(sample2);
86 int N1 = sample1.Count;
87 int N2 = sample2.Count;
88 double df = N1 + N2 - 2;
89 double svar = ((N1 - 1) * meanVar1.Item2 + (N2 - 1) * meanVar2.Item2) / df;
90 double t = Math.Abs((meanVar1.Item1 - meanVar2.Item1) / Math.Sqrt(svar * (1.0d / N1 + 1.0d / N2)));
91 var tdist = new StudentT(df);
92 return (1d - tdist.CDF(t)) * 2d;
93 }

102 public static double UnequalVarianceTtest(IList<double> sample1, IList<double> sample2)
103 {
104 var meanVar1 = Statistics.MeanVariance(sample1);
105 var meanVar2 = Statistics.MeanVariance(sample2);
106 double ave1 = meanVar1.Item1, var1 = meanVar1.Item2, ave2 = meanVar2.Item1, var2 = meanVar2.Item2;
107 int n1 = sample1.Count;
108 int n2 = sample2.Count;
109 double t = Math.Abs((ave1 - ave2) / Math.Sqrt(var1 / n1 + var2 / n2));
110 double df = Tools.Sqr(var1 / n1 + var2 / n2) / (Tools.Sqr(var1 / n1) / (n1 - 1) + Tools.Sqr(var2 / n2) / (n2 - 1));
111 var tdist = new StudentT((int)df);
112 return (1d - tdist.CDF(t)) * 2d;
113 }

122 public static double PairedTtest(IList<double> sample1, IList<double> sample2)
123 {
124 if (sample1.Count != sample2.Count) throw new ArgumentException("The two data samples must be the sample length.");
125
126 var meanVar1 = Statistics.MeanVariance(sample1);
127 var meanVar2 = Statistics.MeanVariance(sample2);
128 int n = sample1.Count;
129 double ave1 = meanVar1.Item1, var1 = meanVar1.Item2, ave2 = meanVar2.Item1, var2 = meanVar2.Item2, sd, df, cov = 0.0;
130 for (int j = 0; j < n; j++) cov += (sample1[j] - ave1) * (sample2[j] - ave2);
131 cov /= (df = n - 1);
132 sd = Math.Sqrt((var1 + var2 - 2.0 * cov) / n);
133 double t = Math.Abs((ave1 - ave2) / sd);
134 var tdist = new StudentT((int)df);
135 return (1d - tdist.CDF(t)) * 2d;
136 }

144 public static double Ftest(IList<double> sample1, IList<double> sample2)
145 {
146 var meanVar1 = Statistics.MeanVariance(sample1);
147 var meanVar2 = Statistics.MeanVariance(sample2);
148 int n1 = sample1.Count, n2 = sample2.Count;
149 double ave1 = meanVar1.Item1, var1 = meanVar1.Item2, ave2 = meanVar2.Item1, var2 = meanVar2.Item2, df1, df2, f, pVal;
150 if (var1 > var2)
151 {
152 f = var1 / var2;
153 df1 = n1 - 1;
154 df2 = n2 - 1;
155 }
156 else
157 {
158 f = var2 / var1;
159 df1 = n2 - 1;
160 df2 = n1 - 1;
161 }
162 pVal = 2.0 * Beta.Incomplete(0.5 * df2, 0.5 * df1, df2 / (df2 + df1 * f));
163 if (pVal > 1.0) pVal = 2d - pVal;
164 return pVal;
165 }

190 public static double JarqueBeraTest(IList<double> sample)
191 {
192 var moments = Statistics.ProductMoments(sample);
193 double S2 = moments[2] * moments[2];
194 double K2 = moments[3] * moments[3];
195 int N = sample.Count;
196 double JB = N / 6.0d * (S2 + K2 / 4.0d);
197 var chi = new ChiSquared(2);
198 return 1d - chi.CDF(JB);

206 public static double WaldWolfowitzTest(IList<double> sample)
207 {
208 double xn = sample.Last(), x1 = sample.First();
209 double R = 0, eR, varR, U, S1 = 0, S2 = 0, S3 = 0, S4 = 0;
210 int n = sample.Count;
211 for (int i = 0; i < n; i++)
212 {
213
214 S1 += sample[i];
215 S2 += Tools.Sqr(sample[i]);
216 S3 += Tools.Pow(sample[i], 3);
217 S4 += Tools.Pow(sample[i], 4);
218
219 if (i < n - 1)
220 {
221 R += sample[i] * sample[i + 1];
222 }
223 }
224 R += xn * x1;
225 double s12 = Tools.Sqr(S1);
226 double s22 = Tools.Sqr(S2);
227 double s14 = Tools.Pow(S1,4);
228
229 eR = (s12 - S2) / (n - 1);
230 varR = (s22 - S4) / (n - 1) - Tools.Sqr(eR);
231 varR += (s14 - 4 * s12 * S2 + 4 * S1 * S3 + s22 - 2 * S4) / ((n - 1) * (n - 2));
232 U = Math.Abs((R - eR) / Math.Sqrt(varR));
233 return (1 - Normal.StandardCDF(U)) * 2d;
234 }

242 public static double LjungBoxTest(IList<double> sample, int lagMax = -1)
243 {
244 int n = sample.Count;
245 if (lagMax < 0) lagMax = (int)Math.Floor(Math.Min(10d * Math.Log10(n), n - 1));
246 var acf = Autocorrelation.Function(sample, lagMax, Autocorrelation.Type.Correlation);
247 double Q = 0;
248 for (int k = 1; k <= lagMax; k++)
249 Q += Tools.Sqr(acf[k, 1]) / (n - k);
250 Q *= n * (n + 2);
251 var chi2 = new ChiSquared(lagMax);
252 return 1d - chi2.CDF(Q);
253 }

261 public static double MannWhitneyTest(IList<double> sample1, IList<double> sample2)
262 {
263 int n1 = sample1.Count, n2 = sample2.Count, n = n1 + n2;
264 if (n1 > n2) throw new ArgumentException("The first sample must have a length less than or equal to sample 2.");
265 if (n1 <= 3 || n2 <= 3) throw new ArgumentException("Each sample must have a length greater than 3.");
266 if (n <= 20) throw new ArgumentException("The combined sample 1 & 2 must have a length greater than 20.");
267
268 var sample = new List<double>();
269 sample.AddRange(sample1.ToList());
270 sample.AddRange(sample2.ToList());
271 var ties = new double[n1]; double R = 0, T = 0;
272
273 var ranks = Statistics.RanksInPlace(sample.ToArray(), out ties);
274
275 for (int i = 0; i < sample1.Count; i++) R += ranks[i];
276 for (int i = 0; i < ties.Length; i++) T += (Tools.Pow(ties[i], 3) - ties[i]) / (n * (n - 1));
277
278 double V = R - n1 * (n1 + 1d) / 2d;
279 double W = n1 * n2 - V;
280 double U = Math.Min(V, W);
281 double eU = n1 * n2 / 2;
282 double varU = n1 * n2 / 12d * ((n + 1) - T);
283 double z = Math.Abs((U - eU) / Math.Sqrt(varU));
284 return (1 - Normal.StandardCDF(z)) * 2d;
285 }

292 public static double MannKendallTest(IList<double> sample)
293 {
294 int i, j, n = sample.Count;
295 if (n < 10) throw new ArgumentException("The sample size must be greater than or equal to 10.");
296
297 double S = 0, T = 0, varS, z;
298
299 for (i = 0; i < n - 1; i++)
300 {
301 for (j = i + 1; j < n; j++)
302 {
303 S += Math.Sign(sample[j] - sample[i]);
304 }
305 }
306
307 var ties = new double[n];
308 var R = Statistics.RanksInPlace(sample.ToArray(), out ties);
309 for (i = 0; i < ties.Length; i++) T += ties[i] * (ties[i] - 1) * (2 * ties[i] + 5);
310 varS = (n * (n - 1) * (2 * n + 5) - T) / 18;
311 z = Math.Abs((S - Math.Sign(S)) / Math.Sqrt(varS));
312
313 return (1 - Normal.StandardCDF(z)) * 2d;
314 }

322 public static double LinearTrendTest(IList<double> indices, IList<double> sample)
323 {
324 if (indices.Count != sample.Count)
325 throw new ArgumentException("The indices must be the same length as the sample data");
326 var xVals = new Matrix(indices.ToArray());
327 var yVals = new Vector(sample.ToArray());
328 var lm = new LinearRegression(xVals, yVals, true);
329 var tdist = new StudentT(lm.DegreesOfFreedom);
330 double d = Math.Abs(lm.Parameters[1] / lm.ParameterStandardErrors[1]);
331 return (1 - tdist.CDF(Math.Abs(lm.Parameters[1] / lm.ParameterStandardErrors[1]))) * 2;
332 }
