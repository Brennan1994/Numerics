456 public double[] ParametersFromLinearMoments(IList<double> moments)
457 {
458 double L1 = moments[0];
459 double L2 = moments[1];
460 double T3 = moments[2];
461 double alpha = double.NaN;
462 double z;
463 // The following approximation has relative accuracy better than 5x10-5 for all values of alpha.
464 if (Math.Abs(T3) > 0.0d && Math.Abs(T3) < 1d / 3d)
465 {
466 z = 3.0d * Math.PI * Math.Pow(T3, 2d);
467 alpha = (1.0d + 0.2906d * z) / (z + 0.1882d * Math.Pow(z, 2d) + 0.0442d * Math.Pow(z, 3d));
468 }
469 else if (Math.Abs(T3) >= 1d / 3d && Math.Abs(T3) < 1.0d)
470 {
471 z = 1.0d - Math.Abs(T3);
472 alpha = (0.36067d * z - 0.59567d * Math.Pow(z, 2d) + 0.25361d * Math.Pow(z, 3d)) / (1.0d - 2.78861d * z + 2.56096d * Math.Pow(z, 2d) - 0.77045d * Math.Pow(z, 3d));
473 }
474
475 double gamma = 2.0d * Math.Pow(alpha, -0.5d) * Math.Sign(T3);
476 double sigma = L2 * Math.Pow(Math.PI, 0.5d) * Math.Pow(alpha, 0.5d) * Mathematics.SpecialFunctions.Gamma.Function(alpha) / Mathematics.SpecialFunctions.Gamma.Function(alpha + 0.5d);
477 double mu = L1;
478 return [mu, sigma, gamma];
479 }

482 public double[] LinearMomentsFromParameters(IList<double> parameters)
483 {
484 double mu = parameters[0];
485 double sigma = parameters[1];
486 double gamma = parameters[2];
487 double xi = mu - 2.0d * sigma / gamma;
488 double alpha = 4.0d / Math.Pow(gamma, 2d);
489 double beta = 0.5d * sigma * gamma;
490 double L1 = xi + alpha * beta;
491 double L2 = Math.Abs(Math.Pow(Math.PI, -0.5d) * beta * Mathematics.SpecialFunctions.Gamma.Function(alpha + 0.5d) / Mathematics.SpecialFunctions.Gamma.Function(alpha));
492 // The following approximations are accurate to 10-6.
493 double A0 = 0.32573501d;
494 double A1 = 0.1686915d;
495 double A2 = 0.078327243d;
496 double A3 = -0.0029120539d;
497 double B1 = 0.46697102d;
498 double B2 = 0.24255406d;
499 double C0 = 0.12260172d;
500 double C1 = 0.05373013d;
501 double C2 = 0.043384378d;
502 double C3 = 0.011101277d;
503 double D1 = 0.18324466d;
504 double D2 = 0.20166036d;
505 double E1 = 2.3807576d;
506 double E2 = 1.5931792d;
507 double E3 = 0.11618371d;
508 double F1 = 5.1533299d;
509 double F2 = 7.142526d;
510 double F3 = 1.9745056d;
511 double G1 = 2.1235833d;
512 double G2 = 4.1670213d;
513 double G3 = 3.1925299d;
514 double H1 = 9.0551443d;
515 double H2 = 26.649995d;
516 double H3 = 26.193668d;
517 double T3;
518 double T4;
519 if (alpha >= 1d)
520 {
521 T3 = Math.Pow(alpha, -0.5d) * (A0 + A1 * Math.Pow(alpha, -1) + A2 * Math.Pow(alpha, -2) + A3 * Math.Pow(alpha, -3)) / (1d + B1 * Math.Pow(alpha, -1) + B2 * Math.Pow(alpha, -2));
522 T4 = (C0 + C1 * Math.Pow(alpha, -1) + C2 * Math.Pow(alpha, -2) + C3 * Math.Pow(alpha, -3)) / (1d + D1 * Math.Pow(alpha, -1) + D2 * Math.Pow(alpha, -2));
523 }
524 else
525 {
526 T3 = (1d + E1 * alpha + E2 * Math.Pow(alpha, 2d) + E3 * Math.Pow(alpha, 3d)) / (1d + F1 * alpha + F2 * Math.Pow(alpha, 2d) + F3 * Math.Pow(alpha, 3d));
527 T4 = (1d + G1 * alpha + G2 * Math.Pow(alpha, 2d) + G3 * Math.Pow(alpha, 3d)) / (1d + H1 * alpha + H2 * Math.Pow(alpha, 2d) + H3 * Math.Pow(alpha, 3d));
528 }
529
530 return [L1, L2, T3, T4];
531 }

534 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
535 {
536 var initialVals = new double[NumberOfParameters];
537 var lowerVals = new double[NumberOfParameters];
538 var upperVals = new double[NumberOfParameters];
539 // Get initial values
540 var moments = Statistics.ProductMoments(sample);
541 initialVals = moments.Subset(0, 2);
542 // Get bounds of mean
543 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[0]) + 1d));
544 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[0]) + 1d));
545 // Get bounds of standard deviation
546 lowerVals[1] = Tools.DoubleMachineEpsilon;
547 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[1]) + 1d));
548 // Get bounds of skew
549 lowerVals[2] = -2d;
550 upperVals[2] = 2d;
551
552 // Correct initial value of skew if necessary
553 if (initialVals[2] <= lowerVals[2] || initialVals[2] >= upperVals[2])
554 {
555 initialVals[2] = 0d;
556 }
557 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
558 }

561 public double[] MLE(IList<double> sample)
562 {
563 // Set constraints
564 var tuple = GetParameterConstraints(sample);
565 var Initials = tuple.Item1;
566 var Lowers = tuple.Item2;
567 var Uppers = tuple.Item3;
568
569 // Solve using Nelder-Mead (Downhill Simplex)
570 double logLH(double[] x)
571 {
572 var P3 = new PearsonTypeIII();
573 P3.SetParameters(x);
574 return P3.LogLikelihood(sample);
575 }
576 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
577 solver.ReportFailure = true;
578 solver.Maximize();
579 return solver.BestParameterSet.Values;
580
581 }

584 public override double PDF(double x)
585 {
586 // Validate parameters
587 if (_parametersValid == false)
588 ValidateParameters(Mu, Sigma, Gamma, true);
589 if (x < Minimum || x > Maximum) return 0.0d;
590
591 if (Math.Abs(Gamma) <= NearZero)
592 {
593 // Use Normal distribution
594 double z = (x - Mu) / Sigma;
595 return Math.Exp(-0.5d * z * z) / (Tools.Sqrt2PI * Sigma);
596 }
597 // Use Gamma distribution
598 if (Beta > 0d)
599 {
600 double shiftedX = x - Xi;
601 return Math.Exp(-shiftedX / Math.Abs(Beta) + (Alpha - 1.0d) * Math.Log(shiftedX) - Alpha * Math.Log(Math.Abs(Beta)) - Mathematics.SpecialFunctions.Gamma.LogGamma(Alpha));
602 }
603 else
604 {
605 double shiftedX = Xi - x;
606 return Math.Exp(-shiftedX / Math.Abs(Beta) + (Alpha - 1.0d) * Math.Log(shiftedX) - Alpha * Math.Log(Math.Abs(Beta)) - Mathematics.SpecialFunctions.Gamma.LogGamma(Alpha));
607 }
608 }

611 public override double CDF(double x)
612 {
613 // Validate parameters
614 if (_parametersValid == false)
615 ValidateParameters(Mu, Sigma, Gamma, true);
616 if (x <= Minimum)
617 return 0d;
618 if (x >= Maximum)
619 return 1d;
620 if (Math.Abs(Gamma) <= NearZero)
621 {
622 return 0.5d * (1.0d + Mathematics.SpecialFunctions.Erf.Function((x - Mu) / (Sigma * Math.Sqrt(2.0d))));
623 }
624 else if (Beta > 0d)
625 {
626 double shiftedX = x - Xi;
627 return Mathematics.SpecialFunctions.Gamma.LowerIncomplete(Alpha, shiftedX / Math.Abs(Beta));
628 }
629 else
630 {
631 double shiftedX = Xi - x;
632 return 1.0d - Mathematics.SpecialFunctions.Gamma.LowerIncomplete(Alpha, shiftedX / Math.Abs(Beta));
633 }
634 }

637 public override double InverseCDF(double probability)
638 {
639 // Validate probability
640 if (probability < 0.0d || probability > 1.0d)
641 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
642 if (probability == 0.0d)
643 return Minimum;
644 if (probability == 1.0d)
645 return Maximum;
646 // Validate parameters
647 if (_parametersValid == false)
648 ValidateParameters(Mu, Sigma, Gamma, true);
649 if (Math.Abs(Gamma) <= NearZero)
650 {
651 return Mu + Sigma * Normal.StandardZ(probability);
652 }
653 else if (Beta > 0d)
654 {
655 return Xi + Mathematics.SpecialFunctions.Gamma.InverseLowerIncomplete(Alpha, probability) * Math.Abs(Beta);
656 }
657 else
658 {
659 return Xi - Mathematics.SpecialFunctions.Gamma.InverseLowerIncomplete(Alpha, 1d - probability) * Math.Abs(Beta);
660 }

693 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
694 {
695 if (estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
696 {
697 throw new NotImplementedException();
698 }
699 // Validate parameters
700 if (_parametersValid == false)
701 ValidateParameters(_mu, _sigma, _gamma, true);
702 // Compute covariance
703 double alpha = 1d / Beta;
704 double lambda = Alpha;
705 double A = 2d * Mathematics.SpecialFunctions.Gamma.Trigamma(lambda) - 2d / (lambda - 1d) + 1d / Math.Pow(lambda - 1d, 2d);
706 var covar = new double[3, 3];
707 covar[0, 0] = (lambda - 2d) / (sampleSize * A) * (1d / Math.Pow(alpha, 2d)) * (Mathematics.SpecialFunctions.Gamma.Trigamma(lambda) * lambda - 1d); // location
708 covar[1, 1] = (lambda - 2d) * Math.Pow(alpha, 2d) / (sampleSize * A) * (Mathematics.SpecialFunctions.Gamma.Trigamma(lambda) / (lambda - 2d) - 1d / Math.Pow(lambda - 1d, 2d)); // scale
709 covar[2, 2] = 2d / (sampleSize * A); // shape
710 covar[0, 1] = 1d / sampleSize * ((lambda - 2d) / A) * (Mathematics.SpecialFunctions.Gamma.Trigamma(lambda) - 1d / (lambda - 1d)); // location & scale
711 covar[1, 0] = covar[0, 1];
712 covar[0, 2] = (2d - lambda) / (sampleSize * alpha * A * (lambda - 1d)); // location & shape
713 covar[2, 0] = covar[0, 2];
714 covar[1, 2] = alpha / (sampleSize * A * (lambda - 1d)); // scale & shape
715 covar[2, 1] = covar[1, 2];
716 return covar;
717 }

720 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
721 {
722 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
723 {
724 double u2 = _sigma;
725 double c = _gamma;
726 double c2 = Math.Pow(c, 2d);
727 int N = sampleSize;
728 return Math.Pow(u2, 2d) / N * (1d + Math.Pow(GammaDistribution.FrequencyFactorKp(c, probability), 2d) / 2d * (1d + 0.75d * c2) + GammaDistribution.FrequencyFactorKp(c, probability) * c + 6d * (1d + 0.25d * c2) * GammaDistribution.PartialKp(c, probability) * (GammaDistribution.PartialKp(c, probability) * (1d + 5d * c2 / 4d) + GammaDistribution.FrequencyFactorKp(c, probability) / 2d * c));
729 }
730 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
731 {
732 var covar = ParameterCovariance(sampleSize, estimationMethod);
733 var grad = QuantileGradient(probability);
734 double varA = covar[0, 0];
735 double varB = covar[1, 1];
736 double varG = covar[2, 2];
737 double covAB = covar[1, 0];
738 double covAG = covar[2, 0];
739 double covBG = covar[2, 1];
740 double dQx1 = grad[0];
741 double dQx2 = grad[1];
742 double dQx3 = grad[2];
743 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + Math.Pow(dQx3, 2d) * varG + 2d * dQx1 * dQx2 * covAB + 2d * dQx1 * dQx3 * covAG + 2d * dQx2 * dQx3 * covBG;
744 }
745
746 return double.NaN;
747 }

750 public double[] QuantileGradient(double probability)
751 {
752 // Validate parameters
753 if (_parametersValid == false)
754 ValidateParameters(Mu, Sigma, Gamma, true);
755 double alpha = 1d / Beta;
756 double lambda = Alpha;
757 double eps = Math.Sign(alpha);
758 var gradient = new double[]
759 {
760 1.0d, // location
761 -lambda / Math.Pow(alpha, 2d) * (1.0d + eps / Math.Sqrt(lambda) * GammaDistribution.FrequencyFactorKp(Gamma, probability)), // scale
762 1.0d / alpha * (1.0d + eps / Math.Sqrt(lambda) * (GammaDistribution.FrequencyFactorKp(Gamma, probability) / 2.0d) - 1.0d / lambda * GammaDistribution.PartialKp(Gamma, probability)) // shape
763 };
764 return gradient;
765 }

786 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
787 {
788 if (probabilities.Count != NumberOfParameters)
789 {
790 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
791 }
792 // Get gradients
793 var dQp1 = QuantileGradientForMoments(probabilities[0]);
794 var dQp2 = QuantileGradientForMoments(probabilities[1]);
795 var dQp3 = QuantileGradientForMoments(probabilities[2]);
796 // Compute determinant
797 // |a b c|
798 // |d e f|
799 // |g h i|
800 // |A| = a(ei − fh) − b(di − fg) + c(dh − eg)
801 double a = dQp1[0];
802 double b = dQp1[1];
803 double c = dQp1[2];
804 double d = dQp2[0];
805 double e = dQp2[1];
806 double f = dQp2[2];
807 double g = dQp3[0];
808 double h = dQp3[1];
809 double i = dQp3[2];
810 determinant = a * (e * i - f * h) - b * (d * i - f * g) + c * (d * h - e * g);
811 // Return Jacobian
812 var jacobian = new double[,] { { a, b, c }, { d, e, f }, { g, h, i } };
