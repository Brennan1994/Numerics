61 public static List<StratificationBin> XValues(StratificationOptions options, bool isLogarithmic = false)
62 {
63 var bins = new List<StratificationBin>();
64 if (options.IsValid == false)
65 return bins;
66 if (options.IsProbability == true)
67 return bins;
68 double delta, offset, min, max, xl, xu;
69 //
70 if (isLogarithmic == true)
71 {
72 // first determine the offset
73 if (options.LowerBound <= 0d)
74 {
75 offset = 1.0d - options.LowerBound;
76 }
77 else
78 {
79 offset = 0.0d;
80 }
81 // get delta
82 min = Math.Log10(options.LowerBound + offset);
83 max = Math.Log10(options.UpperBound + offset);
84 delta = (max - min) / options.NumberOfBins;
85 // stratify first bin
86 xl = Math.Pow(10d, min) - offset;
87 xu = Math.Pow(10d, Math.Log10(xl + offset) + delta) - offset;
88 bins.Add(new StratificationBin(xl, xu));
89 // stratify remaining bins
90 for (int i = 1; i < options.NumberOfBins; i++)
91 {
92 xl = xu;
93 xu = Math.Pow(10d, Math.Log10(xl + offset) + delta) - offset;
94 bins.Add(new StratificationBin(xl, xu));
95 }
96 }
97 else
98 {
99 // get delta
100 delta = (options.UpperBound - options.LowerBound) / options.NumberOfBins;
101 // stratify first bin
102 xl = options.LowerBound;
103 xu = xl + delta;
104 bins.Add(new StratificationBin(xl, xu));
105 // stratify remaining bins
106 for (int i = 1; i < options.NumberOfBins; i++)
107 {
108 xl = xu;
109 xu = xl + delta;
110 bins.Add(new StratificationBin(xl, xu));
111 }
112 }
113 //
114 return bins;
115 }

140 public static List<StratificationBin> XToExceedanceProbability(List<StratificationBin> xValues, Func<double, double> transformFunction, bool isExhaustive = true)
141 {
142 var bins = new List<StratificationBin>();
143 if (xValues.Count == 0)
144 return bins;
145 double xl, xu;
146 xu = transformFunction(xValues.First().LowerBound);
147 xl = transformFunction(xValues.First().UpperBound);
148 bins.Add(new StratificationBin(xl, xu));
149 // transform the bins
150 for (int i = 1; i < xValues.Count; i++)
151 {
152 xu = xl;
153 xl = transformFunction(xValues[i].UpperBound);
154 bins.Add(new StratificationBin(xl, xu));
155 }
156
157 if (isExhaustive == true)
158 {
159 // adjust end bins to ensure exhaustivity
160 bins.First().Weight = 1.0d - bins.First().LowerBound;
161 bins.Last().Weight = bins.Last().UpperBound;
162 }
163
164 return bins;
165 }

175 public static List<StratificationBin> XToProbability(List<StratificationBin> xValues, Func<double, double> transformFunction, bool isExhaustive = true)
176 {
177 var bins = new List<StratificationBin>();
178 double xl, xu, w;
179 // transform first bin
180 xl = transformFunction(xValues.First().LowerBound);
181 xu = transformFunction(xValues.First().UpperBound);
182 // determine if the first bin weight should ensure exhaustivity
183 if (isExhaustive == true)
184 {
185 // The weight for the first interval is evaluated
186 // as the non-exceedance probability of its upper bound.
187 w = xu;
188 bins.Add(new StratificationBin(xl, xu, w));
189 }
190 else
191 {
192 bins.Add(new StratificationBin(xl, xu));
193 }
194 // transform the remaining inner bins
195 for (int i = 1; i < xValues.Count - 1; i++)
196 {
197 xl = xu;
198 xu = transformFunction(xValues[i].UpperBound);
199 bins.Add(new StratificationBin(xl, xu));
200 }
201 // transform the last bin
202 xl = xu;
203 xu = transformFunction(xValues.Last().UpperBound);
204 // determine if the last bin weight should ensure exhaustivity
205 if (isExhaustive == true)
206 {
207 // The weight for the last interval is evaluated
208 // as the exceedance probability of its lower bound.
209 w = 1.0d - xl;
210 bins.Add(new StratificationBin(xl, xu, w));
211 }
212 else
213 {
214 bins.Add(new StratificationBin(xl, xu));
215 }
216 //
217 return bins;

384 public static List<List<StratificationBin>> MultivariateProbabilities(StratificationOptions options, ImportanceDistribution distributionType = ImportanceDistribution.Uniform, bool isExhaustive = true, int dimension = 1, int seed = -1, double[,] correlation = null)
385 {
386 var bins = new List<List<StratificationBin>>();
387 var output = new List<List<StratificationBin>>();
388
389 // Create random number generators
390 var rnd = seed > 0 ? new MersenneTwister(seed) : new MersenneTwister();
391 var mu = new double[dimension];
392 for (int i = 0; i < dimension; i++)
393 {
394 mu[i] = 0;
395 bins.Add(Probabilities(options, distributionType, isExhaustive));
396 output.Add(new List<StratificationBin>());
397 }
398
399 var _mvn = new MultivariateNormal(dimension);
400 if (correlation == null)
401 {
402 correlation = new double[dimension, dimension];
403 for (int i = 0; i < dimension; i++)
404 for (int j = 0; j < dimension; j++)
405 correlation[i, j] = i == j ? 1d : 0d;
406 }
407 _mvn.SetParameters(mu, correlation);
408 var rvals = _mvn.LatinHypercubeRandomValues(options.NumberOfBins, seed);
409
410 // Sample random bin without replacement
411 for (int row = 0; row < options.NumberOfBins; row++)
412 {
413 for (int col = 0; col < dimension; col++)
414 {
415 int r = (int)Math.Floor(Normal.StandardCDF(rvals[row, col]) * bins[col].Count);
416 output[col].Add((StratificationBin)bins[col][r].Clone());
417 bins[col].RemoveAt(r);
418 }
419 }
420
421 return output;
422 }
