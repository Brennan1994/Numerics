505 public void SetParameters(ref double[] parameters)
506 {
507 if (parameters == null) return;
508 if (Weights == null || Weights.Length == 0) return;
509 if (Distributions == null || Distributions.Count() == 0) return;
510 if (Distributions.Count() == 1 && parameters.Length == Distributions[0].NumberOfParameters)
511 {
512 if (IsZeroInflated)
513 {
514 Weights[0] = 1 - ZeroWeight;
515 }
516 else
517 {
518 Weights[0] = 1;
519 }
520 Distributions[0].SetParameters(parameters);
521 }
522 else
523 {
524
525 // Get the weights
526 int K = Distributions.Count();
527 int t = 0; // keep track of parameter index
528
529 // Get weights
530 double sum = 0.0;
531 for (int i = 0; i < K; i++)
532 {
533 Weights[i] = parameters[i];
534 sum += Weights[i];
535 t++;
536 }
537
538 // Check if weights need to be normalized
539 if (sum <= 0.0)
540 {
541 // If weights sum to 0, reset to be uniformly distributed
542 double w = IsZeroInflated ? (1d - ZeroWeight) / K : 1d / K;
543 for (int i = 0; i < K; i++)
544 {
545 Weights[i] = w;
546 parameters[i] = Weights[i];
547 }
548 }
549 else
550 {
551 // Normalize weights to sum to 1.
552 var c = IsZeroInflated ? (1d - ZeroWeight) / sum : 1d / sum;
553 // Normalize weights
554 for (int i = 0; i < K; i++)
555 {
556 Weights[i] *= c;
557 parameters[i] = Weights[i];
558 }
559 }
560
561
562 // Set distribution parameters
563 for (int i = 0; i < Distributions.Count(); i++)
564 {
565 var parms = new List<double>();
566 for (int j = t; j < t + Distributions[i].NumberOfParameters; j++)
567 {
568 parms.Add(parameters[j]);
569 }
570 Distributions[i].SetParameters(parms);
571 t += Distributions[i].NumberOfParameters;
572 }
573 }
574
575 // Validate parameters
576 _parametersValid = ValidateParameters(parameters, false) is null;
577 _momentsComputed = false;
578 _inverseCDFCreated = false;
579 }

582 public override ArgumentOutOfRangeException ValidateParameters(IList<double> parameters, bool throwException)
583 {
584 // Check if weights are between 0 and 1.
585 if (IsZeroInflated && (ZeroWeight < 0.0 || ZeroWeight > 1.0))
586 {
587 if (throwException)
588 throw new ArgumentOutOfRangeException(nameof(ZeroWeight), "The zero value weight must be between 0 and 1.");
589 return new ArgumentOutOfRangeException(nameof(ZeroWeight), "The zero value weight must be between 0 and 1.");
590 }
591 for (int i = 0; i < Distributions.Count(); i++)
592 {
593 if (Weights[i] < 0.0 || Weights[i] > 1.0)
594 {
595 if (throwException)
596 throw new ArgumentOutOfRangeException(nameof(Weights), "The weights must be between 0 and 1.");
597 return new ArgumentOutOfRangeException(nameof(Weights), "The weights must be between 0 and 1.");
598 }
599 }
600 // Check if weights sum to 1.
601 double sum = IsZeroInflated ? ZeroWeight : 0.0;
602 for (int i = 0; i < Distributions.Count(); i++)
603 sum += Weights[i];
604 if (sum.AlmostEquals(1d, 1E-8) == false)
605 {
606 if (throwException)
607 throw new ArgumentOutOfRangeException(nameof(Weights), "The weights must sum to 1.0.");
608 return new ArgumentOutOfRangeException(nameof(Weights), "The weights must sum to 1.0.");
609 }
610 // Check if distributions are valid
611 for (int i = 0; i < Distributions.Count(); i++)
612 {
613 if (Distributions[i].ParametersValid == false)
614 {
615 if (throwException)
616 throw new ArgumentOutOfRangeException(nameof(Distributions), "Distribution " + (i + 1).ToString() + " has invalid parameters.");
617 return new ArgumentOutOfRangeException(nameof(Distributions), "Distribution " + (i + 1).ToString() + " has invalid parameters.");
618 }
619 }
620 return null;
621 }

624 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
625 {
626 var initialVals = new double[NumberOfParameters];
627 var lowerVals = new double[NumberOfParameters];
628 var upperVals = new double[NumberOfParameters];
629
630 // Weights are first
631 int t = 0;
632 for (int i = 0; i < Distributions.Count(); i++)
633 {
634 initialVals[i] = IsZeroInflated ? (1d - ZeroWeight) / Distributions.Count() : 1d / Distributions.Count();
635 lowerVals[i] = 0.0;
636 upperVals[i] = 1.0;
637 t += 1;
638 }
639
640 for (int i = 0; i < Distributions.Count(); i++)
641 {
642 var tuple = ((IMaximumLikelihoodEstimation)Distributions[i]).GetParameterConstraints(sample);
643 var initials = tuple.Item1;
644 var lowers = tuple.Item2;
645 var uppers = tuple.Item3;
646
647 for (int j = t; j < t + Distributions[i].NumberOfParameters; j++)
648 {
649 initialVals[j] = initials[j - t];
650 lowerVals[j] = lowers[j - t];
651 upperVals[j] = uppers[j - t];
652 }
653 t += Distributions[i].NumberOfParameters;
654 }
655 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
656 }

659 public double[] MLE(IList<double> sample)
660 {
661
662 int N = sample.Count;
663 int Np = Distributions.Sum(x => x.NumberOfParameters);
664 int K = Distributions.Count();
665
666 // Set constraints
667 var tuple = GetParameterConstraints(sample);
668 var Initials = tuple.Item1.Subset(K);
669 var Lowers = tuple.Item2.Subset(K);
670 var Uppers = tuple.Item3.Subset(K);
671
672 // Set up EM parameters
673 var mleWeights = tuple.Item1.Subset(0 , K - 1);
674 var mleParameters = Initials;
675 var likelihood = new double[N, K];
676 double oldLogLH = double.MinValue, newLogLH = double.MinValue;
677
678 // The Expectation step.
679 double EStep(double[] x)
680 {
681 var dist = (Mixture)Clone();
682 dist.SetParameters(mleWeights, x);
683 // Outer loop for computing the likelihoods
684 for (int k = 0; k < K; k++)
685 {
686 for (int i = 0; i < N; i++)
687 {
688 if (IsZeroInflated && sample[i] <= 0.0)
689 {
690 likelihood[i, k] = Math.Log(ZeroWeight);
691 }
692 else
693 {
694 likelihood[i, k] = Math.Log(mleWeights[k]) + dist.Distributions[k].LogPDF(sample[i]);
695 }
696 }
697 }
698 // At this point we have unnormalized log likelihoods.
699 // We need to normalize using log-sum-exp and compute the true log-likelihoods.
700 double logLH = 0;
701 for (int i = 0; i < N; i++)
702 {
703 // Get max likelihood
704 double max = double.MinValue;
705 for (int k = 0; k < K; k++)
706 {
707 if (likelihood[i, k] > max)
708 {
709 max = likelihood[i, k];
710 }
711 }
712 // log-sum-exp trick begins here
713 double sum = 0;
714 for (int k = 0; k < K; k++)
715 sum += Math.Exp(likelihood[i, k] - max);
716 double tmp = max + Math.Log(sum);
717 for (int k = 0; k < K; k++)
718 likelihood[i, k] = Math.Exp(likelihood[i, k] - tmp);
719 logLH += tmp;
720 }
721 return logLH;
722 }
723
724 // The Maximization step
725 double[] MStep(double[] x)
726 {
727 // Get updated weights
728 for (int k = 0; k < K; k++)
729 {
730 double wgt = 0d;
731 for (int i = 0; i < N; i++)
732 {
733 if (!IsZeroInflated || sample[i] > 0.0 )
734 {
735 wgt += likelihood[i, k];
736 }
737 }
738 mleWeights[k] = wgt / N;
739 }
740 // MLE
741 var solver = new NelderMead(logLH, Np, x, Lowers, Uppers);
742 solver.Maximize();
743 return solver.BestParameterSet.Values;
744 }
745
746 // The log-likelihood to maximize in the M-Step
747 // Weights are held fixed, only the distribution parameters are solved.
748 double logLH(double[] x)
749 {
750 var dist = (Mixture)Clone();
751 dist.SetParameters(mleWeights, x);
752 double lh = dist.LogLikelihood(sample);
753 if (double.IsNaN(lh) || double.IsInfinity(lh)) return double.MinValue;
754 return lh;
755 }
756
757 // Estimate using the EM Algorithm
758 for (Iterations = 1; Iterations <= MaxIterations; Iterations++)
759 {
760 // Perform the expectation step
761 newLogLH = EStep(mleParameters);
762
763 // Check convergence
764 if (Math.Abs((oldLogLH - newLogLH) / oldLogLH) < Tolerance)
765 break;
766
767 // Perform the maximization step
768 mleParameters = MStep(mleParameters);
769
770 // Update log-likelihood state
771 oldLogLH = newLogLH;
772
773 }
774
775 // Return the full list of distribution parameters
776 var result = new List<double>();
777 result.AddRange(mleWeights);
778 result.AddRange(mleParameters);
779 return result.ToArray();
780 }

905 if (probability < 0.0d || probability > 1.0d)
906 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
907 if (probability == 0.0d) return Minimum;
908 if (probability == 1.0d) return Maximum;
909 if (IsZeroInflated && probability <= ZeroWeight) return 0;
910
911 // Validate parameters
912 if (_parametersValid == false)
913 ValidateParameters(GetParameters, true);
914
915 // If there is only one distribution, return its inverse CDF
916 if (Distributions.Count() == 1)
917 {
918 return Distributions[0].InverseCDF(probability);
919 }
920
921 // Otherwise use a root finder to solve the inverse CDF
922 var xVals = Distributions.Select(d => d.InverseCDF(probability));
923 double minX = xVals.Min();
924 double maxX = xVals.Max();
925 double x = 0;
926 try
927 {
928 if (IsZeroInflated)
929 {
930 Brent.Bracket((y) => { return probability - CDF(y); }, ref minX, ref maxX, out var f1, out var f2);
931 }
932 x = Brent.Solve((y) => { return probability - CDF(y); }, minX, maxX, 1E-6, 100, true);
933 }
934 catch (Exception)
935 {
936 // If the root finder fails, create an empirical inverse CDF
937 if (_inverseCDFCreated == false)
938 CreateInverseCDF();
939 x = _inverseCDF.InverseCDF(probability);
940 }
941 double min = Minimum;
942 double max = Maximum;
943 return x < min ? min : x > max ? max : x;
944 }

949 private void CreateInverseCDF()
950 {
951 // Get min & max
952 double minP = 1E-16;
953 double maxP = 1 - 1E-16;
954 double minX = Distributions.Min(d => d.InverseCDF(minP));
955 double maxX = Distributions.Max(d => d.InverseCDF(maxP));
956 // Get number of bins
957 double shift = 0;
958 if (minX <= 0) shift = Math.Abs(minX) + 1d;
959 double min = minX + shift;
960 double max = maxX + shift;
961 int order = (int)Math.Floor(Math.Log10(max) - Math.Log10(min));
962 int binN = Math.Max(200, 100 * order) - 1;
963 // Create bins
964 var bins = Stratify.XValues(new StratificationOptions(minX, maxX, binN, false), true);
965 var xValues = new List<double>();
966 var pValues = new List<double>();
967 var x = bins.First().LowerBound;
968 var p = CDF(bins.First().LowerBound);
969 xValues.Add(x);
970 pValues.Add(p);
971 for (int i = 1; i < bins.Count; i++)
972 {
973 x = bins[i].LowerBound;
974 p = CDF(x);
975 if (x > xValues.Last() && p > pValues.Last())
976 {
977 xValues.Add(x);
978 pValues.Add(p);
979 }
980 }
981 x = maxX;
982 p = CDF(x);
983 if (x > xValues.Last() && p > pValues.Last())
984 {
985 xValues.Add(x);
986 pValues.Add(p);
987 }
988 _inverseCDF = new EmpiricalDistribution(xValues, pValues) { XTransform = XTransform, ProbabilityTransform = ProbabilityTransform };
989 _inverseCDFCreated = true;
990 }

993 public override double[] GenerateRandomValues(int sampleSize, int seed = -1)
994 {
995 // Create PRNG for generating random numbers
996 var rnd = seed > 0 ? new MersenneTwister(seed) : new MersenneTwister();
997 var weights = new List<double>();
998 var distributions = new List<UnivariateDistributionBase>();
999 if (IsZeroInflated)
1000 {
1001 weights.Add(ZeroWeight);
1002 distributions.Add(new Deterministic(0.0));
1003 }
1004 weights.AddRange(Weights);
1005 distributions.AddRange(Distributions);
1006
1007 var sample = new double[sampleSize];
1008 // Generate values
1009 for (int i = 0; i < sampleSize; i++)
1010 {
1011 var u = rnd.NextDouble();
1012 var cdfW = new double[distributions.Count()];
1013 for (int j = 0; j < distributions.Count(); j++)
1014 {
1015 cdfW[j] = j == 0 ? weights[j] : cdfW[j - 1] + weights[j];
1016 if (u <= cdfW[j])
1017 {
1018 sample[i] = distributions[j].InverseCDF(rnd.NextDouble());
1019 break;
1020 }
1021 }
1022 }
1023 // Return array of random values
1024 return sample;
