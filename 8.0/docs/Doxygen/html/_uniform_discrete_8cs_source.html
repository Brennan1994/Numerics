272 public override double PDF(double k)
273 {
274 // Validate parameters
275 if (_parametersValid == false)
276 ValidateParameters(Min, Max, true);
277 if (k < Minimum || k > Maximum) return 0.0d;
278 return 1d / N;
279 }

282 public override double CDF(double k)
283 {
284 // Validate parameters
285 if (_parametersValid == false)
286 ValidateParameters(Min, Max, true);
287 if (k < Minimum)
288 return 0d;
289 if (k > Maximum)
290 return 1d;
291 return (k - Min + 1.0d) / N;
292 }
293
295 public override double InverseCDF(double probability)
296 {
297 // Validate probability
298 if (probability < 0.0d || probability > 1.0d)
299 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
300 if (probability == 0.0d)
301 return Minimum;
302 if (probability == 1.0d)
303 return Maximum;
304 // Validate parameters
305 if (_parametersValid == false)
306 ValidateParameters(Min, Max, true);
307 return Math.Floor(Min + probability * N);
308 }
