195 protected override void Optimize()
196 {
197 if (SampleSize < 4) throw new ArgumentOutOfRangeException(nameof(SampleSize), "The sample size must be greater than or equal to 4.");
198 if (Gamma <= 0 || Gamma >= 1) throw new ArgumentOutOfRangeException(nameof(Gamma), "The reduction parameter must be between 0 and 1.");
199 if (Sigma <= 0) throw new ArgumentOutOfRangeException(nameof(Sigma), "The scale parameter must be positive.");
200 if (MinNoImprovement < 1) throw new ArgumentOutOfRangeException(nameof(MinNoImprovement), "The minimum number of iterations to carry out with no improvement must be greater than 0.");
201 if (MaxNoImprovement < 1) throw new ArgumentOutOfRangeException(nameof(MaxNoImprovement), "The maximum number of iterations to carry out with no improvement must be greater than 0.");
202 if (MaxNoImprovement < MinNoImprovement) throw new ArgumentOutOfRangeException(nameof(MaxNoImprovement), "The maximum number of iterations must be greater than or equal to the minimum number.");
203
204
205 int i, j;
206 double D = NumberOfParameters, N = SampleSize;
207 double oldFit = double.MaxValue;
208 int noImprovement = 0;
209 bool cancel = false;
210 Optimizer solver = null;
211
212 // Set lower and upper bounds and
213 // create uniform distributions for each parameter
214 var uniformDists = new List<Uniform>();
215 for (i = 0; i < D; i++)
216 uniformDists.Add(new Uniform(LowerBounds[i], UpperBounds[i]));
217
218 // Set solver parameters
219 var r = new MersenneTwister(PRNGSeed);
220
221 // Critical distance
222 double rkfactor = Math.Sqrt(Math.PI) * Math.Pow(SpecialFunctions.Gamma.Function(D) * Sigma, 1.0 / D);
223 for (i = 0; i < D; i++)
224 rkfactor *= Math.Pow(UpperBounds[i] - LowerBounds[i], 1.0 / D);
225
226 // Create lists for sample points
227 SampledPoints = new List<SamplePoint>();
228 LocalMinimums = new List<SamplePoint>();
229
230 while (Iterations < MaxIterations)
231 {
232 // Step 1. Generate sample points and function values
233 // Add N points, drawn from a uniform distribution over S, to the (initially empty) set of sample points,
234 // and evaluate f(x) at each new sample point.
235
236 if (Iterations == 0)
237 {
238 // On the first iteration, add the user-defined initial starting points
239 // This can often be very close to the true minimum
240 var x = new SamplePoint();
241 x.ParameterSet = new ParameterSet(InitialValues, Evaluate(InitialValues, ref cancel));
242 x.Minimized = true;
243 SampledPoints.Add(x);
244
245 // Perform local minimizations from initial values
246 solver = GetLocalOptimizer(InitialValues, LocalRelativeTolerance, LocalAbsoluteTolerance, ref cancel);
247 solver.Minimize();
248 if (cancel) return;
249
250 var y = new SamplePoint();
251 y.ParameterSet = solver.BestParameterSet;
252 y.Minimized = true;
253 LocalMinimums.Add(y);
254 }
255
256 // Sample a point from the uniform distribution over S.
257 for (i = 0; i < (Iterations == 0 ? N - 1 : N); i++)
258 {
259 var values = new double[(int)D];
260 for (j = 0; j < D; j++)
261 values[j] = uniformDists[j].InverseCDF(r.NextDouble());
262
263 var x = new SamplePoint();
264 x.ParameterSet = new ParameterSet(values, Evaluate(values, ref cancel));
265 if (cancel) return;
266 SampledPoints.Add(x);
267 }
268
269 // Step 2. Reduce the sample points
270 // Sort the entire sample of kN points in order of increasing object function values.
271 // Select the γkN points with the lowest objective function values.
272 // This resultant set, Rk, is called the reduced sample.
273
274 SampledPoints.Sort((x, y) => x.ParameterSet.Fitness.CompareTo(y.ParameterSet.Fitness));
275 int gkN = (int)Math.Ceiling(Gamma * (Iterations + 1) * N);
276 var Rk = new List<SamplePoint>();
277 for (i = 0; i <gkN; i++)
278 Rk.Add(SampledPoints[i]);
279
280 // Step 3. Select start points for local searches
281 // Determine a (possibly empty) subset of the sample points from which to start local searches.
282 //
283 // At iteration k, each sample point x is selected as a start point for a local minimization if it has:
284 // -> not been used as a start point at a previous iteration
285 // -> there is no sample point y within the critical distance rk of x with a lower function value:
286 // ||x - y|| <= rk and also f(x) < f(y)
287
288 double rk = rkfactor * Math.Pow(Math.Log((Iterations + 1) * N) / ((Iterations + 1) * N), 1.0 / D);
289
290 for (i = 0; i < Rk.Count; i++)
291 {
292 bool canMinimize = true;
293 // Check if local search has already been performed
294 if (Rk[i].Minimized == true) canMinimize = false;
295
296 if (canMinimize == true)
297 {
298 // Check if there is any other point y in Rk such that ||xi - y|| <= rk and also f(x) < f(y)
299 for (j = 0; j < Rk.Count; j++)
300 {
301 if (i != j)
302 {
303 var dist = Tools.Distance(Rk[i].ParameterSet.Values, Rk[j].ParameterSet.Values);
304 if (dist <= rk && Rk[j].ParameterSet.Fitness < Rk[i].ParameterSet.Fitness)
305 {
306 canMinimize = false;
307 break;
308 }
309 }
310 }
311 // Check if there is any other local minimum y such that ||xi - y|| <= rk and also f(x) < f(y)
312 for (j = 0; j < LocalMinimums.Count; j++)
313 {
314 var dist = Tools.Distance(Rk[i].ParameterSet.Values, LocalMinimums[j].ParameterSet.Values);
315 if (dist <= rk && LocalMinimums[j].ParameterSet.Fitness < Rk[i].ParameterSet.Fitness)
316 {
317 canMinimize = false;
318 break;
319 }
320 }
321 }
322
323 // Check if point is a potential minimizer
324 if (canMinimize == true)
325 {
326 // Perform local minimizations from all start points
327 if (Status != OptimizationStatus.None)
328 {
329 return;
330 }
331 solver = GetLocalOptimizer(Rk[i].ParameterSet.Values, LocalRelativeTolerance, LocalAbsoluteTolerance, ref cancel);
332 solver.Minimize();
333 if (cancel) return;
334
335 Rk[i].Minimized = true;
336
337 var x = new SamplePoint();
338 x.ParameterSet = solver.BestParameterSet.Clone();
339 x.Minimized = true;
340 LocalMinimums.Add(x);
341 }
342
343 }
344
345 // Step 4. Decide whether to stop
346 // If stopping rule is satisfied, regard the lowest local minimizer as the global minimizer, otherwise go to step 1.
347 double w = LocalMinimums.Count;
348 double s = SampledPoints.Count;
349 double B1 = (w * (s - 1d)) / (s - w - 2d);
350 double B2 = (s - w - 1d) * (s + w) / (s * (s - 1d));
351
352
353 if (Iterations >= 1 && w >= 1 && oldFit == BestParameterSet.Fitness)
354 {
355 noImprovement += 1;
356 }
357 else
358 {
359 noImprovement = 0;
360 oldFit = BestParameterSet.Fitness;
361 }
362
363 // And the Bayesian stopping rule is satisfied
364 if (noImprovement >= MaxNoImprovement || (noImprovement >= MinNoImprovement && w >= 1 && B1 - w < 0.5 && B2 >= 0.995))
365 {
366 // Polish the final result
367 if (Polish == true)
368 {
369 solver = GetLocalOptimizer(BestParameterSet.Values, RelativeTolerance, AbsoluteTolerance, ref cancel);
370 solver.Minimize();
371 UpdateStatus(solver.Status);
372 return;
373 }
374
375 // Stop and return
376 UpdateStatus(OptimizationStatus.Success);
377 return;
378 }
379
380
381 Iterations += 1;
382 }
383
384 // If we made it to here, the maximum iterations were reached.
385 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
386
387 }

396 private Optimizer GetLocalOptimizer(IList<double> initialValues, double relativeTolerance, double absoluteTolerance, ref bool cancel)
397 {
398 bool _cancel = false;
399 Optimizer solver = null;
400
401 // Make sure the parameters are within the bounds.
402 for (int i = 0; i < NumberOfParameters; i++)
403 initialValues[i] = RepairParameter(initialValues[i], LowerBounds[i], UpperBounds[i]);
404
405 if (Method == LocalMethod.BFGS)
406 {
407 solver = new BFGS((x) => Evaluate(x, ref _cancel), NumberOfParameters, initialValues, LowerBounds, UpperBounds) { RelativeTolerance = relativeTolerance, AbsoluteTolerance = absoluteTolerance, MaxFunctionEvaluations = MaxFunctionEvaluations - FunctionEvaluations };
408 }
409 else if (Method == LocalMethod.NelderMead)
410 {
411 solver = new NelderMead((x) => Evaluate(x, ref _cancel), NumberOfParameters, initialValues, LowerBounds, UpperBounds) { RelativeTolerance = relativeTolerance, AbsoluteTolerance = absoluteTolerance, MaxFunctionEvaluations = MaxFunctionEvaluations - FunctionEvaluations };
412 }
413 else if (Method == LocalMethod.Powell)
414 {
415 solver = new Powell((x) => Evaluate(x, ref _cancel), NumberOfParameters, initialValues, LowerBounds, UpperBounds) { RelativeTolerance = relativeTolerance, AbsoluteTolerance = absoluteTolerance, MaxFunctionEvaluations = MaxFunctionEvaluations - FunctionEvaluations };
416 }
417 cancel = _cancel;
418 return solver;
419 }
