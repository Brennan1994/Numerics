265 _dimension = mean.Length;

268 _cholesky = new CholeskyDecomposition(_covariance);
269 double lndet = _cholesky.LogDeterminant();
270 _lnconstant = -(Math.Log(2d * Math.PI) * _mean.Length + lndet) * 0.5d;

294 private void CreateCorrelationMatrix()
295 {
296 // Save correlation matrix
297 var D = Matrix.Diagonal(_covariance);
298 D.Sqrt();
299 var invSqrtD = !D;
300 _correlation = (invSqrtD * _covariance) * invSqrtD;
301 // Create collapsed matrix
302 _correl = new double[NL];
303 int t = 0;
304 for (int i = 1; i < Dimension; i++)
305 {
306 for (int j = 0; j < i; j++)
307 {
308 _correl[t] = _correlation[i, j];
309 t++;
310 }

367 public double Mahalanobis(double[] x)
368 {
369 if (x.Length != Dimension)
370 throw new ArgumentOutOfRangeException(nameof(x), "The vector must be the same dimension as the distribution.");
371 //
372 var z = new double[_mean.Length];
373 for (int i = 0; i < x.Length; i++)
374 z[i] = x[i] - _mean[i];
375 var a = _cholesky.Solve(new Vector(z));
376 double b = 0d;
377 for (int i = 0; i < z.Length; i++)
378 b += a[i] * z[i];
379 return b;

386 public override double CDF(double[] x)
387 {
388 if (Dimension == 1)
389 {
390 double stdDev = Math.Sqrt(Covariance[0, 0]);
391 if (stdDev == 0d)
392 return x[0] == _mean[0] ? 1 : 0;
393 double z = (x[0] - _mean[0]) / stdDev;
394 return Normal.StandardCDF(z);
395 }
396 else if (Dimension == 2)
397 {
398 double sigma1 = Math.Sqrt(Covariance[0, 0]);
399 double sigma2 = Math.Sqrt(Covariance[1, 1]);
400 double rho = Covariance[0, 1] / (sigma1 * sigma2);
401 if (double.IsNaN(rho)) return x.Equals(_mean) ? 1 : 0;
402 double z = (x[0] - _mean[0]) / sigma1;
403 double w = (x[1] - _mean[1]) / sigma2;
404 return BivariateCDF(-z, -w, rho);
405 }
406 else
407 {
408 // Construct inputs for GenzBretz method
409 for (int i = 0; i < x.Length; i++)
410 _upper[i] = (x[i] - _mean[i]) / Math.Sqrt(Covariance[i, i]);
411
412 double ERROR = 0;
413 double VAL = 0;
414 int INFORM = 0;
415
416 if (_correlationMatrixCreated == false)
417 CreateCorrelationMatrix();
418 MVNDST(Dimension, _lower, _upper, _infin, _correl, _maxEvaluations, _absoluteError, _relativeError, ref ERROR, ref VAL, ref INFORM);
419 return VAL;
420 }

594 public static double bivnor(double ah, double ak, double r)
595
596 //****************************************************************************80
597 //
598 // Purpose:
599 //
600 // BIVNOR computes the bivariate normal CDF.
601 //
602 // Discussion:
603 //
604 // BIVNOR computes the probability for two normal variates X and Y
605 // whose correlation is R, that AH <= X and AK <= Y.
606 //
607 // Licensing:
608 //
609 // This code is distributed under the MIT license.
610 //
611 // Modified:
612 //
613 // 13 April 2012
614 //
615 // Author:
616 //
617 // Original FORTRAN77 version by Thomas Donnelly.
618 // C++ version by John Burkardt.
619 //
620 // Reference:
621 //
622 // Thomas Donnelly,
623 // Algorithm 462: Bivariate Normal Distribution,
624 // Communications of the ACM,
625 // October 1973, Volume 16, Number 10, page 638.
626 //
627 // Parameters:
628 //
629 // Input, double AH, AK, the lower limits of integration.
630 //
631 // Input, double R, the correlation between X and Y.
632 //
633 // Output, double BIVNOR, the bivariate normal CDF.
634 //
635 // Local Parameters:
636 //
637 // Local, int IDIG, the number of significant digits
638 // to the right of the decimal point desired in the answer.
639 //
640 {
641 double a2;
642 double ap;
643 double b;
644 double cn;
645 double con;
646 double conex;
647 double ex;
648 double g2;
649 double gh;
650 double gk;
651 double gw = 0;
652 double h2;
653 double h4;
654 int i;
655 int idig = 15;
656 int i5 = 0;
657 double rr;
658 double s1;
659 double s2;
660 double sgn;
661 double sn;
662 double sp;
663 double sqr;
664 double t;
665 double twopi = 6.283185307179587;
666 double w2=0;
667 double wh = 0;
668 double wk = 0;
669
670 b = 0.0;
671
672 gh = gauss(-ah) / 2.0;
673 gk = gauss(-ak) / 2.0;
674
675 if (r == 0.0)
676 {
677 b = 4.00 * gh * gk;
678 b = Math.Max(b, 0.0);
679 b = Math.Min(b, 1.0);
680 return b;
681 }
682
683 rr = (1.0 + r) * (1.0 - r);
684
685 if (rr < 0.0)
686 {
687 //cerr << "\n";
688 //cerr << "BIVNOR - Fatal error!\n";
689 //cerr << " 1 < |R|.\n";
690 //exit(0);
691 }
692
693 if (rr == 0.0)
694 {
695 if (r < 0.0)
696 {
697 if (ah + ak < 0.0)
698 {
699 b = 2.0 * (gh + gk) - 1.0;
700 }
701 }
702 else
703 {
704 if (ah - ak < 0.0)
705 {
706 b = 2.0 * gk;
707 }
708 else
709 {
710 b = 2.0 * gh;
711 }
712 }
713 b = Math.Max(b, 0.0);
714 b = Math.Min(b, 1.0);
715 return b;
716 }
717
718 sqr = Math.Sqrt(rr);
719
720 if (idig == 15)
721 {
722 con = twopi * 1.0E-15 / 2.0;
723 }
724 else
725 {
726 con = twopi / 2.0;
727 for (i = 1; i <= idig; i++)
728 {
729 con = con / 10.0;
730 }
731 }
732 //
733 // (0,0)
734 //
735 if (ah == 0.0 && ak == 0.0)
736 {
737 b = 0.25 + Math.Asin(r) / twopi;
738 b = Math.Max(b, 0.0);
739 b = Math.Min(b, 1.0);
740 return b;
741 }
742 //
743 // (0,nonzero)
744 //
745 if (ah == 0.0 && ak != 0.0)
746 {
747 b = gk;
748 wh = -ak;
749 wk = (ah / ak - r) / sqr;
750 gw = 2.0 * gk;
751 i5 = 1;
752 }
753 //
754 // (nonzero,0)
755 //
756 else if (ah != 0.0 && ak == 0.0)
757 {
758 b = gh;
759 wh = -ah;
760 wk = (ak / ah - r) / sqr;
761 gw = 2.0 * gh;
762 i5 = -1;
763 }
764 //
765 // (nonzero,nonzero)
766 //
767 else if (ah != 0.0 && ak != 0.0)
768 {
769 b = gh + gk;
770 if (ah * ak < 0.0)
771 {
772 b = b - 0.5;
773 }
774 wh = -ah;
775 wk = (ak / ah - r) / sqr;
776 gw = 2.0 * gh;
777 i5 = -1;
778 }
779
780 for (; ; )
781 {
782 sgn = -1.0;
783 t = 0.0;
784
785 if (wk != 0.0)
786 {
787 if (Math.Abs(wk) == 1.0)
788 {
789 t = wk * gw * (1.0 - gw) / 2.0;
790 b = b + sgn * t;
791 }
792 else
793 {
794 if (1.0 < Math.Abs(wk))
795 {
796 sgn = -sgn;
797 wh = wh * wk;
798 g2 = gauss(wh);
799 wk = 1.0 / wk;
800
801 if (wk < 0.0)
802 {
803 b = b + 0.5;
804 }
805 b = b - (gw + g2) / 2.0 + gw * g2;
806 }
807 h2 = wh * wh;
808 a2 = wk * wk;
809 h4 = h2 / 2.0;
810 ex = Math.Exp(-h4);
811 w2 = h4 * ex;
812 ap = 1.0;
813 s2 = ap - ex;
814 sp = ap;
815 s1 = 0.0;
816 sn = s1;
817 conex = Math.Abs(con / wk);
818
819 for (; ; )
820 {
821 cn = ap * s2 / (sn + sp);
822 s1 = s1 + cn;
823
824 if (Math.Abs(cn) <= conex)
825 {
826 break;
827 }
828 sn = sp;
829 sp = sp + 1.0;
830 s2 = s2 - w2;
831 w2 = w2 * h4 / sp;
832 ap = -ap * a2;
833 }
834 t = (Math.Atan(wk) - wk * s1) / twopi;
835 b = b + sgn * t;
836 }
837 }
838 if (0 <= i5 )
839 {
840 break;
841 }
842 if (ak == 0.0)
843 {
844 break;
845 }
846 wh = -ak;
847 wk = (ah / ak - r) / sqr;
848 gw = 2.0 * gk;
849 i5 = 1;
850 }
851
852 b = Math.Max(b, 0.0);
853 b = Math.Min(b, 1.0);
854
855 return b;
856 }
