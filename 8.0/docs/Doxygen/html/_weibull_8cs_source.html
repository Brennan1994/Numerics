317 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
318 {
319 var initialVals = new double[NumberOfParameters];
320 var lowerVals = new double[NumberOfParameters];
321 var upperVals = new double[NumberOfParameters];
322 // Get initial values
323 initialVals = SolveMLE(sample);
324 // Get bounds of scale
325 lowerVals[0] = Tools.DoubleMachineEpsilon;
326 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[0]) + 1d));
327 // Get bounds of shape
328 lowerVals[1] = Tools.DoubleMachineEpsilon;
329 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[1]) + 1d));

334 public double[] MLE(IList<double> sample)
335 {
336 // Set constraints
337 var tuple = GetParameterConstraints(sample);
338 var Initials = tuple.Item1;
339 var Lowers = tuple.Item2;
340 var Uppers = tuple.Item3;
341
342 // Solve using Nelder-Mead (Downhill Simplex)
343 double logLH(double[] x)
344 {
345 var W = new Weibull();
346 W.SetParameters(x);
347 return W.LogLikelihood(sample);
348 }
349 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
350 solver.ReportFailure = true;
351 solver.Maximize();
352 return solver.BestParameterSet.Values;
353 }

371 public double[] SolveMLE(IList<double> samples)
372 {
373 double n = samples.Count;
374 if (n <= 1d)
375 {
376 throw new Exception("Observations not sufficient. There must be more than 1 data point.");
377 }
378
379 double s1 = 0d;
380 double s2 = 0d;
381 double s3 = 0d;
382 double previousC = int.MinValue;
383 double QofC = 0d;
384 double c = 10d; // shape
385 double b = 0d; // scale
386
387 // solve for the shape parameter
388 while (Math.Abs(c - previousC) >= 0.0001d)
389 {
390 s1 = 0d;
391 s2 = 0d;
392 s3 = 0d;
393 foreach (double x in samples)
394 {
395 if (x > 0d)
396 {
397 s1 += Math.Log(x);
398 s2 += Math.Pow(x, c);
399 s3 += Math.Pow(x, c) * Math.Log(x);
400 }
401 }
402
403 QofC = n * s2 / (n * s3 - s1 * s2);
404 previousC = c;
405 c = (c + QofC) / 2d;
406 }
407
408 // solve for scale
409 foreach (double x in samples)
410 {
411 if (x > 0d)
412 {
413 b += Math.Pow(x, c);
414 }
415 }
416
417 b = Math.Pow(b / n, 1d / c);
418
419 // return parameters
420 return [b, c];
421 }

424 public override double PDF(double x)
425 {
426 // Validate parameters
427 if (_parametersValid == false)
428 ValidateParameters(Lambda, Kappa, true);
429 if (x < Minimum) return 0.0d;
430 if (x == 0.0d && Kappa == 1.0d)
431 {
432 return Kappa / Lambda;
433 }
434 else
435 {
436 return Kappa / Lambda * Math.Pow(x / Lambda, Kappa - 1.0d) * Math.Exp(-Math.Pow(x / Lambda, Kappa));
437 }
438 }

442 {
443 // Validate parameters
444 if (_parametersValid == false)
445 ValidateParameters(Lambda, Kappa, true);
446 if (x < Minimum)
447 return 0d;

452 public override double InverseCDF(double probability)
453 {
454 // Validate probability
455 if (probability < 0.0d || probability > 1.0d)
456 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
457 if (probability == 0.0d)
458 return Minimum;
459 if (probability == 1.0d)
460 return Maximum;
461 // Validate parameters
462 if (_parametersValid == false)
463 ValidateParameters(Lambda, Kappa, true);
464 // Compute the inverse CDF
465 return Lambda * Math.Pow(Math.Log(1d / (1d - probability)), 1d / Kappa);

475 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
476 {
477 if (estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
478 {
479 throw new NotImplementedException();
480 }
481 // Validate parameters
482 if (_parametersValid == false)
483 ValidateParameters(_lambda, _kappa, true);
484
485 // Compute covariance
486 double a = Lambda;
487 double b = Kappa;
488 var covar = new double[2, 2];
489 covar[0, 0] = 1.108665d * a * a / (sampleSize * b * b); // scale
490 covar[1, 1] = 0.607927d * b * b / sampleSize; // shape
491 covar[0, 1] = 0.257022d * a / sampleSize;
492 covar[1, 0] = covar[0, 1];
493 return covar;
494 }

497 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
498 {
499 var covar = ParameterCovariance(sampleSize, estimationMethod);
500 var grad = QuantileGradient(probability);
501 double varA = covar[0, 0];
502 double varB = covar[1, 1];
503 double covAB = covar[1, 0];
504 double dQx1 = grad[0];
505 double dQx2 = grad[1];
506 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + 2d * dQx1 * dQx2 * covAB;
507 }

510 public double[] QuantileGradient(double probability)
511 {
512 // Validate parameters
513 if (_parametersValid == false)
514 ValidateParameters(_lambda, _kappa, true);
515 double a = Lambda;
516 double b = Kappa;
517 var gradient = new double[]
518 {
519 Math.Log(Math.Pow(1d / (1d - probability), 1d / Kappa)), // scale
520 a * Math.Log(1d - probability) / (b * b) // shape
521 };
522 return gradient;
523 }

526 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
527 {
528 if (probabilities.Count != NumberOfParameters)
529 {
530 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
531 }
532 // Get gradients
533 var dQp1 = QuantileGradient(probabilities[0]);
534 var dQp2 = QuantileGradient(probabilities[1]);
535 // Compute determinant
536 // |a b|
537 // |c d|
538 // |A| = ad − bc
539 double a = dQp1[0];
540 double b = dQp1[1];
541 double c = dQp2[0];
542 double d = dQp2[1];
543 determinant = a * d - b * c;
544 // Return Jacobian
545 var jacobian = new double[,] { { a, b }, { c, d } };
546 return jacobian;
547 }
