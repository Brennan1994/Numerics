188 public void Train()
189 {
190 IsTrained = false;
191 Features = Math.Min(Features, Dimensions);
192 DecisionTrees = new DecisionTree[NumberOfTrees];
193 var seeds = Random.NextIntegers(NumberOfTrees);
194
195 // Estimate trees in parallel
196 Parallel.For(0, NumberOfTrees, idx =>
197 {
198 DecisionTrees[idx] = BootstrapDecisionTree(seeds[idx]);
199 DecisionTrees[idx].Train();
200 });
201
202 IsTrained = true;
203 }
204
209 private DecisionTree BootstrapDecisionTree(int seed = -1)
210 {
211 var rnd = seed > 0 ? new MersenneTwister(seed) : new MersenneTwister();
212 var idxs = rnd.NextIntegers(0, X.NumberOfRows, X.NumberOfRows);
213 var bootX = new Matrix(X.NumberOfRows, X.NumberOfColumns);
214 var bootY = new Vector(Y.Length);
215 for (int i = 0; i < X.NumberOfRows; i++)
216 {
217 for (int j = 0; j < X.NumberOfColumns; j++)
218 {
219 bootX[i, j] = X[idxs[i], j];
220 }
221 bootY[i] = Y[idxs[i]];
222 }
223 return new DecisionTree(bootX, bootY, seed) { MinimumSplitSize = MinimumSplitSize, MaxDepth = MaxDepth, Features = Features, IsRegression = IsRegression };
224 }

251 public double[,] Predict(Matrix X, double alpha = 0.1)
252 {
253 if (!IsTrained) return null;
254
255 var percentiles = new double[] { alpha / 2d, 0.5, 1d - alpha / 2d };
256 var output = new double[X.NumberOfRows, 4]; // lower, median, upper, mean
257
258 var bootResults = new double[X.NumberOfRows, NumberOfTrees];
259
260 // Bootstrap the predictions
261 Parallel.For(0, NumberOfTrees, idx => { bootResults.SetColumn(idx, DecisionTrees[idx].Predict(X)); });
262
263 // Process results
264 Parallel.For(0, X.NumberOfRows, idx =>
265 {
266 var values = bootResults.GetRow(idx);
267 Array.Sort(values);
268
269 if (IsRegression)
270 {
271 // Record percentiles for CIs
272 for (int j = 0; j < percentiles.Length; j++)
273 output[idx, j] = Statistics.Percentile(values, percentiles[j], true);
274 output[idx, 3] = Statistics.ParallelMean(values);
275 }
276 else
277 {
278 // Record percentiles for CIs
279 for (int j = 0; j < percentiles.Length; j++)
280 output[idx, j] = Math.Floor(Statistics.Percentile(values, percentiles[j], true));
281 output[idx, 3] = Math.Floor(Statistics.ParallelMean(values));
282 }
283
284 });
285
286 return output;
287 }
