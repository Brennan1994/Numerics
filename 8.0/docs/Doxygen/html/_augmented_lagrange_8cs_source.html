68 public AugmentedLagrange(Func<double[], double> objectiveFunction, Optimizer optimizer, IList<IConstraint> constraints) : base(objectiveFunction, optimizer.NumberOfParameters)
69 {
70 // Check if there are constraints
71 if (constraints == null || constraints.Count == 0)
72 {
73 throw new ArgumentOutOfRangeException(nameof(constraints), "There must be at least one constraint.");
74 }
75 // Constraints
76 _constraints = constraints.ToArray();
77 _lambda = new double[_constraints.Where((x) => x.Type == ConstraintType.EqualTo).ToArray().Length];
78 _mu = new double[_constraints.Where((x) => x.Type == ConstraintType.LesserThanOrEqualTo).ToArray().Length];
79 _nu = new double[_constraints.Where((x) => x.Type == ConstraintType.GreaterThanOrEqualTo).ToArray().Length];
80
81 // Set up objective functions and optimizer
82 if (optimizer.GetType() == typeof(AugmentedLagrange)) throw new ArgumentException(nameof(optimizer), "The inner optimizer cannot also be an Augmented Lagrange optimizer.");
83 _primaryObjectiveFunction = objectiveFunction;
84 this.Optimizer = optimizer;
85 this.Optimizer.ObjectiveFunction = augmentedLagrangianFunction;
86

126 private double augmentedLagrangianFunction(double[] x)
127 {
128 double phi = _primaryObjectiveFunction(x);
129 double rho2 = 0.5 * rho;
130
131 // For each equality constraint
132 for (int i = 0; i < _constraints.Length; i++)
133 {
134 double actual = _constraints[i].Function(x);
135 double c = 0;
136
137 switch (_constraints[i].Type)
138 {
139 case ConstraintType.EqualTo:
140 c = actual - _constraints[i].Value;
141 phi += rho2 * Math.Pow(c + _lambda[i] / rho, 2d);
142 break;
143
144 case ConstraintType.LesserThanOrEqualTo:
145 c = actual - _constraints[i].Value;
146 if (c > 0) phi += rho2 * Math.Pow(c + _mu[i] / rho, 2d);
147 break;
148
149 case ConstraintType.GreaterThanOrEqualTo:
150 c = _constraints[i].Value - actual;
151 if (c > 0) phi += rho2 * Math.Pow(c + _nu[i] / rho, 2d);
152 break;
153 }
154 }
155
156 return phi;
157 }

160 protected override void Optimize()
161 {
162 double ICM = Double.PositiveInfinity;
163 double minPenalty = Double.PositiveInfinity;
164 double penalty;
165 double minFitness = Double.PositiveInfinity;
166 double currentFitness;
167 bool minFeasible = false;
168 bool cancel = false;
169
170 // magic parameters from Birgin & Martinez
171 const double tau = 0.5, gam = 10;
172
173 this.Optimizer.Minimize();
174 var currentValues = this.Optimizer.BestParameterSet.Values.ToArray();
175
176 Array.Clear(_lambda, 0, _lambda.Length);
177 Array.Clear(_mu, 0, _mu.Length);
178 Array.Clear(_nu, 0, _nu.Length);
179 rho = 1;
180
181 // Starting rho suggested by B & M
182 if (_lambda.Length > 0 || _mu.Length > 0 || _nu.Length > 0)
183 {
184 bool feasible = true;
185 double con2 = 0;
186 penalty = 0;
187
188 // Evaluate function
189 currentFitness = Evaluate(currentValues, ref cancel);
190
191 // For each constraint
192 for (int i = 0; i < _constraints.Length; i++)
193 {
194 double actual = _constraints[i].Function(currentValues);
195 double c = 0;
196
197 switch (_constraints[i].Type)
198 {
199 case ConstraintType.EqualTo:
200 c = actual - _constraints[i].Value;
201 penalty += Math.Abs(c);
202 con2 += c * c;
203 feasible = feasible && Math.Abs(c) <= _constraints[i].Tolerance;
204 break;
205
206 case ConstraintType.GreaterThanOrEqualTo:
207 c = _constraints[i].Value - actual;
208 if (c > 0)
209 {
210 penalty += c;
211 con2 += c * c;
212 }
213 feasible = feasible && c <= _constraints[i].Tolerance;
214 break;
215
216 case ConstraintType.LesserThanOrEqualTo:
217 c = actual - _constraints[i].Value;
218 if (c > 0)
219 {
220 penalty += c;
221 con2 += c * c;
222 }
223 feasible = feasible && c <= _constraints[i].Tolerance;
224 break;
225 }
226 }
227
228 minFitness = currentFitness;
229 minPenalty = penalty;
230 minFeasible = feasible;
231 double num = 2.0 * Math.Abs(minFitness);
232 if (num < 1e-300)
233 {
234 rho = rhoMin;
235 }
236 else if (con2 < 1e-300)
237 {
238 rho = rhoMax;
239 }
240 else
241 {
242 rho = num / con2;
243 if (rho < rhoMin)
244 rho = rhoMin;
245 else if (rho > rhoMax)
246 rho = rhoMax;
247 }
248
249 }
250
251 while (Iterations < MaxIterations)
252 {
253 double prevICM = ICM;
254
255 this.Optimizer.Minimize();
256 currentValues = this.Optimizer.BestParameterSet.Values.ToArray();
257 currentFitness = Evaluate(currentValues, ref cancel);
258 if (cancel) return;
259
260 ICM = 0;
261 penalty = 0;
262 bool feasible = true;
263
264 // Update lambdas
265 for (int i = 0; i < _constraints.Length; i++)
266 {
267 double actual = _constraints[i].Function(currentValues);
268 double c = 0;
269 double newLambda = 0;
270
271 switch (_constraints[i].Type)
272 {
273 case ConstraintType.EqualTo:
274 c = actual - _constraints[i].Value;
275 newLambda = _lambda[i] + rho * c;
276 penalty += Math.Abs(c);
277 feasible = feasible && Math.Abs(c) <= _constraints[i].Tolerance;
278 ICM = Math.Max(ICM, Math.Abs(c));
279 _lambda[i] = Math.Min(Math.Max(-1e20, newLambda), 1e20);
280 break;
281
282 case ConstraintType.LesserThanOrEqualTo:
283 c = actual - _constraints[i].Value;
284 newLambda = _mu[i] + rho * c;
285 penalty += c > 0 ? c : 0;
286 feasible = feasible && c <= _constraints[i].Tolerance;
287 ICM = Math.Max(ICM, Math.Abs(Math.Max(c, -_mu[i] / rho)));
288 _mu[i] = Math.Min(Math.Max(0.0, newLambda), 1e20);
289 break;
290
291 case ConstraintType.GreaterThanOrEqualTo:
292 c = _constraints[i].Value - actual;
293 newLambda = _nu[i] + rho * c;
294 penalty += c > 0 ? c : 0;
295 feasible = feasible && c <= _constraints[i].Tolerance;
296 ICM = Math.Max(ICM, Math.Abs(Math.Max(c, -_nu[i] / rho)));
297 _nu[i] = Math.Min(Math.Max(0.0, newLambda), 1e20);
298 break;
299 }
300 }
301 // Update rho
302 if (ICM > tau * prevICM)
303 {
304 rho *= gam;
305 }
306
307 // Determine if the solution as converged.
308 bool a = !minFeasible || penalty < minPenalty || currentFitness < minFitness;
309 bool b = !minFeasible && penalty < minPenalty;
310 if ((feasible && a) || b)
311 {
312 if (feasible && CheckConvergence(minFitness, currentFitness))
313 {
314 UpdateStatus(OptimizationStatus.Success);
315 return;
316 }
317
318 BestParameterSet = new ParameterSet(currentValues, currentFitness);
319 minFitness = currentFitness;
320 minPenalty = penalty;
321 minFeasible = feasible;
322
323 }
324 else if (ICM == 0)
325 {
326 UpdateStatus(OptimizationStatus.Success);
327 return;
328 }
329
330 Iterations += 1;
331 }
332
333 // If we made it to here, the maximum iterations were reached.
334 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
335 }
336
337 }
