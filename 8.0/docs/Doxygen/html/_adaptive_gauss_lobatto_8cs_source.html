102 public override void Integrate()
103 {
104 ClearResults();
105 Validate();
106
107 try
108 {
109 double m, h, fa, fb, i1, i2, iS, erri1, erri2, r;
110 var y = new double[13];
111 m = 0.5 * (a + b);
112 h = 0.5 * (b - a);
113 fa = y[0] = Function(a);
114 fb = y[12] = Function(b);
115 FunctionEvaluations += 2;
116 for (int i = 1; i < 12; i++)
117 {
118 y[i] = Function(m + x[i] * h);
119 FunctionEvaluations++;
120 }
121
122 // 4-point Gauss-Lobatto formula
123 i2 = (h / 6.0) * (y[0] + y[12] + 5.0 * (y[4] + y[8]));
124 // 7-point Kronrod extension
125 i1 = (h / 1470.0) * (77.0 * (y[0] + y[12]) + 432.0 * (y[2] + y[10]) + 625.0 * (y[4] + y[8]) + 672.0 * y[6]);
126 // 13- point Kronrod extension
127 iS= h * (0.0158271919734802 * (y[0] + y[12]) + 0.0942738402188500 * (y[1] + y[11]) + 0.155071987336585 * (y[2] + y[10]) + 0.188821573960182 * (y[3] + y[9]) + 0.199773405226859 * (y[4] + y[8]) + 0.224926465333340 * (y[5] + y[7]) + 0.242611071901408 * y[6]);
128 erri1 = Math.Abs(i1 -iS);
129 erri2 = Math.Abs(i2 -iS);
130 r = (erri2 != 0.0) ? erri1 / erri2 : 1.0;
131 toler = (r > 0.0 && r < 1.0) ? RelativeTolerance / r : RelativeTolerance;
132 if (iS == 0.0) iS = b - a;
133 iS= Math.Abs(iS);
134
135 Result = adaptlob(Function, a, b, fa, fb, iS);
136
137 }
138 catch (Exception ex)
139 {
140 Status = IntegrationStatus.Failure;
141 if (ReportFailure) throw ex;
142 }

149 private double adaptlob(Func<double, double> function, double a, double b, double fa, double fb, double iS)
150 {
151 double m, h, mll, ml, mr, mrr, fmll, fml, fm, fmrr, fmr, i1, i2;
152 m = 0.5 * (a + b);
153 h = 0.5 * (b - a);
154 mll = m - alpha * h;
155 ml = m - beta * h;
156 mr = m + beta * h;
157 mrr = m + alpha * h;
158 fmll = function(mll);
159 fml = function(ml);
160 fm = function(m);
161 fmr = function(mr);
162 fmrr = function(mrr);
163
164 Iterations++;
165 FunctionEvaluations += 5;
166
167 // 4-point Gauss-Lobatto formula
168 i2 = h / 6.0 * (fa + fb + 5.0 * (fml + fmr));
169 // 7-point Kronrod extension
170 i1 = h / 1470.0 * (77.0 * (fa + fb) + 432.0 * (fmll + fmrr) + 625.0 * (fml + fmr) + 672.0 * fm);
171
172 // Check iterations
173 if (Iterations >= MaxIterations)
174 {
175 // Terminate recursion
176 UpdateStatus(IntegrationStatus.MaximumIterationsReached);
177 return i1;
178 }
179 // Check function evaluations
180 if (FunctionEvaluations >= MaxFunctionEvaluations)
181 {
182 // Terminate recursion
183 UpdateStatus(IntegrationStatus.MaximumFunctionEvaluationsReached);
184 return i1;
185 }
186 // Check tolerance
187 if (Math.Abs(i1 - i2) <= toler * iS || mll <= a || b <= mrr)
188 {
189 if ((mll <= a || b <= mrr) && terminate)
190 {
191 // Interval contains no more machine numbers
192 outOfTolerance = true;
193 terminate = false;
194 }
195 // Terminate recursion
196 UpdateStatus(IntegrationStatus.Success);
197 return i1;
198 }
199 else
200 // Subdivide interval
201 return adaptlob(function, a, mll, fa, fmll,iS) +
202 adaptlob(function, mll, ml, fmll, fml,iS) +
203 adaptlob(function, ml, m, fml, fm,iS) +
204 adaptlob(function, m, mr, fm, fmr,iS) +
205 adaptlob(function, mr, mrr, fmr, fmrr,iS) +
206 adaptlob(function, mrr, b, fmrr, fb,iS);
207 }
