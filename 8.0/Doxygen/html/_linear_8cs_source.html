Numerics 1.0.0
Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

77 public override double BaseInterpolate(double x, int index)
78 {
79 // See if x is out of range
80 if ((SortOrder == SortOrder.Ascending && x <= XValues[0]) ||
81 (SortOrder == SortOrder.Descending && x >= XValues[0])) return YValues[0];
82 if ((SortOrder == SortOrder.Ascending && x >= XValues[Count - 1]) ||
83 (SortOrder == SortOrder.Descending && x <= XValues[Count - 1])) return YValues[Count - 1];
84
85 double y, x1 = 0, x2 = 0, y1 = 0, y2 = 0;
86 int xlo = index, xhi = index + 1;
87
88 // Get X transform
89 if (XTransform == Transform.None)
90 {
91 x1 = XValues[xlo];
92 x2 = XValues[xhi];
93 }
94 else if (XTransform == Transform.Logarithmic)
95 {
96 x = Tools.Log10(x);
97 x1 = Tools.Log10(XValues[xlo]);
98 x2 = Tools.Log10(XValues[xhi]);
99 }
100 else if (XTransform == Transform.NormalZ)
101 {
102 x = Normal.StandardZ(x);
103 x1 = Normal.StandardZ(XValues[xlo]);
104 x2 = Normal.StandardZ(XValues[xhi]);
105 }
106
107 // Get Y transform
108 if (YTransform == Transform.None)
109 {
110 y1 = YValues[xlo];
111 y2 = YValues[xhi];
112 }
113 else if (YTransform == Transform.Logarithmic)
114 {
115 y1 = Tools.Log10(YValues[xlo]);
116 y2 = Tools.Log10(YValues[xhi]);
117 }
118 else if (YTransform == Transform.NormalZ)
119 {
120 y1 = Normal.StandardZ(YValues[xlo]);
121 y2 = Normal.StandardZ(YValues[xhi]);
122 }
123
124 // Interpolate
125 // Check for division by zero
126 if ((x2 - x1) == 0)
127 {
128 y = y1;
129 }
130 else
131 {
132 y = y1 + (x - x1) / (x2 - x1) * (y2 - y1);
133 }
134 //
135 if (YTransform == Transform.None)
136 {
137 return y;
138 }
139 else if (YTransform == Transform.Logarithmic)
140 {
141 return Math.Pow(10d, y);
142 }
143 else if (YTransform == Transform.NormalZ)
144 {
145 return Normal.StandardCDF(y);
146 }
147
148 // return NaN if we get to here
149 return double.NaN;
150 }

156 public double Extrapolate(double x)
157 {
158 double y, x1, x2, y1, y2;
159
160 // See if x is out of range
161 if ((SortOrder == SortOrder.Ascending && x < XValues[0]) || (SortOrder == SortOrder.Descending && x > XValues[0]))
162 {
163 x1 = XValues[0];
164 x2 = XValues[1];
165 y1 = YValues[0];
166 y2 = YValues[1];
167 }
168 else if ((SortOrder == SortOrder.Ascending && x > XValues[Count - 1]) || (SortOrder == SortOrder.Descending && x < XValues[Count - 1]))
169 {
170 x1 = XValues[Count - 1];
171 x2 = XValues[Count - 2];
172 y1 = YValues[Count - 1];
173 y2 = YValues[Count - 2];
174 }
175 else
176 {
177 return Interpolate(x);
178 }
179
180 // Get X transform
181 if (XTransform == Transform.Logarithmic)
182 {
183 x = Tools.Log10(x);
184 x1 = Tools.Log10(x1);
185 x2 = Tools.Log10(x2);
186 }
187 else if (XTransform == Transform.NormalZ)
188 {
189 x = Normal.StandardZ(x);
190 x1 = Normal.StandardZ(x1);
191 x2 = Normal.StandardZ(x2);
192 }
193
194 // Get Y transform
195 if (YTransform == Transform.Logarithmic)
196 {
197 y1 = Tools.Log10(y1);
198 y2 = Tools.Log10(y2);
199 }
200 else if (YTransform == Transform.NormalZ)
201 {
202 y1 = Normal.StandardZ(y1);
203 y2 = Normal.StandardZ(y2);
204 }
205
206 // Extrapolate
207 // Check for division by zero
208 if ((y2 - y1) == 0)
209 {
210 y = y1;
211 }
212 else
213 {
214 y = y1 - (x1 - x) * (y2 - y1) / (x2 - x1);
215 }
216 //
217 if (YTransform == Transform.None)
218 {
219 return y;
220 }
221 else if (YTransform == Transform.Logarithmic)
222 {
223 return Math.Pow(10d, y);
224 }
225 else if (YTransform == Transform.NormalZ)
226 {
227 return Normal.StandardCDF(y);
228 }
229
230 // return NaN if we get to here
231 return double.NaN;
232 }
