Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

264 public override double PDF(double x)
265 {
266 // Validate parameters
267 if (_parametersValid == false)
268 ValidateParameters(Min, Max, true);
269 //
270 if (Min == Max) return 0d;
271 if (x < Minimum || x > Maximum) return 0.0d;
272 return 1.0d / (Max - Min);
273 }
274
276 public override double CDF(double x)
277 {
278 // Validate parameters
279 if (_parametersValid == false)
280 ValidateParameters(Min, Max, true);
281 if (Min == Max) return 1d;
282 if (x <= Minimum) return 0d;
283 if (x >= Maximum) return 1d;
284 return (x - Min) / (Max - Min);
285 }
286
288 public override double InverseCDF(double probability)
289 {
290 // Validate probability
291 if (probability < 0.0d || probability > 1.0d)
292 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
293 if (Min == Max) return Min;
294 if (probability == 0.0d) return Minimum;
295 if (probability == 1.0d) return Maximum;
296 // Validate parameters
297 if (_parametersValid == false)
298 ValidateParameters(Min, Max, true);
299 return Min + probability * (Max - Min);
300 }
