177 public double[,] GenerateRandomValues(int sampleSize, int seed = -1)
178 {
179 var rand = LatinHypercube.Random(sampleSize, 2, seed);
180 var sample = new double[sampleSize, 2];
181 for (int i = 0; i < sampleSize; i++)
182 {
183 var vals = InverseCDF(rand[i, 0], rand[i, 1]);
184 if (MarginalDistributionX != null && MarginalDistributionY != null)
185 {
186 sample[i, 0] = MarginalDistributionX.InverseCDF(vals[0]);
187 sample[i, 1] = MarginalDistributionY.InverseCDF(vals[1]);
188 }
189 else
190 {
191 sample[i, 0] = vals[0];
192 sample[i, 1] = vals[1];
193 }
194 }
195 return sample;
196 }

203 public double PseudoLogLikelihood(IList<double> sampleDataX, IList<double> sampleDataY)
204 {
205 if (sampleDataX.Count < 2 || sampleDataY.Count < 2) throw new ArgumentOutOfRangeException("sample data", "There must be at least two items in each sample.");
206 if (sampleDataX.Count != sampleDataY.Count) throw new ArgumentOutOfRangeException("sample data", "The sample data arrays must be the same length.");
207
208 double LogLH = 0;
209 for (int i = 0; i < sampleDataX.Count; i++)
210 LogLH += LogPDF(sampleDataX[i], sampleDataY[i]);
211 if (double.IsNaN(LogLH) || double.IsInfinity(LogLH)) return double.MinValue;
212 return LogLH;
213 }

220 public double IFMLogLikelihood(IList<double> sampleDataX, IList<double> sampleDataY)
221 {
222 if (sampleDataX.Count < 2 || sampleDataY.Count < 2) throw new ArgumentOutOfRangeException("sample data", "There must be at least two items in each sample.");
223 if (sampleDataX.Count != sampleDataY.Count) throw new ArgumentOutOfRangeException("sample data", "The sample data arrays must be the same length.");
224 if (MarginalDistributionX == null || MarginalDistributionY == null) throw new ArgumentOutOfRangeException("marginal distributions", "There must be 2 marginal distributions to evaluate.");
225
226 double LogLH = 0;
227 for (int i = 0; i < sampleDataX.Count; i++)
228 LogLH += LogPDF(MarginalDistributionX.CDF(sampleDataX[i]), MarginalDistributionY.CDF(sampleDataY[i]));
229 if (double.IsNaN(LogLH) || double.IsInfinity(LogLH)) return double.MinValue;
230 return LogLH;
231 }

238 public double LogLikelihood(IList<double> sampleDataX, IList<double> sampleDataY)
239 {
240 if (sampleDataX.Count < 2 || sampleDataY.Count < 2) throw new ArgumentOutOfRangeException("sample data", "There must be at least two items in each sample.");
241 if (sampleDataX.Count != sampleDataY.Count) throw new ArgumentOutOfRangeException("sample data", "The sample data arrays must be the same length.");
242 if (MarginalDistributionX == null || MarginalDistributionY == null) throw new ArgumentOutOfRangeException("marginal distributions", "There must be 2 marginal distributions to evaluate.");
243
244 double LogLH = 0;
245 for (int i = 0; i < sampleDataX.Count; i++)
246 LogLH += LogPDF(MarginalDistributionX.CDF(sampleDataX[i]), MarginalDistributionY.CDF(sampleDataY[i])) + MarginalDistributionX.LogPDF(sampleDataX[i]) + MarginalDistributionY.LogPDF(sampleDataY[i]);
247 if (double.IsNaN(LogLH) || double.IsInfinity(LogLH)) return double.MinValue;
248 return LogLH;
249 }
