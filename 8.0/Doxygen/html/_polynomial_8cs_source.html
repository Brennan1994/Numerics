Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

90 public override double BaseInterpolate(double x, int index)
91 {
92 // Given a value x, this routine returns an interpolate value y, and stores an error estimate.
93 // The return value is obtained by Degree-point polynomial interpolation on the subrange x[start..start + Order]
94 if (index < 0 || index >= Count) index = 0;
95 int mm = Order + 1, ns = 0;
96 double y, den, dif, dift, ho, hp, w;
97 int jl = index + mm > Count ? 0 : index;
98 var xa = XValues.ToArray().Subset(jl);
99 var ya = YValues.ToArray().Subset(jl);
100 var c = new double[mm];
101 var d = new double[mm];
102 dif = Math.Abs(x - xa[0]);
103 // Here we find the index ns of the closest table entry
104 for (int i = 0; i < mm; i++)
105 {
106 if ((dift = Math.Abs(x - xa[i])) < dif)
107 {
108 ns = i;
109 dif = dift;
110 }
111 // and initialize the tableau of c's and d's.
112 c[i] = ya[i];
113 d[i] = ya[i];
114 }
115 // This is the initial approximation to y.
116 y = ya[ns--];
117 // For each column in the tableau,
118 // we loop over the current c's and d's and update them.
119 for (int m = 1; m < mm; m++)
120 {
121 for (int i = 0; i < mm - m; i++)
122 {
123 ho = xa[i] - x;
124 hp = xa[i + m] - x;
125 w = c[i + 1] - d[i];
126 den = ho - hp;
127 den = w / den;
128 // Here the c's and d's are updated.
129 d[i] = hp * den;
130 c[i] = ho * den;
131 }
132 y += (Error = (2 * (ns + 1) < (mm - m) ? c[ns + 1] : d[ns--]));
133 // After each column in the tableau is completed, we decide which correction, c or d, we
134 // want to add to our accumulating value of y, i.e., which path to take through the tableau
135 // - forking up or down. TWe do this in such a way as to take the most "straight line"
136 // route through the tableau to its apex, updating ns accordingly to keep track of where
137 // we are. This route keeps the partial approximations centered (insofar as possible) on
138 // the target x. The last dy added is thus the error indication.
139 }
140 return y;
141 }
