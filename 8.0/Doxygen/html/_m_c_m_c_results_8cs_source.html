Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

118 private void ProcessParameterResults(MCMCSampler sampler, double alpha = 0.1)
119 {
120 // Compute the Gelman-Rubin diagnostic using the post-warm up period
121 var GR = MCMCDiagnostics.GelmanRubin(sampler.MarkovChains, sampler.WarmupIterations);
122
123 // Compute parameter summary statistics
124 ParameterResults = new ParameterResults[sampler.NumberOfParameters];
125 for (int i = 0; i < sampler.NumberOfParameters; i++)
126 {
127 // Compute the Autocorrelation Function (ACF) and Effective Sample Size (ESS)
128 // Average the ACF and sum the ESS
129 var x = new List<double>();
130 var avgACF = new double[51, 2];
131 double ess = 0;
132
133 for (int j = 0; j < sampler.NumberOfChains; j++)
134 {
135 // Get list of parameters
136 var N = sampler.Output[j].Count;
137 var y = new List<double>();
138 for (int k = 0; k < N; k++)
139 y.Add(sampler.Output[j][k].Values[i]);
140 x.AddRange(y);
141
142 // Get ACF
143 var acf = Fourier.Autocorrelation(y, (int)Math.Ceiling((double)N / 2));
144 for (int k = 0; k < acf.GetLength(0); k++)
145 {
146 if (k > 50) break;
147 avgACF[k, 1] += acf[k, 1] / sampler.NumberOfChains;
148 }
149 // Get ESS
150 double rho = 0;
151 for (int k = 1; k < acf.GetLength(0); k++)
152 {
153 if (acf[k, 1] < 0.05) break;
154 rho += acf[k, 1];
155 }
156 ess += Math.Min(N / (1d + 2d * rho), N);
157 }
158
159 ParameterResults[i] = new ParameterResults(x, alpha);
160 ParameterResults[i].SummaryStatistics.Rhat = GR[i];
161 ParameterResults[i].SummaryStatistics.ESS = ess;
162 ParameterResults[i].Autocorrelation = avgACF;
163 }
164 }
