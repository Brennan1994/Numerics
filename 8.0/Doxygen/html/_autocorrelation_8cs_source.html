153 private static double[,] Covariance(IList<double> data, int lagMax = -1)
154 {
155 int n = data.Count;
156 if (lagMax < 0) lagMax = (int)Math.Floor(Math.Min(10d * Math.Log10(n), n - 1));
157 if (lagMax < 1 || n < 2) return null;
158 var acvf = new double[lagMax + 1, 2];
159 double mean = Statistics.Mean(data);
160 for (int lag = 0; lag <= lagMax; lag++)
161 {
162 double cov = 0d;
163 for (int t = lag; t < n; t++)
164 cov += (data[t] - mean) * (data[t - lag] - mean);
165 acvf[lag, 0] = lag;
166 acvf[lag, 1] = cov / n;
167 }
168
169 return acvf;
170 }

250 private static double[,] Partial(IList<double> data, int lagMax = -1)
251 {
252 int n = data.Count;
253 if (lagMax < 0) lagMax = (int)Math.Floor(Math.Min(10d * Math.Log10(n), n - 1));
254 if (lagMax < 1 || n < 2) return null;
255 // First compute the ACVF
256 var acvf = Covariance(data, lagMax);
257 // Then compute PACF using the Durbin-Levinson algorithm
258 int i, j;
259 var phis = new double[lagMax + 1];
260 var phis2 = new double[lagMax + 1];
261 var pacf = new double[lagMax, 2];
262 double vi, phinn;
263 phis[0] = acvf[1, 1] / acvf[0, 1];
264 pacf[0, 0] = 1d;
265 pacf[0, 1] = phis[0];
266 vi = acvf[0, 1];
267 vi *= 1d - phis[0] * phis[0];
268 for (i = 2; i <= lagMax; i++)
269 {
270 for (j = 0; j < i - 1; j++)
271 phis2[j] = phis[i - j - 2];
272 phinn = acvf[i, 1];
273 for (j = 1; j < i; j++)
274 phinn -= phis[j - 1] * acvf[i - j, 1];
275 phinn /= vi;
276 for (j = 0; j < i - 1; j++)
277 phis[j] -= phinn * phis2[j];
278 vi *= 1d - phinn * phinn;
279 phis[i - 1] = phinn;
280 pacf[i - 1, 0] = i;
281 pacf[i - 1, 1] = phis[i - 1];
282 }
283
284 return pacf;
285 }

296 private static double[,] Partial(TimeSeries timeSeries, int lagMax = -1)
297 {
298 int n = timeSeries.Count;
299 if (lagMax < 0) lagMax = (int)Math.Floor(Math.Min(10d * Math.Log10(n), n - 1));
300 if (lagMax < 1 || n < 2) return null;
301 // First compute the ACVF
302 var acvf = Covariance(timeSeries, lagMax);
303 // Then compute PACF using the Durbin-Levinson algorithm
304 int i, j;
305 var phis = new double[lagMax + 1];
306 var phis2 = new double[lagMax + 1];
307 var pacf = new double[lagMax, 2];
308 double vi, phinn;
309 phis[0] = acvf[1, 1] / acvf[0, 1];
310 pacf[0, 0] = 1d;
311 pacf[0, 1] = phis[0];
312 vi = acvf[0, 1];
313 vi *= 1d - phis[0] * phis[0];
314 for (i = 2; i <= lagMax; i++)
315 {
316 for (j = 0; j < i - 1; j++)
317 phis2[j] = phis[i - j - 2];
318 phinn = acvf[i, 1];
319 for (j = 1; j < i; j++)
320 phinn -= phis[j - 1] * acvf[i - j, 1];
321 phinn /= vi;
322 for (j = 0; j < i - 1; j++)
323 phis[j] -= phinn * phis2[j];
324 vi *= 1d - phinn * phinn;
325 phis[i - 1] = phinn;
326 pacf[i - 1, 0] = i;
327 pacf[i - 1, 1] = phis[i - 1];
328 }

341 public static double[] CorrelationConfidenceInterval(int sampleSize, double interval = 0.95d)
342 {
343 double alpha = 0.5d * (1.0d - interval);
344 double lo = Normal.StandardZ(alpha) / Math.Sqrt(sampleSize);
345 double hi = Normal.StandardZ(1d - alpha) / Math.Sqrt(sampleSize);
346 return new[] { lo, hi };
