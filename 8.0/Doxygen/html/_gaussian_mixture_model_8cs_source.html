182 public void Train(int seed = -1, bool kMeansPlusPlus = true)
183 {
184 // 1. Initialize clusters from k-Means
185 var kMeans = new KMeans(X, K);
186 kMeans.Train(seed, kMeansPlusPlus);
187 Means = kMeans.Means;
188
189 // Give equal weight to each cluster
190 // And initialize the covariance matrix
191 Weights = new double[K];
192 Likelihood = new double[X.NumberOfRows, K];
193 Sigmas = new Matrix[K];
194 for (int k = 0; k < K; k++)
195 {
196 Weights[k] = 1d / K;
197 Sigmas[k] = new Matrix(Dimension);
198 for (int i = 0; i < Dimension; i++)
199 Sigmas[k][i, i] = 1E-10;
200 }
201
202 // 2. Optimize clusters
203 double oldLogLH = double.MinValue, newLogLH = double.MinValue;
204 for (Iterations = 1; Iterations <= MaxIterations; Iterations++)
205 {
206 // Perform the expectation step
207 newLogLH = EStep();
208
209 // Check convergence
210 if (Math.Abs((oldLogLH - newLogLH) / oldLogLH) < Tolerance)
211 break;
212
213 // Perform the maximization step
214 MStep();
215
216 // Update log-likelihood state
217 oldLogLH = newLogLH;
218
219 }
220
221 }

228 private double EStep()
229 {
230 var logDet = new double[K];
231
232 // Outer loop for computing the likelihoods
233 for (int k = 0; k < K; k++)
234 {
235 // Decompose the covariance in the outer loop
236 var cholesky = new CholeskyDecomposition(Sigmas[k]);
237 logDet[k] = cholesky.LogDeterminant();
238 for (int i = 0; i < X.NumberOfRows; i++)
239 {
240 // Inner loop for likelihoods
241 var u = new Vector(Dimension);
242 for (int d = 0; d < Dimension; d++)
243 u[d] = X[i, d] - Means[k, d];
244 // Solve L*v=u
245 var v = cholesky.Forward(u);
246 double sum = 0;
247 for (int d = 0; d < Dimension; d++)
248 sum += Tools.Sqr(v[d]);
249 Likelihood[i, k] = -0.5 * (sum + logDet[k]) + Math.Log(Weights[k]);
250 }
251 }
252 // At this point we have unnormalized logs of the likelihoods.
253 // We need to normalize using log-sum-exp and compute the log-likelihoods.
254 double logLH = 0;
255 for (int i = 0; i < X.NumberOfRows; i++)
256 {
257 // Get max likelihood and index
258 double max = double.MinValue;
259 int idx = -1;
260 for (int k = 0; k < K; k++)
261 {
262 if (Likelihood[i, k] > max)
263 {
264 max = Likelihood[i, k];
265 idx = k;
266 }
267 }
268 // update labels given the likelihoods
269 Labels[i] = idx;
270
271 // log-sum-exp trick begins here
272 double sum = 0;
273 for (int k = 0; k < K; k++)
274 sum += Math.Exp(Likelihood[i, k] - max);
275 double tmp = max + Math.Log(sum);
276 for (int k = 0; k < K; k++)
277 Likelihood[i, k] = Math.Exp(Likelihood[i, k] - tmp);
278 logLH += tmp;
279 }
280 return logLH;
281 }

286 private void MStep()
287 {
288 for (int k = 0; k < K; k++)
289 {
290 double wgt = 0d;
291 for (int i = 0; i < X.NumberOfRows; i++)
292 wgt += Likelihood[i, k];
293 Weights[k] = wgt / X.NumberOfRows;
294 for (int d = 0; d < Dimension; d++)
295 {
296 // Compute centroids
297 double sum = 0;
298 for (int i = 0; i < X.NumberOfRows; i++)
299 sum += Likelihood[i, k] * X[i, d];
300 Means[k, d] = sum / wgt;
301 // Compute covariance
302 for (int j = 0; j < Dimension; j++)
303 {
304 sum = 0;
305 for (int i = 0; i < X.NumberOfRows; i++)
306 {
307 sum += Likelihood[i, k] * (X[i, d] - Means[k, d]) * (X[i, j] - Means[k, j]);
308 }
309 Sigmas[k][d, j] = sum / wgt;
310 }
311 }
312 }
