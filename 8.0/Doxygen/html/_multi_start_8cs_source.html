160 protected override void Optimize()
161 {
162 int i, j, D = NumberOfParameters;
163 bool cancel = false;
164 Optimizer solver = null;
165
166 // Set lower and upper bounds and
167 // create uniform distributions for each parameter
168 var uniformDists = new List<Uniform>();
169 for (i = 0; i < D; i++)
170 uniformDists.Add(new Uniform(LowerBounds[i], UpperBounds[i]));
171
172 // Set solver parameters
173 var r = new MersenneTwister(PRNGSeed);
174 var values = new double[D];
175
176 while (Iterations < MaxIterations)
177 {
178 if (Iterations == 0)
179 {
180 values = InitialValues;
181 }
182 else
183 {
184 // Step 1. Draw a point from the uniform distribution over S.
185 for (j = 0; j < D; j++)
186 values[j] = uniformDists[j].InverseCDF(r.NextDouble());
187 }
188
189 // Step 2. Apply P to the new sample point.
190 solver = GetLocalOptimizer(values, LocalRelativeTolerance, LocalAbsoluteTolerance, ref cancel);
191 solver.Minimize();
192 if (cancel) return;
193
194 Iterations += 1;
195 }
196
197 // Polish the final result
198 if (Polish == true)
199 {
200 solver = GetLocalOptimizer(BestParameterSet.Values, RelativeTolerance, AbsoluteTolerance, ref cancel);
201 solver.Minimize();
202 if (cancel) return;
203 }
204
205 // If we made it to here, the maximum iterations were reached.
206 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
207
208 }

217 private Optimizer GetLocalOptimizer(IList<double> initialValues, double relativeTolerance, double absoluteTolerance, ref bool cancel)
218 {
219 bool _cancel = false;
220 Optimizer solver = null;
221
222 // Make sure the parameters are within the bounds.
223 for (int i = 0; i < NumberOfParameters; i++)
224 initialValues[i] = RepairParameter(initialValues[i], LowerBounds[i], UpperBounds[i]);
225
226 if (Method == LocalMethod.BFGS)
227 {
228 solver = new BFGS((x) => Evaluate(x, ref _cancel), NumberOfParameters, initialValues, LowerBounds, UpperBounds) { RelativeTolerance = relativeTolerance, AbsoluteTolerance = absoluteTolerance };
229 }
230 else if (Method == LocalMethod.NelderMead)
231 {
232 solver = new NelderMead((x) => Evaluate(x, ref _cancel), NumberOfParameters, initialValues, LowerBounds, UpperBounds) { RelativeTolerance = relativeTolerance, AbsoluteTolerance = absoluteTolerance };
233 }
234 else if (Method == LocalMethod.Powell)
235 {
236 solver = new Powell((x) => Evaluate(x, ref _cancel), NumberOfParameters, initialValues, LowerBounds, UpperBounds) { RelativeTolerance = relativeTolerance, AbsoluteTolerance = absoluteTolerance };
237 }
238 cancel = _cancel;
239 return solver;
240 }
