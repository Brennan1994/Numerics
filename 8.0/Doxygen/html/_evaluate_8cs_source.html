Numerics 1.0.0
Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

74 public static double Polynomial(double[] coefficients, double x)
75 {
76 int n = coefficients.Length;
77 double value = coefficients[n - 1];
78 for (int i = n - 2; i >= 0; i -= 1)
79 {
80 value *= x;
81 value += coefficients[i];
82 }
83
84 return value;
85 }

100 public static double PolynomialRev(double[] coefficients, double x, int n = -1)
101 {
102 if (n > coefficients.Length)
103 {
104 throw new ArgumentOutOfRangeException("n cannot be greater than the number of coefficients");
105 }
106 else if (n == -1 || n == coefficients.Length)
107 {
108 n = coefficients.Length - 1;
109 }
110
111 double value = coefficients[0];
112 for (int i = 1; i <= n; i++)
113 {
114 value *= x;
115 value += coefficients[i];
116 }
117
118 return value;
119 }
