209 public void Estimate(IList<double> sample, ParameterEstimationMethod estimationMethod)
210 {
211 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
212 {
213 SetParameters(Statistics.StandardDeviation(sample));
214 }
215 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
216 {
217 // https://en.wikipedia.org/wiki/Rayleigh_distribution
218 // compute the biased estimator first
219 double sum = 0d;
220 for (int i = 0; i < sample.Count; i++)
221 sum += Math.Pow(sample[i], 2d);
222 double n = sample.Count;
223 double c = 1d / (2d * n);
224 double biased = Math.Sqrt(c * sum);
225 // then compute the biased corrected estimator
226 double num = Gamma.LogGamma(n) + 0.5d * Math.Log(n);
227 double den = Gamma.LogGamma(n + 0.5d);
228 double correction = Math.Exp(num - den);
229 if (double.IsPositiveInfinity(num) && double.IsPositiveInfinity(den))
230 {
231 correction = 1d;
232 }
233
234 SetParameters(biased * correction);
235 }
236 else
237 {
238 throw new NotImplementedException();
239 }
240 }

280 public override double PDF(double x)
281 {
282 // Validate parameters
283 if (_parametersValid == false)
284 ValidateParameters(Sigma, true);
285 if (x < Minimum) return 0.0d;
286 return x / (Sigma * Sigma) * Math.Exp(-x * x / (2.0d * Sigma * Sigma));
287 }
288
290 public override double CDF(double x)
291 {
292 // Validate parameters
293 if (_parametersValid == false)
294 ValidateParameters(Sigma, true);
295 if (x <= Minimum)
296 return 0d;
297 return 1.0d - Math.Exp(-x * x / (2.0d * Sigma * Sigma));
298 }
299
301 public override double InverseCDF(double probability)
302 {
303 // Validate probability
304 if (probability < 0.0d || probability > 1.0d)
305 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
306 if (probability == 0.0d)
307 return Minimum;
308 if (probability == 1.0d)
309 return Maximum;
310 // Validate parameters
311 if (_parametersValid == false)
312 ValidateParameters(Sigma, true);
313 return Sigma * Math.Sqrt(-2 * Math.Log(1d - probability));
314 }
