76 public static double Function(int x)
77 {
78 if (x < 0)
79 {
80 throw new ArgumentOutOfRangeException(nameof(x), "X must be non-negative.");
81 }
82
83 if (x < _factorialCache.Length)
84 return _factorialCache[x];
85 return double.PositiveInfinity;
86 }

95 public static double Function(long x)
96 {
97 if (x < 0L)
98 {
99 throw new ArgumentOutOfRangeException(nameof(x), "X must be non-negative.");
100 }
101
102 if (x == 0L)
103 return 1.0d;
104 long r = x;
105 while (System.Threading.Interlocked.Decrement(ref x) > 1L)
106 r *= x;
107 return r;
108 }

117 public static double LogFactorial(int x)
118 {
119 if (x < 0)
120 {
121 throw new ArgumentOutOfRangeException(nameof(x), "X must be non-negative.");
122 }
123
124 if (x <= 0)
125 return 0.0d;
126 if (x < _factorialCache.Length)
127 return Math.Log(_factorialCache[x]);
128 return Gamma.LogGamma(x + 1.0d);
129 }

145 public static double BinomialCoefficient(int n, int k)
146 {
147 if (k < 0 || n < 0 || k > n)
148 return 0.0d;
149 return Math.Floor(0.5d + Math.Exp(LogFactorial(n) - LogFactorial(k) - LogFactorial(n - k)));
150 }

160 private static IEnumerable<int[]> FindCombosRecursive(int[] buffer, int done, int begin, int end)
161 {
162 for (int i = begin; i < end; i++)
163 {
164 buffer[done] = i;
165
166 if (done == buffer.Length - 1)
167 yield return buffer;
168 else
169 foreach (int[] child in FindCombosRecursive(buffer, done + 1, i + 1, end))
170 yield return child;
171 }
172 }

192 public static int[,] AllCombinations(int n)
193 {
194 int f = (int)Math.Pow(2, n) - 1;
195 var output = new int[f, n];
196 int t = 0;
197 for (int i = 1; i <= n; i++)
198 {
199 foreach (int[] c in FindCombinations(i, n))
200 {
201 for (int j = 0; j < c.Length; j++)
202 {
203 output[t, c[j]] = 1;
204 }
205 t++;
206 }
207 }
208 return output;
209 }
