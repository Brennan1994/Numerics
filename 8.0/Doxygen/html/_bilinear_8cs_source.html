Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

148 public double Interpolate(double x1, double x2)
149 {
150 // Handle edge cases where x1 and/or x2 are out of range.
151 // Both out of range
152 if ((SortOrder == SortOrder.Ascending && x1 < X1Values[0] && x2 < X2Values[0]) ||
153 (SortOrder == SortOrder.Descending && x1 > X1Values[0] && x2 > X2Values[0])) return YValues[0, 0]; // Top left
154 if ((SortOrder == SortOrder.Ascending && x1 < X1Values[0] && x2 > X2Values[X2LI.Count - 1]) ||
155 (SortOrder == SortOrder.Descending && x1 > X1Values[0] && x2 < X2Values[X2LI.Count - 1])) return YValues[0, X2LI.Count - 1]; // Top Right
156 if ((SortOrder == SortOrder.Ascending && x1 > X1Values[X1LI.Count - 1] && x2 > X2Values[X2LI.Count - 1]) ||
157 (SortOrder == SortOrder.Descending && x1 < X1Values[X1LI.Count - 1] && x2 < X2Values[X2LI.Count - 1])) return YValues[X1LI.Count - 1, X2LI.Count - 1]; // Bottom Right
158 if ((SortOrder == SortOrder.Ascending && x1 > X1Values[X1LI.Count - 1] && x2 < X2Values[0]) ||
159 (SortOrder == SortOrder.Descending && x1 < X1Values[X1LI.Count - 1] && x2 > X2Values[0])) return YValues[X1LI.Count - 1, 0]; // Bottom Left
160
161 double y, t, u, x1i, x1ii, x2j, x2jj, yij, yiij, yiijj, yijj, x1lb, x1ub, x2lb, x2ub;
162 int i = X1LI.Search(x1);
163 int j = X2LI.Search(x2);
164
165 // Get x1 transform
166 x1i = X1LI.XValues[i];
167 x1ii = X1LI.XValues[i + 1];
168 x1lb = X1Values[0];
169 x1ub = X1Values[X1LI.Count - 1];
170 if (X1Transform == Transform.Logarithmic)
171 {
172 x1 = Math.Log10(x1);
173 x1i = Math.Log10(x1i);
174 x1ii = Math.Log10(x1ii);
175 x1lb = Math.Log10(x1lb);
176 x1ub = Math.Log10(x1ub);
177 }
178 else if (X1Transform == Transform.NormalZ)
179 {
180 x1 = Normal.StandardZ(x1);
181 x1i = Normal.StandardZ(x1i);
182 x1ii = Normal.StandardZ(x1ii);
183 x1lb = Normal.StandardZ(x1lb);
184 x1ub = Normal.StandardZ(x1ub);
185 }
186
187 // Get x2 transform
188 x2j = X2LI.YValues[j];
189 x2jj = X2LI.YValues[j + 1];
190 x2lb = X2Values[0];
191 x2ub = X2Values[X2LI.Count - 1];
192 if (X2Transform == Transform.Logarithmic)
193 {
194 x2 = Math.Log10(x2);
195 x2j = Math.Log10(x2j);
196 x2jj = Math.Log10(x2jj);
197 x2lb = Math.Log10(x2lb);
198 x2ub = Math.Log10(x2ub);
199 }
200 else if (X2Transform == Transform.NormalZ)
201 {
202 x2 = Normal.StandardZ(x2);
203 x2j = Normal.StandardZ(x2j);
204 x2jj = Normal.StandardZ(x2jj);
205 x2lb = Normal.StandardZ(x2lb);
206 x2ub = Normal.StandardZ(x2ub);
207 }
208
209 // Get y transform
210 yij = YValues[i, j];
211 yiij = YValues[i + 1, j];
212 yiijj = YValues[i + 1, j + 1];
213 yijj = YValues[i, j + 1];
214 if (YTransform == Transform.Logarithmic)
215 {
216 yij = Math.Log10(yij);
217 yiij = Math.Log10(yiij);
218 yiijj = Math.Log10(yiijj);
219 yijj = Math.Log10(yijj);
220 }
221 else if (YTransform == Transform.NormalZ)
222 {
223 yij = Normal.StandardZ(yij);
224 yiij = Normal.StandardZ(yiij);
225 yiijj = Normal.StandardZ(yiijj);
226 yijj = Normal.StandardZ(yijj);
227 }
228
229 // Interpolate
230 t = (x1 - x1i) / (x1ii - x1i);
231 u = (x2 - x2j) / (x2jj - x2j);
232 y = (1d - t) * (1d - u) * yij + t * (1d - u) * yiij + t * u * yiijj + (1d - t) * u * yijj;
233
234 // x1 out of range - 1D linear interpolation
235 if ((SortOrder == SortOrder.Ascending && x1 < x1lb) ||
236 (SortOrder == SortOrder.Descending && x1 > x1lb)) y = yij + (x2 - x2j) / (x2jj - x2j) * (yijj - yij);
237 if ((SortOrder == SortOrder.Ascending && x1 > x1ub) ||
238 (SortOrder == SortOrder.Descending && x1 < x1ub)) y = yiij + (x2 - x2j) / (x2jj - x2j) * (yiijj - yiij);
239
240 // x2 out of range - 1D linear interpolation
241 if ((SortOrder == SortOrder.Ascending && x2 < x2lb) ||
242 (SortOrder == SortOrder.Descending && x2 > x2lb)) y = yij + (x1 - x1i) / (x1ii - x1i) * (yiij - yij);
243 if ((SortOrder == SortOrder.Ascending && x2 > x2ub) ||
244 (SortOrder == SortOrder.Descending && x2 < x2ub)) y = yijj + (x1 - x1i) / (x1ii - x1i) * (yiijj - yijj);
245
246 // Back transform y
247 if (YTransform == Transform.Logarithmic)
248 {
249 y = Math.Pow(10d, y);
250 }
251 else if (YTransform == Transform.NormalZ)
252 {
253 y = Normal.StandardCDF(y);
254 }
255
256 return y;
257 }
