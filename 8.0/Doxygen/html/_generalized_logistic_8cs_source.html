178 public override double Mean
179 {
180 get
181 {
182 if (Math.Abs(Kappa) <= NearZero)
183 {
184 return Xi;
185 }
186 else if (Math.Abs(Kappa) < 1d)
187 {
188 double U1 = Gamma.Function(1d + Kappa) * Gamma.Function(1d - Kappa);
189 return Xi + Alpha / Kappa * (1d - U1);

221 public override double StandardDeviation
222 {
223 get
224 {
225 if (Math.Abs(Kappa) <= NearZero)
226 {
227 return Alpha * Math.PI / Math.Sqrt(3d);
228 }
229 else if (Math.Abs(Kappa) < 0.5d)
230 {
231 double U1 = Gamma.Function(1d + Kappa) * Gamma.Function(1d - Kappa);
232 double U2 = Gamma.Function(1d + 2d * Kappa) * Gamma.Function(1d - 2d * Kappa);
233 return Math.Sqrt(Math.Pow(Alpha, 2d) / Math.Pow(Kappa, 2d) * (U2 - Math.Pow(U1, 2d)));

243 public override double Skewness
244 {
245 get
246 {
247 if (Math.Abs(Kappa) <= NearZero)
248 {
249 return 0.0d;
250 }
251 else if (Math.Abs(Kappa) < 1d / 3d)
252 {
253 double U1 = Gamma.Function(1d + Kappa) * Gamma.Function(1d - Kappa);
254 double U2 = Gamma.Function(1d + 2d * Kappa) * Gamma.Function(1d - 2d * Kappa);
255 double U3 = Gamma.Function(1d + 3d * Kappa) * Gamma.Function(1d - 3d * Kappa);
256 return Math.Sign(Kappa) * (-U3 + 3d * U1 * U2 - 2d * Math.Pow(U1, 3d)) / Math.Pow(U2 - Math.Pow(U1, 2d), 3d / 2d);
257 }
258 else
259 {
260 return double.NaN;
261 }

266 public override double Kurtosis
267 {
268 get
269 {
270 if (Math.Abs(Kappa) <= NearZero)
271 {
272 return 3d + 6d / 5d;
273 }
274 else if (Math.Abs(Kappa) < 0.25d)
275 {
276 double U1 = Gamma.Function(1d + Kappa) * Gamma.Function(1d - Kappa);
277 double U2 = Gamma.Function(1d + 2d * Kappa) * Gamma.Function(1d - 2d * Kappa);
278 double U3 = Gamma.Function(1d + 3d * Kappa) * Gamma.Function(1d - 3d * Kappa);
279 double U4 = Gamma.Function(1d + 4d * Kappa) * Gamma.Function(1d - 4d * Kappa);
280 double kNum = U4 - 4d * U3 * U1 - 3d * Math.Pow(U2, 2d) + 12d * U2 * Math.Pow(U1, 2d) - 6d * Math.Pow(U1, 4d);
281 double kDen = Math.Pow(U2 - Math.Pow(U1, 2d), 2d);
282 return 3 + kNum / kDen;
283 }
284 else
285 {
286 return double.NaN;
287 }

428 public double[] DirectMethodOfMoments(IList<double> moments)
429 {
430 // Solve for kappa
431 double k = SolveForKappa(moments[2]);
432 double a;
433 double x;
434 if (Math.Abs(k) <= NearZero)
435 {
436 x = moments[0];
437 a = moments[1] * Math.Sqrt(3d) / Math.PI;
438 }
439 else
440 {
441 double U1 = Gamma.Function(1d + k) * Gamma.Function(1d - k);
442 double U2 = Gamma.Function(1d + 2d * k) * Gamma.Function(1d - 2d * k);
443 a = Math.Sqrt(moments[1] * moments[1] * k * k / (U2 - Math.Pow(U1, 2d)));
444 x = moments[0] - a / k * (1d - U1);
445 }
446 // return parameters
447 return [x, a, k];
448 }

469 public double SolveForKappa(double skew)
470 {
471 if (Math.Abs(skew) < 10d)
472 {
473 // Kappa must be solved for. The Brent method is used here.
474 return Brent.Solve((x) =>
475 {
476 double U1 = Gamma.Lanczos(1d + x) * Gamma.Lanczos(1d - x);
477 double U2 = Gamma.Lanczos(1d + 2d * x) * Gamma.Lanczos(1d - 2d * x);
478 double U3 = Gamma.Lanczos(1d + 3d * x) * Gamma.Lanczos(1d - 3d * x);
479 double k = Math.Sign(x) * (-U3 + 3d * U1 * U2 - 2d * Math.Pow(U1, 3d)) / Math.Pow(U2 - Math.Pow(U1, 2d), 3d / 2d);
480 return k - skew;
481
482 }, -(1d / 3d), 1d / 3d);
483 }
484 else
485 {
486 return double.NaN;
487 }

491 public double[] ParametersFromLinearMoments(IList<double> moments)
492 {
493 double L1 = moments[0];
494 double L2 = moments[1];
495 double T3 = moments[2];
496 double T4 = moments[3];
497 double kappa = -T3;
498 double alpha = L2 * Math.Sin(kappa * Math.PI) / (kappa * Math.PI);
499 double xi = L1 - alpha * (1.0d / kappa - Math.PI / Math.Sin(kappa * Math.PI));
500 return [xi, alpha, kappa];
501 }

504 public double[] LinearMomentsFromParameters(IList<double> parameters)
505 {
506 double xi = parameters[0];
507 double alpha = parameters[1];
508 double kappa = parameters[2];
509 if (Math.Abs(kappa) >= 1.0d)
510 throw new ArgumentOutOfRangeException(nameof(Kappa), "L-moments can only be defined for -1 < kappa < 1.");
511 double L1 = xi + alpha * (1.0d / kappa - Math.PI / Math.Sin(kappa * Math.PI));
512 double L2 = alpha * kappa * Math.PI / Math.Sin(kappa * Math.PI);
513 double T3 = -kappa;
514 double T4 = (1.0d + 5.0d * Math.Pow(kappa, 2.0d)) / 6.0d;
515 return [L1, L2, T3, T4];
516 }

519 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
520 {
521 // Estimate initial values using the method of moments (a.k.a product moments).
522 var initialVals = new double[NumberOfParameters];
523 var lowerVals = new double[NumberOfParameters];
524 var upperVals = new double[NumberOfParameters];
525 // Get initial values
526 // initialVals = DirectMethodOfMoments(Statistics.ComputeProductMoments(sample))
527 initialVals = ParametersFromLinearMoments(Statistics.LinearMoments(sample));
528 // Get bounds of location
529 if (initialVals[0] == 0d) initialVals[0] = Tools.DoubleMachineEpsilon;
530 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
531 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
532 // Get bounds of scale
533 lowerVals[1] = Tools.DoubleMachineEpsilon;
534 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[1]))) + 1d);
535 // Get bounds of shape
536 lowerVals[2] = -10;
537 upperVals[2] = 10d;
538 // Correct initial value of kappa if necessary
539 if (initialVals[2] <= lowerVals[2] || initialVals[2] >= upperVals[2])
540 {
541 initialVals[2] = 0d;
542 }
543 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
544 }

547 public double[] MLE(IList<double> sample)
548 {
549 // Set constraints
550 var tuple = GetParameterConstraints(sample);
551 var Initials = tuple.Item1;
552 var Lowers = tuple.Item2;
553 var Uppers = tuple.Item3;
554
555 // Solve using Nelder-Mead (Downhill Simplex)
556 double logLH(double[] x)
557 {
558 var GLO = new GeneralizedLogistic();
559 GLO.SetParameters(x);
560 return GLO.LogLikelihood(sample);
561 }
562 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
563 solver.ReportFailure = true;
564 solver.Maximize();
565 return solver.BestParameterSet.Values;
566 }

569 public override double PDF(double x)
570 {
571 if (_parametersValid == false)
572 ValidateParameters(Xi, Alpha, Kappa, true);
573 if (x < Minimum || x > Maximum) return 0.0d;
574 double y = (x - Xi) / Alpha;
575 if (Math.Abs(Kappa) > NearZero)
576 y = -Math.Log(1d - Kappa * y) / Kappa;
577 return 1d / Alpha * Math.Exp(-(1d - Kappa) * y) / Math.Pow(1d + Math.Exp(-y), 2d);
578 }

581 public override double CDF(double x)
582 {
583 if (_parametersValid == false)
584 ValidateParameters(Xi, Alpha, Kappa, true);
585 if (x <= Minimum)
586 return 0d;
587 if (x >= Maximum)
588 return 1d;
589 double y = (x - Xi) / Alpha;
590 if (Math.Abs(Kappa) > NearZero)
591 y = -Math.Log(1d - Kappa * y) / Kappa;
592 return 1d / (1d + Math.Exp(-y));
593 }

596 public override double InverseCDF(double probability)
597 {
598 // Validate probability
599 if (probability < 0.0d || probability > 1.0d)
600 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
601 if (probability == 0.0d)
602 return Minimum;
603 if (probability == 1.0d)
604 return Maximum;
605 // Validate parameters
606 if (_parametersValid == false)
607 ValidateParameters(Xi, Alpha, Kappa, true);
608 if (Math.Abs(Kappa) <= NearZero)
609 {
610 return Xi - Alpha * Math.Log((1d - probability) / probability);
611 }
612 else
613 {
614 return Xi + Alpha / Kappa * (1d - Math.Pow((1d - probability) / probability, Kappa));
615 }

637 public double[] QuantileGradient(double probability)
638 {
639 if (_parametersValid == false)
640 ValidateParameters(Xi, _alpha, Kappa, true);
641 double a = Alpha;
642 double k = Kappa;
643 var gradient = new double[3];
644 gradient[0] = 1.0d; // location
645 gradient[1] = 1d / k * (1d - Math.Pow((1d - probability) / probability, k)); // scale
646 gradient[2] = -(a / (k * k)) * (1d - Math.Pow((1d - probability) / probability, k)) - a / k * Math.Pow((1d - probability) / probability, k) * Math.Log((1d - probability) / probability); // shape
647 return gradient;
648 }

651 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
652 {
653 if (probabilities.Count != NumberOfParameters)
654 {
655 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
656 }
657 // Get gradients
658 var dQp1 = QuantileGradient(probabilities[0]);
659 var dQp2 = QuantileGradient(probabilities[1]);
660 var dQp3 = QuantileGradient(probabilities[2]);
661 // Compute determinant
662 // |a b c|
663 // |d e f|
664 // |g h i|
665 // |A| = a(ei − fh) − b(di − fg) + c(dh − eg)
666 double a = dQp1[0];
667 double b = dQp1[1];
668 double c = dQp1[2];
669 double d = dQp2[0];
670 double e = dQp2[1];
671 double f = dQp2[2];
672 double g = dQp3[0];
673 double h = dQp3[1];
674 double i = dQp3[2];
675 determinant = a * (e * i - f * h) - b * (d * i - f * g) + c * (d * h - e * g);
676 // Return Jacobian
677 var jacobian = new double[,] { { a, b, c }, { d, e, f }, { g, h, i } };
678 return jacobian;
679 }
