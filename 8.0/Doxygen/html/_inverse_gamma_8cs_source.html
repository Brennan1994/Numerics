The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

262 public override double PDF(double x)
263 {
264 // Validate parameters
265 if (_parametersValid == false)
266 ValidateParameters([Beta, Alpha], true);
267 if (x < Minimum || x > Maximum) return 0.0d;
268 return Math.Pow(Beta, Alpha) / Gamma.Function(Alpha) * Math.Pow(x, -Alpha - 1d) * Math.Exp(-Beta / x);
269 }

272 public override double CDF(double x)
273 {
274 // Validate parameters
275 if (_parametersValid == false)
276 ValidateParameters([Beta, Alpha], true);
277 if (x <= Minimum)
278 return 0d;
279 if (x >= Maximum)
280 return 1d;
281 return Gamma.UpperIncomplete(Alpha, Beta / x);
282 }

285 public override double InverseCDF(double probability)
286 {
287 // Validate probability
288 if (probability < 0.0d || probability > 1.0d)
289 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
290 if (probability == 0.0d)
291 return Minimum;
292 if (probability == 1.0d)
293 return Maximum;
294 // Validate parameters
295 if (_parametersValid == false)
296 ValidateParameters([Beta, Alpha], true);
297 return Beta / Gamma.InverseUpperIncomplete(Alpha, probability);
298 }
