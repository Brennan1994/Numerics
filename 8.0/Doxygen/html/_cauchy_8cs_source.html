218 public override ArgumentOutOfRangeException ValidateParameters(IList<double> parameters, bool throwException)

237 public override double PDF(double x)
238 {
239 // Validate parameters
240 if (_parametersValid == false)
241 ValidateParameters([X0, Gamma], true);
242 double z = (x - X0) / Gamma;
243 return 1.0d / (Math.PI * Gamma * (1.0d + z * z));
244 }
245
247 public override double CDF(double x)
248 {
249 // Validate parameters
250 if (_parametersValid == false)
251 ValidateParameters([X0, Gamma], true);
252 return 1.0d / Math.PI * Math.Atan2(x - X0, Gamma) + 0.5d;
253 }
254
256 public override double InverseCDF(double probability)
257 {
258 // Validate probability
259 if (probability < 0.0d || probability > 1.0d)
260 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
261 if (probability == 0.0d)
262 return Minimum;
263 if (probability == 1.0d)
264 return Maximum;
265 // Validate parameters
266 if (_parametersValid == false)
267 ValidateParameters([X0, Gamma], true);
268 return X0 + Gamma * Math.Tan(Math.PI * (probability - 0.5d));
269 }
