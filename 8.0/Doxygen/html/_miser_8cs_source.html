76 public Miser(Func<double[], double> function, int dimensions, IList<double> min, IList<double> max)
77 {
78 if (dimensions < 1) throw new ArgumentOutOfRangeException(nameof(dimensions), "There must be at least 1 dimension to evaluate.");
79
80 // Check if the length of the min and max values equal the number of dimensions
81 if (min.Count != dimensions || max.Count != dimensions)
82 {
83 throw new ArgumentOutOfRangeException(nameof(min), "The minimum and maximum values must be the same length as the number of dimensions.");
84 }
85
86 // Check if the minimum values are less than the maximum values
87 for (int i = 0; i < min.Count; i++)
88 {
89 if (max[i] <= min[i])
90 {
91 throw new ArgumentOutOfRangeException(nameof(max), "The maximum values cannot be less than or equal to the minimum values.");
92 }
93 }
94
95 Function = function ?? throw new ArgumentNullException(nameof(function), "The function cannot be null.");
96 Dimensions = dimensions;
97 Min = min.ToArray();
98 Max = max.ToArray();
99 Random = new MersenneTwister();
100 _sobol = new SobolSequence(Dimensions);
101
102 }

162 public override void Integrate()
163 {
164 ClearResults();
165 Validate();
166
167 try
168 {
169 // Create region array
170 var regn = new double[2 * Dimensions];
171 for (int i = 0; i < Dimensions; i++)
172 regn[i] = Min[i];
173 for (int i = Dimensions; i < 2 * Dimensions; i++)
174 regn[i] = Max[i - Dimensions];
175
176 double volume = 1;
177 for (int i = 0; i < Dimensions; i++)
178 volume *= (Max[i] - Min[i]);
179
180 // Compute
181 double result = 0;
182 double error = 0;
183
184 miser(Function, regn, MaxFunctionEvaluations, Dither, ref result, ref error);
185 Result = result * volume;
186 StandardError = Math.Sqrt(error) * volume;
187
188 }
189 catch (Exception ex)
190 {
191 UpdateStatus(IntegrationStatus.Failure, ex);
192 }
193
194 }

207 private void miser(Func<double[], double> function, double[] regn, int npts, double dith, ref double ave, ref double var)
208 {
209 // Monte Carlo samples a user-supplied ndim-dimensional function func in a rectangular volume
210 // specified by regn[0..2 * ndim - 1], a vector consisting of ndim “lower - left” coordinates of the
211 // region followed by ndim “upper - right” coordinates. The function is sampled a total of npts
212 // times, at locations determined by the method of recursive stratified sampling. The mean value
213 // of the function in the region is returned as ave; an estimate of the statistical uncertainty of ave
214 // (square of standard deviation) is returned as var. The input parameter dith should normally
215 // be set to zero, but can be set to(e.g.) 0.1 if func’s active region falls on the boundary of a
216 // power-of-two subdivision of region.
217
218 int MNPT = MinimumNumberOfSubregionPoints, MNBS = MinimumNumberOfBisections;
219 double PFAC = Fraction, TINY = 1.0e-30, BIG = 1.0e30;
220
221 // Here PFAC is the fraction of remaining function evaluations used at each stage to explore
222 // the variance of func. At least MNPT function evaluations are performed in any terminal
223 // subregion; a subregion is further bisected only if at least MNBS function evaluations are
224 // available. We take MNBS = 4 ␃* MNPT.
225
226 int iran = 0;
227 int j, jb, n, ndim, npre, nptl, nptr;
228 double avel = 0, varl = 0, fracl, fval, rgl, rgm, rgr, s, sigl, siglb, sigr, sigrb;
229 double sum, sumb, summ, summ2;
230
231 ndim = Dimensions;
232 var pt = new double[ndim];
233 if (npts < MNBS)
234 {
235 // Too few points to bisect, do straight Monte Carlo
236 summ = summ2 = 0.0;
237 for (n = 0; n < npts; n++)
238 {
239 ranpt(pt, regn);
240 fval = function(pt);
241 FunctionEvaluations++;
242 summ += fval;
243 summ2 += fval * fval;
244 }
245 ave = summ / npts;
246 var = Math.Max(TINY, (summ2 - summ * summ / npts) / (npts * npts));
247 }
248 else
249 {
250 // Do the preliminary (uniform sampling)
251 var rmid = new double[ndim];
252 npre = Math.Max((int)(npts * PFAC), MNPT);
253 var fmaxl = new double[ndim];
254 var fmaxr = new double[ndim];
255 var fminl = new double[ndim];
256 var fminr = new double[ndim];
257 // Initialize the left and right bounds for each dimension
258 for (j = 0; j < ndim; j++)
259 {
260 iran = (iran * 2661 + 36979) % 175000;
261 s = Tools.Sign(dith, (double)(iran - 87500));
262 rmid[j] = (0.5 + s) * regn[j] + (0.5 - s) * regn[ndim + j];
263 fminl[j] = fminr[j] = BIG;
264 fmaxl[j] = fmaxr[j] = (-BIG);
265 }
266 // Loop over the points in the sample
267 for (n = 0; n < npre; n++)
268 {
269 ranpt(pt, regn);
270 fval = function(pt);
271 FunctionEvaluations++;
272 // Find the left and right bounds for each dimension
273 for (j = 0; j < ndim; j++)
274 {
275 if (pt[j] <= rmid[j])
276 {
277 fminl[j] = Math.Min(fminl[j], fval);
278 fmaxl[j] = Math.Max(fmaxl[j], fval);
279 }
280 else
281 {
282 fminr[j] = Math.Min(fminr[j], fval);
283 fmaxr[j] = Math.Max(fmaxr[j], fval);
284 }
285 }
286 }
287 // Choose which dimension jb to bisect
288 sumb = BIG;
289 jb = -1;
290 siglb = sigrb = 1.0;
291 for (j = 0; j < ndim; j++)
292 {
293 if (fmaxl[j] > fminl[j] && fmaxr[j] > fminr[j])
294 {
295 sigl = Math.Max(TINY, Math.Pow(fmaxl[j] - fminl[j], 2.0 / 3.0));
296 sigr = Math.Max(TINY, Math.Pow(fmaxr[j] - fminr[j], 2.0 / 3.0));
297 sum = sigl + sigr; // Equation (7.9.24), see text.
298 if (sum <= sumb)
299 {
300 sumb = sum;
301 jb = j;
302 siglb = sigl;
303 sigrb = sigr;
304 }
305 }
306 }
307 if (jb == -1) jb = (ndim * iran) / 175000; // MNPT may be too small
308 // Apportion the remain points between left and right
309 rgl = regn[jb];
310 rgm = rmid[jb];
311 rgr = regn[ndim + jb];
312 fracl = Math.Abs((rgm - rgl) / (rgr - rgl));
313 nptl = (int)(MNPT + (npts - npre - 2 * MNPT) * fracl * siglb / (fracl * siglb + (1.0 - fracl) * sigrb)); // Equation (7.9.23)
314 nptr = npts - npre - nptl;
315 // Now allocate and integrate the two sub-regions
316 var regn_temp = new double[2 * ndim];
317 for (j = 0; j < ndim; j++)
318 {
319 regn_temp[j] = regn[j];
320 regn_temp[ndim + j] = regn[ndim + j];
321 }
322 regn_temp[ndim + jb] = rmid[jb];
323 miser(function, regn_temp, nptl, dith, ref avel, ref varl);
324 regn_temp[jb] = rmid[jb];
325 regn_temp[ndim + jb] = regn[ndim + jb];
326 miser(function, regn_temp, nptr, dith, ref ave, ref var);
327 ave = fracl * avel + (1 - fracl) * ave;
328 var = fracl * fracl * varl + (1 - fracl) * (1 - fracl) * var;
329 // Combine left and right regions by equation (7.9.11)
330 }
331 }

339 private void ranpt(double[] pt, double[] regn)
340 {
341 int j, n = pt.Length;
342 double[] rnd = null;
343 if (UseSobolSequence)
344 rnd = _sobol.NextDouble();
345
346 for (j = 0; j < n; j++)
347 pt[j] = regn[j] + (regn[n + j] - regn[j]) * (UseSobolSequence ? rnd[j] : Random.NextDouble());
348 }
