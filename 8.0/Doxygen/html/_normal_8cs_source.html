245 public void Estimate(IList<double> sample, ParameterEstimationMethod estimationMethod)
246 {
247 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
248 {
249 SetParameters(Statistics.ProductMoments(sample));
250 }
251 else if (estimationMethod == ParameterEstimationMethod.MethodOfLinearMoments)
252 {
253 SetParameters(ParametersFromLinearMoments(Statistics.LinearMoments(sample)));
254 }
255 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
256 {
257 // The only difference between this method and MOM is that the population standard deviation is used
258 // rather than the sample; Note: the maximum likelihood estimate of the standard deviation
259 // is a biased estimator.
260 SetParameters(MLE(sample));
261 }
262 else
263 {
264 throw new NotImplementedException();
265 }
266 }

362 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
363 {
364 var initialVals = new double[NumberOfParameters];
365 var lowerVals = new double[NumberOfParameters];
366 var upperVals = new double[NumberOfParameters];
367 // Estimate initial values using the method of moments (a.k.a product moments).
368 var moments = Statistics.ProductMoments(sample);
369 initialVals[0] = moments[0];
370 initialVals[1] = moments[1];
371 // Get bounds of mean
372 if (initialVals[0] == 0d) initialVals[0] = Tools.DoubleMachineEpsilon;
373 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
374 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
375 // Get bounds of standard deviation
376 lowerVals[1] = Tools.DoubleMachineEpsilon;
377 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[1]) + 1d));
378 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
379 }

382 public double[] MLE(IList<double> sample)
383 {
384 // Set constraints
385 var tuple = GetParameterConstraints(sample);
386 var Initials = tuple.Item1;
387 var Lowers = tuple.Item2;
388 var Uppers = tuple.Item3;
389
390 // Solve using Nelder-Mead (Downhill Simplex)
391 double logLH(double[] x)
392 {
393 var N = new Normal();
394 N.SetParameters(x);
395 return N.LogLikelihood(sample);
396 }
397 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
398 solver.ReportFailure = true;
399 solver.Maximize();
400 return solver.BestParameterSet.Values;
401 }

433 private static double r8_normal_01_cdf_inverse(double p)
434 {
435 //****************************************************************************80
436 //
437 // Purpose:
438 //
439 // R8_NORMAL_01_CDF_INVERSE inverts the standard normal CDF.
440 //
441 // Discussion:
442 //
443 // The result is accurate to about 1 part in 10**16.
444 //
445 // Licensing:
446 //
447 // This code is distributed under the GNU LGPL license.
448 //
449 // Modified:
450 //
451 // 19 March 2010
452 //
453 // Author:
454 //
455 // Original FORTRAN77 version by Michael Wichura.
456 // C++ version by John Burkardt.
457 //
458 // Reference:
459 //
460 // Michael Wichura,
461 // The Percentage Points of the Normal Distribution,
462 // Algorithm AS 241,
463 // Applied Statistics,
464 // Volume 37, Number 3, pages 477-484, 1988.
465 //
466 // Parameters:
467 //
468 // Input, double P, the value of the cumulative probability
469 // density function. 0 < P < 1. If P is outside this range, an "infinite"
470 // value is returned.
471 //
472 // Output, double R8_NORMAL_01_CDF_INVERSE, the normal deviate value
473 // with the property that the probability of a standard normal deviate being
474 // less than or equal to this value is P.
475 //
476
477 double q;
478 double r;
479 double value;
480
481 if (p <= 0.0)
482 {
483 return double.NegativeInfinity;
484 }
485 if (1.0 <= p)
486 {
487 return double.PositiveInfinity;
488 }
489
490 q = p - 0.5;
491
492 // 0.075 <= p <= 0.925
493 if (Math.Abs(q) <= 0.425)
494 {
495 r = 0.180625 - q * q;
496 value = q * r8poly_value(8, a, r) / r8poly_value(8, b, r);
497 }
498 else
499 {
500 if (q < 0.0)
501 {
502 r = p;
503 }
504 else
505 {
506 r = 1.0 - p;
507 }
508
509 r = Math.Sqrt(-Math.Log(r));
510
511 if (r <= 5.0)
512 {
513 r = r - 1.6;
514 value = r8poly_value(8, c, r) / r8poly_value(8, d, r);
515 }
516 else
517 {
518 r = r - 5.0;
519 value = r8poly_value(8, e, r) / r8poly_value(8, f, r);
520 }
521
522 if (q < 0.0)
523 {
524 value = -value;
525 }
526
527 }
528
529 return value;
530 }

538 private static double r8poly_value(int n, double[] a, double x)
539 {
540 //****************************************************************************80
541 //
542 // Purpose:
543 //
544 // R8POLY_VALUE evaluates a double precision polynomial.
545 //
546 // Discussion:
547 //
548 // For sanity's sake, the value of N indicates the NUMBER of
549 // coefficients, or more precisely, the ORDER of the polynomial,
550 // rather than the DEGREE of the polynomial. The two quantities
551 // differ by 1, but cause a great deal of confusion.
552 //
553 // Given N and A, the form of the polynomial is:
554 //
555 // p(x) = a[0] + a[1] * x + ... + a[n-2] * x^(n-2) + a[n-1] * x^(n-1)
556 //
557 // Licensing:
558 //
559 // This code is distributed under the GNU LGPL license.
560 //
561 // Modified:
562 //
563 // 13 August 2004
564 //
565 // Author:
566 //
567 // John Burkardt
568 //
569 // Parameters:
570 //
571 // Input, int N, the order of the polynomial.
572 //
573 // Input, double A[N], the coefficients of the polynomial.
574 // A[0] is the constant term.
575 //
576 // Input, double X, the point at which the polynomial is to be evaluated.
577 //
578 // Output, double R8POLY_VALUE, the value of the polynomial at X.
579 //
580
581 int i;
582 double value = 0.0;
583 for (i = n - 1; 0 <= i; i--)
584 {
585 value = value * x + a[i];
586 }
587
588 return value;
589 }

592 public override double InverseCDF(double probability)
593 {
594 // Validate probability
595 if (probability < 0.0d || probability > 1.0d)
596 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
597 if (probability == 0.0d)
598 return Minimum;
599 if (probability == 1.0d)
600 return Maximum;
601 // Validate parameters
602 if (_parametersValid == false)
603 ValidateParameters(Mu, Sigma, true);
604 return Mu + Sigma * r8_normal_01_cdf_inverse(probability);
605 }

699 public double[,] NormalConfidenceIntervals(int sampleSize, IList<double> quantiles, IList<double> percentiles)
700 {
701 // validate parameters
702 if (_parametersValid == false)
703 ValidateParameters(Mu, _sigma, true);
704 // Dimension output array
705 int q = quantiles.Count;
706 int p = percentiles.Count;
707 var Output = new double[q, p];
708 for (int i = 0; i < q; i++)
709 {
710 double Xp = InverseCDF(quantiles[i]);
711 double Zp = StandardZ(quantiles[i]);
712 // Record percentiles for user-defined probabilities
713 for (int j = 0; j < p; j++)
714 {
715 double Z = StandardZ(percentiles[j]);
716 double VARp = Sigma * Sigma / sampleSize * (1.0d + 0.5d * Zp * Zp);
717 Output[i, j] = Xp + Z * Math.Sqrt(VARp);
718 }
719 }
720 // Return confidence percentile output
721 return Output;
722 }

766 public double[,] MonteCarloConfidenceIntervals(int sampleSize, int realizations, IList<double> quantiles, IList<double> percentiles)
767 {
768 // validate parameters
769 if (_parametersValid == false)
770 ValidateParameters(Mu, _sigma, true);
771 // Dimension output array
772 int q = quantiles.Count;
773 int p = percentiles.Count;
774 var Output = new double[q, p];
775
776 // Variables
777 double OriginalMean = Mu;
778 double OriginalStdDev = Sigma;
779
780 // Create random numbers for mean and standard deviation
781 var r = new MersenneTwister(12345);
782 var rndMean = r.NextDoubles(realizations);
783 r = new MersenneTwister(45678);
784 var rndStdDev = r.NextDoubles(realizations);
785
786 // Create list of Monte Carlo distributions
787 var MonteCarloDistributions = new UnivariateDistributionBase[realizations];
788 Parallel.For(0, realizations, idx =>
789 {
790
791 // Generate new mean
792 var Normal = new Normal(OriginalMean, OriginalStdDev / Math.Sqrt(sampleSize));
793 double NewMu = Normal.InverseCDF(rndMean[idx]);
794 // Generate new standard deviation
795 var Chi = new ChiSquared(sampleSize - 1);
796 double NewSigma = Math.Sqrt((sampleSize - 1) * Math.Pow(OriginalStdDev, 2d) / Chi.InverseCDF(rndStdDev[idx]));
797 // Create a new distribution with the new parameters
798 MonteCarloDistributions[idx] = new Normal(NewMu, NewSigma);
799 });
800 // Create confidence intervals
801 for (int i = 0; i < q; i++)
802 {
803 // Create array of X values across user-defined probabilities
804 var XValues = new double[realizations];
805 // Record X values
806 Parallel.For(0, realizations, idx => XValues[idx] = MonteCarloDistributions[idx].InverseCDF(quantiles[i]));
807 // Record percentiles for user-defined probabilities
808 for (int j = 0; j < p; j++)
809 Output[i, j] = Statistics.Percentile(XValues, percentiles[j]);
810 }
811 // Return confidence percentile output
812 return Output;
813 }

834 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
835 {
836 if (estimationMethod != ParameterEstimationMethod.MethodOfMoments &&
837 estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
838 {
839 throw new NotImplementedException();
840 }
841 // Compute covariance
842 double u2 = Sigma;
843 var covar = new double[2, 2];
844 covar[0, 0] = Math.Pow(u2, 2d) / sampleSize; // location
845 covar[1, 1] = 2d * Math.Pow(u2, 4d) / sampleSize; // scale
846 covar[0, 1] = 0.0;
847 covar[1, 0] = covar[0, 1];
848 return covar;
849 }

852 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
853 {
854 var covar = ParameterCovariance(sampleSize, estimationMethod);
855 var grad = QuantileGradient(probability);
856 double varA = covar[0, 0];
857 double varB = covar[1, 1];
858 double covAB = covar[1, 0];
859 double dQx1 = grad[0];
860 double dQx2 = grad[1];
861 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + 2d * dQx1 * dQx2 * covAB;
862 }

882 {

884 {
885 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
886 }
887 // Get gradients
888 var dQp1 = QuantileGradient(probabilities[0]);
889 var dQp2 = QuantileGradient(probabilities[1]);
890 // Compute determinant
891 // |a b|
892 // |c d|
893 // |A| = ad − bc
894 double a = dQp1[0];
895 double b = dQp1[1];
896 double c = dQp2[0];
897 double d = dQp2[1];
898 determinant = a * d - b * c;
899 // Return Jacobian
900 var jacobian = new double[,] { { a, b }, { c, d } };
