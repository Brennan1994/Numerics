76 public static int Function(double[] X)
77 {
78 double Alphaout = 0.005d;
79 double Alphain = 0.0d;
80 double Alphazeroin = 0.1d;
81 double maxFracLO = 0.5d;
82 int N = X.Count();
83 var zt = new double[N];
84 var pvaluew = new double[N];
85 for (int i = 0; i < N; i++)
86 {
87 zt[i] = Math.Log10(Math.Max(1.0E-88d, X[i]));
88 pvaluew[i] = -99.0d;
89 }
90
91 // sort log flows from smallest to largest
92 Array.Sort(zt);
93
94 // set starting point for MGBT search at approximate median position (1/2 N)
95 int n2 = (int)(N * maxFracLO);
96 double S1 = 0d;
97 double S2 = 0d;
98 for (int i = N; i >= n2 + 2; i -= 1)
99 {
100 S1 += zt[i - 1];
101 S2 += Math.Pow(zt[i - 1], 2d);
102 }
103
104 int NC;
105 double XV;
106 double XM;
107 var W = new double[N];
108 for (int i = n2; i >= 1; i -= 1)
109 {
110 S1 += zt[i];
111 S2 += Math.Pow(zt[i], 2d);
112 NC = N - i;
113 XM = S1 / NC;
114 XV = (S2 - NC * Math.Pow(XM, 2d)) / (NC - 1);
115 W[i - 1] = (zt[i - 1] - XM) / Math.Sqrt(XV);
116 pvaluew[i - 1] = GGBCRITP(N, i, W[i - 1]);
117 }
118
119 // Determine Number of low outliers in 2 Or 3 steps.
120 // Based on TAC original code And JRS recommendations.
121 //
122 // Step 1. Outward sweep from median (always done).
123 // alpha level of test = Alphaout
124 // number of outliers = J1
125 //
126 // Step 2. Inward sweep from largest low outlier identified in Step 1.
127 // alpha level of test = Alphain
128 // number of outliers = J2
129 //
130 // Step 3. Inward sweep from smallest observation
131 // alpha level of test = Alphazeroin
132 // number of outliers = J3
133
134 // Initialize counters
135 int J1 = 0; // Outward sweep number of low outliers
136 int J2 = 0; // Inward sweep number of low outliers
137 int J3 = 0; // Oth Inward sweep number of low outliers
138
139 // 1) Outward sweep check: Loop over low flows up to median
140
141 for (int i = n2; i >= 1; i -= 1)
142 {
143 if (pvaluew[i - 1] < Alphaout)
144 {
145 J1 = i;
146 break;
147 }
148 }
149
150 // 2) Inward sweep check with Alphain
151 J2 = J1;
152 for (int i = J1 + 1; i <= n2; i++)
153 {
154 if (pvaluew[i - 1] >= Alphain)
155 {
156 J2 = i - 1;
157 break;
158 }
159 }
160
161 // 3) Inward sweep check with Alphazeroin
162 for (int i = 1; i <= n2; i++)
163 {
164 if (pvaluew[i - 1] >= Alphazeroin)
165 {
166 J3 = i - 1;
167 break;
168 }
169 }
170
171 // Set number of low outliers as max of 3 sweeps
172 int MGBTP = Math.Max(J1, Math.Max(J2, J3));
173 return MGBTP;
174 }

184 private static double GGBCRITP(int N, int R, double ETA)
185 {
186 if (N < 10 | R > N / 2d)
187 {
188 return 0.5d;
189 }
190 else
191 {
192 _nIn = N;
193 _rIn = R;
194 _etaIn = ETA;
195 }
196
197 // The original FORTRAN source code utilized a globally adaptive Gauss-Kronrod integration method.
198 // This implementation, however, employs the adaptive Simpson's rule for integration.
199 // The number of low outliers computed by this method is consistent with the results from the FORTRAN code.
200 // Further testing is necessary to identify any edge cases where the adaptive Simpson's rule might prove insufficient.
201 var sr = new AdaptiveSimpsonsRule(FGGB, 1E-16, 1 - 1E-16);
202 sr.RelativeTolerance = 1E-4;
203 sr.MaxDepth = 50;
204 sr.ReportFailure = false;
205 sr.Integrate();
206 return sr.Status != IntegrationStatus.Failure ? sr.Result : double.NaN;
207 }

212 private static double FGGB(double PZR)
213 {
214 double df, MuM, MuS2, VarM, VarS2, CovMS2;
215 double EX1, EX2, EX3, EX4;
216 double CovMS, VarS, alpha, beta;
217 double MuMP, EtaP, H, Lambda, MuS, ncp, q, VarMP, PR, ZR, N2;
218 double ANS;
219 int N = _nIn;
220 int R = _rIn;
221 double ETA = _etaIn;
222
223 // Compute the value of the r-th smallest obs. based on its order statistic
224 N2 = N - R;
225 var betaDist = new BetaDistribution(R, N + 1 - R);
226 PR = betaDist.InverseCDF(PZR);
227 ZR = Normal.StandardZ(PR);
228
229 // Calculate the expected values of M, S2, S and their variances/covariances
230 H = Normal.StandardPDF(ZR) / Math.Max(0.0000000001d, 1.0d - PR);
231 EX1 = H;
232 EX2 = 1d + H * ZR;
233 EX3 = 2d * EX1 + H * Math.Pow(ZR, 2d);
234 EX4 = 3d * EX2 + H * Math.Pow(ZR, 3d);
235 MuM = EX1;
236 MuS2 = EX2 - Math.Pow(EX1, 2d);
237 VarM = MuS2 / N2;
238 VarS2 = (EX4 - 4d * EX3 * EX1 + 6d * EX2 * Math.Pow(EX1, 2d) - 3d * Math.Pow(EX1, 4d) - Math.Pow(MuS2, 2d)) / N2 + 2.0d / ((N2 - 1.0d) * N2) * Math.Pow(MuS2, 2d);
239 alpha = MuS2 * MuS2 / VarS2;
240 beta = MuS2 / alpha;
241 CovMS2 = (EX3 - 3d * EX2 * EX1 + 2d * Math.Pow(EX1, 3d)) / Math.Sqrt(N2 * (N2 - 1.0d));
242 MuS = Math.Sqrt(beta) * Math.Exp(Gamma.LogGamma(alpha + 0.5d) - Gamma.LogGamma(alpha));
243 CovMS = CovMS2 / (2d * MuS);
244 VarS = MuS2 - Math.Pow(MuS, 2d);
245 Lambda = CovMS / VarS;
246 EtaP = ETA + Lambda;
247 MuMP = MuM - Lambda * MuS;
248 VarMP = VarM - CovMS * CovMS / VarS;
249 df = 2.0d * alpha;
250 ncp = (MuMP - ZR) / Math.Sqrt(VarMP);
251 q = -Math.Sqrt(MuS2 / VarMP) * EtaP;
252 var NCTDist = new NoncentralT(df, ncp);
253 ANS = 1.0d - NCTDist.CDF(q);
254 return ANS;
255 }

263 public static void GrubbsBeckTest(IList<double> sample, out double XHi, out double XLo)
264 {
265 // The following polynomial approximation proposed by Pilon et al. (1985)
266 int n = sample.Count;
267 var logSample = new double[n];
268 for (int i = 0; i < n; i++) logSample[i] = Math.Log(sample[i]);
269 double mean = Statistics.Mean(logSample);
270 double sd = Statistics.StandardDeviation(logSample);
271 double Kn = -3.62201 + 6.28446 * Math.Pow(n, 0.25) - 2.49835 * Math.Pow(n, 0.5) + 0.491436 * Math.Pow(n, 0.75) - 0.037911 * n;
272 XHi = Math.Exp(mean + Kn * sd);
273 XLo = Math.Exp(mean - Kn * sd);
274 }
275 }
