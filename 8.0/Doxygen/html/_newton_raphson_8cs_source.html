96 public static double Solve(Func<double, double> f, Func<double, double> df, double firstGuess, double tolerance = 1E-8, int maxIterations = 1000, bool reportFailure = true)
97 {
98
99 // Define variables
100 double root = firstGuess;
101 bool solutionFound = false;
102 double y;
103 double yPrime;
104 double x0;
105 double x1;
106 double eps = Tools.DoubleMachineEpsilon; // Don't want to divide by a number smaller than this
107
108 // Newton-Raphson loop
109 for (int i = 1; i <= maxIterations; i++)
110 {
111 // do Newton's computation
112 x0 = root;
113 y = f(x0);
114 yPrime = df(x0);
115 if (Math.Abs(yPrime) < eps)
116 {
117 // the denominator is too small
118 solutionFound = false;
119 break;
120 }
121 x1 = x0 - y / yPrime;
122 root = x1;
123 // check if the result is within the desired tolerance
124 if (Math.Abs(x1 - x0) < tolerance)
125 {
126 solutionFound = true;
127 break;
128 }
129 }
130
131 // return results of solver
132 if (solutionFound == false && reportFailure == true)
133 {
134 throw new ArgumentException("Newton-Raphson method failed to find root.");
135 }
136 else
137 {
138 return root;
139 }

163 public static double RobustSolve(Func<double, double> f, Func<double, double> df, double firstGuess, double lowerBound, double upperBound, double tolerance = 1E-8, int maxIterations = 1000, bool reportFailure = true)
164 {
165
166 // Define variables
167 double root = firstGuess;
168 bool solutionFound = false;
169 double xl = lowerBound;
170 double xh = upperBound;
171 double fl = f(xl);
172 double fh = f(xh);
173
174 // validate inputs
175 if (upperBound < lowerBound)
176 {
177 throw new ArgumentOutOfRangeException(nameof(upperBound), "The upper bound (b) cannot be less than the lower bound (a).");
178 }
179 if (root < lowerBound || root > upperBound)
180 {
181 throw new ArgumentOutOfRangeException(nameof(firstGuess), "The first guess must be between the upper and lower bound.");
182 }
183 if (fl * fh >= 0d)
184 {
185 throw new ArgumentException("Robust Newton-Raphson method failed because the root is not bracketed.");
186 }
187
188 if (fl < 0d)
189 {
190 // Orient the search so that f(lowerBound)<0
191 xl = lowerBound;
192 xh = upperBound;
193 }
194 else
195 {
196 xh = lowerBound;
197 xl = upperBound;
198 }
199 // The "step-size before last"
200 double dxold = Math.Abs(upperBound - lowerBound);
201 // The last step
202 double dx = dxold;
203 double F = f(root);
204 double dF = df(root);
205 // Robust Newton-Raphson loop
206 for (int i = 1; i <= maxIterations; i++)
207 {
208 // Bisect if Newton is out of range, or not decreasing fast enough.
209 if ((((root - xh) * dF - F) * ((root - xl) * dF - F) > 0.0)
210 || (Math.Abs(2.0 * F) > Math.Abs(dxold * dF)))
211 {
212 dxold = dx;
213 dx = 0.5 * (xh - xl);
214 root = xl + dx;
215 if (xl == root)
216 {
217 // Change in root is negligible.
218 // Bisection step acceptable.
219 solutionFound = true;
220 break;
221 }
222 }
223 else
224 {
225 dxold = dx;
226 dx = F / dF;
227 double temp = root;
228 root -= dx;
229 if (temp == root)
230 {
231 // Change in root is negligible.
232 // Newton step acceptable.
233 }
234 }
235 // Check convergence
236 if (Math.Abs(dx) < tolerance)
237 {
238 solutionFound = true;
239 break;
240 }
241 // The one new function evaluation per iteration.
242 F = f(root);
243 dF = df(root);
244 // Maintain the bracket on the root.
245 if (F < 0.0)
246 xl = root;
247 else
248 xh = root;
249 }
250
251 // return results of solver
252 if (solutionFound == false && reportFailure == true)
253 {
254 throw new ArgumentException("Robust Newton-Raphson method failed to find root.");
255 }
256 else
257 {
258 return root;
259 }
260
261 }
