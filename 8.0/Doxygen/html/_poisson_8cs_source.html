Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

224 public override double PDF(double k)
225 {
226 // Validate parameters
227 if (_parametersValid == false)
228 ValidateParameters([Lambda], true);
229 k = Math.Floor(k);
230 if (k < Minimum || k > Maximum) return 0.0d;
231 return Math.Exp(-Lambda + k * Math.Log(Lambda) - Factorial.LogFactorial((int)k));
232 }
233
235 public override double CDF(double k)
236 {
237 // Validate parameters
238 if (_parametersValid == false)
239 ValidateParameters([Lambda], true);
240 k = Math.Floor(k);
241 if (k < Minimum)
242 return 0.0d;
243 if (k > Maximum)
244 return 1.0d;
245 return Gamma.UpperIncomplete(k + 1d, Lambda);
246 }
247
249 public override double InverseCDF(double probability)
250 {
251 // Validate probability
252 if (probability < 0.0d || probability > 1.0d)
253 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
254 if (probability == 0.0d)
255 return Minimum;
256 if (probability == 1.0d)
257 return Maximum;
258 // Validate parameters
259 if (_parametersValid == false)
260 ValidateParameters([Lambda], true);
261
262 double lower = Minimum;
263 double upper = Mean;
264 Brent.Bracket(x => CDF(x) - probability, ref lower, ref upper, out var f1, out var f2);
265 return Brent.Solve(x => CDF(x) - probability, lower, upper);
266 }
