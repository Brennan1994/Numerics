Numerics 1.0.0
Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

218 public static UnivariateDistributionBase CreateDistribution(XElement xElement)
219 {
220 UnivariateDistributionType type;
221 UnivariateDistributionBase dist = null;
222 if (xElement.Attribute(nameof(UnivariateDistributionBase.Type)) != null)
223 {
224 Enum.TryParse(xElement.Attribute(nameof(UnivariateDistributionBase.Type)).Value, out type);
225
226 if (type == UnivariateDistributionType.Mixture)
227 {
228 dist = Mixture.FromXElement(xElement);
229 return dist;
230 }
231 else
232 {
233 dist = CreateDistribution(type);
234 }
235
236 }
237 var names = dist.GetParameterPropertyNames;
238 var parms = dist.GetParameters;
239 var vals = new double[dist.NumberOfParameters];
240 for (int i = 0; i < dist.NumberOfParameters; i++)
241 {
242 if (xElement.Attribute(names[i]) != null)
243 {
244 double.TryParse(xElement.Attribute(names[i]).Value, System.Globalization.NumberStyles.Any, System.Globalization.CultureInfo.InvariantCulture, out vals[i]);
245 }
246 }
247 dist.SetParameters(vals);
248 return dist;
249 }
