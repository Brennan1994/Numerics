55 public static double EffectiveSampleSize(IList<double> series)
56 {
57 //https://www.rdocumentation.org/packages/LaplacesDemon/versions/16.1.4/topics/ESS
58 int N = series.Count;
59 var acf = Autocorrelation.Function(series, (int)Math.Ceiling((double)N / 2));
60 double rho = 0;
61 for (int i = 1; i < acf.GetLength(0); i++)
62 {
63 if (acf[i, 1] < 0.05) break;
64 rho += acf[i, 1];
65 }
66 return Math.Min(N / (1d + 2d * rho), N);
67 }

82 public static double[] GelmanRubin(IList<List<ParameterSet>> markovChains, int warmupIterations = 0)
83 {
84
85 int i, j, k;
86 // Get number of chains
87 int M = markovChains.Count;
88 // Get the minimum number of iterations
89 int N = int.MaxValue;
90 for (i = 0; i < M; i++)
91 if (markovChains[i].Count < N) N = markovChains[i].Count;
92 N -= warmupIterations;
93 // Get number of parameters
94 int P = markovChains[0][0].Values.Length;
95
96 // Validation checks
97 if (M < 2)
98 {
99 var result = new double[P];
100 result.Fill(double.NaN);
101 return result;
102 }
103 if (N < 2) throw new ArgumentOutOfRangeException(nameof(markovChains), "There must be at least two iterations to evaluate.");
104 if (P < 1) throw new ArgumentOutOfRangeException(nameof(markovChains), "There must be at least one parameter to evaluate.");
105 if (warmupIterations < 0) throw new ArgumentOutOfRangeException(nameof(warmupIterations), "The warm up iterations must be non-negative.");
106 if (warmupIterations == 0) warmupIterations = 1;
107
108 // Gelman-Rubin array
109 var GR = new double[P];
110
111 // Compute GR or each parameter
112 for (i = 0; i < P; i++)
113 {
114 // Compute between- and within-chain mean
115 var mean_chain = new double[M];
116 double mean_all = 0;
117 for (j = 0; j < M; j++)
118 {
119 for (k = warmupIterations - 1; k < N; k++)
120 mean_chain[j] += markovChains[j][k].Values[i];
121 // Get within-chain mean
122 mean_chain[j] /= N;
123 mean_all += mean_chain[j];
124 }
125 // Get between-chain mean
126 mean_all /= M;
127
128 // Compute between- and within-chain variance
129 var var_chain = new double[M];
130 double B = 0, W = 0;
131 for (j = 0; j < M; j++)
132 {
133 for (k = warmupIterations - 1; k < N; k++)
134 var_chain[j] += Tools.Sqr(markovChains[j][k].Values[i] - mean_chain[j]);
135 // within-chain variance
136 var_chain[j] *= 1d / (N - 1);
137 W += var_chain[j];
138 // between-chain variance
139 B += Tools.Sqr(mean_chain[j] - mean_all);
140 }
141 // Set between- and within-chain variance
142 W /= M;
143 B *= (double)N / (M - 1);
144
145 // Compute the pooled variance
146 double V = ((N - 1) * W + B) / N;
147
148 // Compute Rhat
149 GR[i] = Math.Sqrt(V / W);
150 }
151
152 return GR;
153 }

162 public static int MinimumSampleSize(double quantile, double tolerance, double probability)
163 {
164 double q = quantile;
165 double r = tolerance;
166 double s = probability;
167 double N = (q * (1d - q) * Math.Pow(Normal.StandardZ(0.5d * (s + 1d)), 2d)) / Math.Pow(r, 2d);
168 int Nmin = (int)Math.Round(decimal.Parse(N.ToString()) / 100M, 0) * 100;
169 return Nmin;
170 }
