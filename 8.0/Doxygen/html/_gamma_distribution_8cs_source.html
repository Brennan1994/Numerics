130 /*
131 * There are three different parameterizations in common use:
132 * 1. With a shape parameter κ and a scale parameter θ.
133 * 2. With a shape parameter α = κ and an inverse scale parameter β = 1/θ, called a rate parameter.
134 * 3. With a shape parameter κ and a mean parameter μ = kθ = α/β.
135 *
136 * This class only uses the parameterization described in #1.
137 */

402 public double[] ParametersFromMoments(IList<double> moments)
403 {
404 var parms = new double[NumberOfParameters];
405 parms[0] = 1d / (moments[0] / Math.Pow(moments[1], 2d));
406 parms[1] = Math.Pow(moments[0], 2d) / Math.Pow(moments[1], 2d);

411 public double[] MomentsFromParameters(IList<double> parameters)
412 {
413 var dist = new GammaDistribution();
414 dist.SetParameters(parameters);
415 var m1 = dist.Mean;
416 var m2 = dist.StandardDeviation;
417 var m3 = dist.Skewness;
418 var m4 = dist.Kurtosis;
419 return [m1, m2, m3, m4];

423 public double[] ParametersFromLinearMoments(IList<double> moments)
424 {
425 double A1 = -0.3080d;
426 double A2 = -0.05812d;
427 double A3 = 0.01765d;
428 double B1 = 0.7213d;
429 double B2 = -0.5947d;
430 double B3 = -2.1817d;
431 double B4 = 1.2113d;
432 double L1 = moments[0];
433 double L2 = moments[1];
434 double CV = L2 / L1;
435 double T, theta, kappa;
436 if (CV < 0.5)
437 {
438 T = Math.PI * CV * CV;
439 kappa = (1d + A1 * T) / (T * (1d + T * (A2 + T * A3)));
440 }
441 else
442 {
443 T = 1d - CV;
444 kappa = T * (B1 + T * B2) / (1d + T * (B3 + T * B4));
445 }
446 theta = L1 / kappa;
447 return [theta, kappa];
448 }

451 public double[] LinearMomentsFromParameters(IList<double> parameters)
452 {
453 double alpha = parameters[1];
454 double beta = parameters[0];
455 double L1 = alpha * beta;
456 double L2 = Math.Abs(Math.Pow(Math.PI, -0.5d) * beta * Gamma.Function(alpha + 0.5d) / Gamma.Function(alpha));
457 // The following approximations are accurate to 10-6.
458 double A0 = 0.32573501d;
459 double A1 = 0.1686915d;
460 double A2 = 0.078327243d;
461 double A3 = -0.0029120539d;
462 double B1 = 0.46697102d;
463 double B2 = 0.24255406d;
464 double C0 = 0.12260172d;
465 double C1 = 0.05373013d;
466 double C2 = 0.043384378d;
467 double C3 = 0.011101277d;
468 double D1 = 0.18324466d;
469 double D2 = 0.20166036d;
470 double E1 = 2.3807576d;
471 double E2 = 1.5931792d;
472 double E3 = 0.11618371d;
473 double F1 = 5.1533299d;
474 double F2 = 7.142526d;
475 double F3 = 1.9745056d;
476 double G1 = 2.1235833d;
477 double G2 = 4.1670213d;
478 double G3 = 3.1925299d;
479 double H1 = 9.0551443d;
480 double H2 = 26.649995d;
481 double H3 = 26.193668d;
482 double T3;
483 double T4;
484 if (alpha >= 1d)
485 {
486 T3 = Math.Pow(alpha, -0.5d) * (A0 + A1 * Math.Pow(alpha, -1) + A2 * Math.Pow(alpha, -2) + A3 * Math.Pow(alpha, -3)) / (1d + B1 * Math.Pow(alpha, -1) + B2 * Math.Pow(alpha, -2));
487 T4 = (C0 + C1 * Math.Pow(alpha, -1) + C2 * Math.Pow(alpha, -2) + C3 * Math.Pow(alpha, -3)) / (1d + D1 * Math.Pow(alpha, -1) + D2 * Math.Pow(alpha, -2));
488 }
489 else
490 {
491 T3 = (1d + E1 * alpha + E2 * Math.Pow(alpha, 2d) + E3 * Math.Pow(alpha, 3d)) / (1d + F1 * alpha + F2 * Math.Pow(alpha, 2d) + F3 * Math.Pow(alpha, 3d));
492 T4 = (1d + G1 * alpha + G2 * Math.Pow(alpha, 2d) + G3 * Math.Pow(alpha, 3d)) / (1d + H1 * alpha + H2 * Math.Pow(alpha, 2d) + H3 * Math.Pow(alpha, 3d));
493 }
494
495 return [L1, L2, T3, T4];
496 }

499 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
500 {
501 var initialVals = new double[NumberOfParameters];
502 var lowerVals = new double[NumberOfParameters];
503 var upperVals = new double[NumberOfParameters];
504 // Get initial values
505 initialVals = ParametersFromMoments(Statistics.ProductMoments(sample));
506 // Get bounds of scale
507 lowerVals[0] = Tools.DoubleMachineEpsilon;
508 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[0]) + 1d));
509 // Get bounds of shape
510 lowerVals[1] = Tools.DoubleMachineEpsilon;
511 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[1]) + 1d));
512 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
513 }

516 public double[] MLE(IList<double> sample)
517 {
518 // Set constraints
519 var tuple = GetParameterConstraints(sample);
520 var Initials = tuple.Item1;
521 var Lowers = tuple.Item2;
522 var Uppers = tuple.Item3;
523
524 // Solve using Nelder-Mead (Downhill Simplex)
525 double logLH(double[] x)
526 {
527 var GM = new GammaDistribution();
528 GM.SetParameters(x);
529 return GM.LogLikelihood(sample);
530 }
531 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
532 solver.ReportFailure = true;
533 solver.Maximize();
534 return solver.BestParameterSet.Values;
535 }

541 public void MLE_NR(IList<double> sample)
542 {
543 double lnsum = 0d;
544 for (int i = 0; i < sample.Count; i++)
545 lnsum += Math.Log(sample[i]);
546 double mean = Statistics.Mean(sample);
547 double s = Math.Log(mean) - lnsum / sample.Count;
548 if (double.IsNaN(s))
549 {
550 throw new ArgumentException("Observation vector contains negative values.", "observations");
551 }
552 double K_hat = (3d - s + Math.Sqrt((s - 3d) * (s - 3d) + 24d * s)) / (12d * s);
553 double newK = NewtonRaphson.Solve((k) => Math.Log(k) - Gamma.Digamma(k) - s, (k) => 1d / k - Gamma.Trigamma(k), K_hat, 0.000001d, 100, true);
554 double newTheta = mean / Kappa;
555 SetParameters(newTheta, newK);
556 }

562 public void MLE_Bobee(IList<double> sample)
563 {
564 double A = Statistics.Mean(sample);
565 double G = Statistics.GeometricMean(sample);
566 double U = Math.Log(A) - Math.Log(G);
567 double La = (1d + Math.Sqrt(1d + 4d * U / 3d)) / (4d * U);
568 double dL = 0.04475d * Math.Pow(0.26d, La);
569 double K = La - dL;
570 double T = A / K;
571 SetParameters(T, K);
572 }

575 public override double PDF(double X)
576 {
577 // Validate parameters
578 if (_parametersValid == false)
579 ValidateParameters(Theta, Kappa, true);
580 if (X < Minimum || X > Maximum) return 0.0d;
581 return Math.Exp(-X / Theta + (Kappa - 1.0d) * Math.Log(X) - Kappa * Math.Log(Theta) - Gamma.LogGamma(Kappa));
582 }

585 public override double CDF(double X)
586 {
587 // Validate parameters
588 if (_parametersValid == false)
589 ValidateParameters(Theta, Kappa, true);
590 if (X <= Minimum)
591 return 0d;
592 if (X >= Maximum)
593 return 1d;
594 return Gamma.LowerIncomplete(Kappa, X / Theta);

598 public override double InverseCDF(double probability)
599 {
600 // Validate probability
601 if (probability < 0.0d || probability > 1.0d)
602 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
603 if (probability == 0.0d)
604 return Minimum;
605 if (probability == 1.0d)
606 return Maximum;

608 if (_parametersValid == false)
609 ValidateParameters(Theta, Kappa, true);
610 return Gamma.InverseLowerIncomplete(Kappa, probability) * Theta;

641 public static double FrequencyFactorKp(double skewness, double probability)
642 {
643 double C = skewness;
644 double absC = Math.Abs(C);
645 // If skew is sufficiently close to zero, return standard Normal Z variate.
646 if (absC < 0.0001d) return Normal.StandardZ(probability);
647
648 // If abs(skew) is less than or equal to 2, use Cornish-Fisher transformation (Fisher and Cornish, 1960)
649 if (absC <= 2d)
650 {
651 double C2 = Math.Pow(C, 2d);
652 double C3 = Math.Pow(C, 3d);
653 double C4 = Math.Pow(C, 4d);
654 double C5 = Math.Pow(C, 5d);
655 double C6 = Math.Pow(C, 6d);
656 double U = Normal.StandardZ(probability);
657 double U2 = Math.Pow(U, 2d);
658 double U3 = Math.Pow(U, 3d);
659 double U4 = Math.Pow(U, 4d);
660 double U5 = Math.Pow(U, 5d);
661 double U6 = Math.Pow(U, 6d);
662 double U7 = Math.Pow(U, 7d);
663 double Kterm0 = U;
664 double Kterm1 = C / 2d * ((U2 - 1d) / 3d);
665 double Kterm2 = C2 / Math.Pow(2d, 4d) * ((U3 - 7d * U) / 9d);
666 double Kterm3 = C3 / Math.Pow(2d, 5d) * ((6d * U4 + 14d * U2 - 32d) / 405d);
667 double Kterm4 = C4 / Math.Pow(2d, 7d) * ((9d * U5 + 256d * U3 - 433d * U) / 4860d);
668 double Kterm5 = C5 / Math.Pow(2d, 8d) * ((12d * U6 - 143d * U4 - 923d * U2 + 1472d) / 25515d);
669 double Kterm6 = C6 / Math.Pow(2d, 10d) * ((3753d * U7 + 4353d * U5 - 289517d * U3 - 289717d * U) / 9185400d);
670 return Kterm0 + Kterm1 + Kterm2 - Kterm3 + Kterm4 + Kterm5 - Kterm6;
671 }
672 else
673 {
674 // If abs(skew) is greater than 2, use Modified Wilson-Hilferty transformation (Kirby, 1972)
675 // Only, valid if abs(skew) <= 9.75. Enforce limits.
676 if (C < -9.75d)
677 C = -9.75d;
678 if (C > 9.75d)
679 C = 9.75d;
680
681 // Hoshi and Burges (1981b) gave polynomial expressions for 1/A, B, G and H^3 as a function of Cs
682 // Compute skew orders
683 double C2 = Math.Pow(absC, 2d);
684 double C3 = Math.Pow(absC, 3d);
685 double C4 = Math.Pow(absC, 4d);
686 double C5 = Math.Pow(absC, 5d);
687 // Compute A
688 double a0 = 0.00199447d;
689 double a1 = 0.48489d;
690 double a2 = 0.0230935d;
691 double a3 = -0.0152435d;
692 double a4 = 0.00160597d;
693 double a5 = -0.000055869d;
694 double A = 1d / (a0 + a1 * C + a2 * C2 + a3 * C3 + a4 * C4 + a5 * C5);
695 // Compute B
696 double b0 = 0.990562d;
697 double b1 = 0.0319647d;
698 double b2 = -0.0274231d;
699 double b3 = 0.00777405d;
700 double b4 = -0.000571184d;
701 double b5 = 0.0000142077d;
702 double B = b0 + b1 * C + b2 * C2 + b3 * C3 + b4 * C4 + b5 * C5;
703 // Compute G
704 double g0 = -0.00385205d;
705 double g1 = 1.00426d;
706 double g2 = 0.00651207d;
707 double g3 = -0.0149166d;
708 double g4 = 0.00163945d;
709 double g5 = -0.0000583804d;
710 double G = g0 + g1 * C + g2 * C2 + g3 * C3 + g4 * C4 + g5 * C5;
711 // Compute H
712 double H = Math.Pow(B - 2.0d / absC / A, 1d / 3d);
713 return Math.Sign(C) * A * (Math.Pow(Math.Max(H, 1.0d - Math.Pow(G / 6.0d, 2d) + G / 6.0d * Normal.StandardZ(probability)), 3d) - B);
714 }
715 }

722 public static double PartialKp(double skewness, double probability)
723 {
724 double C = skewness;
725 double absC = Math.Abs(C);
726 // If skew is sufficiently close to zero, return standard Normal Z variate.
727 if (absC < 0.0001d) return Normal.StandardZ(probability);
728
729 // If abs(skew) is less than or equal to 2, use Cornish-Fisher transformation (Fisher and Cornish, 1960)
730 if (absC <= 2d)
731 {
732 double C2 = Math.Pow(C, 2d);
733 double C3 = Math.Pow(C, 3d);
734 double C4 = Math.Pow(C, 4d);
735 double C5 = Math.Pow(C, 5d);
736 double C6 = Math.Pow(C, 6d);
737 double U = Normal.StandardZ(probability);
738 double U2 = Math.Pow(U, 2d);
739 double U3 = Math.Pow(U, 3d);
740 double U4 = Math.Pow(U, 4d);
741 double U5 = Math.Pow(U, 5d);
742 double U6 = Math.Pow(U, 6d);
743 double U7 = Math.Pow(U, 7d);
744 // Determine the first derivative of K with respect to skew
745 double dKterm0 = 0d;
746 double dKterm1 = 1d / 2d * ((U2 - 1d) / 3d);
747 double dKterm2 = 2d * (C / Math.Pow(2d, 4d)) * ((U3 - 7d * U) / 9d);
748 double dKterm3 = 3d * (C2 / Math.Pow(2d, 5d)) * ((6d * U4 + 14d * U2 - 32d) / 405d);
749 double dKterm4 = 4d * (C3 / Math.Pow(2d, 7d)) * ((9d * U5 + 256d * U3 - 433d * U) / 4860d);
750 double dKterm5 = 5d * (C4 / Math.Pow(2d, 8d)) * ((12d * U6 - 143d * U4 - 923d * U2 + 1472d) / 25515d);
751 double dKterm6 = 6d * (C5 / Math.Pow(2d, 10d)) * ((3753d * U7 + 4353d * U5 - 289517d * U3 - 289717d * U) / 9185400d);
752 //
753 return dKterm0 + dKterm1 + dKterm2 - dKterm3 + dKterm4 + dKterm5 - dKterm6;
754 }
755 else
756 {
757 // If abs(skew) is greater than 2, use Modified Wilson-Hilferty transformation (Kirby, 1972)
758 return NumericalDerivative.Derivative((x) => FrequencyFactorKp(x, probability), skewness, 0.0001d);
759 }

771 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
772 {
773 if (estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
774 {
775 throw new NotImplementedException();
776 }
777 // Validate parameters
778 if (_parametersValid == false)
779 ValidateParameters(_theta, _kappa, true);
780 // Compute covariance
781 double alpha = 1d / Theta;
782 double lambda = Kappa;
783 double NA = Gamma.Trigamma(lambda) - 1d / lambda;
784 var covar = new double[2, 2];
785 covar[0, 0] = Math.Pow(alpha, 2d) * Gamma.Trigamma(lambda) / (sampleSize * lambda * NA); // scale
786 covar[1, 1] = 1d / (sampleSize * NA); // shape
787 covar[0, 1] = alpha / (sampleSize * lambda * NA); // scale & shape
788 covar[1, 0] = covar[0, 1];
789 return covar;
790 }

798 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
799 {
800 if (estimationMethod != ParameterEstimationMethod.MethodOfMoments &&
801 estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
802 {
803 throw new NotImplementedException();
804 }
805 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
806 {
807 double CV = CoefficientOfVariation;
808 double V = Variance;
809 int N = sampleSize;
810 return V / N * (Math.Pow(1d + FrequencyFactorKp(Skewness, probability) * CV, 2d) + 0.5d * Math.Pow(FrequencyFactorKp(Skewness, probability) + 2d * CV * PartialKp(Skewness, probability), 2d) * (1d + Math.Pow(CV, 2d)));
811 }
812 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
813 {
814 var covar = ParameterCovariance(sampleSize, estimationMethod);
815 var grad = QuantileGradient(probability);
816 double varA = covar[0, 0];
817 double varB = covar[1, 1];
818 double covAB = covar[1, 0];
819 double dQx1 = grad[0];
820 double dQx2 = grad[1];
821 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + 2d * dQx1 * dQx2 * covAB;
822 }
823 return double.NaN;
824 }

827 public double[] QuantileGradient(double probability)
828 {
829 // Validate parameters
830 if (_parametersValid == false)
831 ValidateParameters(_theta, _kappa, true);
832 double alpha = 1d / Theta;
833 double lambda = Kappa;
834 double eps = Math.Sign(alpha);
835 var gradient = new double[]
836 {
837 -lambda / Math.Pow(alpha, 2d) * (1.0d + eps / Math.Sqrt(lambda) * FrequencyFactorKp(Skewness, probability)), // scale
838 1.0d / alpha * (1.0d + eps / Math.Sqrt(lambda) * FrequencyFactorKp(Skewness, probability) / 2.0d - 1.0d / lambda * PartialKp(Skewness, probability)) // shape
839 };
840 return gradient;
841 }
842
847 private double PartialforTheta(double probability)
848 {
849 return FrequencyFactorKp(Skewness, probability) * Math.Sqrt(Kappa) + Kappa;
850 }
851
856 private double PartialforKappa(double probability)
857 {
858 return Theta * (FrequencyFactorKp(Skewness, probability) / (2.0d * Math.Sqrt(Kappa)) + 1.0d - PartialKp(Skewness, probability) / Kappa);
859 }

862 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
863 {
864 if (probabilities.Count != NumberOfParameters)
865 {
866 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
867 }
868 // Compute determinant
869 // |a b|
870 // |c d|
871 // |A| = ad − bc
872 double a = PartialforTheta(probabilities[0]);
873 double b = PartialforKappa(probabilities[0]);
874 double c = PartialforTheta(probabilities[1]);
875 double d = PartialforKappa(probabilities[1]);
876 determinant = a * d - b * c;
877 // Return Jacobian
878 var jacobian = new double[,] { { a, b }, { c, d } };
879 return jacobian;
880 }
