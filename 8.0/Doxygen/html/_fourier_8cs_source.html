79 public static void FFT(double[] data, bool inverse = false)
80 {
81 // create variables
82 int n, nn, mmax, m, j, istep, i, isign;
83 double wtemp, wr, wpr, wpi, wi, theta, tempr, tempi;
84 n = (int)(data.Count() / 2d);
85 // Check if n is a power of 2.
86 if (n < 2 || IsPowerOfTwo(n) == false)
87 {
88 throw new ArgumentOutOfRangeException(nameof(data), "The data array length must be a power of 2.");
89 }
90 nn = n << 1;
91 isign = inverse == false ? 1 : -1;
92 j = 1;
93 // This is the bit-reversal section of the routine.
94 for (i = 1; i < nn; i += 2)
95 {
96 if (j > i)
97 {
98 // swap values
99 tempr = data[j - 1];
100 tempi = data[j];
101 data[j - 1] = data[i - 1];
102 data[j] = data[i];
103 data[i - 1] = tempr;
104 data[i] = tempi;
105 }
106 m = n;
107 while (m >= 2 && j > m)
108 {
109 j -= m;
110 m >>= 1;
111 }
112 j += m;
113 }
114 // Here begins the Danielson-Lanczos section of the routine.
115 mmax = 2;
116 // Outer loop executed log2 n times
117 while (nn > mmax)
118 {
119 istep = mmax << 1;
120 // Initialize the trigonometric recurrence.
121 theta = isign * (2d * Math.PI / mmax);
122 wpr = -2.0d * Math.Pow(Math.Sin(0.5d * theta), 2d);
123 wpi = Math.Sin(theta);
124 wr = 1.0d;
125 wi = 0.0d;
126 // Here are the two nested inner loops.
127 for (m = 1; m < mmax; m += 2)
128 {
129 for (i = m; istep >= 0 ? i <= nn : i >= nn; i += istep)
130 {
131 j = i + mmax;
132 // This is the Danielson-Lanczos formula:
133 tempr = wr * data[j - 1] - wi * data[j];
134 tempi = wr * data[j] + wi * data[j - 1];
135 data[j - 1] = data[i - 1] - tempr;
136 data[j] = data[i] - tempi;
137 data[i - 1] += tempr;
138 data[i] += tempi;
139 }
140 // Trigonometric recurrence
141 wtemp = wr;
142 wr = wr * wpr - wi * wpi + wr;
143 wi = wi * wpr + wtemp * wpi + wi;
144 }
145
146 mmax = istep;
147 }
148 }

161 public static void RealFFT(double[] data, bool inverse = false)
162 {
163 int i, i1, i2, i3, i4, n, isign;
164 double c1, c2, h1r, h1i, h2r, h2i, wr, wi, wpr, wpi, wtemp, theta;
165 n = data.Count();
166 isign = inverse == false ? 1 : -1;
167 // Initial the recurrence
168 theta = Math.PI / (n >> 1);
169 c1 = 0.5d;
170 if (isign == 1)
171 {
172 c2 = -0.5d;
173 // The forward transform is here.
174 FFT(data);
175 }
176 else
177 {
178 // Otherwise set up for an inverse transform.
179 c2 = 0.5d;
180 theta = -theta;
181 }
182 wpr = -2.0d * Math.Pow(Math.Sin(0.5d * theta), 2d);
183 wpi = Math.Sin(theta);
184 wr = 1.0d + wpr;
185 wi = wpi;
186 // Case i = 0 done separately below.
187 for (i = 1; i < (n >> 2); i++)
188 {
189 i1 = i + i;
190 i2 = 1 + i1;
191 i3 = n - i1;
192 i4 = 1 + i3;
193 // The two separate transforms are separated out of data.
194 h1r = c1 * (data[i1] + data[i3]);
195 h1i = c1 * (data[i2] - data[i4]);
196 h2r = -c2 * (data[i2] + data[i4]);
197 h2i = c2 * (data[i1] - data[i3]);
198 // Here they are recombined to form the true transform of
199 // the original real data.
200 data[i1] = h1r + wr * h2r - wi * h2i;
201 data[i2] = h1i + wr * h2i + wi * h2r;
202 data[i3] = h1r - wr * h2r + wi * h2i;
203 data[i4] = -h1i + wr * h2i + wi * h2r;
204 // The recurrence.
205 wtemp = wr;
206 wr = wr * wpr - wi * wpi + wr;
207 wi = wi * wpr + wtemp * wpi + wi;
208 }
209
210 if (isign == 1)
211 {
212 // Squeeze the first and last data together to get them
213 // all within the original array.
214 h1r = data[0];
215 data[0] = h1r + data[1];
216 data[1] = h1r - data[1];
217 }
218 else
219 {
220 // This is the inverse transform for the case isign = -1.
221 h1r = data[0];
222 data[0] = c1 * (h1r + data[1]);
223 data[1] = c1 * (h1r - data[1]);
224 FFT(data, true);
225 }
226 }

241 public static double[] Correlation(double[] data1, double[] data2)
242 {
243 int no2, i;
244 int n = data1.Count();
245 double tmp;
246 var temp = new double[n];
247 var ans = new double[n];
248 for (i = 0; i < n; i++)
249 {
250 ans[i] = data1[i];
251 temp[i] = data2[i];
252 }
253 // Transform both data vectors
254 RealFFT(ans);
255 RealFFT(temp);
256 // Normalization for inverse FFT.
257 // Multiply to find FFT of their correlation.
258 no2 = n >> 1;
259 for (i = 2; i < n; i += 2)
260 {
261 tmp = ans[i];
262 ans[i] = (ans[i] * temp[i] + ans[i + 1] * temp[i + 1]) / no2;
263 ans[i + 1] = (ans[i + 1] * temp[i] - tmp * temp[i + 1]) / no2;
264 }
265 ans[0] = ans[0] * temp[0] / no2;
266 ans[1] = ans[1] * temp[1] / no2;
267 // Inverse transform gives correlation.
268 RealFFT(ans, true);
269 return ans;
270 }

280 public static double[,] Autocorrelation(IList<double> series, int lagMax = -1)
281 {
282 int n = series.Count;
283 if (lagMax < 0)
284 lagMax = (int)Math.Floor(Math.Min(10d * Math.Log10(n), n - 1));
285 if (lagMax < 1 || n < 2)
286 return null;
287 // Pad the length to be the power of 2 to facilitate FFT speed.
288 int newLength = Convert.ToInt32(Math.Pow(2d, Math.Ceiling(Math.Log(series.Count, 2d))));
289 // Normalize the data series
290 var normalizedSeries = new double[newLength];
291 double mean = Statistics.Mean(series);
292 for (int i = 0; i < newLength; i++)
293 {
294 if (i < series.Count)
295 {
296 normalizedSeries[i] = series[i] - mean;
297 }
298 }
299 // Get correlation function
300 var correlation = Correlation(normalizedSeries, normalizedSeries);
301 // Convert to normalized correlation coefficient
302 double maxValue = correlation.Max();
303 var acf = new double[lagMax + 1, 2];
304 for (int i = 0; i <= lagMax; i++)
305 {
306 acf[i, 0] = i;
307 acf[i, 1] = correlation[i] / maxValue;
308 }
309
310 return acf;
311 }
