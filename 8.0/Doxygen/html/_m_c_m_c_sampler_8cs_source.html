369 protected virtual ParameterSet[] InitializeChains()
370 {
371 if (Initialize == InitializationType.UserDefined)
372 {
373 // If user-defined, return the last states of the chains.
374 var chainStates = new ParameterSet[NumberOfChains];
375 for (int i = 0; i < NumberOfChains; i++)
376 {
377 chainStates[i] = MarkovChains[i].Last();
378 }
379 return chainStates;
380 }
381
382 var prng = new Random(PRNGSeed);
383 var rnds = LatinHypercube.Random(InitialIterations, NumberOfParameters, prng.Next());
384 var parameters = new double[NumberOfParameters];
385 var tempPopulation = new List<ParameterSet>();
386 var initials = new ParameterSet[NumberOfChains];
387 double logLH = 0;
388
389 if (Initialize == InitializationType.MAP)
390 {
391 // Use differential evolution to find a global optimum
392 var lowerBounds = PriorDistributions.Select(x => x.Minimum).ToArray();
393 var upperBounds = PriorDistributions.Select(x => x.Maximum).ToArray();
394 var inititals = lowerBounds.Add(upperBounds).Divide(2d);
395 var DE = new DifferentialEvolution((x) => { return LogLikelihoodFunction(x); }, NumberOfParameters, lowerBounds, upperBounds);
396 DE.ReportFailure = false;
397 DE.Maximize();
398 if (DE.Status == OptimizationStatus.Success)
399 {
400 try
401 {
402 _mapSuccessful = true;
403 // Get MAP
404 MAP = DE.BestParameterSet.Clone();
405 // Get Fisher Information Matrix (or Hessian)
406 var hessian = DE.Hessian.Clone();
407 Matrix fisher = hessian * -1d;
408 // Invert it to get the covariance matrix
409 fisher = fisher.Inverse();
410 // Scale it to give wider coverage
411 fisher = fisher * 3;
412
413 // Set up proposal distribution
414 _MVN = new MultivariateNormal(MAP.Values, fisher.ToArray());
415 // Then randomly sample from the proposal
416 for (int i = 0; i < InitialIterations; i++)
417 {
418 parameters = _MVN.InverseCDF(rnds.GetRow(i));
419 logLH = LogLikelihoodFunction(parameters);
420 if (IsPopulationSampler) PopulationMatrix.Add(new ParameterSet(parameters, logLH));
421 tempPopulation.Add(new ParameterSet(parameters, logLH));
422 }
423
424 // Set the initial vectors randomly from the MVN proposal
425 for (int i = 0; i < NumberOfChains; i++)
426 initials[i] = tempPopulation[i].Clone();
427
428 return initials;
429
430 }
431 catch (Exception)
432 {
433 // If this fails go to naive initialization below
434 Initialize = InitializationType.Randomize;
435 }
436 }
437 }
438
439
440 // *** If not using MAP or if MAP fails, then use random initialization *** //
441
442 // First add the mean of the priors
443 for (int j = 0; j < NumberOfParameters; j++)
444 parameters[j] = PriorDistributions[j].Mean;
445 logLH = LogLikelihoodFunction(parameters);
446 if (IsPopulationSampler) PopulationMatrix.Add(new ParameterSet(parameters, logLH));
447 tempPopulation.Add(new ParameterSet(parameters, logLH));
448
449 // If the initial population and the number of chains is 1,
450 // then just take the mean of the priors
451 if (InitialIterations == 1 && NumberOfChains == 1)
452 {
453 initials[0] = tempPopulation.First().Clone();
454 return initials;
455 }
456
457 // Then randomly sample from the priors
458 for (int i = 1; i < InitialIterations; i++)
459 {
460 for (int j = 0; j < NumberOfParameters; j++)
461 parameters[j] = PriorDistributions[j].InverseCDF(rnds[i, j]);
462 logLH = LogLikelihoodFunction(parameters);
463 if (IsPopulationSampler) PopulationMatrix.Add(new ParameterSet(parameters, logLH));
464 tempPopulation.Add(new ParameterSet(parameters, logLH));
465 }
466
467 // Sort temp population by log-likelihood in descending order
468 tempPopulation.Sort((x, y) => -1 * x.Fitness.CompareTo(y.Fitness));
469
470 // Set the initial vectors to the best performing parameter sets
471 for (int i = 0; i < NumberOfChains; i++)
472 initials[i] = tempPopulation[i].Clone();
473
474 return initials;
475 }

502 public virtual void Sample()
503 {
504 // Validate the input settings
505 ValidateSettings();
506
507 CancellationTokenSource = new CancellationTokenSource();
508
509 // Setup the sampler
510 if (ResumeSimulation = false || _simulations < 1)
511 {
512 // Initialize the chains
513 _chainStates = InitializeChains();
514
515 // Initialize custom settings
516 InitializeCustomSettings();
517 }
518
519 // Output settings
520 int outputIterations = (int)Math.Ceiling(OutputLength / (double)NumberOfChains);
521 int totalIterations = Iterations + outputIterations;
522 int outputCount = 0;
523 Output = new List<ParameterSet>[NumberOfChains];
524 for (int i = 0; i < NumberOfChains; i++)
525 Output[i] = new List<ParameterSet>();
526
527 // progress counter
528 int progress = 0;
529
530 // Sample chains
531 for (int i = 1; i <= totalIterations; i++)
532 {
533
534 if (ParallelizeChains)
535 {
536 Parallel.For(0, NumberOfChains, (j) => { _chainStates[j] = SampleChain(j, _chainStates[j]); });
537 }
538 else
539 {
540 for (int j = 0; j < NumberOfChains; j++)
541 _chainStates[j] = SampleChain(j, _chainStates[j]);
542 }
543
544 // Save chain states
545 if (i <= Iterations)
546 MeanLogLikelihood.Add(0);
547
548 // Record output
549 for (int j = 0; j < NumberOfChains; j++)
550 {
551 // Update population
552 if (IsPopulationSampler == true)
553 PopulationMatrix.Add(_chainStates[j].Clone());
554
555 if (i <= Iterations)
556 {
557 // Record mean log-likelihood
558 MeanLogLikelihood[i - 1] += _chainStates[j].Fitness / NumberOfChains;
559
560 // Save chain state
561 MarkovChains[j].Add(_chainStates[j].Clone());
562 }
563 else if (i > Iterations && outputCount < OutputLength)
564 {
565 // Record the output and keep track of MAP
566 Output[j].Add(_chainStates[j].Clone());
567 outputCount++;
568 if (_chainStates[j].Fitness > MAP.Fitness)
569 MAP = _chainStates[j].Clone();
570 }
571 }
572
573 // Check for cancellation
574 if (CancellationTokenSource.Token.IsCancellationRequested)
575 break;
576
577 // Update progress
578 progress += 1;
579 if (progress % (int)(totalIterations * ProgressChangedRate) == 0)
580 {
581 ReportProgress((double)progress / totalIterations);
582 }
583
584 }
585
586 _simulations += 1;
587 }

612 public void Reset()
613 {
614 _simulations = 0;
615 // Clear old memory and re-instantiate the result storage
616 _masterPRNG = new Random(PRNGSeed);
617 _chainPRNGs = new Random[NumberOfChains];
618 PopulationMatrix = new List<ParameterSet>();
619 MarkovChains = new List<ParameterSet>[NumberOfChains];
620 for (int i = 0; i < NumberOfChains; i++)
621 {
622 _chainPRNGs[i] = new Random(_masterPRNG.Next());
623 MarkovChains[i] = new List<ParameterSet>();
624 }
625 AcceptCount = new int[NumberOfChains];
626 SampleCount = new int[NumberOfChains];
627 MeanLogLikelihood = new List<double>();
628 MAP = new ParameterSet([], double.MinValue);
629 Output = new List<ParameterSet>[NumberOfChains];
630 }
