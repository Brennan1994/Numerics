157 public override double PDF(double u, double v)
158 {
159 // Validate parameters
160 if (_parametersValid == false) ValidateParameter(Theta, true);
161 double num = Theta * (Math.Exp(Theta) - 1d) * Math.Exp(Theta * (1d + u + v));
162 double den = Math.Pow(Math.Exp(Theta * (u + v)) - Math.Exp(Theta * (1d + u)) - Math.Exp(Theta * (1d + v)) + Math.Exp(Theta), 2d);
163 return num / den;
164 }
165
167 public override double CDF(double u, double v)
168 {
169 // Validate parameters
170 if (_parametersValid == false) ValidateParameter(Theta, true);
171 return -(1d / Theta) * Math.Log(1d + (Math.Exp(-Theta * u) - 1d) * (Math.Exp(-Theta * v) - 1d) / (Math.Exp(-Theta) - 1d));
172 }
173
175 public override double[] InverseCDF(double u, double v)
176 {
177 // Validate parameters
178 if (_parametersValid == false) ValidateParameter(Theta, true);
179 double a = -Math.Abs(Theta);
180 v = -1d / a * Math.Log((-v * (Math.Exp(-a) - 1d) / (Math.Exp(-a * u) * (v - 1d) - v)) + 1d);
181 v = Theta > 0d ? 1d - v : v;
182 return [u, v];
183 }
