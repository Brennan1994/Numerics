Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

175 public bool OrdinateValid(UncertainOrdinate ordinateToCompare, bool strictX, bool strictY, SortOrder xOrder, SortOrder yOrder, bool compareOrdinateIsNext, bool allowDifferentTypes = false)
176 {
177 //
178 if (IsValid == false)
179 return false;
180 if (ordinateToCompare.IsValid == false)
181 return false;
182 // Check for equivalent distribution types
183 if (allowDifferentTypes == false && ordinateToCompare.Y.Type != Y.Type)
184 return false;
185
186 double minPercentile = Y.Type == UnivariateDistributionType.PertPercentile || Y.Type == UnivariateDistributionType.PertPercentileZ ? 0.05 : 1E-5;
187
188 // Test reasonable lower bound
189 if (GetOrdinate(minPercentile).OrdinateValid(ordinateToCompare.GetOrdinate(minPercentile), strictX, strictY, xOrder, yOrder, compareOrdinateIsNext) == false)
190 return false;
191 // Test central tendency
192 if (GetOrdinate().OrdinateValid(ordinateToCompare.GetOrdinate(), strictX, strictY, xOrder, yOrder, compareOrdinateIsNext) == false)
193 return false;
194 // Test reasonable upper bound
195 if (GetOrdinate(1 - minPercentile).OrdinateValid(ordinateToCompare.GetOrdinate(1 - minPercentile), strictX, strictY, xOrder, yOrder, compareOrdinateIsNext) == false)
196 return false;
197 // Passed the test
198 return true;
199 }

212 public List<string> OrdinateErrors(UncertainOrdinate ordinateToCompare, bool strictX, bool strictY, SortOrder xOrder, SortOrder yOrder, bool compareOrdinateIsNext, bool allowDifferentTypes = false)
213 {
214 var result = new List<string>();
215 // Validate the target ordinate
216 result.AddRange(OrdinateErrors());
217 //
218 if (ordinateToCompare == null)
219 return result;
220 //
221 if (ordinateToCompare.IsValid == false)
222 {
223 if (double.IsInfinity(ordinateToCompare.X))
224 result.Add("Ordinate X value can not be infinity.");
225 if (double.IsNaN(ordinateToCompare.X))
226 result.Add("Ordinate X value must be a valid number.");
227 if (ordinateToCompare.Y == null)
228 {
229 result.Add("Ordinate Y value must be defined.");
230 }
231 else if (ordinateToCompare.Y.ParametersValid == false)
232 {
233 var argError = ordinateToCompare.Y.ValidateParameters(ordinateToCompare.Y.GetParameters, false);
234 if (argError != null) result.Add(argError.Message);
235 }
236 }
237 // Check for equivalent distribution types
238 if (allowDifferentTypes == false && ordinateToCompare.Y.Type != Y.Type)
239 result.Add("Can't compare two ordinates with different distribution types."); // Return False
240 //
241 if (IsValid == true && ordinateToCompare.IsValid == true)
242 {
243 double minPercentile = Y.Type == UnivariateDistributionType.PertPercentile || Y.Type == UnivariateDistributionType.PertPercentileZ ? 0.05 : 1E-5;
244
245 // Test reasonable lower bound
246 result.AddRange(GetOrdinate(minPercentile).OrdinateErrors(ordinateToCompare.GetOrdinate(minPercentile), strictX, strictY, xOrder, yOrder, compareOrdinateIsNext));
247 // Test central tendency
248 result.AddRange(GetOrdinate(0.5d).OrdinateErrors(ordinateToCompare.GetOrdinate(0.5d), strictX, strictY, xOrder, yOrder, compareOrdinateIsNext));
249 // Test reasonable upper bound
250 result.AddRange(GetOrdinate(1 - minPercentile).OrdinateErrors(ordinateToCompare.GetOrdinate(1 - minPercentile), strictX, strictY, xOrder, yOrder, compareOrdinateIsNext));
251 }
252 // Finished Checking
253 return result;
254 }
