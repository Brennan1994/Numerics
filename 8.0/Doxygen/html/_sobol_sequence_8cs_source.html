123 private void initialize()
124 {
125
126 // special case: dimension 1 -> use unit initialization
127 for (int i = 1; i <= BITS; i++)
128 {
129 _direction[0, i] = 1L << (BITS - i);
130 }
131
132 using (StreamReader reader = new StreamReader(new MemoryStream(Properties.Resources.new_joe_kuo_6)))
133 {
134 // ignore the first line
135 reader.ReadLine();
136
137 int index = 1;
138 string line = null;
139 while ((line = reader.ReadLine()) != null)
140 {
141 var st = line.Split(' ');
142
143 try
144 {
145 int dim;
146 int.TryParse(st[0], out dim);
147
148 if (dim >= 2 && dim <= Dimension)
149 {
150 // we have found the right dimension
151 int i, s = 0, a = 0;
152 for (i = 1; i < st.Length; i++)
153 {
154 if (st[i] != "")
155 {
156 int.TryParse(st[i], out s);
157 break;
158 }
159 }
160 i++;
161 for (; i < st.Length; i++)
162 {
163 if (st[i] != "")
164 {
165 int.TryParse(st[i], out a);
166 break;
167 }
168 }
169 i++;
170 int[] m = new int[s + 1];
171 for (; i < st.Length; i++)
172 {
173 if (st[i] != "")
174 {
175 for (int j = 1; j <= s; j++)
176 {
177 int.TryParse(st[i + j - 1], out m[j]);
178 }
179 break;
180 }
181 }
182 initDirectionVector(index++, a, m);
183 }
184
185 if (dim > Dimension)
186 {
187 return;

202 private void initDirectionVector(int d, int a, int[] m)
203 {
204 int s = m.Length - 1;
205 for (int i = 1; i <= s; i++)
206 {
207 _direction[d, i] = ((long)m[i]) << (BITS - i);
208 }
209 for (int i = s + 1; i <= BITS; i++)
210 {
211 _direction[d, i] = _direction[d, i - s] ^ (_direction[d, i - s] >> s);
212 for (int k = 1; k <= s - 1; k++)
213 {
214 _direction[d, i] ^= ((a >> (s - 1 - k)) & 1) * _direction[d, i - k];
215 }
216 }

223 public double[] NextDouble()
224 {
225 double[] v = new double[Dimension];
226 if (count == 0)
227 {
228 count++;
229 //return v;
230 }
231
232 // find the index c of the rightmost 0
233 int c = 1;
234 int value = count - 1;
235 while ((value & 1) == 1)
236 {
237 value >>= 1;
238 c++;
239 }
240
241 for (int i = 0; i < Dimension; i++)
242 {
243 x[i] ^= _direction[i, c];
244 v[i] = (double)x[i] / SCALE;
245 }
246 count++;
247 return v;
248 }

255 public double[] SkipTo(int index)
256 {
257 if (index == 0)
258 {
259 // reset x vector
260 x.Fill(0);
261 }
262 else
263 {
264 int i = index - 1;
265 long grayCode = i ^ (i >> 1); // compute the gray code of i = i XOR floor(i / 2)
266 for (int j = 0; j < Dimension; j++)
267 {
268 long result = 0;
269 for (int k = 1; k <= BITS; k++)
270 {
271 long shift = grayCode >> (k - 1);
272 if (shift == 0)
273 {
274 // stop, as all remaining bits will be zero
275 break;
276 }
277 // the k-th bit of i
278 long ik = shift & 1;
279 result ^= ik * _direction[j, k];
280 }
281 x[j] = result;
282 }
283 }
284 count = index;
285 return NextDouble();
286 }
