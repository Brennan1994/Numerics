The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

121 protected override void InitializeCustomSettings()
122 {
123 // Set up multivariate Normal distributions and
124 // adaptive covariance matrix for each chain
125 mvn = new MultivariateNormal[NumberOfChains];
126 sigma = new RunningCovarianceMatrix[NumberOfChains];
127 for (int i = 0; i < NumberOfChains; i++)
128 {
129 mvn[i] = new MultivariateNormal(NumberOfParameters);
130 sigma[i] = new RunningCovarianceMatrix(NumberOfParameters);
131
132 if (Initialize == InitializationType.MAP && _mapSuccessful)
133 {
134 // Hot start the covariance matrix
135 for (int j = 0; j < NumberOfParameters; j++)
136 {
137 for (int k = 0; k < NumberOfParameters; k++)
138 {
139 sigma[i].Covariance[j, k] = _MVN.Covariance[j, k];
140 }
141 }
142 }
143 }

147 protected override ParameterSet ChainIteration(int index, ParameterSet state)
148 {
149
150 // Update the sample count
151 SampleCount[index] += 1;
152
153 if (_chainPRNGs[index].NextDouble() <= Beta || SampleCount[index] <= 100 * NumberOfParameters)
154 {
155 // Use the identity matrix the first 100*D samples
156 mvn[index].SetParameters(state.Values.ToArray(), sigmaIdentity.ToArray());
157 }
158 else
159 {
160 // Use the adaptive covariance matrix
161 mvn[index].SetParameters(state.Values.ToArray(), (sigma[index].Covariance * (1d / (sigma[index].N - 1)) * Scale).ToArray());
162 }
163
164 int c1 = index;
165 if (SampleCount[index] > 100 * NumberOfParameters && _chainPRNGs[index].NextDouble() <= CrossoverProbability)
166 {
167 // Sample from a different chain's covariance matrix
168 do c1 = _chainPRNGs[index].Next(0, NumberOfChains); while (c1 == index);
169 }
170
171 // Get proposal vector
172 var xp = mvn[c1].InverseCDF(_chainPRNGs[index].NextDoubles(NumberOfParameters));
173
174 // Check if the parameter is feasible (within the constraints)
175 for (int i = 0; i < NumberOfParameters; i++)
176 {
177 if (xp[i] < PriorDistributions[i].Minimum || xp[i] > PriorDistributions[i].Maximum)
178 {
179 // The proposed parameter vector was infeasible, so leave xi unchanged.
180 // Adapt Covariance Matrix after warmup
181 if (SampleCount[index] > ThinningInterval * WarmupIterations)
182 sigma[index].Push(state.Values);
183 return state;
184 }
185 }
186
187 // Evaluate fitness
188 var logLHp = LogLikelihoodFunction(xp);
189 var logLHi = state.Fitness;
190
191 // Calculate the Metropolis ratio
192 var logRatio = logLHp - logLHi;
193
194 // Accept the proposal with probability min(1,r)
195 // otherwise leave xi unchanged
196 var logU = Math.Log(_chainPRNGs[index].NextDouble());
197 if (logU <= logRatio)
198 {
199 // The proposal is accepted
200 AcceptCount[index] += 1;
201 // Adapt Covariance Matrix
202 sigma[index].Push(xp);
203 return new ParameterSet(xp, logLHp);
204 }
205 else
206 {
207 // Adapt Covariance Matrix after warmup
208 if (SampleCount[index] > ThinningInterval * WarmupIterations)
209 sigma[index].Push(state.Values);
210 return state;
211 }
212 }
