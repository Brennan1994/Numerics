429 public double[] DirectMethodOfMoments(IList<double> moments)
430 {
431
432 // Solve for kappa
433 double k = SolveForKappa(moments[2]);
434 double a;
435 double x;
436 if (Math.Abs(k) <= NearZero)
437 {
438 x = moments[0] - moments[1];
439 a = moments[1];
440 }
441 else
442 {
443 a = Math.Sqrt(moments[1] * moments[1] * Math.Pow(1d + k, 2d) * (1d + 2d * k));
444 x = moments[0] - a / (1d + k);
445 }
446
447 // return parameters
448 return [x, a, k];
449 }

476 public double[] ModifiedMethodOfMoments(IList<double> sample)
477 {
478 int N = sample.Count;
479 var moments = Statistics.ProductMoments(sample);
480 double min = Statistics.Minimum(sample);
481 double m1 = moments[0];
482 double m2 = Math.Pow(moments[1], 2d);
483 double b = (N - 1) * m2 / (m1 - min) - m1;
484 double c = m1 * m1 - m2 + 2d * m2 * (m1 - N * min) / (m1 - min);
485 double x = -b + Math.Sqrt(b * b - c);
486 double k = 0.5d * (Math.Pow(m1 - x, 2d) / m2 - 1d);
487 double a = 0.5d * ((m1 - x) * (Math.Pow(m1 - x, 2d) / m2 + 1d));
488 // return parameters
489 return [x, a, k];
490 }

500 public double SolveForKappa(double skew)
501 {
502 if (Math.Abs(skew) < 10d)
503 {
504 // Kappa must be solved for. The Brent method is used here.
505 return Brent.Solve((x) =>
506 {
507 double k = 2d * (1d - x) * Math.Sqrt(1d + 2d * x) / (1d + 3d * x);
508 return k - skew;
509 }, -(1d / 3d), 1d / 3d);
510 }
511 else
512 {
513 return double.NaN;
514 }

518 public double[] ParametersFromLinearMoments(IList<double> moments)
519 {
520 double L1 = moments[0];
521 double L2 = moments[1];
522 double T3 = moments[2];
523 double kappa = (1.0d - 3.0d * T3) / (1.0d + T3);
524 double alpha = (1.0d + kappa) * (2.0d + kappa) * L2;
525 double xi = L1 - (2.0d + kappa) * L2;
526 return [xi, alpha, kappa];
527 }

530 public double[] LinearMomentsFromParameters(IList<double> parameters)
531 {
532 double xi = parameters[0];
533 double alpha = parameters[1];
534 double kappa = parameters[2];
535 if (kappa <= -1.0d)
536 throw new ArgumentOutOfRangeException(nameof(Kappa), "L-moments can only be defined for kappa > -1.");
537 double L1 = xi + alpha / (1.0d + kappa);
538 double L2 = alpha / ((1.0d + kappa) * (2.0d + kappa));
539 double T3 = (1.0d - kappa) / (3.0d + kappa);
540 double T4 = (1.0d - kappa) * (2.0d - kappa) / ((3.0d + kappa) * (4.0d + kappa));
541 return [L1, L2, T3, T4];
542 }

545 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
546 {
547 var initialVals = new double[NumberOfParameters];
548 var lowerVals = new double[NumberOfParameters];
549 var upperVals = new double[NumberOfParameters];
550 //
551 // Get initial values
552 initialVals = ParametersFromLinearMoments(Statistics.LinearMoments(sample));
553 double minData = Statistics.Minimum(sample);
554 // Get bounds of location
555 if (initialVals[0] == 0d) initialVals[0] = Tools.DoubleMachineEpsilon;
556 lowerVals[0] = initialVals[0] - Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0]))));
557 upperVals[0] = minData + Tools.DoubleMachineEpsilon;
558
559 // Get bounds of scale
560 lowerVals[1] = Tools.DoubleMachineEpsilon;
561 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[1])) + 1d));
562 // Get bounds of shape
563 lowerVals[2] = -10d;
564 upperVals[2] = 10d;
565 // Correct initial values if necessary
566 if (initialVals[0] <= lowerVals[0] || initialVals[0] >= upperVals[0])
567 {
568 initialVals[0] = Statistics.Mean(new[] { lowerVals[0], upperVals[0] });
569 }
570 if (initialVals[1] <= lowerVals[1] || initialVals[1] >= upperVals[1])
571 {
572 initialVals[1] = Statistics.Mean(new[] { lowerVals[1], upperVals[1] });
573 }
574 if (initialVals[2] <= lowerVals[2] || initialVals[2] >= upperVals[2])
575 {
576 initialVals[2] = 0d;
577 }
578 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
579 }

582 public double[] MLE(IList<double> sample)
583 {
584 // Set constraints
585 var tuple = GetParameterConstraints(sample);
586 var Initials = tuple.Item1;
587 var Lowers = tuple.Item2;
588 var Uppers = tuple.Item3;
589
590 var xi = Statistics.Minimum(sample);
591
592 // Solve using Nelder-Mead (Downhill Simplex)
593 double logLH(double[] x)
594 {
595 var GPA = new GeneralizedPareto();
596 GPA.SetParameters(new[] { xi, x[0], x[1] });
597 //GPA.SetParameters(x);
598 return GPA.LogLikelihood(sample);
599 }
600 var solver = new NelderMead(logLH, 2, Initials.Subset(1), Lowers.Subset(1), Uppers.Subset(1));
601 solver.ReportFailure = true;
602 solver.Maximize();
603 return [xi, solver.BestParameterSet.Values[0], solver.BestParameterSet.Values[1]];
604 }

607 public override double PDF(double x)
608 {
609 // Validate parameters
610 if (_parametersValid == false)
611 ValidateParameters(Xi, Alpha, Kappa, true);
612 if (x < Minimum || x > Maximum) return 0.0d;
613 double y = (x - Xi) / Alpha;
614 if (Math.Abs(Kappa) > NearZero)
615 y = -Math.Log(1d - Kappa * y) / Kappa;
616 return Math.Exp(-(1d - Kappa) * y) / Alpha;
617 }

620 public override double CDF(double x)
621 {
622 // Validate parameters
623 if (_parametersValid == false)
624 ValidateParameters(Xi, Alpha, Kappa, true);
625 if (x <= Minimum)
626 return 0d;
627 if (x >= Maximum)
628 return 1d;
629 double y = (x - Xi) / Alpha;
630 if (Math.Abs(Kappa) > NearZero)
631 y = -Math.Log(1d - Kappa * y) / Kappa;
632 return 1d - Math.Exp(-y);

636 public override double InverseCDF(double probability)
637 {
638 // Validate probability
639 if (probability < 0.0d || probability > 1.0d)
640 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
641 if (probability == 0.0d)
642 return Minimum;
643 if (probability == 1.0d)
644 return Maximum;
645 // Validate parameters
646 if (_parametersValid == false)
647 ValidateParameters(Xi, Alpha, Kappa, true);
648 if (Math.Abs(Kappa) <= NearZero)
649 {
650 return Xi - Alpha * Math.Log(1d - probability);
651 }
652 else
653 {
654 return Xi + Alpha / Kappa * (1d - Math.Pow(1d - probability, Kappa));

665 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
666 {
667 if (estimationMethod != ParameterEstimationMethod.MethodOfMoments &&
668 estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
669 {
670 throw new NotImplementedException();
671 }
672 // Validate parameters
673 if (_parametersValid == false)
674 ValidateParameters(new[] { Xi, _alpha }, true);
675 double a = Alpha;
676 double k = Kappa;
677 int N = sampleSize;
678 var covar = new double[3, 3];
679 covar[0, 0] = N * a * a / ((N + 2d * k) * Math.Pow(N + k, 2d)); // location
680 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
681 {
682 double num = Math.Pow(1d + k, 2d) * (1d + 6d * k + 12d * Math.Pow(k, 2d));
683 double den = (1d + 2d * k) * (1d + 3d * k) * (1d + 4d * k);
684 covar[1, 1] = 2d * a * a / N * num / den; // scale
685 //
686 num = Math.Pow(1d + k, 2d) * Math.Pow(1d + 2d * k, 2d) * (1d + k + 6d * Math.Pow(k, 2d));
687 covar[2, 2] = 1d / N * num / den; // shape
688 //
689 covar[0, 1] = 0.0;
690 covar[0, 2] = 0.0;
691 covar[1, 0] = 0.0;
692 covar[2, 0] = 0.0;
693 //
694 num = Math.Pow(1d + k, 2d) * (1d + 2d * k) * (1d + 4d * k + 12d * Math.Pow(k, 2d));
695 den = (1d + 2d * k) * (1d + 3d * k) * (1d + 4d * k);
696 covar[2, 1] = a / N * num / den; // scale & shape
697 covar[1, 2] = covar[2, 1];
698 }
699 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
700 {
701 covar[1, 1] = (1d - k) * (2d * a * a) / N; // scale
702 covar[2, 2] = 1d / N * Math.Pow(1d - k, 2d); // shape
703 //
704 covar[0, 1] = 0.0;
705 covar[0, 2] = 0.0;
706 covar[1, 0] = 0.0;
707 covar[2, 0] = 0.0;
708 //
709 covar[2, 1] = a / N * (1d - k); // scale & shape
710 covar[1, 2] = covar[2, 1];
711 }
712
713 return covar;
714 }

717 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
718 {
719 // The variance in the location parameter is of order N-2 and thus
720 // can be neglected relative to the variances of scale and shape (order N-1)
721 // to obtain approximate confidence intervals.
722 var covar = ParameterCovariance(sampleSize, estimationMethod);
723 var grad = QuantileGradient(probability);
724 double varA = covar[1, 1];
725 double varB = covar[2, 2];
726 double covAB = covar[2, 1];
727 double dQx1 = grad[1];
728 double dQx2 = grad[2];
729 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + 2d * dQx1 * dQx2 * covAB;
730 }

733 public double[] QuantileGradient(double probability)
734 {
735 if (_parametersValid == false)
736 ValidateParameters(Xi, _alpha, Kappa, true);
737 double p = probability;
738 double a = Alpha;
739 double k = Kappa;
740 var gradient = new double[]
741 {
742 1.0d, // location
743 1d / k * (1d - Math.Pow(1d - p, Kappa)), // scale
744 -a / (k * k) * (1d - Math.Pow(1d - p, k)) - a / k * Math.Log(1d - p) * Math.Pow(1d - p, k) // shape
745 };
746 return gradient;
747 }

750 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
751 {
752 if (probabilities.Count != NumberOfParameters)
753 {
754 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
755 }
756
757 // Get gradients
758 var dQp1 = QuantileGradient(probabilities[0]);
759 var dQp2 = QuantileGradient(probabilities[1]);
760 var dQp3 = QuantileGradient(probabilities[2]);
761 // Compute determinant
762 // |a b c|
763 // |d e f|
764 // |g h i|
765 // |A| = a(ei − fh) − b(di − fg) + c(dh − eg)
766 double a = dQp1[0];
767 double b = dQp1[1];
768 double c = dQp1[2];
769 double d = dQp2[0];
770 double e = dQp2[1];
771 double f = dQp2[2];
772 double g = dQp3[0];
773 double h = dQp3[1];
774 double i = dQp3[2];
775 determinant = a * (e * i - f * h) - b * (d * i - f * g) + c * (d * h - e * g);
776 // Return Jacobian
777 var jacobian = new double[,] { { a, b, c }, { d, e, f }, { g, h, i } };
778 return jacobian;
779 }
