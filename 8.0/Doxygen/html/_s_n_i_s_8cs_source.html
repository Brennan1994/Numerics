Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

111 public override void Sample()
112 {
113 InitializeChains();
114 InitializeCustomSettings();
115 ValidateSettings();
116 ParallelizeChains = true;
117 ResumeSimulation = false;
118 CancellationTokenSource = new CancellationTokenSource();
119
120 // Create inputs
121 _masterPRNG = new MersenneTwister(PRNGSeed);
122 MarkovChains = new List<ParameterSet>[NumberOfChains];
123 var sets = new ParameterSet[Iterations];
124 MarkovChains[0] = sets.ToList();
125 Output = new List<ParameterSet>[NumberOfChains];
126 Output[0] = new List<ParameterSet>();
127
128 // Create sample & accept counts
129 AcceptCount = new int[NumberOfChains];
130 SampleCount = new int[NumberOfChains];
131
132 // Create mean log-likelihood list
133 MeanLogLikelihood = new List<double>();
134
135 // Keeps track of best parameter set
136 MAP = new ParameterSet([], double.MinValue, double.MinValue);
137
138 // Create parameter random values
139 var rnds = _masterPRNG.NextDoubles(Iterations, NumberOfParameters);
140
141 // Perform sampling
142 Parallel.For(0, Iterations, (idx) =>
143 {
144 // Sample parameters
145 var parameters = new double[NumberOfParameters];
146 double logLH = 0;
147 double weight = 0;
148 if (useImportanceSampling == true)
149 {
150 parameters = mvn.InverseCDF(rnds.GetRow(idx));
151 logLH = LogLikelihoodFunction(parameters);
152 weight = logLH - mvn.LogPDF(parameters);
153 }
154 else
155 {
156 for (int i = 0; i < NumberOfParameters; i++)
157 parameters[i] = PriorDistributions[i].InverseCDF(rnds[idx, i]);
158 logLH = LogLikelihoodFunction(parameters);
159 weight = logLH;
160 }
161 // Record sample
162 MarkovChains[0][idx] = new ParameterSet(parameters, logLH, weight);
163 });
164
165 // Get the maximum a posteriori
166 for (int i = 0; i < Iterations; i++)
167 if (MarkovChains[0][i].Weight > MAP.Weight)
168 MAP = MarkovChains[0][i].Clone();
169
170 // Get the normalization factor
171 double max = MAP.Weight;
172 double sum = 0;
173 for (int i = 0; i < Iterations; i++)
174 {
175 if (MarkovChains[0][i].Weight != double.MinValue)
176 {
177 sum += Math.Exp(MarkovChains[0][i].Weight - max);
178 }
179 }
180 double normalization = max + Math.Log(sum);
181
182 // Compute the posterior weights
183 Parallel.For(0, Iterations, (idx) =>
184 {
185 double w = MarkovChains[0][idx].Weight != double.MinValue ? Math.Exp(MarkovChains[0][idx].Weight - normalization) : 0d;
186 MarkovChains[0][idx] = new ParameterSet(MarkovChains[0][idx].Values, MarkovChains[0][idx].Fitness, w);
187 });
188
189 // Sort list in ascending order of posterior weights
190 MarkovChains[0].Sort((x, y) => x.Fitness.CompareTo(y.Fitness));
191 var cdf = new double[Iterations];
192 cdf[0] = MarkovChains[0][0].Weight;
193
194 // Create CDF
195 for (int i = 1; i < Iterations; i++)
196 cdf[i] = cdf[i-1] + MarkovChains[0][i].Weight;
197
198 // Record output
199 var rndOut = PlottingPositions.Weibull(OutputLength);
200 int idx = 0;
201 for (int i = 0; i < OutputLength; i++)
202 {
203 idx = Search.Sequential(rndOut[i], cdf, idx);
204 Output[0].Add(MarkovChains[0][idx].Clone());
205 }
206 }
207
208 }
