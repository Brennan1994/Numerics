80 public void Push(IList<double> values)
81 {
82 N += 1;
83 var x = new Matrix(values.ToArray());
84 // Update mean
85 var oldMean = Mean;
86 Mean += N == 1 ? x : (x - Mean) * (1d / N);
87 // Update covariance
88 Covariance += (x - oldMean) * Matrix.Transpose(x - Mean);
