124 public static TimeSeries FromGHCN(string siteNumber, GHCNTimeSeriesType timeSeriesType = GHCNTimeSeriesType.DailyPrecipitation, DepthUnit unit = DepthUnit.Inches)
125 {
126 var timeSeries = new TimeSeries(TimeInterval.OneDay);
127
128 try
129 {
130 // Check if there is an Internet connection
131 if (IsConnectedToInternet() == false)
132 throw new Exception("There is no Internet connection!");
133
134 // Setup url parameters
135 string url = "https://ncei.noaa.gov/pub/data/ghcn/daily/by_station/" + siteNumber + ".csv.gz";
136
137 // Temp file names
138 string zipFileName = "C:/Temp/" + siteNumber + ".csv.gz";
139
140 // Download data to temp directory
141 using (var client = new WebClient())
142 {
143 client.DownloadFile(url, zipFileName);
144 }
145
146 // Unzip the file
147 FileInfo fileToDecompress = new FileInfo(zipFileName);
148 using (FileStream originalFileStream = fileToDecompress.OpenRead())
149 {
150 string currentFileName = fileToDecompress.FullName;
151 string newFileName = currentFileName.Remove(currentFileName.Length - fileToDecompress.Extension.Length);
152 using (FileStream decompressedFileStream = File.Create(newFileName))
153 {
154 using (GZipStream decompressionStream = new GZipStream(originalFileStream, CompressionMode.Decompress))
155 {
156 decompressionStream.CopyTo(decompressedFileStream);
157 }
158 }
159 }
160
161 // Parse the file and create time series
162 // import at-site data
163 string filePath = "C:/Temp/" + siteNumber + ".csv";
164 StreamReader reader = null;
165 if (File.Exists(filePath))
166 {
167 reader = new StreamReader(File.OpenRead(filePath));
168 while (!reader.EndOfStream)
169 {
170 var line = reader.ReadLine().Split(',');
171
172 // Get date
173 var dateString = line[1];
174 int.TryParse(dateString.Substring(0, 4), out var year);
175 int.TryParse(dateString.Substring(4, 2), out var month);
176 int.TryParse(dateString.Substring(6, 2), out var day);
177 var dateTime = new DateTime(year, month, day);
178
179 // Get type
180 var typeString = line[2];
181
182 // Get value
183 double.TryParse(line[3], out var value);
184 if ((typeString == "PRCP" && timeSeriesType == GHCNTimeSeriesType.DailyPrecipitation) || (typeString == "SNOW" && timeSeriesType == GHCNTimeSeriesType.DailySnow))
185 {
186 value /= 10d;
187 if (unit == DepthUnit.Millimeters)
188 {
189 timeSeries.Add(new SeriesOrdinate<DateTime, double>(dateTime, value));
190 }
191 else if (unit == DepthUnit.Centimeters)
192 {
193 timeSeries.Add(new SeriesOrdinate<DateTime, double>(dateTime, value / 10d));
194 }
195 else if (unit == DepthUnit.Inches)
196 {
197 timeSeries.Add(new SeriesOrdinate<DateTime, double>(dateTime, value / 25.4d));
198 }
199 }
200 }
201 reader.Close();
202 }
203 else
204 {
205 throw new Exception("File doesn't exist");
206 }
207
208 // Delete the zip file and .csv file
209 File.Delete(zipFileName);
210 File.Delete(filePath);
211
212 }
213 catch (Exception ex)
214 {
215 throw ex;
216 }
217
218 return timeSeries;
219 }

226 public static TimeSeries FromUSGS(string siteNumber, USGSTimeSeriesType timeSeriesType = USGSTimeSeriesType.DailyDischarge)
227 {
228 var timeSeries = new TimeSeries();
229
230 try
231 {
232 // Check if there is an Internet connection
233 if (IsConnectedToInternet() == false)
234 throw new Exception("There is no Internet connection!");
235
236 // Setup url parameters
237 string timeInterval = "dv";
238 string startDate = "1800-01-01";
239 string endDate = DateTime.Now.ToString("yyyy-MM-dd");
240 string statCode = "&statCd=00003";
241 string parameterCode = "00060";
242 string url = "";
243
244 if (timeSeriesType == USGSTimeSeriesType.DailyDischarge || timeSeriesType == USGSTimeSeriesType.DailyStage)
245 {
246 timeInterval = "dv";
247 statCode = "&statCd=00003";
248 parameterCode = timeSeriesType == USGSTimeSeriesType.DailyDischarge ? "00060" : "00065";
249 timeSeries = new TimeSeries(TimeInterval.OneDay);
250 url = "https://waterservices.usgs.gov/nwis/" + timeInterval + "/?format=waterml,2.0&sites=" + siteNumber + "&startDT=" + startDate + "&endDT=" + endDate + statCode + "&parameterCd=" + parameterCode + "&siteStatus=all";
251 }
252 else if (timeSeriesType == USGSTimeSeriesType.InstantaneousDischarge || timeSeriesType == USGSTimeSeriesType.InstantaneousStage)
253 {
254 timeInterval = "iv";
255 statCode = "";
256 parameterCode = timeSeriesType == USGSTimeSeriesType.InstantaneousDischarge ? "00060" : "00065";
257 timeSeries = new TimeSeries(TimeInterval.FifteenMinute);
258 url = "https://waterservices.usgs.gov/nwis/" + timeInterval + "/?format=waterml,2.0&sites=" + siteNumber + "&startDT=" + startDate + "&endDT=" + endDate + statCode + "&parameterCd=" + parameterCode + "&siteStatus=all";
259 }
260 else if (timeSeriesType == USGSTimeSeriesType.PeakDischarge || timeSeriesType == USGSTimeSeriesType.PeakStage)
261 {
262 timeInterval = "peak";
263 statCode = "";
264 timeSeries = new TimeSeries(TimeInterval.Irregular);
265 url = "https://nwis.waterdata.usgs.gov/nwis/peak?site_no=" + siteNumber + "&agency_cd=USGS&format=rdb";
266 }
267
268 // Download data
269 string textDownload;
270 using (var client = new WebClient())
271 textDownload = client.DownloadString(url);
272
273 if (timeSeriesType == USGSTimeSeriesType.DailyDischarge || timeSeriesType == USGSTimeSeriesType.DailyStage)
274 {
275 // Convert to XElement and check if the download was valid
276 var xElement = XElement.Parse(textDownload);
277 var points = xElement.Descendants("{http://www.opengis.net/waterml/2.0}point");
278
279 // Create time series
280 foreach (XElement point in points.Elements())
281 {
282 if (point.Element("{http://www.opengis.net/waterml/2.0}time") != null && point.Element("{http://www.opengis.net/waterml/2.0}value") != null)
283 {
284 // Get date
285 DateTime index = DateTime.Now;
286 DateTime.TryParse(point.Element("{http://www.opengis.net/waterml/2.0}time").Value, out index);
287
288 // See if this date is
289 if (timeSeries.Count > 0 && index != TimeSeries.AddTimeInterval(timeSeries.Last().Index, TimeInterval.OneDay))
290 {
291 while (timeSeries.Last().Index < TimeSeries.SubtractTimeInterval(index, TimeInterval.OneDay))
292 timeSeries.Add(new SeriesOrdinate<DateTime, double>(TimeSeries.AddTimeInterval(timeSeries.Last().Index, TimeInterval.OneDay), double.NaN));
293 }
294
295 // Get value
296 double value = 0;
297 string valueStg = point.Element("{http://www.opengis.net/waterml/2.0}value").Value;
298 if (valueStg == "" || valueStg == " " || valueStg == " " || string.IsNullOrEmpty(valueStg))
299 value = double.NaN;
300 else
301 {
302 double.TryParse(valueStg, out value);
303 }
304 timeSeries.Add(new SeriesOrdinate<DateTime, double>(index, value));
305 }
306 }
307 }
308 else if (timeSeriesType == USGSTimeSeriesType.PeakDischarge || timeSeriesType == USGSTimeSeriesType.PeakStage)
309 {
310 var delimiters = new char[] { '\t' };
311 var lines = textDownload.Split(new string[] { Environment.NewLine }, StringSplitOptions.RemoveEmptyEntries);
312 foreach (string line in lines)
313 {
314 var segments = line.Split(delimiters, StringSplitOptions.RemoveEmptyEntries);
315 if (segments.Count() >= 4 && segments.First() == "USGS")
316 {
317 // Get date
318 DateTime index = DateTime.Now;
319 var dateString = segments[2].Split('-');
320 if (dateString[1] == "00")
321 {
322 int year = 2000;
323 int.TryParse(dateString[0], out year);
324 index = new DateTime(year, 1, 1, 0, 0, 0);
325 }
326 else
327 {
328 DateTime.TryParse(segments[2], out index);
329 }
330
331 // Get value
332 double value = 0;
333
334 // see if the 4th column has a time
335 int offset = 0;
336 var timeString = segments[3].Split(':');
337 if (timeString.Length == 2)
338 offset = 1;
339
340 int idx = timeSeriesType == USGSTimeSeriesType.PeakDischarge ? 3 + offset : 4 + offset;
341 if (segments[idx] == "" || segments[idx] == " " || segments[idx] == " " || string.IsNullOrEmpty(segments[idx]))
342 value = double.NaN;
343 else
344 {
345 double.TryParse(segments[idx], out value);
346 }
347 timeSeries.Add(new SeriesOrdinate<DateTime, double>(index, value));
348 }
349 }
350 }
351 }
352 catch (Exception ex)
353 {
354 throw ex;
355 }
356
357 return timeSeries;
358 }
