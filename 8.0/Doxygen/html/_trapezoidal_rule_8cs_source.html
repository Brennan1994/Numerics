107
108 // Check function evaluations
109 if (FunctionEvaluations >= MaxFunctionEvaluations)
110 {
111 UpdateStatus(IntegrationStatus.MaximumFunctionEvaluationsReached);
112 return;
113 }
114
115 // Check convergence
116 if (i > 2)
117 {
118 if (EvaluateConvergence(s1, s0) || (s0 == 0.0 && s1 == 0.0))
119 {
120 Result = s0;
121 UpdateStatus(IntegrationStatus.Success);
122 return;
123 }
124 }
125 s1 = s0;
126 }
127 // If we get to here, then the maximum number of steps were reached before converging.
128 UpdateStatus(IntegrationStatus.MaximumIterationsReached);
129 }
130 catch (Exception ex)
131 {
132 UpdateStatus(IntegrationStatus.Failure, ex);
133 }
134 }

139 private double Next()
140 {
141 double x, tnm, sum, del;
142 int it, j;
143 Iterations++;
144 if (Iterations == 1)
145 {
146 _s = 0.5 * (b - a) * (Function(a) + Function(b));
147 FunctionEvaluations += 2;
148 }
149 else
150 {
151 for (it = 1, j = 1; j < Iterations - 1; j++) it <<= 1;
152 tnm = it;
153 del = (b - a) / tnm;
154 x = a + 0.5 * del;
155 for (sum = 0.0, j = 0; j < it; j++, x += del)
156 {
157 sum += Function(x);
158 FunctionEvaluations += 1;
159 }
160 _s = 0.5 * (_s + (b - a) * sum / tnm);
161 }
162 return _s;
163 }
