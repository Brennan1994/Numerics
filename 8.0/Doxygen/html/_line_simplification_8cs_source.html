63 public static void RamerDouglasPeucker(List<Ordinate> ordinates, double epsilon, ref List<Ordinate> output)
64 {
65 if (ordinates.Count < 2)
66 throw new ArgumentOutOfRangeException("Not enough points to simplify");
67
68 // Find the point with the maximum distance from line between the start and end
69 double dmax = 0.0;
70 int index = 0;
71 int end = ordinates.Count - 1;
72 for (int i = 1; i < end; ++i)
73 {
74 double d = PerpendicularDistance(ordinates[i], ordinates[0], ordinates[end]);
75 if (d > dmax)
76 {
77 index = i;
78 dmax = d;
79 }
80 }
81
82 // If max distance is greater than epsilon, recursively simplify
83 if (dmax > epsilon)
84 {
85 List<Ordinate> recResults1 = new List<Ordinate>();
86 List<Ordinate> recResults2 = new List<Ordinate>();
87 List<Ordinate> firstLine = ordinates.Take(index + 1).ToList();
88 List<Ordinate> lastLine = ordinates.Skip(index).ToList();
89 RamerDouglasPeucker(firstLine, epsilon, ref recResults1);
90 RamerDouglasPeucker(lastLine, epsilon, ref recResults2);
91
92 // build the result list
93 output.AddRange(recResults1.Take(recResults1.Count - 1));
94 output.AddRange(recResults2);
95 if (output.Count < 2) throw new Exception("Problem assembling output");
96 }
97 else
98 {
99 // Just return start and end points
100 output.Clear();
101 output.Add(ordinates[0]);
102 output.Add(ordinates[ordinates.Count - 1]);
103 }
104
105 }

114 public static double PerpendicularDistance(Ordinate pt, Ordinate lineStart, Ordinate lineEnd)
115 {
116 double dx = lineEnd.X - lineStart.X;
117 double dy = lineEnd.Y - lineStart.Y;
118
119 // Normalize
120 double mag = Math.Sqrt(dx * dx + dy * dy);
121 if (mag > 0.0)
122 {
123 dx /= mag;
124 dy /= mag;
125 }
126 double pvx = pt.X - lineStart.X;
127 double pvy = pt.Y - lineStart.Y;
128
129 // Get dot product (project pv onto normalized direction)
130 double pvdot = dx * pvx + dy * pvy;
131
132 // Scale line direction vector and subtract it from pv
133 double ax = pvx - pvdot * dx;
134 double ay = pvy - pvdot * dy;
135
136 return Math.Sqrt(ax * ax + ay * ay);
137 }
138 }
