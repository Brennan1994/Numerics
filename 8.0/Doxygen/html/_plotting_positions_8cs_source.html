61 public static double[] Function(int N, double alpha)
62 {
63 if (N <= 2)
64 {
65 throw new ArgumentOutOfRangeException(nameof(N), "The sample size N must be greater than 2.");
66 }
67
68 if (alpha < 0d || alpha > 1d)
69 {
70 throw new ArgumentOutOfRangeException(nameof(alpha), "The alpha coefficient must be between 0 and 1.");
71 }
72
73 var PP = new double[N];
74 for (int i = 1; i <= N; i++)
75 PP[i - 1] = (i - alpha) / (N + 1 - 2d * alpha);
76 return PP;
77 }

185 public static double[] KaplanMeier(int N)
186 {

188 {
189 throw new ArgumentOutOfRangeException(nameof(N), "The sample size N must be greater than 2.");

191 var PP = new double[N];
