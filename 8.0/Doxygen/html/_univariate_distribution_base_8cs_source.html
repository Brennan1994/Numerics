313 public virtual double[] CentralMoments(double tolerance = 1E-8)
314 {
315 double u1 = 0, u2 = 0, u3 = 0, u4 = 0;
316 double a = InverseCDF(1E-16);
317 double b = InverseCDF(1-1E-16);
318 if (a >= b) return new double[] { a, double.NaN, double.NaN, double.NaN };
319
320 // Mean
321 var sr = new AdaptiveSimpsonsRule(x => { return x * PDF(x); }, a, b) { RelativeTolerance = tolerance, ReportFailure = false };
322 sr.Integrate();
323 u1 = sr.Status != IntegrationStatus.Failure ? sr.Result : double.NaN;
324 // Standard Deviation
325 sr = new AdaptiveSimpsonsRule(x => { return Math.Pow(x - u1, 2d) * PDF(x); }, a, b) { RelativeTolerance = tolerance, ReportFailure = false };
326 sr.Integrate();
327 u2 = sr.Status != IntegrationStatus.Failure ? sr.Result : double.NaN;
328 u2 = Math.Sqrt(u2);
329 // Skewness
330 sr = new AdaptiveSimpsonsRule(x => { return Math.Pow((x - u1) / u2, 3d) * PDF(x); }, a, b) { RelativeTolerance = tolerance, ReportFailure = false };
331 sr.Integrate();
332 u3 = sr.Status != IntegrationStatus.Failure ? sr.Result : double.NaN;
333 // Kurtosis
334 sr = new AdaptiveSimpsonsRule(x => { return Math.Pow((x - u1) / u2, 4d) * PDF(x); }, a, b) { RelativeTolerance = tolerance, ReportFailure = false };
335 sr.Integrate();
336 u4 = sr.Status != IntegrationStatus.Failure ? sr.Result : double.NaN;
337
338 return new double[] { u1, u2, u3, u4 };
339 }

345 public virtual double[] CentralMoments(int steps = 200)
346 {
347 double a = InverseCDF(1E-8);
348 double b = InverseCDF(1 - 1E-8);
349 if (a >= b) return [a, double.NaN, double.NaN, double.NaN];
350
351 var bins = Stratify.XValues(new StratificationOptions(a, b, steps));
352 var dFx = new double[steps];
353 double u1, u2, u3, u4;
354 double sumU1 = 0;
355 double sumU2 = 0;
356 double sumU3 = 0;
357 double sumU4 = 0;
358
359 // First compute the mean and standard deviation
360 dFx[0] = CDF(bins[0].UpperBound);
361 sumU1 += bins[0].UpperBound * dFx[0];
362 sumU2 += Math.Pow(bins[0].UpperBound, 2d) * dFx[0];
363 for (int i = 1; i < steps - 1; i++)
364 {
365 dFx[i] = CDF(bins[i].UpperBound) - CDF(bins[i].LowerBound);
366 sumU1 += bins[i].Midpoint * dFx[i];
367 sumU2 += Math.Pow(bins[i].Midpoint, 2d) * dFx[i];
368 }
369 dFx[steps - 1] = 1 - CDF(bins.Last().LowerBound);
370 sumU1 += bins.Last().LowerBound * dFx[steps - 1];
371 sumU2 += Math.Pow(bins.Last().LowerBound, 2d) * dFx[steps - 1];
372 u1 = sumU1;
373 u2 = Math.Sqrt(sumU2 - Math.Pow(u1, 2d));
374
375 // Then compute skewness and kurtosis
376 sumU3 += Math.Pow((bins[0].UpperBound - u1) / u2, 3d) * dFx[0];
377 sumU4 += Math.Pow((bins[0].UpperBound - u1) / u2, 4d) * dFx[0];
378 for (int i = 1; i < steps - 1; i++)
379 {
380 sumU3 += Math.Pow((bins[i].Midpoint - u1) / u2, 3d) * dFx[i];
381 sumU4 += Math.Pow((bins[i].Midpoint - u1) / u2, 4d) * dFx[i];
382 }
383 sumU3 += Math.Pow((bins.Last().LowerBound - u1) / u2, 3d) * dFx[steps - 1];
384 sumU4 += Math.Pow((bins.Last().LowerBound - u1) / u2, 4d) * dFx[steps - 1];
385 u3 = sumU3;
386 u4 = sumU4;
387 return [u1, u2, u3, u4];
388 }

692 public static bool operator ==(UnivariateDistributionBase left, UnivariateDistributionBase right)
693 {
694 // Check for null arguments. Keep in mind null == null
695 if (left is null && right is null)
696 {
697 return true;
698 }
699 else if (left is null)
700 {
701 return false;
702 }
703 else if (right is null)
704 {
705 return false;
706 }
707 // Check to see if they are the same distribution
708 if (left.Type != right.Type)
709 return false;
710 // Test for equality of parameters
711 // Compare parameters here.
712 // Need to have special cases for kernel density, univariate, and bivariate distributions. Everything else can use the GetParameters method.
713 if (left.Type == UnivariateDistributionType.Empirical)
714 {
715 EmpiricalDistribution leftTyped = (EmpiricalDistribution)left;
716 EmpiricalDistribution rightTyped = (EmpiricalDistribution)right;
717 // Check X values for equality
718 var leftValues = leftTyped.XValues;
719 var rightValues = rightTyped.XValues;
720 if (leftValues.Count != rightValues.Count)
721 return false;
722 for (int i = 0; i < leftValues.Count; i++)
723 {
724 if (Math.Abs(leftValues[i] - rightValues[i]) > Tools.DoubleMachineEpsilon)
725 return false;
726 }
727 // Check P values for equality, no need to test count since it was tested with X values
728 leftValues = leftTyped.ProbabilityValues;
729 rightValues = rightTyped.ProbabilityValues;
730 for (int i = 0; i < leftValues.Count; i++)
731 {
732 if (Math.Abs(leftValues[i] - rightValues[i]) > Tools.DoubleMachineEpsilon)
733 return false;
734 }
735 }
736 else if (left.Type == UnivariateDistributionType.KernelDensity)
737 {
738 throw new NotImplementedException();
739 }
740 else
741 {
742 var leftParams = left.GetParameters;
743 var rightParams = right.GetParameters;
744 for (int i = 0; i < leftParams.Count(); i++)
745 {
746 if (Math.Abs(leftParams[i] - rightParams[i]) > Tools.DoubleMachineEpsilon)
747 return false;
748 }
749 }
750 // All tests passed
751 return true;
752 //
753 }
