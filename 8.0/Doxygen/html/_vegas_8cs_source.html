295 private void vegas(Func<double[], double, double> fxn, double[] regn, int init, int ncall, int itmx, ref double tgral, ref double sd, ref double chi2a)
296 {
297 // Performs Monte Carlo integration of a user - supplied ndim - dimensional function fxn over a
298 // rectangular volume specified by regn[0..2 * ndim - 1], a vector consisting of ndim “lower left”
299 // coordinates of the region followed by ndim “upper right” coordinates. The integration consists
300 // of itmx iterations, each with approximately ncall calls to the function. After each iteration
301 // the grid is refined; more than 5 or 10 iterations are rarely useful. The input flag init signals
302 // whether this call is a new start or a subsequent call for additional iterations (see comments in the
303 // code). The input flag nprn (normally 0) controls the amount of diagnostic output. Returned
304 // answers are tgral (the best estimate of the integral), sd(its standard deviation), and chi2a
305 // (Chi␟2 per degree of freedom, an indicator of whether consistent results are being obtained). See
306 // text for further details.
307
308 int ndim = regn.Length / 2;
309 if (init <= 0)
310 {
311 // Normal entry. Enter here on a cold start.
312 // Change mds = 0 to disable stratified sampling, i.e. use importance sampling only.
313 mds = ndo = 1;
314 for (j = 0; j < ndim; j++) xi[j, 0] = 1.0;
315 }
316 if (init <= 1)
317 {
318 // Enter here to inherit the grid from a previous call, but not its answers.
319 si = swgt = schi = 0.0;
320 }
321 if (init <= 2)
322 {
323 // Enter here to inherit the previous grid and its answers.
324 nd = Bins;
325 ng = 1;
326 // Set up for stratification
327 if (mds != 0)
328 {
329 ng = (int)Math.Pow(ncall / 2.0 + 0.25, 1.0 / ndim);
330 mds = 1;
331 if ((2 * ng - Bins) >= 0)
332 {
333 mds = -1;
334 npg = ng / Bins + 1;
335 nd = ng / npg;
336 ng = npg * nd;
337 }
338 }
339 for (k = 1, i = 0; i < ndim; i++) k *= ng;
340 npg = Math.Max((int)(ncall / k), 2);
341 calls = (double)(npg) * (double)(k);
342 dxg = 1.0 / ng;
343 for (dv2g = 1, i = 0; i < ndim; i++) dv2g *= dxg;
344 dv2g = Tools.Sqr(calls * dv2g) / npg / npg / (npg - 1.0);
345 xnd = nd;
346 dxg *= xnd;
347 xjac = 1.0 / calls;
348 for (j = 0; j < ndim; j++)
349 {
350 dx[j] = regn[j + ndim] - regn[j];
351 xjac *= dx[j];
352 }
353 // Do binning if necessary
354 if (nd != ndo)
355 {
356 for (i = 0; i < Math.Max(nd, ndo); i++) r[i] = 1.0;
357 for (j = 0; j < ndim; j++)
358 rebin(ndo / xnd, nd, r, xin, xi, j);
359 ndo = nd;
360 }
361
362 }
363 // Main iteration loop. Can enter here (init >= 3) to do an additional itmx iteration with all other parameters unchanged.
364 for (it = 0; it < itmx; it++)
365 {
366 Iterations++;
367
368 ti = tsi = 0.0;
369 for (j = 0; j < ndim; j++)
370 {
371 kg[j] = 1;
372 for (i = 0; i < nd; i++) d[i, j] = di[i, j] = 0.0;
373 }
374 for (; ; )
375 {
376 fb = f2b = 0.0;
377 for (k = 0; k < npg; k++)
378 {
379 wgt = xjac;
380
381 // Sobol Sequence quasi-random numbers
382 double[] rnd = null;
383 if (UseSobolSequence)
384 rnd = _sobol.NextDouble();
385
386 for (j = 0; j < ndim; j++)
387 {
388 xn = (kg[j] - (UseSobolSequence ? rnd[j] : Random.NextDouble())) * dxg + 1.0;
389 ia[j] = Math.Max(Math.Min((int)xn, Bins), 1);
390 if (ia[j] > 1)
391 {
392 xo = xi[j, ia[j] - 1] - xi[j, ia[j] - 2];
393 rc = xi[j, ia[j] - 2] + (xn - ia[j]) * xo;
394 }
395 else
396 {
397 xo = xi[j, ia[j] - 1];
398 rc = (xn - ia[j]) * xo;
399 }
400 x[j] = regn[j] + rc * dx[j];
401 wgt *= xo * xnd;
402 }
403 f = wgt * fxn(x, wgt);
404 // Keep track of function evaluations
405 FunctionEvaluations++;
406 f2 = f * f;
407 fb += f;
408 f2b += f2;
409 for (j = 0; j < ndim; j++)
410 {
411 di[ia[j] - 1, j] += f;
412 if (mds >= 0) d[ia[j] - 1, j] += f2;
413 }
414 }
415 f2b = Math.Sqrt(f2b * npg);
416 f2b = (f2b - fb) * (f2b + fb);
417 if (f2b <= 0.0) f2b = TINY;
418 ti += fb;
419 tsi += f2b;
420 if (mds < 0)
421 {
422 // Use stratified sampling
423 for (j = 0; j < ndim; j++) d[ia[j] - 1, j] += f2b;
424 }
425 for (k = ndim - 1; k >= 0; k--)
426 {
427 kg[k] %= ng;
428 if (++kg[k] != 1) break;
429 }
430 if (k < 0) break;
431 }
432 // Compute final results for this iteration
433 tsi *= dv2g;
434 wgt = 1.0 / tsi;
435 si += wgt * ti;
436 schi += wgt * ti * ti;
437 swgt += wgt;
438 tgral = si / swgt;
439
440 chi2a = (schi - si * tgral) / (it + 0.0001);
441 if (chi2a < 0.0) chi2a = 0.0;
442 sd = Math.Sqrt(1.0 / swgt);
443 tsi = Math.Sqrt(tsi);
444
445 // check convergence
446 if ((CheckConvergence && Iterations > 1 && Math.Abs(sd/tgral) < RelativeTolerance) || FunctionEvaluations >= MaxFunctionEvaluations)
447 {
448 break;
449 }
450
451 // Refine the grid. Consult references to understand the subtlety of this procedure. The refinement is damped.
452 // to avoid rapid, destabilizing changes, and also compressed in range by the exponent ALPH.
453 for (j = 0; j < ndim; j++)
454 {
455 xo = d[0, j];
456 xn = d[1, j];
457 d[0, j] = (xo + xn) / 2.0;
458 dt[j] = d[0, j];
459 for (i = 2; i < nd; i++)
460 {
461 rc = xo + xn;
462 xo = xn;
463 xn = d[i, j];
464 d[i - 1, j] = (rc + xn) / 3.0;
465 dt[j] += d[i - 1, j];
466 }
467 d[nd - 1, j] = (xo + xn) / 2.0;
468 dt[j] += d[nd - 1, j];
469 }
470 for (j = 0; j < ndim; j++)
471 {
472 rc = 0.0;
473 for (i = 0; i < nd; i++)
474 {
475 if (d[i, j] < TINY) d[i, j] = TINY;
476 r[i] = Math.Pow((1.0 - d[i, j] / dt[j]) /
477 (Math.Log(dt[j]) - Math.Log(d[i, j])), Alpha);
478 rc += r[i];
479 }
480 rebin(rc / xnd, nd, r, xin, xi, j);
481 }
482
483 }
484
485 }

494 private void rebin(double rc, int nd, double[] r, double[] xin, double[,] xi, int j)
495 {
496 int i, k = 0;
497 double dr = 0.0, xn = 0.0, xo = 0.0;
498 for (i = 0; i < nd - 1; i++)
499 {
500 while (rc > dr)
501 dr += r[(++k) - 1];
502 if (k > 1) xo = xi[j, k - 2];
503 xn = xi[j, k - 1];
504 dr -= rc;
505 xin[i] = xn - (xn - xo) * dr / r[k - 1];
506 }
507 for (i = 0; i < nd - 1; i++) xi[j, i] = xin[i];
508 xi[j, nd - 1] = 1.0;
509 }
