180 {
181 get
182 {

184 {
185 return Xi + Alpha * Tools.Euler;
186 }

232 {
233 get
234 {

236 {

253 public override double Skewness
254 {
255 get
256 {
257 if (Math.Abs(Kappa) <= NearZero)
258 {
259 return 1.1396d;
260 }
261 else if (Math.Abs(Kappa) < 1d / 3d)
262 {
263 double U1 = Gamma.Function(1d + Kappa);
264 double U2 = Gamma.Function(1d + 2d * Kappa);
265 double U3 = Gamma.Function(1d + 3d * Kappa);
266 return Math.Sign(Kappa) * (-U3 + 3d * U1 * U2 - 2d * Math.Pow(U1, 3d)) / Math.Pow(U2 - Math.Pow(U1, 2d), 3d / 2d);

277 {
278 get
279 {
280 if (Math.Abs(Kappa) <= NearZero)
281 {
282 return 3 + 12d / 5d;
283 }
284 else if (Math.Abs(Kappa) < 0.25d)
285 {

436 public double[] DirectMethodOfMoments(IList<double> moments)
437 {
438 // Solve for kappa
439 double k = SolveForKappa(moments[2]);
440 double a;
441 double x;
442 if (Math.Abs(k) <= NearZero)
443 {
444 a = Math.Sqrt(6d) / Math.PI * moments[1];
445 x = moments[0] - a * Tools.Euler;
446 }
447 else
448 {
449 double U1 = Gamma.Function(1d + k);
450 double U2 = Gamma.Function(1d + 2d * k);
451 a = Math.Sqrt(moments[1] * moments[1] * k * k / (U2 - Math.Pow(U1, 2d)));
452 x = moments[0] - a / k * (1d - U1);
453 }
454
455 // return parameters
456 return [x, a, k];
457 }

502 public double SolveForKappa(double skew)
503 {
504 if (skew > 1.14d && skew < 10d)
505 {
506 // Extreme Value Type II
507 return 0.2858221d - 0.357983d * skew + 0.116659d * Math.Pow(skew, 2d) - 0.022725d * Math.Pow(skew, 3d) + 0.002604d * Math.Pow(skew, 4d) - 0.000161d * Math.Pow(skew, 5d) + 0.000004d * Math.Pow(skew, 6d);
508 }
509 else if (skew == 1.14d)
510 {
511 // Extreme Value Type I
512 return 0d;
513 }
514 else if (skew >= 0d && skew < 1.14d)
515 {
516 // Extreme Value Type III
517 // This regression equation works for -2 < Cs < 1.14
518 return 0.277648d - 0.322016d * skew + 0.060278d * Math.Pow(skew, 2d) + 0.016759d * Math.Pow(skew, 3d) - 0.005873d * Math.Pow(skew, 4d) - 0.00244d * Math.Pow(skew, 5d) - 0.00005d * Math.Pow(skew, 6d);
519 }
520 else if (skew < 0d && skew >= -2)
521 {
522 // For negative values of skew, there exists two possible values of kappa.
523 // Either extreme value type II or III
524 // Therefor it must be solved. The Brent method is used here.
525 return Brent.Solve((x) =>
526 {
527 double U1 = Gamma.Lanczos(1d + x);
528 double U2 = Gamma.Lanczos(1d + 2d * x);
529 double U3 = Gamma.Lanczos(1d + 3d * x);
530 double k = Math.Sign(x) * (-U3 + 3d * U1 * U2 - 2d * Math.Pow(U1, 3d)) / Math.Pow(U2 - Math.Pow(U1, 2d), 3d / 2d);
531 return k - skew;
532 }, -(1d / 3d), 1d);
533 }
534 else if (skew < -2)
535 {
536 // EV2(3)
537 // This regression equation works for -10 < Cs < 0
538 return -0.50405d - 0.00861d * skew + 0.015497d * Math.Pow(skew, 2d) + 0.005613d * Math.Pow(skew, 3d) + 0.00087d * Math.Pow(skew, 4d) + 0.000065d * Math.Pow(skew, 5d);
539 }
540 else
541 {
542 return double.NaN;
543 }
544 }

547 public double[] ParametersFromLinearMoments(IList<double> moments)
548 {
549 double L1 = moments[0];
550 double L2 = moments[1];
551 double T3 = moments[2];
552 double T4 = moments[3];
553 // The following approximation given by Hosking et al. (1985b) has accuracy better than 9x10-4 for abs(t3)<=0.5.
554 if (Math.Abs(T3) <= 0.5d)
555 {
556 double c = 2d / (3d + T3) - Math.Log(2d) / Math.Log(3d);
557 double kappa = 7.859d * c + 2.9554d * Math.Pow(c, 2d);
558 double alpha = L2 * kappa / ((1d - Math.Pow(2d, -kappa)) * Gamma.Function(1d + kappa));
559 double xi = L1 - alpha * (1d - Gamma.Function(1d + kappa)) / kappa;
560 return [xi, alpha, kappa];
561 }
562 else
563 {
564 // Solve for kappa
565 double kappa = Brent.Solve(x => T3 - (2.0d * (1.0d - Math.Pow(3.0d, -x)) / (1.0d - Math.Pow(2.0d, -x)) - 3.0d), -1, 10d);
566 double alpha = L2 * kappa / ((1d - Math.Pow(2d, -kappa)) * Gamma.Function(1d + kappa));
567 double xi = L1 - alpha * (1d - Gamma.Function(1d + kappa)) / kappa;
568 return [xi, alpha, kappa];
569 }

573 public double[] LinearMomentsFromParameters(IList<double> parameters)
574 {
575 double xi = parameters[0];
576 double alpha = parameters[1];
577 double kappa = parameters[2];
578 if (kappa <= -1.0d)
579 throw new ArgumentOutOfRangeException(nameof(Kappa), "L-moments can only be defined for kappa > -1.");
580 double L1 = xi + alpha * (1.0d - Gamma.Function(1.0d + kappa)) / kappa;
581 double L2 = alpha * (1.0d - Math.Pow(2.0d, -kappa)) * Gamma.Function(1.0d + kappa) / kappa;
582 double T3 = 2.0d * (1.0d - Math.Pow(3.0d, -kappa)) / (1.0d - Math.Pow(2.0d, -kappa)) - 3.0d;
583 double T4 = (5.0d * (1.0d - Math.Pow(4.0d, -kappa)) - 10.0d * (1.0d - Math.Pow(3.0d, -kappa)) + 6.0d * (1.0d - Math.Pow(2.0d, -kappa))) / (1.0d - Math.Pow(2.0d, -kappa));
584 return [L1, L2, T3, T4];
585 }

588 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
589 {
590 var initialVals = new double[NumberOfParameters];
591 var lowerVals = new double[NumberOfParameters];
592 var upperVals = new double[NumberOfParameters];
593 // Get initial values
594 // initialVals = DirectMethodOfMoments(Statistics.ComputeProductMoments(sample))
595 initialVals = ParametersFromLinearMoments(Statistics.LinearMoments(sample));
596 // Get bounds of location
597 if (initialVals[0] == 0d) initialVals[0] = Tools.DoubleMachineEpsilon;
598 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
599 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
600 // Get bounds of scale
601 lowerVals[1] = Tools.DoubleMachineEpsilon;
602 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[1])) + 1d));
603 // Get bounds of shape
604 lowerVals[2] = -10;
605 upperVals[2] = 10d;
606 // Correct initial value of kappa if necessary
607 if (initialVals[2] <= lowerVals[2] || initialVals[2] >= upperVals[2])
608 {
609 initialVals[2] = 0d;
610 }
611 //
612 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
613 }

616 public double[] MLE(IList<double> sample)
617 {
618 // Set constraints
619 var tuple = GetParameterConstraints(sample);
620 var Initials = tuple.Item1;
621 var Lowers = tuple.Item2;
622 var Uppers = tuple.Item3;
623
624 // Solve using Nelder-Mead (Downhill Simplex)
625 double logLH(double[] x)
626 {
627 var GEV = new GeneralizedExtremeValue();
628 GEV.SetParameters(x);
629 return GEV.LogLikelihood(sample);
630 }
631 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
632 solver.ReportFailure = true;
633 solver.Maximize();
634 return solver.BestParameterSet.Values;
635

639 public override double PDF(double x)
640 {
641 // Validate parameters
642 if (_parametersValid == false)
643 ValidateParameters(Xi, Alpha, Kappa, true);
644 if (x < Minimum || x > Maximum) return 0.0d;
645 double y = (x - Xi) / Alpha;
646 if (Math.Abs(Kappa) > NearZero)
647 y = -Math.Log(1d - Kappa * y) / Kappa;
648 return Math.Exp(-(1d - Kappa) * y - Math.Exp(-y)) / Alpha;
649 }

652 public override double CDF(double x)
653 {
654 // Validate parameters
655 if (_parametersValid == false)
656 ValidateParameters(Xi, Alpha, Kappa, true);
657 if (x <= Minimum) return 0d;
658 if (x >= Maximum) return 1d;
659 double y = (x - Xi) / Alpha;
660 if (Math.Abs(Kappa) > NearZero)
661 y = -Math.Log(1d - Kappa * y) / Kappa;
662 return Math.Exp(-Math.Exp(-y));

666 public override double InverseCDF(double probability)
667 {
668 // Validate probability
669 if (probability < 0.0d || probability > 1.0d)
670 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");

681 }
682 else
683 {
684 return Xi + Alpha / Kappa * (1d - Math.Pow(-Math.Log(probability), Kappa));

692 public Matrix ExpectedInformationMatrix(int sampleSize)
693 {
694 var _matrix = new double[3, 3];
695 int N = sampleSize;
696 double a = Alpha;
697 double k = Kappa;
698 double p = Math.Pow(1d - k, 2d) * Gamma.Function(1d - 2d * k);
699 double q = (1d - k) * Gamma.Function(1d - k) * (Gamma.Digamma(1d - k) - (1d - k) / k);
700 double g = Tools.Euler;
701 double d2du2 = N / (a * a) * p;
702 double d2da2 = N / (a * a * k * k) * (1d - 2d * (1d - k) * Gamma.Function(1d - k) + p);
703 double d2dk2 = N / (k * k) * (Math.PI * Math.PI / 6d + Math.Pow(1d - g - 1d / k, 2d) + 2d * q / k + p / (k * k));
704 double d2duda = N / (a * a * k) * (p - (1d - k) * Gamma.Function(1d - k));
705 double d2dudk = -N / (a * k) * (p / k + q);
706 double d2dadk = N / (a * k * k) * (1d - g - (1d - (1d - k) * Gamma.Function(1d - k)) / k - p / k - q);
707 // Row 1
708 _matrix[0, 0] = d2du2;
709 _matrix[0, 1] = d2duda;
710 _matrix[0, 2] = d2dudk;
711 // Row 2
712 _matrix[1, 0] = d2duda;
713 _matrix[1, 1] = d2da2;
714 _matrix[1, 2] = d2dadk;
715 // Row 3
716 _matrix[2, 0] = d2dudk;
717 _matrix[2, 1] = d2dadk;
718 _matrix[2, 2] = d2dk2;
719 return new Matrix(_matrix);
720 }

744 public double[] QuantileGradient(double probability)
745 {
746 if (_parametersValid == false)
747 ValidateParameters(Xi, _alpha, Kappa, true);
748 double a = Alpha;
749 double k = Kappa;
750 var gradient = new double[]
751 {
752 1.0d, // location
753 1d / k * (1d - Math.Pow(-Math.Log(probability), k)), // scale
754 -(a / (k * k)) * (1d - Math.Pow(-Math.Log(probability), k)) - a / k * Math.Pow(-Math.Log(probability), k) * Math.Log(-Math.Log(probability)) // shape
755 };
756 return gradient;

760 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
761 {
762 var covar = ParameterCovariance(sampleSize, estimationMethod);
763 var grad = QuantileGradient(probability);
764 double varA = covar[0, 0];
765 double varB = covar[1, 1];
766 double varG = covar[2, 2];
767 double covAB = covar[1, 0];
768 double covAG = covar[2, 0];
769 double covBG = covar[2, 1];
770 double dQx1 = grad[0];
771 double dQx2 = grad[1];
772 double dQx3 = grad[2];
773 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + Math.Pow(dQx3, 2d) * varG + 2d * dQx1 * dQx2 * covAB + 2d * dQx1 * dQx3 * covAG + 2d * dQx2 * dQx3 * covBG;
774 }

777 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
778 {
779 if (probabilities.Count != NumberOfParameters)
780 {
781 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
782 }
783 // Get gradients
784 var dQp1 = QuantileGradient(probabilities[0]);
785 var dQp2 = QuantileGradient(probabilities[1]);
786 var dQp3 = QuantileGradient(probabilities[2]);
787 // Compute determinant
788 // |a b c|
789 // |d e f|
790 // |g h i|
791 // |A| = a(ei − fh) − b(di − fg) + c(dh − eg)
792 double a = dQp1[0];
793 double b = dQp1[1];
794 double c = dQp1[2];
795 double d = dQp2[0];
796 double e = dQp2[1];
797 double f = dQp2[2];
798 double g = dQp3[0];
799 double h = dQp3[1];
800 double i = dQp3[2];
801 determinant = a * (e * i - f * h) - b * (d * i - f * g) + c * (d * h - e * g);
802 // Return Jacobian
803 var jacobian = new double[,] { { a, b, c }, { d, e, f }, { g, h, i } };
804 return jacobian;
805 }
