240 private double[] kNNPredict(Matrix xTrain, Vector yTrain, Matrix xTest)
241 {
242 if (xTest.NumberOfColumns != xTrain.NumberOfColumns) return null;
243 int R = xTest.NumberOfRows;
244 var result = new double[R];
245 for (int i = 0; i < R; i++)
246 {
247 var point = xTest.Row(i);
248
249 // Get distances
250 var items = new kNNItem[xTrain.NumberOfRows];
251 Parallel.For(0, xTrain.NumberOfRows, idx =>
252 {
253 items[idx].Index = idx;
254 items[idx].Distance = Tools.Distance(point, xTrain.Row(idx));
255 });
256
257 // Sort items and find the k-nearest neighbors
258 Array.Sort(items, (a, b) => a.Distance.CompareTo(b.Distance));
259 var knn = new double[K];
260
261 // Record results
262 if (IsRegression == true)
263 {
264 // get the inverse distance weighted average of the values of k nearest neighbors
265 double sum = 0;
266 double avg = 0;
267 for (int j = 0; j < K; j++)
268 {
269 double w = items[j].Distance > 0 ? 1d / Tools.Sqr(items[j].Distance) : 1;
270 knn[j] = yTrain[items[j].Index] * w;
271 sum += w;
272 }
273 for (int j = 0; j < K; j++)
274 avg += knn[j] / sum;
275
276 result[i] = avg;
277 }
278 else
279 {
280 // get the most common value
281 for (int j = 0; j < K; j++)
282 knn[j] = yTrain[items[j].Index];
283
284 result[i] = knn.GroupBy(i => i).OrderByDescending(grp => grp.Count()).Select(grp => grp.Key).First();
285 }
286 }
287
288 return result;
289 }

298 private double[] kNNBootstrapPredict(Matrix xTrain, Vector yTrain, Matrix xTest, int seed = -1)
299 {
300 var rnd = seed > 0 ? new Random(seed) : new Random();
301 var idxs = rnd.NextIntegers(0, xTrain.NumberOfRows, xTrain.NumberOfRows);
302 var bootX = new Matrix(xTrain.NumberOfRows, xTrain.NumberOfColumns);
303 var bootY = new Vector(yTrain.Length);
304 for (int i = 0; i < xTrain.NumberOfRows; i++)
305 {
306 for (int j = 0; j < xTrain.NumberOfColumns; j++)
307 {
308 bootX[i, j] = xTrain[idxs[i], j];
309 }
310 bootY[i] = yTrain[idxs[i]];
311 }
312 return kNNPredict(bootX, bootY, xTest);
313 }

324 private double[,] kNNPredictionIntervals(Matrix xTrain, Vector yTrain, Matrix xTest, int seed = -1, int realizations = 1000, double alpha = 0.1)
325 {
326 var percentiles = new double[] { alpha / 2d, 0.5, 1d - alpha / 2d };
327 var output = new double[xTest.NumberOfRows, 4]; // lower, median, upper, mean
328
329 var bootResults = new double[xTest.NumberOfRows, realizations];
330 var rnd = seed > 0 ? new Random(seed) : new Random();
331 var seeds = rnd.NextIntegers(realizations);
332
333 // Bootstrap the predictions
334 Parallel.For(0, realizations, idx => { bootResults.SetColumn(idx, kNNBootstrapPredict(xTrain, yTrain, xTest, seeds[idx]));});
335
336 // Process results
337 Parallel.For(0, xTest.NumberOfRows, idx =>
338 {
339 var values = bootResults.GetRow(idx);
340 Array.Sort(values);
341
342 // Record percentiles for CIs
343 for (int j = 0; j < percentiles.Length; j++)
344 output[idx, j] = Statistics.Percentile(values, percentiles[j], true);
345
346 output[idx, 3] = Statistics.ParallelMean(values);
347 });
348
349 return output;
350 }
