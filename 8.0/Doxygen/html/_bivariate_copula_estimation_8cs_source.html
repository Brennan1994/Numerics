58 public static void Estimate(ref BivariateCopula copula, IList<double> sampleDataX, IList<double> sampleDataY, CopulaEstimationMethod estimationMethod)
59 {
60 switch (estimationMethod)
61 {
62 case CopulaEstimationMethod.PseudoLikelihood:
63 MPL(copula, sampleDataX, sampleDataY);
64 break;
65 case CopulaEstimationMethod.InferenceFromMargins:

67 break;

71 }

80 private static void MPL(BivariateCopula copula, IList<double> sampleDataX, IList<double> sampleDataY)
81 {
82 // Get constraints
83 var LU = copula.ParameterConstraints(sampleDataX, sampleDataY);
84
85 // Solve using Brent method
86 Func<double, double> func = (x) =>
87 {
88 var C = copula.Clone();
89 C.Theta = x;
90 return C.PseudoLogLikelihood(sampleDataX, sampleDataY);
91 };
92 var brent = new BrentSearch(func, LU[0], LU[1]);
93 brent.Maximize();
94 copula.Theta = brent.BestParameterSet.Values[0];
95 }

126 private static void MLE(BivariateCopula copula, IList<double> sampleDataX, IList<double> sampleDataY)
127 {
128 // See if marginals are estimable
129 IMaximumLikelihoodEstimation margin1 = (IMaximumLikelihoodEstimation)copula.MarginalDistributionX;
130 IMaximumLikelihoodEstimation margin2 = (IMaximumLikelihoodEstimation)copula.MarginalDistributionY;
131 if (margin1 == null || margin2 == null) throw new ArgumentOutOfRangeException("marginal distributions", "There marginal distributions must implement the IMaximumLikelihoodEstimation interface to use this method.");
132
133 int np1 = copula.MarginalDistributionX.NumberOfParameters;
134 int np2 = copula.MarginalDistributionY.NumberOfParameters;
135
136 // Get constraints
137 var initials = new double[1 + np1 + np2];
138 var lowers = new double[1 + np1 + np2];
139 var uppers = new double[1 + np1 + np2];
140
141 // Theta
142 // get ranks of data
143 var rank1 = Statistics.RanksInplace(sampleDataX.ToArray());
144 var rank2 = Statistics.RanksInplace(sampleDataY.ToArray());
145 // get plotting positions
146 for (int i = 0; i < rank1.Length; i++)
147 {
148 rank1[i] = rank1[i] / (rank1.Length + 1d);
149 rank2[i] = rank2[i] / (rank2.Length + 1d);
150 }
151
152 // Get constraints
153 var LU = copula.ParameterConstraints(sampleDataX, sampleDataY);
154 lowers[0] = LU[0];
155 uppers[0] = LU[1];
156 // Estimate copula using MPL
157 MPL(copula, rank1, rank2);
158 initials[0] = copula.Theta;
159
160 // Estimate marginals
161 ((IEstimation)copula.MarginalDistributionX).Estimate(sampleDataX, ParameterEstimationMethod.MaximumLikelihood);
162 ((IEstimation)copula.MarginalDistributionY).Estimate(sampleDataY, ParameterEstimationMethod.MaximumLikelihood);
163
164
165 var con = margin1.GetParameterConstraints(sampleDataX);
166 var parms = copula.MarginalDistributionX.GetParameters;
167 for (int i = 0; i < np1; i++)
168 {
169 initials[i + 1] = parms[i];
170 lowers[i + 1] = con.Item2[i];
171 uppers[i + 1] = con.Item3[i];
172 }
173 con = margin2.GetParameterConstraints(sampleDataY);
174 parms = copula.MarginalDistributionY.GetParameters;
175 for (int i = 0; i < np2; i++)
176 {
177 initials[i + 1 + np1] = parms[i];
178 lowers[i + 1 + np1] = con.Item2[i];
179 uppers[i + 1 + np1] = con.Item3[i];
180 }
181
182 // Log-likelihood function
183 Func<double[], double> logLH = (double[] x) => {
184 // Set copula
185 var C = copula.Clone();
186 C.Theta = x[0];
187
188 // marginal 1
189 var m1 = ((UnivariateDistributionBase)copula.MarginalDistributionX).Clone();
190 var p1 = new double[np1];
191 for (int i = 0; i < np1; i++)
192 p1[i] = x[i + 1];
193 m1.SetParameters(p1);
194
195 // marginal 2
196 var m2 = ((UnivariateDistributionBase)copula.MarginalDistributionY).Clone();
197 var p2 = new double[np2];
198 for (int i = 0; i < np2; i++)
199 p2[i] = x[i + 1 + np1];
200 m2.SetParameters(p2);
201
202 C.MarginalDistributionX = m1;
203 C.MarginalDistributionY = m2;
204 return C.LogLikelihood(sampleDataX, sampleDataY);
205 };
206
207 var solver = new NelderMead(logLH, lowers.Length, initials, lowers, uppers);
208 solver.Maximize();
209
210 // Set parameters for copula and marginals
211 copula.Theta = solver.BestParameterSet.Values[0];
212
213 var par = new double[np1];
214 for (int i = 0; i < np1; i++)
215 par[i] = solver.BestParameterSet.Values[i + 1];
216 copula.MarginalDistributionX.SetParameters(par);
217
218 par = new double[np1];
219 for (int i = 0; i < np2; i++)
220 par[i] = solver.BestParameterSet.Values[i + 1 + np1];
221 copula.MarginalDistributionY.SetParameters(par);
222
223 }
