161 public void Train(int seed = -1, bool kMeansPlusPlus= true)
162 {
163
164 // 1. Initialize cluster centers
165 Means = Initialize(X, K, seed, kMeansPlusPlus);
166
167 // 2. Optimize clusters
168 for (Iterations = 1; Iterations <= MaxIterations; Iterations++)
169 {
170 // Perform E-step
171 // Assign samples to closest centroids (create clusters)
172 var oldLabels = Labels.ToArray();
173 Labels = GetLabels(Means);
174
175 // Check if the labels changed
176 bool labelsChanged = false;
177 for (int i = 0; i < Labels.Length; i++)
178 {
179 if (oldLabels[i] != Labels[i])
180 {
181 labelsChanged = true;
182 break;
183 }
184
185 }
186 // Stop when the E-step doesn't change the assignment of any data point
187 if (labelsChanged == false)
188 break;
189
190 // Perform M-step
191 // Calculate new centroids from the clusters
192 Means = GetCentroids(Labels);
193
194 }

205 public static double[,] Initialize(Matrix X, int k, int seed = -1, bool kMeansPlusPlus = true)
206 {
207 var centroids = new double[k, X.NumberOfColumns];
208 var rnd = seed > 0 ? new MersenneTwister(seed) : new MersenneTwister();
209
210 if (kMeansPlusPlus == false)
211 {
212
213 var rndIdxs = rnd.NextIntegers(0, X.NumberOfRows, k, false);
214 Array.Sort(rndIdxs);
215 for (int i = 0; i < k; i++)
216 for (int j = 0; j < X.NumberOfColumns; j++)
217 centroids[i, j] = X[rndIdxs[i], j];
218 }
219 else
220 {
221 // Initialize using K-Means++
222 // http://en.wikipedia.org/wiki/K-means%2B%2B
223
224
225 // 1. Choose one center uniformly at random from among the data points.
226 int idx = rnd.Next(0, X.NumberOfRows);
227 for (int j = 0; j < X.NumberOfColumns; j++)
228 centroids[0, j] = X[idx, j];
229
230 for (int c = 1; c < k; c++)
231 {
232 // 2. For each data point x not chosen yet, compute D(x),
233 // the distance between x and the nearest center that has already been chosen.
234
235 double sum = 0;
236 var D = new double[X.NumberOfRows];
237 for (int i = 0; i < X.NumberOfRows; i++)
238 {
239 var x = X.Row(i);
240
241 double min = Tools.Distance(x, centroids.GetRow(0));
242 for (int j = 1; j < c; j++)
243 {
244 double d = Tools.Distance(x, centroids.GetRow(0));
245
246 if (d < min)
247 min = d;
248 }
249
250 D[i] = min;
251 sum += min;
252 }
253
254 // Following Accord.Net checks:
255 // https://github.com/accord-net/framework/blob/development/Sources/Accord.MachineLearning/Clustering/KMeans/KMeans.cs
256
257 // Note: the following checks could have been avoided if we added
258 // a small value to each distance, but is kept as this to avoid 022
259 // breaking the random pattern in existing code.
260
261 if (sum == 0)
262 {
263 // Degenerate case: all points are the same, chose any of them
264 idx = rnd.Next(0, X.NumberOfRows);
265 }
266 else
267 {
268 // 3. Choose one new data point at random as a new center, using a weighted
269 // probability distribution where a point x is chosen with probability
270 // proportional to D(x)^2.
271 var u = rnd.NextDouble();
272 var cdf = new double[X.NumberOfRows];
273 for (int i = 0; i < D.Length; i++)
274 {
275 D[i] /= sum;
276 cdf[i] = i == 0 ? D[i] : cdf[i - 1] + D[i];
277 if (u <= cdf[i])
278 {
279 idx = i;
280 break;
281 }
282 }
283
284 }
285 for (int j = 0; j < X.NumberOfColumns; j++)
286 centroids[c, j] = X[idx, j];
287 }
288 }
289
290 return centroids;
291 }

297 private int[] GetLabels(double[,] centroids)
298 {
299 // Assign samples to the closest centroids
300 var labels = new int[X.NumberOfRows];
301 Parallel.For(0, X.NumberOfRows, idx => { labels[idx] = GetClosestCentroid(X.Row(idx), centroids); });
302 return labels;
303 }

309 private double[,] GetCentroids(int[] labels)
310 {
311 var count = new double[K];
312 var centroids = new double[K, Dimension];
313
314 // Get sums and counts
315 for (int i = 0; i < X.NumberOfRows; i++)
316 {
317 count[labels[i]]++;
318 for (int j = 0; j < Dimension; j++)
319 centroids[labels[i], j] += X[i, j];
320 };
321
322 // Get mean of clusters
323 for (int k = 0; k < K; k++)
324 for (int j = 0; j < Dimension; j++)
325 centroids[k, j] /= count[k] > 0 ? count[k] : 1;
326
327 return centroids;
328 }

335 private int GetClosestCentroid(double[] sample, double[,] centroids)
336 {
337 double min = double.MaxValue;
338 int minIdx = 0;
339 for (int k = 0; k < K; k++)
340 {
341 var dist = Tools.Distance(sample, centroids.GetRow(k));
342 if (dist < min)
343 {
344 min = dist;
345 minIdx = k;
346 }
347 }
348 return minIdx;
349 }
