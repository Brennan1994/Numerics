140 public override double GeneratorPrime(double t)
141 {

146 public override double GeneratorPrime2(double t)
147 {
148 double num = (Theta - 1.0d) * (Theta * (2.0d * t - 1.0d) + 1.0d);
149 double a = Theta * (t - 1.0d) * t + t;
150 double den = Math.Sign(a) * Math.Pow(Math.Abs(a), 2d);
151 return -num / den;
152 }

161 public override double PDF(double u, double v)
162 {
163 // Validate parameters
164 if (_parametersValid == false) ValidateParameter(Theta, true);
165 return (-1 + Tools.Sqr(Theta) * (-1 + u + v - u * v) - Theta * (-2 + u + v + u * v)) / Math.Pow(-1 + Theta * (-1 + u) * (-1 + v), 3);
166 }

169 public override double[] InverseCDF(double u, double v)
170 {
171 // Validate parameters
172 if (_parametersValid == false) ValidateParameter(Theta, true);
173 //Johnson (1987, p.362).
174 double w = v;
175 double b = 1d - u;
176 double A = w * Math.Pow(Theta * b, 2) - Theta;
177 double B = Theta + 1d - 2d * Theta * b * w;
178 double C = w - 1d;
179 v = (-B + Math.Sqrt(B * B - 4d * A * C)) / 2d / A;
180 v = 1d - v;
181 return [u, v];
182 }

195 public void SetThetaFromTau(IList<double> sampleDataX, IList<double> sampleDataY)
196 {
197 var tau = Correlation.KendallsTau(sampleDataX, sampleDataY);
198
199 if ((tau < (5d - 8d * Math.Log(2d)) / 3d) || (tau > 1d / 3d))
200 throw new Exception("For the AMH copula, tau must be in [(5 - 8 log 2) / 3, 1 / 3] ~= [-0.1817, 0.3333]. The dependency in the data is too strong to use the AMH copula.");
201
202 double L = tau > 0 ? 0.001d : -1d + Tools.DoubleMachineEpsilon;
203 double U = tau > 0 ? 1d - Tools.DoubleMachineEpsilon : -0.001d;
204
205 Theta = Brent.Solve(t =>
206 {
207 var x = 1d - 2d * (Math.Pow(1d - t, 2d) * Math.Log(-t + 1d) + t) / (3d * t * t);
208 return x - tau;
209 }, L, U);
210
211 }
