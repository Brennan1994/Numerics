374 public override double PDF(double x)
375 {
376 // Validate parameters
377 if (_parametersValid == false)
378 ValidateParameters(Mu, Sigma, DegreesOfFreedom, true);
379 double Z = (x - Mu) / Sigma;
380 if (DegreesOfFreedom >= 100000000.0d)
381 {
382 return Normal.StandardPDF(Z);
383 }
384 return Math.Exp(Gamma.LogGamma((DegreesOfFreedom + 1.0d) / 2.0d) - Gamma.LogGamma(DegreesOfFreedom / 2.0d)) * Math.Pow(1.0d + Z * Z / DegreesOfFreedom, -0.5d * (DegreesOfFreedom + 1.0d)) / Math.Sqrt(DegreesOfFreedom * Math.PI);
385 }
386
388 public override double CDF(double x)
389 {
390 // Validate parameters
391 if (_parametersValid == false)
392 ValidateParameters(Mu, Sigma, DegreesOfFreedom, true);
393 //
394 double Z = (x - Mu) / Sigma;
395 if (DegreesOfFreedom >= 100000000.0d)
396 {
397 return Normal.StandardCDF(Z);
398 }
399 double betaX = (Z + Math.Sqrt(Z * Z + DegreesOfFreedom)) / (2d * Math.Sqrt(Z * Z + DegreesOfFreedom));
400 return Beta.Incomplete(DegreesOfFreedom / 2.0d, DegreesOfFreedom / 2.0d, betaX);
401 }

416 public override double InverseCDF(double probability)
417 {
418 // Validate probability
419 if (probability < 0.0d || probability > 1.0d)
420 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
421 if (probability == 0.0d)
422 return Minimum;
423 if (probability == 1.0d)
424 return Maximum;
425 // Validate parameters
426 if (_parametersValid == false)
427 ValidateParameters(Mu, Sigma, DegreesOfFreedom, true);
428 //
429 if (probability > 0.25d && probability < 0.75d)
430 {
431 if (probability == 0.5d)
432 {
433 return Mu;
434 }
435
436 double z = Beta.IncompleteInverse(0.5d, 0.5d * DegreesOfFreedom, Math.Abs(1.0d - 2.0d * probability));
437 double t = Math.Sqrt(DegreesOfFreedom * z / (1.0d - z));
438 if (probability < 0.5d)
439 {
440 t = -t;
441 }
442
443 return Mu + Sigma * t;
444 }
445 else
446 {
447 int rflg = -1;
448 if (probability >= 0.5d)
449 {
450 probability = 1.0d - probability;
451 rflg = 1;
452 }
453
454 double z = Beta.IncompleteInverse(0.5d * DegreesOfFreedom, 0.5d, 2.0d * probability);
455 double t = Math.Sqrt(DegreesOfFreedom / z - DegreesOfFreedom);
456 if (double.MaxValue * z < DegreesOfFreedom)
457 {
458 return rflg * double.MaxValue;
459 }
460
461 return Mu + Sigma * (rflg * t);
462 }
