Numerics is an opensource library for .NET developed by the U.S. Army Corps of Engineers Risk Management Center (USACE-RMC). Numerics provides a comprehensive set of methods and algorithms for numerical computations and statistical analysis, serving as a cornerstone of the RMC software suite. The library includes routines for specialized functions, interpolation, regression, statistics, machine learning, probability distributions, bootstrap uncertainty analysis, Bayesian Markov Chain Monte Carlo, optimization, root finding, and more.

271 public void Push(double value)
272 {
273 _n += 1L;
274 // Update moments
275 double d = value - _m1;
276 double s = d / _n;
277 double s2 = s * s;
278 double t = d * s * (_n - 1L);
279 _m1 += s;
280 _m4 += t * s2 * (_n * _n - 3L * _n + 3L) + 6d * s2 * _m2 - 4d * s * _m3;
281 _m3 += t * s * (_n - 2L) - 3d * s * _m2;
282 _m2 += t;
283 // Update min
284 if (value < _min || double.IsNaN(value))
285 {
286 _min = value;
287 }
288 // Update max
289 if (value > _max || double.IsNaN(value))
290 {
291 _max = value;
292 }
293 }

308 public static RunningStatistics Combine(RunningStatistics a, RunningStatistics b)
309 {
310 if (a._n == 0L)
311 {
312 return b;
313 }
314 else if (b._n == 0L)
315 {
316 return a;
317 }
318
319 long n = a._n + b._n;
320 double d = b._m1 - a._m1;
321 double d2 = d * d;
322 double d3 = d2 * d;
323 double d4 = d2 * d2;
324 double m1 = (a._n * a._m1 + b._n * b._m1) / n;
325 double m2 = a._m2 + b._m2 + d2 * a._n * b._n / n;
326 double m3 = a._m3 + b._m3 + d3 * a._n * b._n * (a._n - b._n) / (n * n) + 3d * d * (a._n * b._m2 - b._n * a._m2) / n;
327 double m4 = a._m4 + b._m4 + d4 * a._n * b._n * (a._n * a._n - a._n * b._n + b._n * b._n) / (n * n * n) + 6d * d2 * (a._n * a._n * b._m2 + b._n * b._n * a._m2) / (n * n) + 4d * d * (a._n * b._m3 - b._n * a._m3) / n;
328 double min = Math.Min(a._min, b._min);
329 double max = Math.Max(a._max, b._max);
330 return new RunningStatistics()
331 {
332 _n = n,
333 _m1 = m1,
334 _m2 = m2,
335 _m3 = m3,
336 _m4 = m4,
337 _min = min,
338 _max = max
339 };
340 }
