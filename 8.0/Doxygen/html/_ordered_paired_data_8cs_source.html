384 private bool OrdinateValid(int index, bool lookBackward = true)
385 {
386 if (index < 0 || index > _ordinates.Count - 1)
387 return true;
388 // Look backward
389 if (lookBackward == true & index > 0)
390 {
391 if (_ordinates[index].OrdinateValid(_ordinates[index - 1], StrictX, StrictY, OrderX, OrderY, false) == false)
392 return false;
393 }
394 // Look forward
395 if (index < _ordinates.Count - 1)
396 {
397 if (_ordinates[index].OrdinateValid(_ordinates[index + 1], StrictX, StrictY, OrderX, OrderY, true) == false)
398 return false;
399 }
400 // Passed the test
401 return true;
402 }

561 public static bool operator ==(OrderedPairedData left, OrderedPairedData right)
562 {
563 // Check for null arguments. Keep in mind null == null
564 if (left is null && right is null)
565 {
566 return true;
567 }
568 else if (left is null)
569 {
570 return false;
571 }
572 else if (right is null)
573 {
574 return false;
575 }
576 if ((left._ordinates == null) && (right._ordinates == null))
577 {
578 return true;
579 }
580 else if (left._ordinates == null)
581 {
582 return false;
583 }
584 else if (right._ordinates == null)
585 {
586 return false;
587 }
588 if (left.Count != right.Count) return false;
589 for (int i = 0; i < left._ordinates.Count; i++)
590 {
591 if (Math.Abs(left._ordinates[i].X - right._ordinates[i].X) > 0.0000000000001d)
592 return false;
593 if (Math.Abs(left._ordinates[i].Y - right._ordinates[i].Y) > 0.0000000000001d)
594 return false;
595 }
596 return true;
597 }

614 public double TrapezoidalAreaUnderY()
615 {
616 double sum = 0d;
617 if (OrderX == SortOrder.Ascending)
618 {
619 for (int i = 1; i < _ordinates.Count; i++)
620 sum += 0.5d * (_ordinates[i].X - _ordinates[i - 1].X) * (_ordinates[i - 1].Y + _ordinates[i].Y);
621 }
622 else if (OrderX == SortOrder.Descending)
623 {
624 for (int i = 1; i < _ordinates.Count; i++)
625 sum += 0.5d * (_ordinates[i - 1].X - _ordinates[i].X) * (_ordinates[i - 1].Y + _ordinates[i].Y);
626 }
627 else
628 {
629 throw new InvalidConstraintException("Unable to calculate area under y with no guarantee of sort order on x.");
630 }
631 return sum;
632 }

693 private double BaseInterpolate(double value, int index, bool givenX = true, Transform xTransform = Transform.None, Transform yTransform = Transform.None)
694 {
695
696 double x = givenX ? value : 0, y = givenX ? 0 : value, x1 = 0, x2 = 0, y1 = 0, y2 = 0;
697 int lo = index, hi = index + 1;
698
699 // Get X transform
700 if (xTransform == Transform.None)
701 {
702 x1 = _ordinates[lo].X;
703 x2 = _ordinates[hi].X;
704 }
705 else if (xTransform == Transform.Logarithmic)
706 {
707 x = Tools.Log10(x);
708 x1 = Tools.Log10(_ordinates[lo].X);
709 x2 = Tools.Log10(_ordinates[hi].X);
710 }
711 else if (xTransform == Transform.NormalZ)
712 {
713 x = Normal.StandardZ(x);
714 x1 = Normal.StandardZ(_ordinates[lo].X);
715 x2 = Normal.StandardZ(_ordinates[hi].X);
716 }
717
718 // Get Y transform
719 if (yTransform == Transform.None)
720 {
721 y1 = _ordinates[lo].Y;
722 y2 = _ordinates[hi].Y;
723 }
724 else if (yTransform == Transform.Logarithmic)
725 {
726 y = Tools.Log10(y);
727 y1 = Tools.Log10(_ordinates[lo].Y);
728 y2 = Tools.Log10(_ordinates[hi].Y);
729 }
730 else if (yTransform == Transform.NormalZ)
731 {
732 y = Normal.StandardZ(y);
733 y1 = Normal.StandardZ(_ordinates[lo].Y);
734 y2 = Normal.StandardZ(_ordinates[hi].Y);
735 }
736
737 // interpolate
738 if (givenX)
739 {
740 // Check for division by zero
741 if ((x2 - x1) == 0)
742 {
743 y = y1;
744 }
745 else
746 {
747 y = y1 + (x - x1) / (x2 - x1) * (y2 - y1);
748 }
749 //
750 if (yTransform == Transform.None)
751 {
752 return y;
753 }
754 else if (yTransform == Transform.Logarithmic)
755 {
756 return Math.Pow(10d, y);
757 }
758 else if (yTransform == Transform.NormalZ)
759 {
760 return Normal.StandardCDF(y);
761 }
762 }
763 else
764 {
765 // Check for division by zero
766 if ((y2 - y1) == 0)
767 {
768 x = x1;
769 }
770 else
771 {
772 x = x1 + (y - y1) / (y2 - y1) * (x2 - x1);
773 }
774 //
775 if (xTransform == Transform.None)
776 {
777 return x;
778 }
779 else if (xTransform == Transform.Logarithmic)
780 {
781 return Math.Pow(10d, x);
782 }
783 else if (xTransform == Transform.NormalZ)
784 {
785 return Normal.StandardCDF(x);
786 }
787 }
788
789 // return NaN if we get to here
790 return double.NaN;
791
792 }

801 public double GetYFromX(double x, Transform xTransform = Transform.None, Transform yTransform = Transform.None)
802 {
803 if (Count == 0) return double.NaN;
804 if (OrderX == SortOrder.None)
805 throw new Exception("Interpolation requires the x-values to be ascending or descending.");
806
807 // First see if value is out of range
808 if (Count == 1) return _ordinates[0].Y;
809 if ((OrderX == SortOrder.Ascending && x <= _ordinates[0].X) || (OrderX == SortOrder.Descending && x >= _ordinates[0].X)) return _ordinates[0].Y;
810 if ((OrderX == SortOrder.Ascending && x >= _ordinates[Count - 1].X) || (OrderX == SortOrder.Descending && x <= _ordinates[Count - 1].X)) return _ordinates[Count - 1].Y;
811
812 // Interpolate
813 return BaseInterpolate(x, SearchX(x), true, xTransform, yTransform);
814 }

1020 public int SequentialSearchX(double x)
1021 {
1022 int jl = XSearchStart;
1023 if ((OrderX == SortOrder.Ascending && x < _ordinates[0].X) ||
1024 (OrderX == SortOrder.Descending && x > _ordinates[0].X))
1025 {
1026 return 0;
1027 }
1028 else if ((OrderX == SortOrder.Ascending && x > _ordinates[Count - 1].X) ||
1029 (OrderX == SortOrder.Descending && x < _ordinates[Count - 1].X))
1030 {
1031 return Count - 2;
1032 }
1033 else if ((OrderX == SortOrder.Ascending && x < _ordinates[XSearchStart].X) ||
1034 (OrderX == SortOrder.Descending && x > _ordinates[XSearchStart].X))
1035 {
1036 jl = 0;
1037 }

1329 private double PerpendicularDistance(double aX, double aY, double bX, double bY, double cX, double cY)
1330 {
1331 // Area = |(1/2)(x1y2 + x2y3 + x3y1 - x2y1 - x3y2 - x1y3)| *Area of triangle
1332 // Base = v((x1-x2)²+(x1-x2)²) *Base of Triangle
1333 // Area = 0.5*Base*H *Solve for height
1334 // Height = Area/.5/Base
1335
1336 double area = Math.Abs((aX * bY + bX * cY + cX * aY - bX * aY - cX * bY - aX * cY) * 0.5);
1337 double triangleBase = Math.Pow(Math.Pow(aX - bX, 2) + Math.Pow(aY - bY, 2), 0.5);
1338 return area * 2 / triangleBase;

1398 public OrderedPairedData LangSimplify(double tolerance, int lookAhead)
1399 {

1401 return this;

1435 private int RecursiveTolerance(int i, int lookAhead, double tolerance)
1436 {
1437 int n = lookAhead;
1438 double angle;
1439
1440 var cp = _ordinates[i];
1441
1442 if (i + n < _ordinates.Count)
1443 {
1444 var v1 = new Ordinate(_ordinates[i + n].X - cp.X, _ordinates[i + n].Y - cp.Y);
1445
1446 for (int j = 1; j <= n; j++)
1447 {
1448 var clp = _ordinates[i + j];
1449
1450 var v2 = new Ordinate(clp.X - cp.X, clp.Y - cp.Y);
1451
1452 angle = Math.Acos((v1.X * v2.X + v1.Y * v2.Y) / (Math.Sqrt(v1.Y * v1.Y + v1.X * v1.X) * Math.Sqrt(v2.Y * v2.Y + v2.X * v2.X)));
1453
1454 if (double.IsNaN(angle) || double.IsInfinity(angle)) angle = 0;
1455
1456 double lH = Math.Sqrt((clp.X - cp.X) * (clp.X - cp.X) + (clp.Y - cp.Y) * (clp.Y - cp.Y));
1457
1458 if (Math.Sin(angle) * lH >= tolerance)
1459 {
1460 n -= 1;
1461
1462 if (n > 0)
1463 return RecursiveTolerance(i, n, tolerance);
1464 else
1465 return 0;
1466 }
1467 }
1468 }
1469
1470 return n;
