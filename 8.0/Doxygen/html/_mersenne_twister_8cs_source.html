130 private const int N = 624;
131 private const int M = 397;
132 private const uint matrixA = 0x9908b0dfU; // constant vector a
133 private const uint upperMask = 0x80000000U; // most significant w-r bits
134 private const uint lowerMask = 0x7fffffffU; // least significant r bits
135 private uint[] mt = new uint[N]; // the array for the state vector
136 int mti = N + 1; // mti==N+1 means mt[N] is not initialized
137 private static readonly uint[] mag01 = { 0x0U, matrixA }; // mag01[x] = x * MATRIX_A for x=0,1
138

140 // The following constants, are used within the ADDITIONAL functions genrand_real2b() and
141 // genrand_real3b(), equivalent to genrand_real() and genrand_real3(), but that return
142 // evenly distributed values in the ranges [0, 1-kMT_Gap] and [0+kMT_Gap, 1-kMT_Gap],
143 // respectively. A similar statement is valid also for genrand_real2c(), genrand_real4b()
144 // and genrand_real5b(). See the section "Functions and procedures implemented" above,
145 // for more details.
146 //
147 // If you want to change the value of kMT_Gap, it is suggested to do it so that:
148 // 5e-15 <= kMT_Gap <= 5e-2
149
150 const double k2_31 = 2147483648.0; // 2^31 == 2147483648 == 80000000
151 const double k2_32 = 2.0 * k2_31; // 2^32 == 4294967296 == 0
152 const double k2_32b = k2_32 - 1.0; // 2^32-1 == 4294967295 == FFFFFFFF == -1
153
154 const double kMT_Gap = 0.0000000000005; // 5.0E-13
155 const double kMT_Gap2 = 2.0 * kMT_Gap; // 1.0E-12
156 const double kMT_GapInterval = 1.0 - kMT_Gap2; // 0.9999999999990
157 const double kMT_2b = kMT_GapInterval / k2_32b;
158 const double kMT_2c = kMT_2b;
159 const double kMT_3b = kMT_2b;

165 public void Initialize(uint seed)
166 {
167 mt[0] = seed & 0xffffffffU;
168 for (mti = 1; mti < N; mti++)
169 {
170 mt[mti] = (uint)(1812433253U * (mt[mti - 1] ^ (mt[mti - 1] >> 30)) + mti);
171 /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
172 /* In the previous versions, MSBs of the seed affect */
173 /* only MSBs of the array _mt[]. */
174 /* 2002/01/09 modified by Makoto Matsumoto */
175 mt[mti] &= 0xffffffffU;
176 /* for >32 bit machines */
177 }
178 }
179
185 private void Initialize(uint[] init_key, int key_length)
186 {
187 Initialize(19650218U);
188 int i = 1, j = 0;
189 int k = (N > key_length) ? N : key_length;
190 for (; k > 0; k--)
191 {
192 mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >> 30)) * 1664525U)) + init_key[j] + (uint)j;
193 mt[i] &= 0xffffffffU; // For WORDSIZE > 32 machines
194 i++; j++;
195 if (i >= N) { mt[0] = mt[N - 1]; i = 1; }
196 if (j >= key_length) j = 0;
197 }
198 for (k = N - 1; k > 0; k--)
199 {
200 mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >> 30)) * 1566083941U)) - (uint)i;
201 mt[i] &= 0xffffffffU; // For WORDSIZE > 32 machines
202 i++;
203 if (i >= N) { mt[0] = mt[N - 1]; i = 1; }
204 }
205
206 mt[0] = upperMask; // MSB is 1; assuring non-zero initial array
207 }

212 public uint GenRandInt32()
213 {
214
215 uint y;
216
217 /* mag01[x] = x * MATRIX_A for x=0,1 */
218
219 if (mti >= N)
220 {
221 /* generate _n words at one time */
222 int kk;
223
224 if (mti == N + 1) /* if init_genrand() has not been called, */
225 {
226 Initialize(5489); /* a default initial seed is used */
227 }
228
229 for (kk = 0; kk < N - M; kk++)
230 {
231 y = (mt[kk] & upperMask) | (mt[kk + 1] & lowerMask);
232 mt[kk] = mt[kk + M] ^ (y >> 1) ^ mag01[y & 0x1];
233 }
234
235 for (; kk < N - 1; kk++)
236 {
237 y = (mt[kk] & upperMask) | (mt[kk + 1] & lowerMask);
238 mt[kk] = mt[kk + (M - N)] ^ (y >> 1) ^ mag01[y & 0x1];
239 }
240
241 y = (mt[N - 1] & upperMask) | (mt[0] & lowerMask);
242 mt[N - 1] = mt[M - 1] ^ (y >> 1) ^ mag01[y & 0x1];
243
244 mti = 0;
245 }
246
247 y = mt[mti++];
248
249 /* Tempering */
250 y ^= y >> 11;
251 y ^= (y << 7) & 0x9d2c5680;
252 y ^= (y << 15) & 0xefc60000;
253 y ^= y >> 18;
254
255 return y;
256 }
