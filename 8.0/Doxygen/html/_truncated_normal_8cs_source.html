204 public override double Mean
205 {
206 get
207 {
208 alpha = (Min - Mu) / Sigma;
209 beta = (Max - Mu) / Sigma;
210 Z = Normal.StandardCDF(beta) - Normal.StandardCDF(alpha);
211 if (Z == 0) return Mu;
212 return Mu + Sigma * ((Normal.StandardPDF(alpha) - Normal.StandardPDF(beta)) / Z);
213 }
214 }

223 Z = Normal.StandardCDF(beta) - Normal.StandardCDF(alpha);
224 if (Z == 0) return Mu;
225 return Mu + Sigma * Normal.StandardZ((Normal.StandardCDF(alpha) + Normal.StandardCDF(beta)) / 2d);
226 }

251 public override double StandardDeviation
252 {
253 get
254 {
255 alpha = (Min - Mu) / Sigma;
256 beta = (Max - Mu) / Sigma;
257 Z = Normal.StandardCDF(beta) - Normal.StandardCDF(alpha);
258 if (Z == 0) return Sigma;
259 return Math.Sqrt(Math.Pow(Sigma, 2d) * (1d + (alpha * Normal.StandardPDF(alpha) - beta * Normal.StandardPDF(beta)) / Z - Math.Pow((Normal.StandardPDF(alpha) - Normal.StandardPDF(beta)) / Z, 2d)));
260 }
261 }
262
267 public override double Skewness
268 {
269 get
270 {
271 double K0 = (Min - Mu) / Sigma;
272 double K1 = (Max - Mu) / Sigma;
273 Z = Normal.StandardCDF(K1) - Normal.StandardCDF(K0);
274 if (Z == 0) return 0;
275 double Z0 = Normal.StandardPDF(K0) / Z;
276 double Z1 = Normal.StandardPDF(K1) / Z;
277 double V = 1d - (K1 * Z1 - K0 * Z0) - Math.Pow(Z1 - Z0, 2d);
278 return -(1d / Math.Pow(V, 3d / 2d)) * (2d * Math.Pow(Z1 - Z0, 3d) + (3d * K1 * Z1 - 3d * K0 * Z0 - 1d) * (Z1 - Z0) + Math.Pow(K1, 2d) * Z1 - Math.Pow(K0, 2d) * Z0);
279 }
280 }
281
283 public override double Kurtosis
284 {
285 get
286 {
287 double K0 = (Min - Mu) / Sigma;
288 double K1 = (Max - Mu) / Sigma;
289 Z = Normal.StandardCDF(K1) - Normal.StandardCDF(K0);
290 if (Z == 0) return 3d;
291 double Z0 = Normal.StandardPDF(K0) / Z;
292 double Z1 = Normal.StandardPDF(K1) / Z;
293 double V = 1d - (K1 * Z1 - K0 * Z0) - Math.Pow(Z1 - Z0, 2d);
294 return (1d / Math.Pow(V, 2d)) * (-3 * Math.Pow(Z1 - Z0, 4d) - 6d * (K1 * Z1 - K0 * Z0) * Math.Pow(Z1 - Z0, 2d) - 2d * Math.Pow(Z1 - Z0, 2d) - 4d * (Math.Pow(K1, 2d) * Z1 - Math.Pow(K0, 2d) * Z0) * (Z1 - Z0) - 3d * (K1 * Z1 - K0 * Z0) - (Math.Pow(K1, 3d) * Z1 - Math.Pow(K0, 3d) * Z0) + 3d);
295 }

399 public override double PDF(double x)
400 {
401 // Validate parameters
402 if (_parametersValid == false)
403 ValidateParameters(Mu, Sigma, Min, Max, true);
404 //
405 if (x < Minimum || x > Maximum) return 0.0d;
406 Xi = (x - Mu) / Sigma;
407 alpha = (Min - Mu) / Sigma;
408 beta = (Max - Mu) / Sigma;
409 Z = Normal.StandardCDF(beta) - Normal.StandardCDF(alpha);
410 if (Z == 0) return Normal.StandardPDF(Xi);
411 return Normal.StandardPDF(Xi) / (Sigma * Z);
412 }
413
415 public override double CDF(double x)
416 {
417 // Validate parameters
418 if (_parametersValid == false)
419 ValidateParameters(Mu, Sigma, Min, Max, true);
420
421 if (x <= Minimum) return 0d;
422 if (x >= Maximum) return 1d;
423 Xi = (x - Mu) / Sigma;
424 alpha = (Min - Mu) / Sigma;
425 beta = (Max - Mu) / Sigma;
426 Z = Normal.StandardCDF(beta) - Normal.StandardCDF(alpha);
427 if (Z == 0) return Normal.StandardCDF(Xi);
428 return (Normal.StandardCDF(Xi) - Normal.StandardCDF(alpha)) / Z;
429 }
430
432 public override double InverseCDF(double probability)
433 {
434 // Validate probability
435 if (probability < 0.0d || probability > 1.0d)
436 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
437 if (probability == 0.0d) return Minimum;
438 if (probability == 1.0d) return Maximum;
439 // Validate parameters
440 if (_parametersValid == false)
441 ValidateParameters(Mu, Sigma, Min, Max, true);
442 alpha = (Min - Mu) / Sigma;
443 beta = (Max - Mu) / Sigma;
444 Z = Normal.StandardCDF(beta) - Normal.StandardCDF(alpha);
445 if (Z == 0) return Mu + Sigma * Normal.StandardZ(probability);
446 return Mu + Sigma * Normal.StandardZ(Normal.StandardCDF(alpha) + probability * Z);
447 }
