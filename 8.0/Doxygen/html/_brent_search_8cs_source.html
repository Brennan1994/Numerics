94 protected override void Optimize()
95 {
96 // Define variables
97 bool cancel = false;
98 double ax = LowerBound, bx = 0.5 * (UpperBound + LowerBound), cx = UpperBound;
99 // Golder ratio and a small number which protects against trying to achieve
100 // fractional accuracy for a minimum that happens to be exactly zero.
101 double CGOLD = 0.381966d;
102 double ZEPS = Tools.DoubleMachineEpsilon * 1.0e-3;
103 double a, b, d = 0.0, etemp, fu, fv, fw, fx;
104 double p, q, r, tol1, tol2, u, v, w, x, xm;
105 double e = 0.0;
106
107 a = (ax < cx ? ax : cx);
108 b = (ax > cx ? ax : cx);
109 x = w = v = bx;
110 fw = fv = fx = Evaluate(new double[] { x }, ref cancel);
111 for (int i = 1; i <= MaxIterations; i++)
112 {
113 xm = 0.5 * (a + b);
114 tol2 = 2.0 * (tol1 = RelativeTolerance * Math.Abs(x) + ZEPS);
115 // Test for done here
116 if (Math.Abs(x - xm) <= tol2 - 0.5d * (b - a))
117 {
118 UpdateStatus(OptimizationStatus.Success);
119 return;
120 }
121 // Construct a trial parabolic fit
122 if (Math.Abs(e) > tol1)
123 {
124 r = (x - w) * (fx - fv);
125 q = (x - v) * (fx - fw);
126 p = (x - v) * q - (x - w) * r;
127 q = 2.0 * (q - r);
128 if (q > 0.0) p = -p;
129 q = Math.Abs(q);
130 etemp = e;
131 e = d;
132 if (Math.Abs(p) >= Math.Abs(0.5 * q * etemp) || p <= q * (a - x) || p >= q * (b - x))
133 d = CGOLD * (e = (x >= xm ? a - x : b - x));
134 // The above conditions determine the acceptability of the parabolic fit.
135 // here we take the golden section step into the larger of the two segments.
136 else
137 {
138 d = p / q;
139 u = x + d;
140 if (u - a < tol2 || b - u < tol2)
141 d = Tools.Sign(tol1, xm - x);
142 }
143 }
144 else
145 {
146 d = CGOLD * (e = (x >= xm ? a - x : b - x));
147 }
148
149 u = (Math.Abs(d) >= tol1 ? x + d : x + Tools.Sign(tol1, d));
150 fu = Evaluate(new double[] { u }, ref cancel);
151 if (cancel == true) return;
152 // This is the one function evaluation per iteration.
153 if (fu <= fx)
154 {
155 if (u >= x) a = x; else b = x;
156 v = w;
157 w = x;
158 x = u;
159 fv = fw;
160 fw = fx;
161 fx = fu;
162 }
163 else
164 {
165 if (u < x) a = u; else b = u;
166 if (fu <= fw || w == x)
167 {
168 v = w;
169 w = u;
170 fv = fw;
171 fw = fu;
172 }
173 else if (fu <= fv || v == x || v == w)
174 {
175 v = u;
176 fv = fu;
177 }
178 }
179 // Done with housekeeping. Back for another iteration
180 }
181
182 // If we made it to here, the maximum iterations were reached.
183 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
184 }

191 public void Bracket(double s = 1E-2, double k = 2d)
192 {
193 double a = LowerBound, b = a + s;
194 double fa = ObjectiveFunction(new double[] { a });
195 double fb = ObjectiveFunction(new double[] { b });
196 double c, fc, temp;
197 if (fb > fa)
198 {
199 temp = a;
200 a = b;
201 b = temp;
202 temp = fa;
203 fa = fb;
204 fb = temp;
205 s *= -1;
206 }
207 while (true)
208 {
209 c = b + s;
210 fc = ObjectiveFunction(new double[] { c });
211 if (fc > fb) break;
212 a = b;
213 b = c;
214 fa = fb;
215 fb = fc;
216 }
217 if (a < c)
218 {
219 LowerBound = a;
220 UpperBound = c;
221 }
222 else
223 {
224 LowerBound = c;
225 UpperBound = a;
226 }
227
228 }
