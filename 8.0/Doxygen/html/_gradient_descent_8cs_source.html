77 public GradientDescent(Func<double[], double> objectiveFunction, int numberOfParameters,
78 IList<double> initialValues, IList<double> lowerBounds, IList<double> upperBounds, double alpha = 0.001,
79 Func<double[], double[]> gradient = null) : base(objectiveFunction, numberOfParameters)
80 {
81 // Check if the length of the initial, lower and upper bounds equal the number of parameters
82 if (initialValues.Count != numberOfParameters || lowerBounds.Count != numberOfParameters || upperBounds.Count != numberOfParameters)
83 {
84 throw new ArgumentOutOfRangeException(nameof(lowerBounds), "The initial values and lower and upper bounds must be the same length as the number of parameters.");
85 }
86 // Check if the initial values are between the lower and upper values
87 for (int j = 0; j < initialValues.Count; j++)
88 {
89 if (upperBounds[j] < lowerBounds[j])
90 {
91 throw new ArgumentOutOfRangeException(nameof(upperBounds), "The upper bound cannot be less than the lower bound.");
92 }
93 if (initialValues[j] < lowerBounds[j] || initialValues[j] > upperBounds[j])
94 {
95 throw new ArgumentOutOfRangeException(nameof(initialValues), "The initial values must be between the upper and lower bounds.");
96 }
97 }
98 InitialValues = initialValues.ToArray();
99 LowerBounds = lowerBounds.ToArray();
100 UpperBounds = upperBounds.ToArray();
101 Alpha = alpha;
102 Gradient = gradient;
103 }

131 protected override void Optimize()
132 {
133 int D = NumberOfParameters;
134 bool cancel = false;
135 var p0 = InitialValues.ToArray();
136 var g = new double[D];
137 double f0 = Evaluate(p0, ref cancel);
138 double f1;
139
140 while (Iterations < MaxIterations)
141 {
142 // Get gradient with respect to objective function
143 g = Gradient != null ? Gradient(p0) : NumericalDerivative.Gradient((x) => Evaluate(x, ref cancel), p0);
144 if (cancel) return;
145
146 // Update parameters
147 for (int i = 0; i < D; i++)
148 {
149 // Update parameter
150 p0[i] = p0[i] - Alpha * g[i];
151 // Make sure the parameter is within the bounds.
152 p0[i] = RepairParameter(p0[i], LowerBounds[i], UpperBounds[i]);
153 }
154
155 // Evaluate the objective function at the new point
156 f1 = Evaluate(p0, ref cancel);
157 if (cancel) return;
158
159 // Check convergence.
160 if (CheckConvergence(f0, f1))
161 {
162 UpdateStatus(OptimizationStatus.Success);
163 return;
164 }
165
166 f0 = f1;
167 Iterations += 1;
168 }
169
170 // If we made it to here, the maximum iterations were reached.
171 UpdateStatus(OptimizationStatus.MaximumIterationsReached);
172 }
173 }
