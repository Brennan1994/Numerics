378 public double[] ParametersFromLinearMoments(IList<double> moments)
379 {
380 double L1 = moments[0];
381 double L2 = moments[1];
382 double T3 = moments[2];
383
384 double E0 = 2.0466534;
385 double E1 = -3.6544371;
386 double E2 = 1.8396733;
387 double E3 = -0.20360244;
388 double F1 = -2.0182173;
389 double F2 = 1.2420401;
390 double F3 = -0.21741801;
391
392 double kappa = -T3 * (E0 + E1 * Math.Pow(T3, 2d) + E2 * Math.Pow(T3, 4d) + E3 * Math.Pow(T3, 6d)) / (1d + F1 * Math.Pow(T3, 2d) + F2 * Math.Pow(T3, 4d) + F3 * Math.Pow(T3, 6d));
393 double alpha = (L2 * kappa * Math.Exp(-(kappa * kappa) / 2d)) / (1d - 2 * Normal.StandardCDF(-kappa / Tools.Sqrt2));
394 double xi = L1 - alpha * (1.0d - Math.Exp(kappa * kappa / 2d)) / kappa;
395 return [xi, alpha, kappa];
396 }

399 public double[] LinearMomentsFromParameters(IList<double> parameters)
400 {
401 double xi = parameters[0];
402 double alpha = parameters[1];
403 double kappa = parameters[2];
404
405 double A0 = 4.8860251 * Math.Pow(10, -1);
406 double A1 = 4.4493076 * Math.Pow(10, -3);
407 double A2 = 8.8027039 * Math.Pow(10, -4);
408 double A3 = 1.1507084 * Math.Pow(10, -6);
409 double B1 = 6.4662924 * Math.Pow(10, -2);
410 double B2 = 3.3090406 * Math.Pow(10, -3);
411 double B3 = 7.4290680 * Math.Pow(10, -5);
412 double C0 = 1.8756590 * Math.Pow(10, -1);
413 double C1 = -2.5352147 * Math.Pow(10, -3);
414 double C2 = 2.6995102 * Math.Pow(10, -4);
415 double C3 = -1.8446680 * Math.Pow(10, -6);
416 double D1 = 8.2325617 * Math.Pow(10, -2);
417 double D2 = 4.2681448 * Math.Pow(10, -3);
418 double D3 = 1.1653690 * Math.Pow(10, -4);
419 double tau40 = 1.2260172 * Math.Pow(10, -1);
420
421 double L1 = xi + alpha * (1.0d - Math.Exp(kappa * kappa / 2d)) / kappa;
422 double L2 = (alpha / kappa) * Math.Exp(kappa * kappa / 2d) * (1d - 2d * Normal.StandardCDF(-kappa / Tools.Sqrt2));
423 double T3 = -kappa * (A0 + A1 * Math.Pow(kappa, 2d) + A2 * Math.Pow(kappa, 4d) + A3 * Math.Pow(kappa, 6d)) / (1d + B1 * Math.Pow(kappa, 2d) + B2 * Math.Pow(kappa, 4d) + B3 * Math.Pow(kappa, 6d));
424 double T4 = tau40 + Math.Pow(kappa, 2d) * (C0 + C1 * Math.Pow(kappa, 2d) + C2 * Math.Pow(kappa, 4d) + C3 * Math.Pow(kappa, 6d)) / (1d + D1 * Math.Pow(kappa, 2d) + D2 * Math.Pow(kappa, 4d) + D3 * Math.Pow(kappa, 6d));
425 return [L1, L2, T3, T4];
426 }

429 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
430 {
431 // Estimate initial values using the method of moments (a.k.a product moments).
432 var initialVals = new double[NumberOfParameters];
433 var lowerVals = new double[NumberOfParameters];
434 var upperVals = new double[NumberOfParameters];
435 // Get initial values
436 initialVals = ParametersFromLinearMoments(Statistics.LinearMoments(sample));
437 // Get bounds of location
438 if (initialVals[0] == 0d) initialVals[0] = Tools.DoubleMachineEpsilon;
439 lowerVals[0] = -Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
440 upperVals[0] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0])) + 1d));
441 // Get bounds of scale
442 lowerVals[1] = Tools.DoubleMachineEpsilon;
443 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[1]))) + 1d);
444 // Get bounds of shape
445 lowerVals[2] = -10d;
446 upperVals[2] = 10d;
447 // Correct initial value of kappa if necessary
448 if (initialVals[2] <= lowerVals[2] || initialVals[2] >= upperVals[2])
449 {
450 initialVals[2] = 0d;
451 }
452 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
453 }

456 public double[] MLE(IList<double> sample)
457 {
458 // Set constraints
459 var tuple = GetParameterConstraints(sample);
460 var Initials = tuple.Item1;
461 var Lowers = tuple.Item2;
462 var Uppers = tuple.Item3;
463
464 // Solve using Nelder-Mead (Downhill Simplex)
465 double logLH(double[] x)
466 {
467 var GLO = new GeneralizedNormal();
468 GLO.SetParameters(x);
469 return GLO.LogLikelihood(sample);
470 }
471 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
472 solver.ReportFailure = true;
473 solver.Maximize();
474 return solver.BestParameterSet.Values;
475 }

478 public override double PDF(double x)
479 {
480 if (_parametersValid == false)
481 ValidateParameters(Xi, Alpha, Kappa, true);
482 if (x < Minimum || x > Maximum) return 0.0d;
483 double y = (x - Xi) / Alpha;
484 if (Math.Abs(Kappa) > NearZero)
485 y = -Math.Log(1d - Kappa * y) / Kappa;
486 return 1d / Alpha * Math.Exp(Kappa * y - y * y / 2d) / Tools.Sqrt2PI;
487 }
488
490 public override double CDF(double x)
491 {
492 if (_parametersValid == false)
493 ValidateParameters(Xi, Alpha, Kappa, true);
494 if (x <= Minimum)
495 return 0d;
496 if (x >= Maximum)
497 return 1d;
498 double y = (x - Xi) / Alpha;
499 if (Math.Abs(Kappa) > NearZero)
500 y = -Math.Log(1d - Kappa * y) / Kappa;
501 return Normal.StandardCDF(y);
502 }

505 public override double InverseCDF(double probability)
506 {
507 // Validate probability
508 if (probability < 0.0d || probability > 1.0d)
509 throw new ArgumentOutOfRangeException("probability", "Probability must be between 0 and 1.");
510 if (probability == 0.0d)
511 return Minimum;
512 if (probability == 1.0d)
513 return Maximum;
514 // Validate parameters
515 if (_parametersValid == false)
516 ValidateParameters(Xi, Alpha, Kappa, true);
517 if (Math.Abs(Kappa) <= NearZero)
518 {
519 return Xi + Alpha * Normal.StandardZ(probability);
520 }
521 else
522 {
523 return Xi - Alpha / Kappa * (Math.Exp(-Kappa * Normal.StandardZ(probability)) - 1d);
524 }

561 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
562 {
563 if (probabilities.Count != NumberOfParameters)
564 {
565 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
566 }
567 // Get gradients
568 var dQp1 = QuantileGradient(probabilities[0]);
569 var dQp2 = QuantileGradient(probabilities[1]);
570 var dQp3 = QuantileGradient(probabilities[2]);
571 // Compute determinant
572 // |a b c|
573 // |d e f|
574 // |g h i|
575 // |A| = a(ei − fh) − b(di − fg) + c(dh − eg)
576 double a = dQp1[0];
577 double b = dQp1[1];
578 double c = dQp1[2];
579 double d = dQp2[0];
580 double e = dQp2[1];
581 double f = dQp2[2];
582 double g = dQp3[0];
583 double h = dQp3[1];
584 double i = dQp3[2];
585 determinant = a * (e * i - f * h) - b * (d * i - f * g) + c * (d * h - e * g);
586 // Return Jacobian
587 var jacobian = new double[,] { { a, b, c }, { d, e, f }, { g, h, i } };
588 return jacobian;
589 }
