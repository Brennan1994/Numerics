335 public Tuple<double[], double[], double[]> GetParameterConstraints(IList<double> sample)
336 {
337 var initialVals = new double[NumberOfParameters];
338 var lowerVals = new double[NumberOfParameters];
339 var upperVals = new double[NumberOfParameters];
340
341 // Get initial values
342 var moments = Statistics.ProductMoments(sample);
343 double minData = Statistics.Minimum(sample);
344 initialVals[0] = (sample.Count * minData - moments[0]) / (sample.Count - 1);
345 initialVals[1] = sample.Count * (moments[0] - minData) / (sample.Count - 1);
346
347 // Get bounds of location
348 if (initialVals[0] == 0d) initialVals[0] = Tools.DoubleMachineEpsilon;
349 lowerVals[0] = initialVals[0] - Math.Pow(10d, Math.Ceiling(Math.Log10(Math.Abs(initialVals[0]))));
350 upperVals[0] = minData;
351
352 // Get bounds of scale
353 lowerVals[1] = Tools.DoubleMachineEpsilon;
354 upperVals[1] = Math.Pow(10d, Math.Ceiling(Math.Log10(initialVals[1]) + 1d));
355
356 // Correct initial values if necessary
357 if (initialVals[0] <= lowerVals[0] || initialVals[0] >= upperVals[0])
358 {
359 initialVals[0] = Statistics.Mean([lowerVals[0], upperVals[0]]);
360 }
361 if (initialVals[1] <= lowerVals[1] || initialVals[1] >= upperVals[1])
362 {
363 initialVals[1] = Statistics.Mean([lowerVals[1], upperVals[1]]);
364 }
365 return new Tuple<double[], double[], double[]>(initialVals, lowerVals, upperVals);
366 }

369 public double[] MLE(IList<double> sample)
370 {
371 // Set constraints
372 var tuple = GetParameterConstraints(sample);
373 var Initials = tuple.Item1;
374 var Lowers = tuple.Item2;
375 var Uppers = tuple.Item3;
376
377 // Solve using Nelder-Mead (Downhill Simplex)
378 double logLH(double[] x)
379 {
380 var EXP = new Exponential();
381 EXP.SetParameters(x);
382 return EXP.LogLikelihood(sample);
383 }
384 var solver = new NelderMead(logLH, NumberOfParameters, Initials, Lowers, Uppers);
385 solver.ReportFailure = true;
386 solver.Maximize();
387 return solver.BestParameterSet.Values;
388 }

413 {
414 // Validate probability
415 if (probability < 0.0d || probability > 1.0d)

434 public double[,] ParameterCovariance(int sampleSize, ParameterEstimationMethod estimationMethod)
435 {
436 if (estimationMethod != ParameterEstimationMethod.MethodOfMoments &&
437 estimationMethod != ParameterEstimationMethod.MaximumLikelihood)
438 {
439 throw new NotImplementedException();
440 }
441 // Validate parameters
442 if (_parametersValid == false)
443 ValidateParameters([Xi, _alpha], true);
444
445 // Compute covariance
446 double a = Alpha;
447 var covar = new double[2, 2];
448 if (estimationMethod == ParameterEstimationMethod.MethodOfMoments)
449 {
450 covar[0, 0] = a * a / sampleSize; // location
451 covar[1, 1] = 2d * a * a / sampleSize; // scale
452 covar[0, 1] = -(a * a) / sampleSize; // location & scale
453 covar[1, 0] = covar[0, 1];
454 }
455 else if (estimationMethod == ParameterEstimationMethod.MaximumLikelihood)
456 {
457 covar[0, 0] = a * a / (sampleSize * (sampleSize - 1)); // location
458 covar[1, 1] = a * a / (sampleSize - 1); // scale
459 covar[0, 1] = -(a * a) / (sampleSize * (sampleSize - 1)); // location & scale
460 covar[1, 0] = covar[0, 1];
461 }
462 return covar;
463 }

466 public double QuantileVariance(double probability, int sampleSize, ParameterEstimationMethod estimationMethod)
467 {
468 var covar = ParameterCovariance(sampleSize, estimationMethod);
469 var grad = QuantileGradient(probability);
470 double varA = covar[0, 0];
471 double varB = covar[1, 1];
472 double covAB = covar[1, 0];
473 double dQx1 = grad[0];
474 double dQx2 = grad[1];
475 return Math.Pow(dQx1, 2d) * varA + Math.Pow(dQx2, 2d) * varB + 2d * dQx1 * dQx2 * covAB;
476 }

493 public double[,] QuantileJacobian(IList<double> probabilities, out double determinant)
494 {
495 if (probabilities.Count != NumberOfParameters)
496 {
497 throw new ArgumentOutOfRangeException(nameof(probabilities), "The number of probabilities must be the same length as the number of distribution parameters.");
498 }
499 // Get gradients
500 var dQp1 = QuantileGradient(probabilities[0]);
501 var dQp2 = QuantileGradient(probabilities[1]);
502 // Compute determinant
503 // |a b|
504 // |c d|
505 // |A| = ad − bc
506 double a = dQp1[0];
507 double b = dQp1[1];
508 double c = dQp2[0];
509 double d = dQp2[1];
510 determinant = a * d - b * c;
511 // Return Jacobian
512 var jacobian = new double[,] { { a, b }, { c, d } };
513 return jacobian;
514 }
